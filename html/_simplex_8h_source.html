13// todo: orthogonal projection to a subspace is a very general linalg operation,
14// and projection_contains() makes use of it. The operation that does this is
15// the pseudoinverse, which can be computed efficiently using a QR decomposition.
16// Probably the best way to do QR is with Householder transformations; the algorithm
17// can be found complete here: http://people.inf.ethz.ch/gander/papers/qrneu.pdf
18//
19// the above should probably be exposed as orthogonal_project(basis[], n, P) in
20// Orthogonal.h, and invoked here by projection_contains(). Also consider adding
21// subspace_projection(bases[], n) -> Mtx, which can amortize the QR bidness.
22//
23// right now we are running two LUP decompositions, and this is likely
24// inefficient/unstable.
25//
26// nb: this might not be useful in lower dimensions
27
28// idea:
29// for the full simplex, solving barycentric coords will give you an implicit
30// face-orthogonal basis: the weight of the vertex "missing" from a face must be 1
31// at the vertex, and 0 across the entire opposing faceâ€” thus that coordinate must
32// define isosurfaces parallel to the opposing face!
33// can we solve once and then inspect the solved coordinate values to figure out
34// sub-simplex containment?
35//
36// the trick is to do this for incomplete simplexes. what constraints can we impose
37// to make the linear system square again? (guess: certain dot products = 0)
38// > might also be possible using an O(N) process analogous to gram-schmidt
39//
40// also of concern: does this work for arbitrary sub-simplexes? i.e. do you
41// orthogonally project to an edge by coordinate clipping, or do you have to
42// re-solve?
43// > it might be. the trick is to express scaled barycentric coordinates in
44// terms of un-scaled coordinates.
45
46// todo: another possible tool is to project the simplex to a lower dimension
47// (by dropping coordinates) and barycentric-solving in that lower space.
48// this amounts to projecting the simplex to an axial (hyper-)plane or an axis.
49// the weightings of the verticies will not be different if you add back in
50// the missing coordinates, so the barycentric solution from this lower space
51// is still valid. (This also explains how keep the linear system square).
52// generally you want to project to the axial plane wherein the simplex
53// has the largest projected area. overall this might be more numerically robust.
54// note that you have to find the coordinates of the projected point `p`
55// in the axial plane; this is sort of a double-projection.
56// edit: actually, this doesn't make sense. this doesn't do an orthogonal projection

223 Rect<T,1> intersect(const Ray<T,N>& r) const {
224 if (n == N + 1) {
225 // where are barycentric coords equal to ray coords?
226 // Î£ w_i * p_i = o + s * v
227 // Î£ w_i = 1
228 // combined:
229 // Î£ (p_i, 1) * w_i = (o, 1) + s * (v, 0)
230 // M * w = B * (1, s)
231 Vec<T,N+1> m[N+1];
232 Vec<T,N+1> B[2] = {Vec<T,N+1>(r.origin, 1), Vec<T,N+1>(r.direction, 0)};
233 for (index_t i = 0; i < N + 1; i++) {
234 m[i] = Vec<T,N+1>(pts[i], 1);
235 }
236 // todo: handle degenerate simplexes
237 // put the solutions (p_i, k_i) for `w_i = p_i + k_i * s` into B:
238 if (linear_solve(m, 2, B)) {
239 // intersect all the ranges of s for which 0 <= w_i <= 1.
240 Rect<T,1> interval = Rect<T,1>::full;
241 for (index_t i = 0; i < N + 1 and interval.hi > interval.lo; ++i) {
242 // w_i = p_i + k_i * s
243 // solve for w_i = 0 and w_i = 1:
244 T p_i = B[0][i];
245 T k_i = B[1][i];
246 if (k_i != 0) {
247 T s0 = -p_i / k_i;
248 T s1 = (1 - p_i) / k_i;
249 interval &= Rect<T,1>::spanning_corners(s0, s1);
250 } else if (p_i < 0 or p_i > 1) {
251 // if p_i is outside (0, 1) then no value of `s` can solve w_i in
252 // (0, 1), because with k = 0 there is no dependency on s!
253 return Rect<T,1>::empty;
254 // (otherwise, all values of `s` solve w_i in (0, 1) for this w_i;
255 // no change to the solution interval).
256 }
257 }
258 return interval;
259 }
260 } else if (n == N) {
261 T s;
262 if (trace_simplex<T,N>(pts, r, nullptr, &s)) {
263 return Rect<T,1>(s, s);
264 }
265 }
266 // no hit
267 return Rect<T,1>();
268 }

281 bool is_facing(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
282 switch (n) {
283 case 0:
284 return false;
285
286 case 1: {
287 return true;
288 } break;
289
290 case 2: {
291 Vec<T,N> v = pts[1] - pts[0];
292 T d = v.dot(p - pts[0]) / v.mag2();
293 if (params) (*params)[0] = d;
294 return d >= 0 and d <= 1;
295 } break;
296
297 default: {
298 // simplex spans a subspace. orthogonally project p along its nullspace.
299 // if the simplex spans the domain, then nullspace() early exits and
300 // everything else works as normal.
301
302 // nullspace comes first (cause we won't solve for its coords),
303 // followed by the simplex basis.
304 const index_t n_bases = n - 1;
305 const index_t n_null = N - n_bases;
306 Vec<T,N> null_bases[N];
307 Vec<T,N>* const bases = null_bases + n_null;
308
309 // compute simplex bases
310 for (index_t i = 1; i < n; ++i) {
311 bases[i - 1] = pts[i] - pts[0];
312 }
313
314 // find the simplex's nullspace
315 nullspace(bases, n_bases, null_bases);
316
317 // write P as a linear combination of the simplex basis and
318 // its nullspace bases. (but don't bother actually solving for
319 // the nullpsace coords; we don't use them).
320 Vec<T,N> x = p - pts[0];
321 if (not linear_solve(null_bases, 1, &x, n_null)) return false;
322
323 // check that the coordinates of `p` are inside the simplex
324 T sum = 0;
325 for (index_t i = n_null; i < n_bases; ++i) {
326 sum += x[i];
327 if (x[i] < 0 or sum > 1) return false;
328 }
329
330 // provide surface parameters to caller, e.g. so they can perform
331 // the projection if they want to.
332 if (params) {
333 for (index_t i = 0; i < n_bases; ++i) {
334 (*params)[i] = x[i + n_null];
335 }
336 }
337
338 return true;
339 }
340 }
341
342 // statically unreachable, but some less-clever compilers may complain without:
343 return false;
344 }

352 T volume() const {
353 if (n < N + 1) { return 0; }
354 T k = 1;
355 Vec<T,N> vs[N];
356 for (index_t i = 1; i <= N; ++i) {
357 vs[i - 1] = pts[i] - pts[0];
358 k *= i;
359 }
360 T* m = vs[0].begin();
361 if constexpr (N == 2) {
362 return det2x2(m) / 2;
363 } else if constexpr (N == 3) {
364 return det3x3(m) / 6;
365 } else {
366 return det_destructive(m, N) / k;
367 }
368 }

379 T measure() const {
380 // special cases (easy to compute):
381 // line segment
382 if (n == 2) return (pts[1] - pts[0]).mag();
383 // 3D triangle
384 if constexpr (N == 3) {
385 if (n == 3) return (pts[1] - pts[0]).cross(pts[2] - pts[0]).mag() / 2;
386 }
387 // full volume
388 if (n == N + 1) return volume();
389 // otherwise, construct the basis of the simplex.
390 // we'll use it to compute the volume
391 Vec<T,N> bases[N];
392 int k = 1;
393 for (index_t i = 1; i < n; ++i) {
394 bases[i - 1] = pts[i] - pts[0];
395 k *= i;
396 }
397 T* m = bases[0].begin();
398 // volume is sqrt of the determinant of the gram matrix,
399 // divided by n factorial. i suspect the gram matrix could probably
400 // be computed more efficiently, because it's symmetric
401 T buf[N * N];
402 WrapperMatrix<T,0,0> b {buf, n, n};
403 WrapperMatrix<T,0,N,MatrixLayout::ROW_MAJOR> m0 {m, n};
404 WrapperMatrix<T,N,0,MatrixLayout::COL_MAJOR> m1 {m, n};
405 // (n x n) = (n x N) * (N x n)
406 mul(&b, m0, m1);
407 return std::sqrt(std::abs(det_destructive(buf, n))) / k;
408 }

422 bool contains(const Vec<T,N>& p) const {
423 if (n < N + 1) return false;
424 // solve for barycentric coordinates x:
425 // Î£ (1, v[i]) * x[i] = (1, p)
426
427 const index_t K = N + 1;
428 T m[K * K];
429 T x[K];
430 for (index_t i = 0; i < K; ++i) {
431 const T* v = pts[i].begin();
432 m[i * K] = 1;
433 std::copy(v, v + N, m + i * K + 1);
434 }
435 x[0] = 1;
436 std::copy(p.begin(), p.end(), x + 1);
437
438 // (no solution means a degenerate simplex, which has zero volume, so empty)
439 if (not linear_solve<T,false>((T*) m, K, 1, (T*) x, 1)) return false;
440 // skip solving for the sum of the weights â¤´ï¸Ž
441 // we don't need it, and we know it's 1.
442
443 T sum = 0;
444 for (index_t i = 1; i < K; ++i) {
445 sum += x[i];
446 if (x[i] < 0 or sum > 1) return false;
447 }
448 return true;
449 }

606template <typename T, index_t N>
607inline bool trace_simplex(const Vec<T,N> verts[N], const Ray<T,N>& ray, Vec<T,N-1>* uv, T* s) {
608 /* Solve for u,v coordinates along the edges radiating from `verts[0]`:
609 A + u(B-A) + v(C-A) = o + sV
610 u(B-A) + v(C-A) - sV = o - A
611 Mx = o - A
612 */
613
614 // todo: cramer's rule might be used in 3d for faster linear solve.
615 // (note: is unstable).
616
617 // compute bases
618 Vec<T,N> bases[N];
619 for (index_t i = 1; i < N; ++i) {
620 bases[i - 1] = verts[i] - verts[0];
621 }
622 bases[N - 1] = -ray.direction;
623
624 // linear solve
625 Vec<T,N> x = ray.origin - verts[0];
626 index_t skip = uv ? 0 : N - 1;
627 if (not linear_solve(bases, 1, &x, skip)) return false;
628
629 // inside simplex?
630 T sum = 0;
631 for (index_t i = 0; i < N - 1; ++i) {
632 if (x[i] < 0) return false;
633 sum += x[i];
634 }
635 if (sum > 1) return false;
636
637 // output result
638 *s = x[N];
639 if (uv) *uv = x.template resized<N-1>();
640 return true;
641}
