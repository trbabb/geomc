15// todo: distinguish project() and clip(). What we call project() is actually clip()
16// here. GJK also needs to use clip(). maybe template this?
17
18// todo: orthogonal projection to a subspace is a very general linalg operation,
19// and projection_contains() makes use of it. The operation that does this is
20// the pseudoinverse, which can be computed efficiently using a QR decomposition.
21// Probably the best way to do QR is with Householder transformations; the algorithm
22// can be found complete here: http://people.inf.ethz.ch/gander/papers/qrneu.pdf
23//
24// the above should probably be exposed as orthogonal_project(basis[], n, P) in
25// Orthogonal.h, and invoked here by projection_contains(). Also consider adding
26// subspace_projection(bases[], n) -> Mtx, which can amortize the QR bidness.
27//
28// right now we are running two LUP decompositions, and this is likely
29// inefficient/unstable.
30
31// idea:
32// for the full simplex, solving barycentric coords will give you an implicit
33// face-orthogonal basis: the weight of the vertex "missing" from a face must be 1
34// at the vertex, and 0 across the entire opposing faceâ€” thus that coordinate must
35// define isosurfaces parallel to the opposing face!
36// can we solve once and then inspect the solved coordinate values to figure out
37// sub-simplex containment?
38//
39// the trick is to do this for incomplete simplexes. what constraints can we impose
40// to make the linear system square again? (guess: certain dot products = 0)
41// > might also be possible using an O(N) process analogous to gram-schmidt
42//
43// also of concern: does this work for arbitrary sub-simplexes? i.e. do you
44// orthogonally project to an edge by coordinate clipping, or do you have to
45// re-solve?
46// > it might be. the trick is to express scaled barycentric coordinates in
47// terms of un-scaled coordinates.
48
49// todo: another possible tool is to project the simplex to a lower dimension
50// (by dropping coordinates) and barycentric-solving in that lower space.
51// this amounts to projecting the simplex to an axial (hyper-)plane or an axis.
52// the weightings of the verticies will not be different if you add back in
53// the missing coordinates, so the barycentric solution from this lower space
54// is still valid. (This also explains how keep the linear system square).
55// generally you want to project to the axial plane wherein the simplex
56// has the largest projected area. overall this might be more numerically robust.
57// note that you have to find the coordinates of the projected point `p`
58// in the axial plane; this is sort of a double-projection.
59// edit: actually, this doesn't make sense. this doesn't do an orthogonal projection
60
61// todo: handle degeneracy. GJK can make use of it.
62// todo: performance check old GJK vs new
63// perf issues:
64// - faster when we know it's always the origin we're testing
65// (can we transform the problem into one where we are checking the origin?)
66// - faster if we know to avoid checking the final face
67// (i.e. the one opposing the excluded vtx)
68// - faster if we indirect the verts
69// todo: performance check projection_contains() vs. project() == this

204 Rect<T,1> intersect(const Ray<T,N>& r) const {
205 if (n == N + 1) {
206 // where are barycentric coords equal to ray coords?
207 // Î£ w_i * p_i = o + s * v
208 // Î£ w_i = 1
209 // combined:
210 // Î£ (p_i, 1) * w_i = (o, 1) + s * (v, 0)
211 // M * w = B * (1, s)
212 Vec<T,N+1> m[N+1];
213 Vec<T,N+1> B[2] = {Vec<T,N+1>(r.origin, 1), Vec<T,N+1>(r.direction, 0)};
214 for (index_t i = 0; i < N + 1; i++) {
215 m[i] = Vec<T,N+1>(pts[i], 1);
216 }
217 // todo: handle degenerate simplexes
218 // put the solutions (p_i, k_i) for `w_i = p_i + k_i * s` into B:
219 if (linear_solve(m, 2, B)) {
220 // intersect all the ranges of s for which 0 <= w_i <= 1.
221 Rect<T,1> interval = Rect<T,1>::full;
222 for (index_t i = 0; i < N + 1 and interval.hi > interval.lo; ++i) {
223 // w_i = p_i + k_i * s
224 // solve for w_i = 0 and w_i = 1:
225 T p_i = B[0][i];
226 T k_i = B[1][i];
227 if (k_i != 0) {
228 T s0 = -p_i / k_i;
229 T s1 = (1 - p_i) / k_i;
230 interval &= Rect<T,1>::spanning_corners(s0, s1);
231 } else if (p_i < 0 or p_i > 1) {
232 // if p_i is outside (0, 1) then no value of `s` can solve w_i in
233 // (0, 1), because with k = 0 there is no dependency on s!
234 return Rect<T,1>::empty;
235 // (otherwise, all values of `s` solve w_i in (0, 1) for this w_i;
236 // no change to the solution interval).
237 }
238 }
239 return interval;
240 }
241 } else if (n == N) {
242 T s;
243 if (trace_simplex<T,N>(pts, r, nullptr, &s)) {
244 return Rect<T,1>(s, s);
245 }
246 }
247 // no hit
248 return Rect<T,1>();
249 }

262 bool projection_contains(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
263 switch (n) {
264 case 0:
265 return false;
266
267 case 1: {
268 return true;
269 } break;
270
271 case 2: {
272 Vec<T,N> v = pts[1] - pts[0];
273 T d = v.dot(p - pts[0]) / v.mag2();
274 if (params) (*params)[0] = d;
275 return d >= 0 and d <= 1;
276 } break;
277
278 default: {
279 // simplex spans a subspace. orthogonally project p along its nullspace.
280 // if the simplex spans the domain, then nullspace() early exits and
281 // everything else works as normal.
282
283 // nullspace comes first (cause we won't solve for its coords),
284 // followed by the simplex basis.
285 const index_t n_bases = n - 1;
286 const index_t n_null = N - n_bases;
287 Vec<T,N> null_bases[N];
288 Vec<T,N>* const bases = null_bases + n_null;
289
290 // compute simplex bases
291 for (index_t i = 1; i < n; ++i) {
292 bases[i - 1] = pts[i] - pts[0];
293 }
294
295 // find the simplex's nullspace
296 nullspace(bases, n_bases, null_bases);
297
298 // write P as a linear combination of the simplex basis and
299 // its nullspace bases. (but don't bother actually solving for
300 // the nullpsace coords; we don't use them).
301 Vec<T,N> x = p - pts[0];
302 if (not linear_solve(null_bases, 1, &x, n_null)) return false;
303
304 // check that the coordinates of `p` are inside the simplex
305 T sum = 0;
306 for (index_t i = n_null; i < n_bases; ++i) {
307 sum += x[i];
308 if (x[i] < 0 or sum > 1) return false;
309 }
310
311 // provide surface parameters to caller, e.g. so they can perform
312 // the projection if they want to.
313 if (params) {
314 for (index_t i = 0; i < n_bases; ++i) {
315 (*params)[i] = x[i + n_null];
316 }
317 }
318
319 return true;
320 }
321 }
322
323 // statically unreachable, but some less-clever compilers may complain without:
324 return false;
325 }

333 T volume() const {
334 if (n < N + 1) { return 0; }
335 T k = 1;
336 Vec<T,N> vs[N];
337 for (index_t i = 1; i <= N; ++i) {
338 vs[i - 1] = pts[i] - pts[0];
339 k *= i;
340 }
341 return det_destructive(vs[0].begin(), N) / k;
342 }

356 bool contains(const Vec<T,N>& p) const {
357 if (n < N + 1) return false;
358 // todo: test performance + stability over projection_contains
359
360 // solve for barycentric coordinates x:
361 // Î£ (1, v[i]) * x[i] = (1, p)
362
363 const index_t K = N + 1;
364 T m[K * K];
365 T x[K];
366 for (index_t i = 0; i < K; ++i) {
367 const T* v = pts[i].begin();
368 m[i * K] = 1;
369 std::copy(v, v + N, m + i * K + 1);
370 }
371 x[0] = 1;
372 std::copy(p.begin(), p.end(), x + 1);
373
374 // (no solution means a degenerate simplex, which has zero volume, so empty)
375 if (not linear_solve<T,false>((T*) m, K, 1, (T*) x, 1)) return false;
376 // skip solving for the sum of the weights â¤´ï¸Ž
377 // we don't need it, and we know it's 1.
378
379 T sum = 0;
380 for (index_t i = 1; i < K; ++i) {
381 sum += x[i];
382 if (x[i] < 0 or sum > 1) return false;
383 }
384 return true;
385 }

420 Vec<T,N> project(const Vec<T,N>& p, Simplex<T,N>* onto=nullptr) const {
421 Vec<T,N> buffer[N];
422 Simplex<T,N> s;
423
424 // why the recursion? why not project as in projection_contains(),
425 // obtaining surface parameters, and then clip them?
426 // A: Because in general, the subspace of the simplex will not
427 // be orthogonal, so clipping the coordinates to the unit triangle
428 // will project the point non-orthogonally along the bases of
429 // that subspace. Instead we have to first find the sub-simplex
430 // which is closest, and then project orthogonally down its nullspace.
431
432 // initialize the nullspace, if necessary
433 if (n - 1 < N) {
434 Vec<T,N>* const basis = buffer + N - n + 1;
435 Vec<T,N>* const null_basis = buffer;
436 for (index_t i = 1; i < n; ++i) {
437 basis[i - 1] = pts[i] - pts[0];
438 }
439 nullspace(basis, n - 1, null_basis);
440 }
441
442 // this puts the simplex which faces `p` into `s`, and
443 // leaves `buffer` containing its nullspace and spanning space.
444 this->_find_nearest_face(p, buffer, false, Vec<T,N>(), &s);
445
446 // choose the smaller basis to project on
447 index_t n_bases = s.n - 1;
448 index_t n_null = N - n_bases;
449 index_t proj_dims;
450 Vec<T,N>* proj_basis;
451 Vec<T,N> out;
452 bool proj_to_null;
453 if (n_bases < n_null) {
454 proj_dims = n_bases;
455 proj_basis = buffer + n_null;
456 proj_to_null = false;
457 } else {
458 proj_dims = n_null;
459 proj_basis = buffer;
460 proj_to_null = true;
461 }
462
463 // project to the relevant subspace
464 if (proj_dims > 0) {
465 if (N > 3 and proj_dims > 1 and (N - n > 0 or not proj_to_null)) {
466 // if we're projecting to the nullspace, it's already orthogonal,
467 // unless the nullspace started with > 1 dimension (in which case
468 // it was made by `nullspace` above; orthogonalize it now).
469 // this will never happen in 3D because there is always a projection
470 // space of dim <= 1 to choose.
471 orthogonalize(proj_basis, proj_dims);
472 }
473 Vec<T,N> p_in_basis = p - s.pts[0];
474 for (index_t i = 0; i < proj_dims; ++i) {
475 out += p_in_basis.project_on(proj_basis[i]);
476 }
477 }
478
479 // offset the subspace projection so it lies on the simplex surface
480 if (proj_to_null) {
481 // `out` is the direction away from the simplex toward `p`.
482 // travel backwards in that direction from `p` to get to the simplex.
483 out = p - out;
484 } else {
485 // `out` is the direction from s's corner to the projected point.
486 // add s's corner to get the absolute point.
487 out += s.pts[0];
488 }
489
490 if (onto) *onto = s;
491
492 return out;
493 }

505 Vec<T,N> convex_support(Vec<T,N> d) const {
506 T best = pts[0].dot(d);
507 index_t k = 0;
508 for (index_t i = 1; i < n; ++i) {
509 T a = pts[i].dot(d);
510 if (a > best) {
511 best = a;
512 k = i;
513 }
514 }
515 return pts[k];
516 }

518private:
519
520 // `all_bases` shall be laid out like:
521 // [normal] [parent null basis] [spanning basis]
522 // where `normal` remains to be computed.
523 // in the base case (not is_sub), all nullspace axes are provided.
524 bool _find_nearest_face(
525 Vec<T,N> p,
526 Vec<T,N> all_bases[],
527 bool is_sub,
528 Vec<T,N> inward_basis,
529 Simplex<T,N>* onto) const
530 {
531 if (n == 1) {
532 if ((p - pts[0]).dot(inward_basis) > 0) return false;
533 // projection onto a single point is trivial
534 // note: we leave `all_bases` in a garbage state,
535 // but we don't use it, since we don't need it!
536 *onto = *this;
537 return true;
538 } else {
539 // orthogonally project p onto this simplex along its nullspace.
540 const index_t n_null = N - n + 1;
541 Vec<T,N>* const bases = all_bases + n_null;
542 Vec<T,N>* const null_bases = all_bases + 1;
543
544 // compute simplex's basis
545 for (index_t i = 1; i < n; ++i) {
546 bases[i - 1] = pts[i] - pts[0];
547 }
548
549 // complete the simplex's nullspace
550 Vec<T,N> normal;
551 if (n_null != 0) {
552 if (is_sub) {
553 // newest nullspace axis not provided; must be computed.
554 all_bases[0] = orthogonal(null_bases); // <- also contains `bases`!
555
556 // `normal` is the vector pointing to `p` along the new axis
557 // (which will be orthogonal both to this simplex, and
558 // to the nullspace of the parent simplex).
559 normal = (p - pts[0]).project_on(all_bases[0]);
560
561 // might `p` be on the "inside" of the parent simplex?
562 if (normal.dot(inward_basis) > 0) {
563 // `p` falls "inward" of this face. `p` therefore does not project
564 // to this simplex; instead it projects to some other simplex,
565 // possibly a parent of this, which *does* face (or envelop) `p`.
566 return false;
567 }
568 }
569 // since there is at least one nullspace axis,
570 // move it aside for the sub-simplex searching code coming below,
571 // which will prepend its own normal.
572 bases[0] = all_bases[0];
573 }
574
575 // does `p` project to some sub-simplex of ours?
576 // try each sub-simplex.
577 for (index_t i = 0; i < n; ++i) {
578 Simplex<T,N> sub = this->exclude(i);
579 if (sub._find_nearest_face(p, all_bases, true, pts[i] - sub[0], onto)) {
580 // some sub-simplex of us both faces `p` and contains
581 // its projection. `all_bases` and `onto` now reflect that sub-simplex.
582 return true;
583 }
584 }
585
586 // restore the state of `all_bases`
587 // nobody messed with the normal because it was in their null space.
588 // and the simplex which just filled in its basis was the one that
589 // excluded vertex `n - 1`.
590 all_bases[0] = bases[0];
591 bases[0] = pts[n - 1] - pts[0];
592
593 // no sub-simplex claimed `p` was "beyond" it and took the projection.
594 // therefore, this is the simplex onto which `p` projects.
595 *onto = *this;
596 return true;
597 }
598 }

616template <typename T, index_t N>
617bool trace_simplex(const Vec<T,N> verts[N], const Ray<T,N>& ray, Vec<T,N-1>* uv, T* s) {
618 /* Solve for u,v coordinates along the edges radiating from `verts[0]`:
619 A + u(B-A) + v(C-A) = o + sV
620 u(B-A) + v(C-A) - sV = o - A
621 Mx = o - A
622 */
623
624 // todo: cramer's rule might be used in 3d for faster linear solve.
625 // (note: is unstable).
626
627 // compute bases
628 Vec<T,N> bases[N];
629 for (index_t i = 1; i < N; ++i) {
630 bases[i - 1] = verts[i] - verts[0];
631 }
632 bases[N - 1] = -ray.direction;
633
634 // linear solve
635 Vec<T,N> x = ray.origin - verts[0];
636 index_t skip = uv ? 0 : N - 1;
637 if (not linear_solve(bases, 1, &x, skip)) return false;
638
639 // inside simplex?
640 T sum = 0;
641 for (index_t i = 0; i < N - 1; ++i) {
642 if (x[i] < 0) return false;
643 sum += x[i];
644 }
645 if (sum > 1) return false;
646
647 // output result
648 *s = x[N];
649 if (uv) *uv = x.template resized<N-1>();
650 return true;
651}
