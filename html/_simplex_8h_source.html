120  bool projection_contains(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
121  switch (n) {
122  case 0:
123  return false;
124 
125  case 1: {
126  return true;
127  } break;
128 
129  case 2: {
130  Vec<T,N> v = pts[1] - pts[0];
131  T d = v.dot(p - pts[0]) / v.mag();
132  if (params) (*params)[0] = d;
133  return d > 0 and d < 1;
134  } break;
135 
136  default: {
137  // simplex spans a subspace. orthogonally project p along its nullspace.
138  // if the simplex spans the domain, then nullspace() early exits and
139  // everything else works as normal.
140 
141  // nullspace comes first (cause we won't solve for its coords),
142  // followed by the simplex basis.
143  const index_t n_bases = n - 1;
144  const index_t n_null = N - n + 1;
145  Vec<T,N> null_bases[N];
146  Vec<T,N>* const bases = null_bases + n_null;
147 
148  // compute simplex bases
149  for (index_t i = 1; i < n; ++i) {
150  bases[i - 1] = pts[i] - pts[0];
151  }
152 
153  // find the simplex's nullspace
154  nullspace(bases, n_bases, null_bases);
155 
156  // write P as a linear combination of the simplex basis and
157  // its nullspace bases. (but don't bother actually solving for
158  // the nullpsace coords; we don't use them).
159  Vec<T,N> x;
160  if (not linearSolve(bases, &x, p - pts[0], n_null)) return false;
161 
162  // check that the coordinates of `p` are inside the simplex
163  T sum = 0;
164  for (index_t i = n_null; i < N; ++i) {
165  if (x[i] < 0) return false;
166  sum += x[i];
167  }
168 
169  // provide surface parameters to caller, e.g. so they can perform
170  // the projection if they want to.
171  if (params) {
172  for (index_t i = 0; i < n_bases; ++i) {
173  (*params)[i] = x[i + n_null];
174  }
175  }
176 
177  return (sum <= 1);
178  }
179  }
180 
181  // for pickier compilers:
182  return false;
183  }

223  Vec<T,N> project(const Vec<T,N>& p, Simplex<T,N>* onto=nullptr) const {
224  Vec<T,N> buffer[N];
225  Simplex<T,N> s;
226 
227  // initialize the nullspace, if necessary
228  if (n - 1 < N) {
229  Vec<T,N>* const basis = buffer + N - n + 1;
230  Vec<T,N>* const null_basis = buffer;
231  for (index_t i = 1; i < n; ++i) {
232  basis[i - 1] = pts[i] - pts[0];
233  }
234  nullspace(basis, n - 1, null_basis);
235  }
236 
237  // this puts the simplex which faces `p` into `s`, and
238  // leaves `buffer` containing its nullspace and spanning space.
239  this->_find_nearest(p, buffer, false, Vec<T,N>::zeros, &s);
240 
241  // choose the smaller basis to project on
242  index_t n_bases = s.n - 1;
243  index_t n_null = N - n_bases;
244  index_t proj_dims;
245  Vec<T,N>* proj_basis;
246  bool proj_to_null;
247  if (n_bases < n_null) {
248  proj_dims = n_bases;
249  proj_basis = buffer + n_null;
250  proj_to_null = false;
251  } else {
252  proj_dims = n_null;
253  proj_basis = buffer;
254  proj_to_null = true;
255  }
256 
257  // project to the relevant subspace
258  Vec<T,N> out;
259  if (proj_dims > 0) {
260  if (N > 3 /*and (not proj_to_null or N - n > 0)*/ and proj_dims > 1) {
261  // if we're projecting to the nullspace, it's already orthogonal,
262  // unless the nullspace started with > 1 dimension (in which case
263  // it was made by `nullspace` above; orthogonalize it now).
264  // this will never happen in 3D because there is always a projection
265  // space of dim <= 1 to choose.
266  orthogonalize(proj_basis, proj_dims);
267  }
268  Vec<T,N> p_in_basis = p - s.pts[0];
269  for (index_t i = 0; i < proj_dims; ++i) {
270  out += p_in_basis.projectOn(proj_basis[i]);
271  }
272  }
273 
274  // offset the subspace projection so it lies on the simplex surface
275  if (proj_to_null) {
276  // `out` is the direction away from the simplex toward `p`.
277  // travel backwards in that direction from `p` to get to the simplex.
278  out = p - out;
279  } else {
280  // `out` is the direction from s's corner to the projected point.
281  // add s's corner to get the absolute point.
282  out += s.pts[0];
283  }
284 
285  if (onto) *onto = s;
286 
287  return out;
288  }

291  Vec<T,N> convexSupport(Vec<T,N> d) const {
292  T best = pts[0].dot(d);
293  index_t k = 0;
294  for (index_t i = 1; i < n; ++i) {
295  T a = pts[i].dot(d);
296  if (a > best) k = i;
297  }
298  return pts[k];
299  }

314  // `all_bases` shall be laid out like:
315  // [normal] [parent null basis] [spanning basis]
316  // where `normal` remains to be computed.
317  // in the base case (not is_sub), all nullspace axes are provided.
318  bool _find_nearest(
319  Vec<T,N> p,
320  Vec<T,N> all_bases[],
321  bool is_sub,
322  Vec<T,N> missing_basis,
323  Simplex<T,N>* onto) const {
324  if (n == 1) {
325  // projection onto a single point is trivial
326  // note: we leave `all_bases` in a garbage state,
327  // but we don't use it, since we don't need it!
328  *onto = *this;
329  return true;
330  } else {
331  // orthogonally project p onto this simplex along its nullspace.
332  const index_t n_null = N - n + 1;
333  Vec<T,N>* const bases = all_bases + n_null;
334  Vec<T,N>* const null_bases = all_bases + 1;
335 
336  // compute simplex's basis
337  for (index_t i = 1; i < n; ++i) {
338  bases[i - 1] = pts[i] - pts[0];
339  }
340 
341  // complete the simplex's nullspace
342  Vec<T,N> normal = all_bases[0];
343  if (n_null != 0) {
344  if (is_sub) {
345  // newest nullspace axis not provided; must be computed.
346  normal = orthogonal(null_bases); // <- also contains `bases`!
347 
348  // `normal` is the vector pointing to `p` along the new axis
349  // (which will be orthogonal both to this simplex, and
350  // to the nullspace of the parent simplex).
351  Vec<T,N> d = (p - pts[0]).projectOn(normal);
352 
353  // might `p` be on the "inside" of the parent simplex?
354  if (d.dot(missing_basis) > 0) {
355  // `p` falls "inward" of the boundary space spanned
356  // by this simplex. `p` therefore does not project to this
357  // simplex; instead it projects to some other simplex,
358  // possibly a parent of this, which *does* face (or envelop) `p`.
359  return false;
360  }
361  }
362  // p is "outward" of us, so the nearest simplex is
363  // either us, or our child. Add `normal` to our
364  // nullspace for our children to use.
365  bases[0] = normal;
366  }
367 
368  // all_bases is now:
369  // [empty space | null axis]? [parent null space]* [normal]? [remaining bases]*
370 
371  // does `p` project to some sub-simplex of ours?
372  // try each sub-simplex.
373  for (index_t i = 0; i < n; ++i) {
374  Simplex<T,N> sub = this->exclude(i);
375  if (sub._find_nearest(p, all_bases, true, pts[i] - sub[0], onto)) {
376  // some sub-simplex of us both faces `p` and contains its projection.
377  // all_bases is now: [null space] [basis space]
378  return true;
379  }
380  }
381 
382  // prepend the normal to our nullspace
383  if (n_null != 0) all_bases[0] = normal;
384  // this was clobbered by the normal. restore the missing axis,
385  // which is the most recent vertex we excluded.
386  bases[0] = pts[n - 1] - pts[0];
387 
388  // all_bases is now: [null space] [basis space]
389 
390  // no sub-simplex claimed `p` was "beyond" it and took the projection.
391  // therefore, this is the simplex onto which `p` projects.
392  *onto = *this;
393  return true;
394  }
395  }
