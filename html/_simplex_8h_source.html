26// inefficient/unstable.

30// idea:
31// for the full simplex, solving barycentric coords will give you an implicit
32// face-orthogonal basis: the weight of the vertex "missing" from a face must be 1
33// at the vertex, and 0 across the entire opposing face— thus that coordinate must
34// define isosurfaces parallel to the opposing face!
35// can we solve once and then inspect the solved coordinate values to figure out
36// sub-simplex containment?

38// the trick is to do this for incomplete simplexes. what constraints can we impose

130 static constexpr Simplex regular_simplex(index_t n=N+1) {
131 Simplex s;
132 Vec<T,N> c = s.pts[0] = {};
133 n = std::min(n, N + 1);
134 for (index_t i = 1; i < n; ++i) {
135 // compute the barycenter of the "base" of the simplex
136 Vec<T,N> c_i = c / i;
137 // remaining coordinate is raised above the center of the base by `h`
138 T h = std::sqrt(1 - c_i.mag2());
139 s.pts[i] = {c_i, h};
140 c += s.pts[i];
141 }
142 s.n = n;
143 return s;
144 }

252 Rect<T,1> intersect(const Ray<T,N>& r) const {
253 if (n == N + 1) {
254 // where are barycentric coords equal to ray coords?
255 // Σ w_i * p_i = o + s * v
256 // Σ w_i = 1
257 // combined:
258 // Σ (p_i, 1) * w_i = (o, 1) + s * (v, 0)
259 // M * w = B * (1, s)
260 Vec<T,N+1> m[N+1];
261 Vec<T,N+1> B[2] = {Vec<T,N+1>(r.origin, 1), Vec<T,N+1>(r.direction, 0)};
262 for (index_t i = 0; i < N + 1; i++) {
263 m[i] = Vec<T,N+1>(pts[i], 1);
264 }
265 // todo: handle degenerate simplexes
266 // put the solutions (p_i, k_i) for `w_i = p_i + k_i * s` into B:
267 if (linear_solve(m, 2, B)) {
268 // intersect all the ranges of s for which 0 <= w_i <= 1.
269 Rect<T,1> interval = Rect<T,1>::full;
270 for (index_t i = 0; i < N + 1 and interval.hi > interval.lo; ++i) {
271 // w_i = p_i + k_i * s
272 // solve for w_i = 0 and w_i = 1:
273 T p_i = B[0][i];
274 T k_i = B[1][i];
275 if (k_i != 0) {
276 T s0 = -p_i / k_i;
277 T s1 = (1 - p_i) / k_i;
278 interval &= Rect<T,1>::spanning_corners(s0, s1);
279 } else if (p_i < 0 or p_i > 1) {
280 // if p_i is outside (0, 1) then no value of `s` can solve w_i in
281 // (0, 1), because with k = 0 there is no dependency on s!
282 return Rect<T,1>::empty;
283 // (otherwise, all values of `s` solve w_i in (0, 1) for this w_i;
284 // no change to the solution interval).
285 }
286 }
287 return interval;
288 }
289 } else if (n == N) {
290 T s;
291 if (trace_simplex<T,N>(pts, r, nullptr, &s)) {
292 return Rect<T,1>(s, s);
293 }
294 }
295 // no hit
296 return Rect<T,1>();
297 }

310 bool is_facing(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
311 switch (n) {
312 case 0:
313 return false;
314
315 case 1: {
316 return true;
317 } break;
318
319 case 2: {
320 Vec<T,N> v = pts[1] - pts[0];
321 T d = v.dot(p - pts[0]) / v.mag2();
322 if (params) (*params)[0] = d;
323 return d >= 0 and d <= 1;
324 } break;
325
326 default: {
327 // simplex spans a subspace. orthogonally project p along its nullspace.
328 // if the simplex spans the domain, then nullspace() early exits and
329 // everything else works as normal.
330
331 // nullspace comes first (cause we won't solve for its coords),
332 // followed by the simplex basis.
333 const index_t n_bases = n - 1;
334 const index_t n_null = N - n_bases;
335 Vec<T,N> null_bases[N];
336 Vec<T,N>* const bases = null_bases + n_null;
337
338 // compute simplex bases
339 for (index_t i = 1; i < n; ++i) {
340 bases[i - 1] = pts[i] - pts[0];
341 }
342
343 // find the simplex's nullspace
344 nullspace(bases, n_bases, null_bases);
345
346 // write P as a linear combination of the simplex basis and
347 // its nullspace bases. (but don't bother actually solving for
348 // the nullpsace coords; we don't use them).
349 Vec<T,N> x = p - pts[0];
350 if (not linear_solve(null_bases, 1, &x, n_null)) return false;
351
352 // check that the coordinates of `p` are inside the simplex
353 T sum = 0;
354 for (index_t i = n_null; i < n_bases; ++i) {
355 sum += x[i];
356 if (x[i] < 0 or sum > 1) return false;
357 }
358
359 // provide surface parameters to caller, e.g. so they can perform
360 // the projection if they want to.
361 if (params) {
362 for (index_t i = 0; i < n_bases; ++i) {
363 (*params)[i] = x[i + n_null];
364 }
365 }
366
367 return true;
368 }
369 }
370
371 // statically unreachable, but some less-clever compilers may complain without:
372 return false;
373 }

381 T volume() const {
382 if (n < N + 1) { return 0; }
383 T k = 1;
384 Vec<T,N> vs[N];
385 for (index_t i = 1; i <= N; ++i) {
386 vs[i - 1] = pts[i] - pts[0];
387 k *= i;
388 }
389 T* m = vs[0].begin();
390 if constexpr (N == 2) {
391 return det2x2(m) / 2;
392 } else if constexpr (N == 3) {
393 return det3x3(m) / 6;
394 } else {
395 return det_destructive(m, N) / k;
396 }
397 }

408 T measure() const {
409 // special cases (easy to compute):
410 // line segment
411 if (n == 2) return (pts[1] - pts[0]).mag();
412 // 3D triangle
413 if constexpr (N == 3) {
414 if (n == 3) return (pts[1] - pts[0]).cross(pts[2] - pts[0]).mag() / 2;
415 }
416 // full volume
417 if (n == N + 1) return volume();
418 // otherwise, construct the basis of the simplex.
419 // we'll use it to compute the volume
420 Vec<T,N> bases[N];
421 int k = 1;
422 for (index_t i = 1; i < n; ++i) {
423 bases[i - 1] = pts[i] - pts[0];
424 k *= i;
425 }
426 T* m = bases[0].begin();
427 // volume is sqrt of the determinant of the gram matrix,
428 // divided by n factorial. i suspect the gram matrix could probably
429 // be computed more efficiently, because it's symmetric
430 T buf[N * N];
431 WrapperMatrix<T,0,0> b {buf, n, n};
432 WrapperMatrix<T,0,N,MatrixLayout::ROW_MAJOR> m0 {m, n};
433 WrapperMatrix<T,N,0,MatrixLayout::COL_MAJOR> m1 {m, n};
434 // (n x n) = (n x N) * (N x n)
435 mul(&b, m0, m1);
436 return std::sqrt(std::abs(det_destructive(buf, n))) / k;
437 }

451 bool contains(const Vec<T,N>& p) const {
452 if (n < N + 1) return false;
453 // solve for barycentric coordinates x:
454 // Σ (1, v[i]) * x[i] = (1, p)
455
456 const index_t K = N + 1;
457 T m[K * K];
458 T x[K];
459 for (index_t i = 0; i < K; ++i) {
460 const T* v = pts[i].begin();
461 m[i * K] = 1;
462 std::copy(v, v + N, m + i * K + 1);
463 }
464 x[0] = 1;
465 std::copy(p.begin(), p.end(), x + 1);
466
467 // (no solution means a degenerate simplex, which has zero volume, so empty)
468 if (not linear_solve<T,false>((T*) m, K, 1, (T*) x, 1)) return false;
469 // skip solving for the sum of the weights ⤴︎
470 // we don't need it, and we know it's 1.
471
472 T sum = 0;
473 for (index_t i = 1; i < K; ++i) {
474 sum += x[i];
475 if (x[i] < 0 or sum > 1) return false;
476 }
477 return true;
478 }

551 Vec<T,N+1> unmap(const Vec<T,N>& p) const {
552 // todo: account for simplexes with fewer than N+1 vertices
553 if (n < N + 1) return {};
554 // solve for barycentric coordinates x:
555 // Σ (1, v[i]) * x[i] = (1, p)
556
557 const index_t K = N + 1;
558 T m[K * K];
559 T x[K];
560 for (index_t i = 0; i < K; ++i) {

562 m[i * K] = 1;
563 std::copy(v, v + N, m + i * K + 1);

565 x[0] = 1;
566 std::copy(p.begin(), p.end(), x + 1);
567

669template <typename T, index_t N>
670inline bool trace_simplex(const Vec<T,N> verts[N], const Ray<T,N>& ray, Vec<T,N-1>* uv, T* s) {
671 /* Solve for u,v coordinates along the edges radiating from `verts[0]`:
672 A + u(B-A) + v(C-A) = o + sV
673 u(B-A) + v(C-A) - sV = o - A
674 Mx = o - A
675 */
676
677 // todo: cramer's rule might be used in 3d for faster linear solve.
678 // (note: is unstable).
679
680 // compute bases
681 Vec<T,N> bases[N];
682 for (index_t i = 1; i < N; ++i) {
683 bases[i - 1] = verts[i] - verts[0];
684 }
685 bases[N - 1] = -ray.direction;
686
687 // linear solve
688 Vec<T,N> x = ray.origin - verts[0];
689 index_t skip = uv ? 0 : N - 1;
690 if (not linear_solve(bases, 1, &x, skip)) return false;
691
692 // inside simplex?
693 T sum = 0;
694 for (index_t i = 0; i < N - 1; ++i) {
695 if (x[i] < 0) return false;
696 sum += x[i];
697 }
698 if (sum > 1) return false;
699
700 // output result
701 *s = x[N];
702 if (uv) *uv = x.template resized<N-1>();
703 return true;
704}
