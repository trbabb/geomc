96  bool projection_contains(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
97  switch (n) {
98  case 0:
99  return false;
100 
101  case 1: {
102  return pts[0] == p;
103  } break;
104 
105  case 2: {
106  Vec<T,N> v = pts[1] - pts[0];
107  T d = v.dot(p - pts[0]) / v.mag();
108  if (params) (*params)[0] = d;
109  return d > 0 and d < 1;
110  } break;
111 
112  default: {
113  // simplex spans a subspace. orthogonally project p along its nullspace.
114  // if the simplex spans the domain, then nullspace() early exits and
115  // everything else works as normal.
116 
117  // nullspace comes first (cause we won't solve for its coords),
118  // followed by the simplex basis.
119  const index_t n_bases = n - 1;
120  const index_t n_null = N - n + 1;
121  Vec<T,N> null_bases[N];
122  Vec<T,N>* const bases = null_bases + n_null;
123 
124  // compute simplex bases
125  for (index_t i = 1; i < n; ++i) {
126  bases[i - 1] = pts[i] - pts[0];
127  }
128 
129  // find the simplex's nullspace
130  nullspace(bases, n_bases, null_bases);
131 
132  // write P as a linear combination of the simplex basis and
133  // its nullspace bases. (but don't bother actually solving for
134  // the nullpsace coords; we don't use them).
135  Vec<T,N> x;
136  if (not linearSolve(bases, &x, p - pts[0], n_null)) return false;
137 
138  // provide surface parameters to caller, e.g. so they can perform
139  // the projection if they want to.
140  if (params) {
141  *params = x;
142  }
143 
144  // check that the coordinates of `p` are inside the simplex
145  T sum = 0;
146  for (index_t i = 0; i < n_bases; ++i) {
147  if (x[i] < 0) return false;
148  sum += x[i];
149  }
150  return (sum <= 1);
151  }
152  }
153 
154  // for pickier compilers:
155  return false;
156  }

195  Vec<T,N> project(const Vec<T,N>& p, Simplex<T,N>* onto=nullptr) const {
196  Vec<T,N> buffer[N];
197  Vec<T,N> out;
198  if (n < N + 1) {
199  // we need to compute the nullspace.
200  Vec<T,N>* const basis = buffer + N - n + 1;
201  Vec<T,N>* const null_basis = buffer;
202  for (index_t i = 1; i < n; ++i) {
203  basis[i - 1] = pts[i] - pts[0];
204  }
205  nullspace(basis, n - 1, null_basis);
206  }
207  this->_project(p, buffer, false, out, &out, onto); // first `out` is dummy
208 
209  // todo: the buffer has your bases + nullspace in a good state.
210  // consider adding option to solve for surface parameters.
211 
212  return out;
213  }

216  Vec<T,N> convexSupport(Vec<T,N> d) const {
217  T best = pts[0].dot(d);
218  index_t k = 0;
219  for (index_t i = 1; i < n; ++i) {
220  T a = pts[i].dot(d);
221  if (a > best) k = i;
222  }
223  return pts[k];
224  }

226  private:
227 
228  // `all_bases` shall be laid out like:
229  // [normal] [parent null basis] [spanning basis]
230  // where `normal` remains to be computed.
231  // in the base case (not is_sub), all nullspace axes are provided.
232  bool _project(
233  Vec<T,N> p,
234  Vec<T,N> all_bases[],
235  bool is_sub,
236  Vec<T,N> missing_basis,
237  Vec<T,N>* proj,
238  Simplex<T,N>* onto) const {
239  switch (n) {
240  case 0:
241  // cannot project onto empty set
242  return false;
243 
244  case 1: {
245  // projection onto a single point is trivial
246  *proj = pts[0];
247  if (onto) *onto = *this;
248  return true;
249  } break;
250 
251  default: {
252  // orthogonally project p onto this simplex along its nullspace.
253  const index_t n_bases = n - 1;
254  const index_t n_null = N - n + 1;
255  Vec<T,N>* const bases = all_bases + n_null;
256  Vec<T,N>* const null_bases = all_bases + 1;
257 
258  // compute simplex's basis
259  for (index_t i = 1; i < n; ++i) {
260  bases[i - 1] = pts[i] - pts[0];
261  }
262 
263  // complete the simplex's nullspace
264  Vec<T,N> normal;
265  if (n_null != 0) {
266  if (is_sub) {
267  // newest nullspace axis not provided; must be computed.
268  all_bases[0] = orthogonal(null_bases); // <- also contains `bases`!
269 
270  // the vector pointing to `p` along the new axis
271  // (which will be orthogonal both to this simplex, and
272  // to the nullspace of the parent simplex).
273  normal = (p - pts[0]).projectOn(all_bases[0]);
274 
275  // might `p` be on the "inside" of the parent simplex?
276  if (is_sub and normal.dot(missing_basis) > 0) {
277  // `p` falls "inward" of the boundary space spanned
278  // by this simplex. `p` therefore does not project to this
279  // simplex; instead it projects to some other simplex,
280  // possibly a parent of this, which *does* face (or envelop) `p`.
281  return false;
282  }
283  }
284  // since there is at least one nullspace axis,
285  // move it aside for the sub-simplex searching code coming below,
286  // which will prepend its own normal.
287  bases[0] = all_bases[0];
288  }
289 
290  // does `p` project to some sub-simplex of ours?
291  // try each sub-simplex.
292  for (index_t i = 0; i < n; ++i) {
293  Simplex<T,N> sub = this->exclude(i);
294  if (sub._project(p, all_bases, true, pts[i] - sub[0], proj, onto)) {
295  // some sub-simplex of us both faces `p` and contains
296  // its projection. `proj` and `onto` now reflect that projection.
297  return true;
298  }
299  }
300 
301  // no sub-simplex claimed `p` was "beyond" it and took the projection.
302  // therefore, this is the simplex onto which `p` projects.
303  *proj = p - normal;
304  if (onto) *onto = *this;
305  return true;
306  }
307  }
308  }
