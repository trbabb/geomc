15 // todo: orthogonal projection to a subspace is a very general linalg operation,
16 // and projection_contains() makes use of it. The operation that does this is
17 // the pseudoinverse, which can be computed efficiently using a QR decomposition.
18 // Probably the best way to do QR is with Householder transformations; the algorithm
19 // can be found complete here: http://people.inf.ethz.ch/gander/papers/qrneu.pdf
20 //
21 // the above should probably be exposed as orthogonal_project(basis[], n, P) in
22 // Orthogonal.h, and invoked here by projection_contains(). Also consider adding
23 // subspace_projection(bases[], n) -> Mtx, which can amortize the QR bidness.
24 //
25 // right now we are running two LUP decompositions, and this is likely
26 // inefficient/unstable.
27 
28 // todo: another possible tool is to project the simplex to a lower dimension
29 // (by dropping coordinates) and barycentric-solving in that lower space.
30 // this amounts to projecting the simplex to an axial (hyper-)plane or an axis.
31 // the weightings of the verticies will not be different if you add back in
32 // the missing coordinates, so the barycentric solution from this lower space
33 // is still valid. (This also explains how keep the linear system square).
34 // generally you want to project to the axial plane wherein the simplex
35 // has the largest projected area. overall this might be more numerically robust.
36 // note that you have to find the coordinates of the projected point `p`
37 // in the axial plane; this is sort of a double-projection.

163  bool projection_contains(const Vec<T,N>& p, Vec<T,N>* params=nullptr) const {
164  switch (n) {
165  case 0:
166  return false;
167 
168  case 1: {
169  return true;
170  } break;
171 
172  case 2: {
173  Vec<T,N> v = pts[1] - pts[0];
174  T d = v.dot(p - pts[0]) / v.mag2();
175  if (params) (*params)[0] = d;
176  return d >= 0 and d <= 1;
177  } break;
178 
179  default: {
180  // simplex spans a subspace. orthogonally project p along its nullspace.
181  // if the simplex spans the domain, then nullspace() early exits and
182  // everything else works as normal.
183 
184  // nullspace comes first (cause we won't solve for its coords),
185  // followed by the simplex basis.
186  const index_t n_bases = n - 1;
187  const index_t n_null = N - n_bases;
188  Vec<T,N> null_bases[N];
189  Vec<T,N>* const bases = null_bases + n_null;
190 
191  // compute simplex bases
192  for (index_t i = 1; i < n; ++i) {
193  bases[i - 1] = pts[i] - pts[0];
194  }
195 
196  // find the simplex's nullspace
197  nullspace(bases, n_bases, null_bases);
198 
199  // write P as a linear combination of the simplex basis and
200  // its nullspace bases. (but don't bother actually solving for
201  // the nullpsace coords; we don't use them).
202  Vec<T,N> x;
203  if (not linear_solve(null_bases, &x, p - pts[0], n_null)) return false;
204 
205  // check that the coordinates of `p` are inside the simplex
206  T sum = 0;
207  for (index_t i = n_null; i < n_bases; ++i) {
208  sum += x[i];
209  if (x[i] < 0 or sum > 1) return false;
210  }
211 
212  // provide surface parameters to caller, e.g. so they can perform
213  // the projection if they want to.
214  if (params) {
215  for (index_t i = 0; i < n_bases; ++i) {
216  (*params)[i] = x[i + n_null];
217  }
218  }
219 
220  return true;
221  }
222  }
223 
224  // statically unreachable, but some less-clever compilers may complain without:
225  return false;
226  }

237  bool contains(const Vec<T,N>& p) const {
238  if (n < N + 1) return false;
239  // todo: test performance + stability over projection_contains
240 
241  // solve for barycentric coordinates x:
242  // Î£ (1, v[i]) * x[i] = (1, p)
243 
244  const index_t K = N + 1;
245  T m[K * K];
246  T x[K];
247  T b[K];
248  for (index_t i = 0; i < K; ++i) {
249  const T* v = pts[i].begin();
250  m[i * K] = 1;
251  std::copy(v, v + N, m + i * K + 1);
252  }
253  b[0] = 1;
254  std::copy(p.begin(), p.end(), b + 1);
255 
256  linear_solve<T,false>(m, K, x, b, 1);
257  // ^^ skip solving for the sum of the weights.
258  // we don't need it, and we know it's 1.
259 
260  T sum = 0;
261  for (index_t i = 1; i < K; ++i) {
262  sum += x[i];
263  if (x[i] < 0 or sum > 1) return false;
264  }
265  return true;
266  }

295  Vec<T,N> project(const Vec<T,N>& p, Simplex<T,N>* onto=nullptr) const {
296  Vec<T,N> buffer[N];
297  Simplex<T,N> s;
298 
299  // why the recursion? why not project as in projection_contains(),
300  // obtaining surface parameters, and then clip them?
301  // A: Because in general, the subspace of the simplex will not
302  // be orthogonal, so clipping the coordinates to the unit triangle
303  // will project the point non-orthogonally along the bases of
304  // that subspace. Instead we have to first find the sub-simplex
305  // which is closest, and then project orthogonally down its nullspace.
306 
307  // initialize the nullspace, if necessary
308  if (n - 1 < N) {
309  Vec<T,N>* const basis = buffer + N - n + 1;
310  Vec<T,N>* const null_basis = buffer;
311  for (index_t i = 1; i < n; ++i) {
312  basis[i - 1] = pts[i] - pts[0];
313  }
314  nullspace(basis, n - 1, null_basis);
315  }
316 
317  // this puts the simplex which faces `p` into `s`, and
318  // leaves `buffer` containing its nullspace and spanning space.
319  this->_find_nearest_face(p, buffer, false, Vec<T,N>(), &s);
320 
321  // choose the smaller basis to project on
322  index_t n_bases = s.n - 1;
323  index_t n_null = N - n_bases;
324  index_t proj_dims;
325  Vec<T,N>* proj_basis;
326  Vec<T,N> out;
327  bool proj_to_null;
328  if (n_bases < n_null) {
329  proj_dims = n_bases;
330  proj_basis = buffer + n_null;
331  proj_to_null = false;
332  } else {
333  proj_dims = n_null;
334  proj_basis = buffer;
335  proj_to_null = true;
336  }
337 
338  // project to the relevant subspace
339  if (proj_dims > 0) {
340  if (N > 3 and proj_dims > 1 and (N - n > 0 or not proj_to_null)) {
341  // if we're projecting to the nullspace, it's already orthogonal,
342  // unless the nullspace started with > 1 dimension (in which case
343  // it was made by `nullspace` above; orthogonalize it now).
344  // this will never happen in 3D because there is always a projection
345  // space of dim <= 1 to choose.
346  orthogonalize(proj_basis, proj_dims);
347  }
348  Vec<T,N> p_in_basis = p - s.pts[0];
349  for (index_t i = 0; i < proj_dims; ++i) {
350  out += p_in_basis.projectOn(proj_basis[i]);
351  }
352  }
353 
354  // offset the subspace projection so it lies on the simplex surface
355  if (proj_to_null) {
356  // `out` is the direction away from the simplex toward `p`.
357  // travel backwards in that direction from `p` to get to the simplex.
358  out = p - out;
359  } else {
360  // `out` is the direction from s's corner to the projected point.
361  // add s's corner to get the absolute point.
362  out += s.pts[0];
363  }
364 
365  if (onto) *onto = s;
366 
367  return out;
368  }

380  Vec<T,N> convex_support(Vec<T,N> d) const {
381  T best = pts[0].dot(d);
382  index_t k = 0;
383  for (index_t i = 1; i < n; ++i) {
384  T a = pts[i].dot(d);
385  if (a > best) {
386  best = a;
387  k = i;
388  }
389  }
390  return pts[k];
391  }

393 private:
394 
395  // `all_bases` shall be laid out like:
396  // [normal] [parent null basis] [spanning basis]
397  // where `normal` remains to be computed.
398  // in the base case (not is_sub), all nullspace axes are provided.
399  bool _find_nearest_face(
400  Vec<T,N> p,
401  Vec<T,N> all_bases[],
402  bool is_sub,
403  Vec<T,N> inward_basis,
404  Simplex<T,N>* onto) const
405  {
406  if (n == 1) {
407  if ((p - pts[0]).dot(inward_basis) > 0) return false;
408  // projection onto a single point is trivial
409  // note: we leave `all_bases` in a garbage state,
410  // but we don't use it, since we don't need it!
411  *onto = *this;
412  return true;
413  } else {
414  // orthogonally project p onto this simplex along its nullspace.
415  const index_t n_null = N - n + 1;
416  Vec<T,N>* const bases = all_bases + n_null;
417  Vec<T,N>* const null_bases = all_bases + 1;
418 
419  // compute simplex's basis
420  for (index_t i = 1; i < n; ++i) {
421  bases[i - 1] = pts[i] - pts[0];
422  }
423 
424  // complete the simplex's nullspace
425  Vec<T,N> normal;
426  if (n_null != 0) {
427  if (is_sub) {
428  // newest nullspace axis not provided; must be computed.
429  all_bases[0] = orthogonal(null_bases); // <- also contains `bases`!
430 
431  // `normal` is the vector pointing to `p` along the new axis
432  // (which will be orthogonal both to this simplex, and
433  // to the nullspace of the parent simplex).
434  normal = (p - pts[0]).projectOn(all_bases[0]);
435 
436  // might `p` be on the "inside" of the parent simplex?
437  if (normal.dot(inward_basis) > 0) {
438  // `p` falls "inward" of this face. `p` therefore does not project
439  // to this simplex; instead it projects to some other simplex,
440  // possibly a parent of this, which *does* face (or envelop) `p`.
441  return false;
442  }
443  }
444  // since there is at least one nullspace axis,
445  // move it aside for the sub-simplex searching code coming below,
446  // which will prepend its own normal.
447  bases[0] = all_bases[0];
448  }
449 
450  // does `p` project to some sub-simplex of ours?
451  // try each sub-simplex.
452  for (index_t i = 0; i < n; ++i) {
453  Simplex<T,N> sub = this->exclude(i);
454  if (sub._find_nearest_face(p, all_bases, true, pts[i] - sub[0], onto)) {
455  // some sub-simplex of us both faces `p` and contains
456  // its projection. `all_bases` and `onto` now reflect that sub-simplex.
457  return true;
458  }
459  }
460 
461  // restore the state of `all_bases`
462  // nobody messed with the normal because it was in their null space.
463  // and the simplex which just filled in its basis was the one that
464  // excluded vertex `n - 1`.
465  all_bases[0] = bases[0];
466  bases[0] = pts[n - 1] - pts[0];
467 
468  // no sub-simplex claimed `p` was "beyond" it and took the projection.
469  // therefore, this is the simplex onto which `p` projects.
470  *onto = *this;
471  return true;
472  }
473  }
