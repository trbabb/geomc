57template <typename T, index_t N>
58class Path {
59public:
60 Path(){}
61
62 Vec<T,N> eval(T t){
63 size_t n = t;
64 size_t sz = knots.size();
65
66 //degenerate cases
67 if (sz < 1){
68 return Vec<T,N>();
69 } else if (t < 0){
70 return knots[0].at_multiple(t);
71 } else if (n >= sz - 1){
72 T t_0 = t - sz + 1;
73 return knots[sz - 1].at_multiple(t_0);
74 }
75
76 T t_0 = t - n;
77 Ray<T,N> &r0 = knots[n];
78 Ray<T,N> &r1 = knots[n+1];
79 Vec<T,N> v0 = r0.at_multiple(t_0);
80 Vec<T,N> v1 = r1.at_multiple(t_0 - 1);
81 return v0.mix(v1, ease(t_0));
82 }
83
84 void appendKnot(Vec<T,N> p, Vec<T,N> v){
85 knots.push_back(Ray<T,N>(p,v));
86 }
87
88 std::vector<Ray<T,N>> knots;
89
90protected:
91 static inline T ease(T t){
92 return t*t*t*(t*(t*6 - 15) + 10);
93 }
94};
