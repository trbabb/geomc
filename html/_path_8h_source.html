56 template <typename T, index_t N>
57 class Path {
58 public:
59  Path(){}
60 
61  Vec<T,N> eval(T t){
62  size_t n = t;
63  size_t sz = knots.size();
64 
65  //degenerate cases
66  if (sz < 1){
67  return Vec<T,N>();
68  } else if (t < 0){
69  return knots[0].atMultiple(t);
70  } else if (n >= sz - 1){
71  T t_0 = t - sz + 1;
72  return knots[sz - 1].atMultiple(t_0);
73  }
74 
75  T t_0 = t - n;
76  Ray<T,N> &r0 = knots[n];
77  Ray<T,N> &r1 = knots[n+1];
78  Vec<T,N> v0 = r0.atMultiple(t_0);
79  Vec<T,N> v1 = r1.atMultiple(t_0 - 1);
80  return v0.mix(v1, ease(t_0));
81  }
