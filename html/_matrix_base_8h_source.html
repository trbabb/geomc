16/*
17 * Matrix typedefs:
18 *
19 * elem_t (T)
20 * reference (T& or AssignmentProxy)
21 * iterator (T* or SubsetIterator)
22 * const_iterator (const T* or SubsetIterator<const T>)
23 * row_iterator (T* or SliceIterator or StrideSliceIterator)
24 * const_row_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
25 * col_iterator (T* or SliceIterator or StrideSliceIterator)
26 * const_col_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
27 * region_iterator (SubsetIterator)
28 * const_region_iterator (SubsetIterator<const>)
29 *
30 * constant iterators cannot be written through.
31 * - They accomplish this by returning `const T` objects, which are not lvals.
32 * - Non-constant iterators, by contrast, return references (or proxy references) to T objects.
33 * - Constant iterators shall call the const version of m.get(), which by nature
34 * does not return writable references. This is done by passing `const M` as
35 * the matrix type to their templates.
36 */

41// here we use the curiously recurring template pattern
42// to provide "template inheritance", i.e. the ability
43// for a whole host of distinct template classes to statically
44// inherit functionality from a "base" template, without virtual
45// inheritance and all the runtime costs associated with it.
46
47// this matters most for element access. virtualizing those
48// functions costs a lot, particularly as they tend to live
49// in inner loop code.
50
51// this creates a bizarre (but useful) case where the base
52// inherits functionality from the derived class.
53
54template <typename T, index_t M, index_t N, typename Derived>
55class MatrixBase {
56
57 inline const Derived* derived() const { return static_cast<const Derived*>(this); }
58
59 // this trait class is necessary because the derived type is incomplete
60 // when the base type is being defined. (More detail in MatrixIterator.h).
61 typedef typename detail::_ImplMtxConstRowtype<Derived,T>::const_row_iterator derived_const_row_iterator;
62
63public:
64
66 static constexpr index_t ROWDIM = M;
68 static constexpr index_t COLDIM = N;
70 typedef T elem_t;
71
73 typedef MtxSubsetIterator<const Derived, const elem_t> const_iterator;
75 typedef const_iterator const_region_iterator;
77 typedef MtxRowIterator<const Derived, const elem_t> const_row_iterator;
79 typedef MtxColIterator<const Derived, const elem_t> const_col_iterator;
80
81 typedef Storage<storage_token_t, _ImplStorageObjCount<Derived>::count> storagebuffer_t;
82
83 // This is accessible from the derived class(es).
84 // We need this so we can check std::is_base_of<>; because
85 // the base we are testing has Derived as a template parameter!
86 // We need some way of know which class it was that
87 // was passed to the curiously-recurring base.
88 // I dub this "rectocranial inversion template pattern".
89 // Obligatory note that c++ is a stinky pile of poo.
90 typedef Derived recurring_t;

95 inline index_t rows() const {
96 // "downward" inheritance is weird.
97 return derived()->rows();
98 }
99
103 inline index_t cols() const {
104 return derived()->cols();
105 }

139 inline elem_t operator()(index_t row, index_t col) const {
140#ifdef GEOMC_MTX_CHECK_BOUNDS
141 if (row >= rows() || col >= cols() || row < 0 || col < 0) {

143 }
144#endif
145 return derived()->operator()(row,col);
146 }

149

188 }

190 // this and its related functions assist in determining
191 // whether two matrices alias memory.
192 inline storagebuffer_t get_storage_token_buffer() const {
193 return storagebuffer_t(derived()->storage_token_count());
