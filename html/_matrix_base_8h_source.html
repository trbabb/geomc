19/*
20 * Matrix typedefs:
21 *
22 * elem_t (T)
23 * reference (T& or AssignmentProxy)
24 * iterator (T* or SubsetIterator)
25 * const_iterator (const T* or SubsetIterator<const T>)
26 * row_iterator (T* or SliceIterator or StrideSliceIterator)
27 * const_row_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
28 * col_iterator (T* or SliceIterator or StrideSliceIterator)
29 * const_col_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
30 * region_iterator (SubsetIterator)
31 * const_region_iterator (SubsetIterator<const>)
32 *
33 * constant iterators cannot be written through.
34 * - They accomplish this by returning `const T` objects, which are not lvals.
35 * - Non-constant iterators, by contrast, return references (or proxy references) to T objects.
36 * - Constant iterators shall call the const version of m.get(), which by nature
37 * does not return writable references. This is done by passing `const M` as
38 * the matrix type to their templates.
39 */

44// here we use the curiously recurring template pattern
45// to provide "template inheritance", i.e. the ability
46// for a whole host of distinct template classes to statically
47// inherit functionality from a "base" template, without virtual
48// inheritance and all the runtime costs associated with it.
49
50// this matters most for element access. virtualizing those
51// functions costs a lot, particularly as they tend to live
52// in inner loop code.
53
54// this creates a bizarre (but useful) case where the base
55// inherits functionality from the derived class.
56
57template <typename T, index_t M, index_t N, typename Derived>
58class MatrixBase {
59
60 inline const Derived* derived() const { return static_cast<const Derived*>(this); }
61
62 // this trait class is necessary because the derived type is incomplete
63 // when the base type is being defined. (More detail in MatrixIterator.h).
64 typedef typename detail::_ImplMtxConstRowtype<Derived,T>::const_row_iterator derived_const_row_iterator;
65
66public:
67
69 static constexpr index_t ROWDIM = M;
71 static constexpr index_t COLDIM = N;
73 typedef T elem_t;
74
76 typedef MtxSubsetIterator<const Derived, const elem_t> const_iterator;
78 typedef const_iterator const_region_iterator;
80 typedef MtxRowIterator<const Derived, const elem_t> const_row_iterator;
82 typedef MtxColIterator<const Derived, const elem_t> const_col_iterator;
83
84 typedef Storage<storage_token_t, _ImplStorageObjCount<Derived>::count> storagebuffer_t;
85
86 // This is accessible from the derived class(es).
87 // We need this so we can check std::is_base_of<>; because
88 // the base we are testing has Derived as a template parameter!
89 // We need some way of know which class it was that
90 // was passed to the curiously-recurring base.
91 // I dub this "rectocranial inversion template pattern".
92 // Obligatory note that c++ is a stinky pile of poo.
93 typedef Derived recurring_t;
94
98 inline index_t rows() const {
99 // "downward" inheritance is weird.
100 return derived()->rows();
101 }
102
106 inline index_t cols() const {
107 return derived()->cols();
108 }
109
114 inline derived_const_row_iterator operator[](index_t i) const {
115 return derived()->row(i);
116 }
117
118 // the following two may be overridden by the deriving class:
119
124 inline const_row_iterator row(index_t i) const {
125 return const_row_iterator(derived(), i);
126 }
127
132 inline const_col_iterator col(index_t i) const {
133 return const_col_iterator(derived(), i);
134 }
135
142 inline elem_t operator()(index_t row, index_t col) const {
143#ifdef GEOMC_MTX_CHECK_BOUNDS
144 if (row >= rows() || col >= cols() || row < 0 || col < 0) {
145 throw std::out_of_range();
146 }
147#endif
148 return derived()->operator()(row,col);
149 }
150
151 // these two may be overridden:
152
157 inline const_iterator begin() const {
158 return const_iterator(derived());
159 }
160
165 inline const_iterator end() const {
166 // TODO: delegate calculation.
167 Vec<index_t,2> p = Vec<index_t,2>(rows(), 0);
168 return const_iterator(derived(), p);
169 }
170
178 inline const_region_iterator region_begin(const MatrixRegion &r) const {
179 return const_region_iterator(derived(), r);
180 }
181
189 inline const_region_iterator region_end(const MatrixRegion &r) const {
190 return const_region_iterator(derived(), r).end();
191 }
192
193 // this and its related functions assist in determining
194 // whether two matrices alias memory.
195 inline storagebuffer_t get_storage_token_buffer() const {
196 return storagebuffer_t(derived()->storage_token_count());
197 }
198
199};
