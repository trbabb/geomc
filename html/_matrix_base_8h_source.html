19 /*
20  * Matrix typedefs:
21  *
22  * elem_t (T)
23  * reference (T& or AssignmentProxy)
24  * iterator (T* or SubsetIterator)
25  * const_iterator (const T* or SubsetIterator<const T>)
26  * row_iterator (T* or SliceIterator or StrideSliceIterator)
27  * const_row_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
28  * col_iterator (T* or SliceIterator or StrideSliceIterator)
29  * const_col_iterator (const T* or SliceIterator<const> or StrideSliceIterator<const>)
30  * region_iterator (SubsetIterator)
31  * const_region_iterator (SubsetIterator<const>)
32  *
33  * constant iterators cannot be written through.
34  * - They accomplish this by returning `const T` objects, which are not lvals.
35  * - Non-constant iterators, by contrast, return references (or proxy references) to T objects.
36  * - Constant iterators shall call the const version of m.get(), which by nature
37  * does not return writable references. This is done by passing `const M` as
38  * the matrix type to their templates.
39  */

44 // here we use the curiously recurring template pattern
45 // to provide "template inheritance", i.e. the ability
46 // for a whole host of distinct template classes to statically
47 // inherit functionality from a "base" template, without virtual
48 // inheritance and all the runtime costs associated with it.
49 
50 // this matters most for element access. virtualizing those
51 // functions costs a lot, particularly as they tend to live
52 // in inner loop code.
53 
54 // this creates a bizarre (but useful) case where the base
55 // inherits functionality from the derived class.
56 
57 template <typename T, index_t M, index_t N, typename Derived>

76  typedef const_iterator const_region_iterator;

85  // We need this so we can check boost::is_base_of<>; because
86  // the base we are testing has Derived as a template parameter!
87  // We need some way of know which class it was that
88  // was passed to the curiously-recurring base.
89  // I dub this "rectocranial inversion template pattern".
90  // Obligatory note that c++ is a stinky pile of poo.
91  typedef Derived recurring_t;

140  inline elem_t operator()(index_t row, index_t col) const {
141 #ifdef GEOMC_MTX_CHECK_BOUNDS
142  if (row >= rows() || col >= cols() || row < 0 || col < 0) {
143  throw std::out_of_range();
144  }
145 #endif
146  return derived()->operator()(row,col);
147  }

164  // TODO: delegate calculation.

177  return const_region_iterator(derived(), r);
178  }
179

189  }

194  return storagebuffer_t(derived()->getStorageIDCount());
