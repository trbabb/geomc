247  template <typename T>
248  void rotmat(SimpleMatrix<T,4,4> *into, T x, T y, T z, T theta) {
249  T c = cos(theta);
250  T s = sin(theta);
251  T oneMcos = 1 - c;
252  T m[4][4] = {
253  {c + oneMcos*x*x, oneMcos*x*y - s*z, oneMcos*x*z + s*y, 0},
254  {oneMcos*y*x + s*z, c + oneMcos*y*y, oneMcos*y*z - s*x, 0},
255  {oneMcos*z*x - s*y, oneMcos*z*y + s*x, c + oneMcos*z*z, 0},
256  {0, 0, 0, 1}
257  };
258  T *from = m[0];
259  std::copy(from, from+16, into->begin());
260  }

269  template <typename T>
270  void rotmat(SimpleMatrix<T,4,4> *into, const Quat<T> &q) {
271  T x = q.x; //for convenience
272  T y = q.y;
273  T z = q.z;
274  T w = q.w;
275  T x2 = x*x; // for speed
276  T y2 = y*y; // ...
277  T z2 = z*z;
278  T w2 = w*w;
279  T twox = 2*x;
280  T twoy = 2*y;
281  T twoz = 2*z;
282  T xy = twox * y;
283  T xz = twox * z;
284  T xw = twox * w;
285  T yz = twoy * z;
286  T yw = twoy * w;
287  T zw = twoz * w;
288  T m[4][4] = {
289  {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
290  {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
291  {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
292  {0, 0, 0, 1}
293  };
294  T *from = m[0];
295  std::copy(from, from+16, into->begin());

329  // rotation aligning `dir` with `alignWith`.
330  // modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
331  template <typename T>
332  inline void rotmat_direction_align(SimpleMatrix<T,4,4> *into,
333  const Vec<T,3> &dir,
334  const Vec<T,3> &alignWith) {
335  const Vec<T,3> v = dir ^ alignWith;
336  const T c = dir.dot(alignWith);
337 
338  // handle poles
339  if (std::abs(c) == 1) {
340  into->setIdentity();
341  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
342  return;
343  }
344 
345  const T k = (1 - c) / (1 - c * c);
346 
347  T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
348  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
349  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
350  0, 0, 0, 1 };
351 
352  std::copy(m, m + 16, into->begin());
353  }

358  const Vec<T,3> &dir,
359  const Vec<T,3> &alignWith) {
360  const Vec<T,3> v = dir ^ alignWith;
361  const T c = dir.dot(alignWith);

367  return;

373  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,

380  // rotation about an arbitrary point (with inverse)
381  template <typename T>
382  void rotmat(SimpleMatrix<T,4,4> *out_mat,
383  SimpleMatrix<T,4,4> *out_inv,
384  T x, T y, T z,
385  T px, T py, T pz,
386  T theta) {
387  T c = cos(theta);
388  T s = sin(theta);
389  T oneMcos = 1 - c;
390  T x2 = x*x;
391  T y2 = y*y;
392  T z2 = z*z;
393  T fac[4][4] = {
394  {x2 + c*(y2 + z2), x*y*oneMcos, x*z*oneMcos, (px*(y2 + z2) - x*(py*y + pz*z))*oneMcos},
395  {x*y*oneMcos, y2 + c*(x2 + z2), y*z*oneMcos, (py*(x2 + z2) - y*(px*x + pz*z))*oneMcos},
396  {x*z*oneMcos, y*z*oneMcos, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*oneMcos},
397  {0, 0, 0, 1}
398  };
399  T sterms[3][4] = {
400  {0, -z*s, y*s, (py*z - pz*y)*s},
401  {z*s, 0, -x*s, (pz*x - px*z)*s},
402  {-y*s, x*s, 0, (px*y - py*x)*s}
403  };
404 
405  // pre-fill output matrices.
406  T *from = fac[0];
407  T *sfrom = sterms[0];
408  T *mat = out_mat->begin();
409  T *inv = out_inv->begin();
410  std::copy(from, from+16, mat);
411  std::copy(from, from+16, inv);
412 
413  // add sine terms in.
414  for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
415  *mat += *sfrom;
416  *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
417  }
418  }

562  AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N> &mat) {
563  SimpleMatrix<T,N,N> m_inv;
564  AffineTransform<T,N> xfnew;
565
