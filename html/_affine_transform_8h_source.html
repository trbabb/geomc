30// todo: We've picked a bad convention. xf3 * xf2 * xf1 * v is counterintuitive.
31// We should have v * xf1 * xf2 * xf3, with v as a row vector; then
32// the operations read in order from left to right.
33// This also means that *= and /= operators will preserve ordering, and
34// things like `v / xf` will be more consistent.
35// It's also consistent with left-to-right function application:
36// thing.apply(xf1).apply(xf2).apply(xf3) aligns with xf1 * xf2 * xf3.
37// It also means that basis vectors will be rows, which means an array
38// of basis vectors can be directly cast to a matrix, without copying,
39// transposing, or tricks with layout templates.
40// >> Can we change or template this choice so that at the very least,
41// the user has the capability to choose their desired convention?

174 template <index_t M, typename U=T>
175 AffineTransform<U,M> resized() const {
176 AffineTransform<U,M> xf;
177 for (index_t i = 0; i < std::min(M,N); i++) {
178 // copy the inner MxM elements (if they exist)
179 for (index_t j = 0; j < std::min(M,N); j++) {
180 xf.mat(i,j) = static_cast<U>(mat(i,j));
181 xf.inv(i,j) = static_cast<U>(inv(i,j));
182 }
183 // keep translation components in the homogenous column
184 xf.mat(i,M) = static_cast<U>(mat(i,N));
185 }
186 // this ought to be 1, but let's be pedantic
187 xf.mat(M,M) = mat(N,N);
188 return xf;
189 }

195 std::copy(mat.begin(), mat.end(), xf.mat.begin());
196 std::copy(inv.begin(), inv.end(), xf.inv.begin());
197 return xf;

214 const Vec<T,N> apply(const Vec<T,N>& p) const {
215 Vec<T,N+1> p_hom(p,1);
216 p_hom = mat * p_hom;
217 return p_hom.template resized<N>(); //c++ is awful
218 }
219
223 const Vec<T,N> apply_vector(const Vec<T,N>& v) const {
224 Vec<T,N> o;
225 for (index_t r = 0; r < N; r++) {
226 for (index_t c = 0; c < N; c++) {
227 o[r] += mat(r,c) * v[c];
228 }
229 }
230 return o;
231 }
232
238 const Vec<T,N> apply_normal(const Vec<T,N>& n) const {
239 // normal matrix = txpose of inverse.
240 Vec<T,N> o;
241 for (index_t r = 0; r < N; r++) {
242 for (index_t c = 0; c < N; c++) {
243 o[r] += inv(c,r) * n[c];
244 }
245 }
246 return o;
247 }
248
252 const Vec<T,N> apply_inverse(const Vec<T,N>& p) const {
253 Vec<T,N+1> p_hom(p,1);
254 p_hom = inv * p_hom; // this is returning a dynamic matrix for some reason.
255 return p_hom.template resized<N>();
256 }

261 const Vec<T,N> apply_inverse_vector(const Vec<T,N>& v) const {
262 Vec<T,N> o;
263 for (index_t r = 0; r < N; r++) {
264 for (index_t c = 0; c < N; c++) {
265 o[r] += inv(r,c) * v[c];
266 }
267 }
268 return o;
269 }
270
274 const Vec<T,N> apply_inverse_normal(const Vec<T,N>& n) const {
275 // txpose of inverse of inverse.
276 Vec<T,N> o;
277 for (index_t r = 0; r < N; r++) {
278 for (index_t c = 0; c < N; c++) {
279 o[r] += mat(c,r) * n[c];
280 }
281 }
282 return o;
283 }

293 xfnew.inv = inv * xf.inv;

338template <typename T>
339void rotmat(SimpleMatrix<T,4,4>* into, T x, T y, T z, T theta) {
340 T c = std::cos(theta);
341 T s = std::sin(theta);
342 T c_1 = 1 - c;
343 T m[4][4] = {
344 {c + c_1*x*x, c_1*x*y - s*z, c_1*x*z + s*y, 0},
345 {c_1*y*x + s*z, c + c_1*y*y, c_1*y*z - s*x, 0},
346 {c_1*z*x - s*y, c_1*z*y + s*x, c + c_1*z*z, 0},
347 {0, 0, 0, 1}
348 };
349 T *from = m[0];
350 std::copy(from, from+16, into->begin());
351}

360template <typename T>
361void rotmat(SimpleMatrix<T,4,4>* into, const Quat<T>& q) {
362 T x = q.x; //for convenience
363 T y = q.y;
364 T z = q.z;
365 T w = q.w;
366 T x2 = x*x; // for speed
367 T y2 = y*y; // ...
368 T z2 = z*z;
369 T w2 = w*w;
370 T twox = 2*x;
371 T twoy = 2*y;
372 T twoz = 2*z;
373 T xy = twox * y;
374 T xz = twox * z;
375 T xw = twox * w;
376 T yz = twoy * z;
377 T yw = twoy * w;
378 T zw = twoz * w;
379 T m[4][4] = {
380 {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
381 {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
382 {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
383 {0, 0, 0, 1}
384 };
385 T *from = m[0];
386 std::copy(from, from+16, into->begin());
387}

389// rotation about an arbitrary point
390template <typename T>
391void rotmat(
392 SimpleMatrix<T,4,4>* into,
393 T x, T y, T z,
394 T px, T py, T pz,
395 T theta)
396{
397 T c = std::cos(theta);
398 T s = std::sin(theta);
399 T c_1 = 1 - c;
400 T x2 = x*x;
401 T y2 = y*y;
402 T z2 = z*z;
403
404 const T q0 = diff_of_products(py, z, pz, y);
405 const T q1 = diff_of_products(pz, x, px, z);
406 const T q2 = diff_of_products(px, y, py, x);
407 const T s0 = sum_of_products( py, y, pz, z);
408 const T s1 = sum_of_products( px, x, pz, z);
409 const T s2 = sum_of_products( px, x, py, y);
410 const T t0 = diff_of_products(px, y2 + z2, x, s0);
411 const T t1 = diff_of_products(py, x2 + z2, y, s1);
412 const T t2 = diff_of_products(pz, x2 + y2, z, s2);
413 const T u0 = sum_of_products( t0, c_1, q0, s);
414 const T u1 = sum_of_products( t1, c_1, q1, s);
415 const T u2 = sum_of_products( t2, c_1, q2, s);
416
417 T m[4][4] = {
418 {x2 + c*(y2 + z2), x*y*c_1 - z*s, x*z*c_1 + y*s, u0},
419 {x*y*c_1 + z*s, y2 + c*(x2 + z2), y*z*c_1 - x*s, u1},
420 {x*z*c_1 - y*s, y*z*c_1 + x*s, z2 + c*(x2 + y2), u2},
421 {0, 0, 0, 1}
422 };
423 T *from = m[0];
424 std::copy(from, from+16, into->begin());
425}

438// rotation aligning `dir` with `alignWith`.
439// modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
440template <typename T>
441inline void rotmat_direction_align(
442 SimpleMatrix<T,4,4>* into,
443 const Vec<T,3>& dir,
444 const Vec<T,3>& alignWith)
445{
446 const Vec<T,3> v = dir ^ alignWith;
447 const T c = dir.dot(alignWith);
448
449 // handle poles
450 if (std::abs(c) == 1) {
451 into->set_identity();
452 if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
453 return;
454 }
455
456 const T k = (1 - c) / (1 - c * c);
457
458 T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
459 v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
460 v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
461 0, 0, 0, 1 };
462
463 std::copy(m, m + 16, into->begin());
464}

466template <typename T>
467SimpleMatrix<T,3,3> rotmat_direction_align(
468 const Vec<T,3>& dir,
469 const Vec<T,3>& alignWith)
470{
471 const Vec<T,3> v = dir ^ alignWith;
472 const T c = dir.dot(alignWith);
473
474 // handle poles
475 if (std::abs(c) == 1) {
476 SimpleMatrix<T,3,3> out;
477 // 180 degree case:
478 if (c < 0) { out.set(0, 0, -1); out.set(2, 2, -1); }
479 return out;
480 }
481
482 const T k = (1 - c) / (1 - c * c);
483
484 return {
485 v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y,
486 v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,
487 v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c
488 };
489}

562 SimpleMatrix<T,3,3>* into,
563 const Vec<T,3>& dir,

586// rotation about an arbitrary point (with inverse)
587template <typename T>
588void rotmat(
589 SimpleMatrix<T,4,4> *out_mat,
590 SimpleMatrix<T,4,4> *out_inv,
591 T x, T y, T z,
592 T px, T py, T pz,
593 T theta)
594{
595 T c = std::cos(theta);
596 T s = std::sin(theta);
597 T c_1 = 1 - c;
598 T x2 = x*x;
599 T y2 = y*y;
600 T z2 = z*z;
601 T xyc = x*y*c_1;
602 T xzc = x*z*c_1;
603 T yzc = y*z*c_1;
604 T da = x2 + c*(y2 + z2);
605 T db = y2 + c*(x2 + z2);
606 T dc = z2 + c*(x2 + y2);
607
608 T fac[4][4] = {
609 {da, xyc, xzc, (px*(y2 + z2) - x*(py*y + pz*z))*c_1},
610 {xyc, db, yzc, (py*(x2 + z2) - y*(px*x + pz*z))*c_1},
611 {xzc, yzc, dc, (pz*(x2 + y2) - z*(px*x + py*y))*c_1},
612 {0, 0, 0, 1}
613 };
614 T sterms[3][4] = {
615 {0, -z*s, y*s, (py*z - pz*y)*s},
616 {z*s, 0, -x*s, (pz*x - px*z)*s},
617 {-y*s, x*s, 0, (px*y - py*x)*s}
618 };
619
620 // pre-fill output matrices.
621 T* from = fac[0];
622 T* sfrom = sterms[0];
623 T* mat = out_mat->begin();
624 T* inv = out_inv->begin();
625 std::copy(from, from + 16, mat);
626 std::copy(from, from + 16, inv);
627
628 // add sine terms in.
629 for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
630 *mat += *sfrom;
631 *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
632 }
633}

718template <typename T>
719AffineTransform<T,2> rotation(T radians) {
720 AffineTransform<T,2> xfnew;
721 T s = std::sin(radians);
722 T c = std::cos(radians);
723
724 xfnew.mat.set(0,0, c);
725 xfnew.mat.set(0,1,-s);
726 xfnew.mat.set(1,0, s);
727 xfnew.mat.set(1,1, c);
728
729 xfnew.inv.set(0,0, c);
730 xfnew.inv.set(0,1, s);
731 xfnew.inv.set(1,0,-s);
732 xfnew.inv.set(1,1, c);
733
734 return xfnew;
735}

813AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N,Lyt,P>& mat) {
814 SimpleMatrix<T,N,N> m_inv;
815 AffineTransform<T,N> xfnew;
816
817 if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
818 throw NonsquareMatrixException(mat.rows(), mat.cols());
819 }
820
821 // calculate the inverse
822 inv(&m_inv, mat);
823
824 // copy the sub-matrices
825 for (index_t r = 0; r < mat.rows(); ++r) {
826 for (index_t c = 0; c < mat.cols(); ++c) {
827 xfnew.inv(r,c) = m_inv(r,c);
828 xfnew.mat(r,c) = mat(r,c);
829 }
830 }
831
832 return xfnew;
833}
