30// todo: We've picked a bad convention. xf3 * xf2 * xf1 * v is counterintuitive.
31// We should have v * xf1 * xf2 * xf3, with v as a row vector; then
32// the operations read in order from left to right.
33// This also means that *= and /= operators will preserve ordering, and
34// things like `v / xf` will be more consistent.
35// It's also consistent with left-to-right function application:
36// thing.apply(xf1).apply(xf2).apply(xf3) aligns with xf1 * xf2 * xf3.
37// It also means that basis vectors will be rows, which means an array
38// of basis vectors can be directly cast to a matrix, without copying,
39// transposing, or tricks with layout templates.
40// >> Can we change or template this choice so that at the very least,
41// the user has the capability to choose their desired convention?

184 template <index_t M, typename U=T>
185 AffineTransform<U,M> resized() const {
186 AffineTransform<U,M> xf;
187 for (index_t i = 0; i < std::min(M,N); i++) {
188 // copy the inner MxM elements (if they exist)
189 for (index_t j = 0; j < std::min(M,N); j++) {
190 xf.mat(i,j) = static_cast<U>(mat(i,j));
191 xf.inv(i,j) = static_cast<U>(inv(i,j));
192 }
193 // keep translation components in the homogenous column
194 xf.mat(i,M) = static_cast<U>(mat(i,N));
195 }
196 // this ought to be 1, but let's be pedantic
197 xf.mat(M,M) = mat(N,N);
198 return xf;
199 }

225 Vec<T,N+1> p_hom(p,1);
226 p_hom = mat * p_hom;

233 const Vec<T,N> apply_direction(const Vec<T,N>& v) const {
234 Vec<T,N> o;
235 for (index_t r = 0; r < N; r++) {
236 for (index_t c = 0; c < N; c++) {
237 o[r] += mat(r,c) * v[c];
238 }
239 }
240 return o;
241 }

248 const Vec<T,N> apply_normal(const Vec<T,N>& n) const {
249 // normal matrix = txpose of inverse.
250 Vec<T,N> o;
251 for (index_t r = 0; r < N; r++) {
252 for (index_t c = 0; c < N; c++) {
253 o[r] += inv(c,r) * n[c];
254 }
255 }
256 return o;
257 }

262 const Vec<T,N> apply_inverse(const Vec<T,N>& p) const {
263 Vec<T,N+1> p_hom(p,1);
264 p_hom = inv * p_hom; // this is returning a dynamic matrix for some reason.
265 return p_hom.template resized<N>();
266 }

348template <typename T>
349void rotmat(SimpleMatrix<T,4,4>* into, T x, T y, T z, T theta) {
350 T c = std::cos(theta);
351 T s = std::sin(theta);
352 T c_1 = 1 - c;
353 T m[4][4] = {
354 {c + c_1*x*x, c_1*x*y - s*z, c_1*x*z + s*y, 0},
355 {c_1*y*x + s*z, c + c_1*y*y, c_1*y*z - s*x, 0},
356 {c_1*z*x - s*y, c_1*z*y + s*x, c + c_1*z*z, 0},
357 {0, 0, 0, 1}
358 };
359 T *from = m[0];
360 std::copy(from, from+16, into->begin());
361}

370template <typename T>
371void rotmat(SimpleMatrix<T,4,4>* into, const Quat<T>& q) {
372 T x = q.x; //for convenience
373 T y = q.y;
374 T z = q.z;
375 T w = q.w;
376 T x2 = x*x; // for speed
377 T y2 = y*y; // ...
378 T z2 = z*z;
379 T w2 = w*w;
380 T twox = 2*x;
381 T twoy = 2*y;
382 T twoz = 2*z;
383 T xy = twox * y;
384 T xz = twox * z;
385 T xw = twox * w;
386 T yz = twoy * z;
387 T yw = twoy * w;
388 T zw = twoz * w;
389 T m[4][4] = {
390 {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
391 {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
392 {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
393 {0, 0, 0, 1}
394 };
395 T *from = m[0];
396 std::copy(from, from+16, into->begin());
397}

399// rotation about an arbitrary point
400template <typename T>
401void rotmat(
402 SimpleMatrix<T,4,4>* into,
403 T x, T y, T z,
404 T px, T py, T pz,
405 T theta)
406{
407 T c = std::cos(theta);
408 T s = std::sin(theta);
409 T c_1 = 1 - c;
410 T x2 = x*x;
411 T y2 = y*y;
412 T z2 = z*z;
413
414 const T q0 = diff_of_products(py, z, pz, y);
415 const T q1 = diff_of_products(pz, x, px, z);
416 const T q2 = diff_of_products(px, y, py, x);
417 const T s0 = sum_of_products( py, y, pz, z);
418 const T s1 = sum_of_products( px, x, pz, z);
419 const T s2 = sum_of_products( px, x, py, y);
420 const T t0 = diff_of_products(px, y2 + z2, x, s0);
421 const T t1 = diff_of_products(py, x2 + z2, y, s1);
422 const T t2 = diff_of_products(pz, x2 + y2, z, s2);
423 const T u0 = sum_of_products( t0, c_1, q0, s);
424 const T u1 = sum_of_products( t1, c_1, q1, s);
425 const T u2 = sum_of_products( t2, c_1, q2, s);
426
427 T m[4][4] = {
428 {x2 + c*(y2 + z2), x*y*c_1 - z*s, x*z*c_1 + y*s, u0},
429 {x*y*c_1 + z*s, y2 + c*(x2 + z2), y*z*c_1 - x*s, u1},
430 {x*z*c_1 - y*s, y*z*c_1 + x*s, z2 + c*(x2 + y2), u2},
431 {0, 0, 0, 1}
432 };
433 T *from = m[0];
434 std::copy(from, from+16, into->begin());
435}

448// rotation aligning `dir` with `alignWith`.
449// modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
450template <typename T>
451inline void rotmat_direction_align(
452 SimpleMatrix<T,4,4>* into,
453 const Vec<T,3>& dir,
454 const Vec<T,3>& alignWith)
455{
456 const Vec<T,3> v = dir ^ alignWith;
457 const T c = dir.dot(alignWith);
458
459 // handle poles
460 if (std::abs(c) == 1) {
461 into->set_identity();
462 if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
463 return;
464 }
465
466 const T k = (1 - c) / (1 - c * c);
467
468 T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
469 v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
470 v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
471 0, 0, 0, 1 };
472
473 std::copy(m, m + 16, into->begin());
474}

476template <typename T>
477SimpleMatrix<T,3,3> rotmat_direction_align(
478 const Vec<T,3>& dir,
479 const Vec<T,3>& alignWith)
480{
481 const Vec<T,3> v = dir ^ alignWith;
482 const T c = dir.dot(alignWith);
483
484 // handle poles
485 if (std::abs(c) == 1) {
486 SimpleMatrix<T,3,3> out;
487 // 180 degree case:
488 if (c < 0) { out.set(0, 0, -1); out.set(2, 2, -1); }
489 return out;
490 }
491
492 const T k = (1 - c) / (1 - c * c);
493
494 return {
495 v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y,
496 v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,
497 v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c
498 };
499}

570template <typename T>
571inline void rotmat_direction_align(
572 SimpleMatrix<T,3,3>* into,
573 const Vec<T,3>& dir,
574 const Vec<T,3>& alignWith)
575{

596// rotation about an arbitrary point (with inverse)
597template <typename T>
598void rotmat(
599 SimpleMatrix<T,4,4> *out_mat,
600 SimpleMatrix<T,4,4> *out_inv,
601 T x, T y, T z,
602 T px, T py, T pz,
603 T theta)
604{
605 T c = std::cos(theta);
606 T s = std::sin(theta);
607 T c_1 = 1 - c;
608 T x2 = x*x;
609 T y2 = y*y;
610 T z2 = z*z;
611 T xyc = x*y*c_1;
612 T xzc = x*z*c_1;
613 T yzc = y*z*c_1;
614 T da = x2 + c*(y2 + z2);
615 T db = y2 + c*(x2 + z2);
616 T dc = z2 + c*(x2 + y2);
617
618 T fac[4][4] = {
619 {da, xyc, xzc, (px*(y2 + z2) - x*(py*y + pz*z))*c_1},
620 {xyc, db, yzc, (py*(x2 + z2) - y*(px*x + pz*z))*c_1},
621 {xzc, yzc, dc, (pz*(x2 + y2) - z*(px*x + py*y))*c_1},
622 {0, 0, 0, 1}
623 };
624 T sterms[3][4] = {
625 {0, -z*s, y*s, (py*z - pz*y)*s},
626 {z*s, 0, -x*s, (pz*x - px*z)*s},
627 {-y*s, x*s, 0, (px*y - py*x)*s}
628 };
629
630 // pre-fill output matrices.
631 T* from = fac[0];
632 T* sfrom = sterms[0];
633 T* mat = out_mat->begin();
634 T* inv = out_inv->begin();
635 std::copy(from, from + 16, mat);
636 std::copy(from, from + 16, inv);
637
638 // add sine terms in.
639 for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
640 *mat += *sfrom;
641 *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
642 }
643}

728template <typename T>
729AffineTransform<T,2> rotation(T radians) {
730 AffineTransform<T,2> xfnew;
731 T s = std::sin(radians);
732 T c = std::cos(radians);
733
734 xfnew.mat.set(0,0, c);
735 xfnew.mat.set(0,1,-s);
736 xfnew.mat.set(1,0, s);
737 xfnew.mat.set(1,1, c);
738
739 xfnew.inv.set(0,0, c);
740 xfnew.inv.set(0,1, s);
741 xfnew.inv.set(1,0,-s);
742 xfnew.inv.set(1,1, c);
743
744 return xfnew;

823inline AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N,Lyt,P>& mat) {
824 SimpleMatrix<T,N,N> m_inv;
825 AffineTransform<T,N> xfnew;
826
827 if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
828 throw NonsquareMatrixException(mat.rows(), mat.cols());
829 }
830
831 // calculate the inverse
832 inv(&m_inv, mat);
833
834 // copy the sub-matrices
835 for (index_t r = 0; r < mat.rows(); ++r) {
836 for (index_t c = 0; c < mat.cols(); ++c) {
837 xfnew.inv(r,c) = m_inv(r,c);
838 xfnew.mat(r,c) = mat(r,c);
839 }
840 }
841
842 return xfnew;
843}
