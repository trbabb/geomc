157  const Vec<T,N> applyVector(const Vec<T,N> &v) const {
158  Vec<T,N> o;
159  for (index_t r = 0; r < N; r++) {
160  for (index_t c = 0; c < N; c++) {
161  o[r] += mat.get(r,c) * v[c];

246  template <typename T>
247  void rotmat(SimpleMatrix<T,4,4> *into, T x, T y, T z, T theta) {
248  T c = cos(theta);
249  T s = sin(theta);
250  T oneMcos = 1 - c;
251  T m[4][4] = {
252  {c + oneMcos*x*x, oneMcos*x*y - s*z, oneMcos*x*z + s*y, 0},
253  {oneMcos*y*x + s*z, c + oneMcos*y*y, oneMcos*y*z - s*x, 0},
254  {oneMcos*z*x - s*y, oneMcos*z*y + s*x, c + oneMcos*z*z, 0},
255  {0, 0, 0, 1}
256  };
257  T *from = m[0];
258  std::copy(from, from+16, into->begin());
259  }

267  // rotation from a quaternion
268  template <typename T>
269  void rotmat(SimpleMatrix<T,4,4> *into, const Quat<T> &q) {
270  T x = q.x; //for convenience
271  T y = q.y;
272  T z = q.z;
273  T w = q.w;
274  T x2 = x*x; // for speed
275  T y2 = y*y; // ...
276  T z2 = z*z;
277  T w2 = w*w;
278  T twox = 2*x;
279  T twoy = 2*y;
280  T twoz = 2*z;
281  T xy = twox * y;
282  T xz = twox * z;
283  T xw = twox * w;
284  T yz = twoy * z;
285  T yw = twoy * w;
286  T zw = twoz * w;
287  T m[4][4] = {
288  {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
289  {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
290  {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
291  {0, 0, 0, 1}
292  };
293  T *from = m[0];
294  std::copy(from, from+16, into->begin());
295  }

299  void rotmat(SimpleMatrix<T,4,4> *into,
300  T x, T y, T z,
301  T px, T py, T pz,
302  T theta) {

304  T s = sin(theta);
305  T oneMcos = 1 - c;

309  T m[4][4] = {
310  {x2 + c*(y2 + z2), x*y*oneMcos - z*s, x*z*oneMcos + y*s, (px*(y2 + z2) - x*(py*y + pz*z))*oneMcos + (py*z - pz*y)*s},
311  {x*y*oneMcos + z*s, y2 + c*(x2 + z2), y*z*oneMcos - x*s, (py*(x2 + z2) - y*(px*x + pz*z))*oneMcos + (pz*x - px*z)*s},
312  {x*z*oneMcos - y*s, y*z*oneMcos + x*s, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*oneMcos + (px*y - py*x)*s},
313  {0, 0, 0, 1}
314  };
315  T *from = m[0];
316  std::copy(from, from+16, into->begin());

328  // rotation aligning `dir` with `alignWith`.
329  // modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
330  template <typename T>
331  inline void rotmat_direction_align(SimpleMatrix<T,4,4> *into,
332  const Vec<T,3> &dir,
333  const Vec<T,3> &alignWith) {
334  const Vec<T,3> v = dir ^ alignWith;
335  const T c = dir.dot(alignWith);
336 
337  // handle poles
338  if (std::abs(c) == 1) {
339  into->setIdentity();
340  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
341  return;
342  }
343 
344  const T k = (1 - c) / (1 - c * c);
345 
346  T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
347  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
348  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
349  0, 0, 0, 1 };
350 
351  std::copy(m, m + 16, into->begin());
352  }

355  template <typename T>
356  inline void rotmat_direction_align(SimpleMatrix<T,3,3> *into,
357  const Vec<T,3> &dir,
358  const Vec<T,3> &alignWith) {
359  const Vec<T,3> v = dir ^ alignWith;
360  const T c = dir.dot(alignWith);
361 
362  // handle poles
363  if (std::abs(c) == 1) {
364  into->setIdentity();
365  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
366  return;
367  }
368 
369  const T k = (1 - c) / (1 - c * c);
370 
371  T m[9] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y,
372  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,
373  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c};
374 
375  std::copy(m, m + 9, into->begin());
376  }

379  // rotation about an arbitrary point (with inverse)
380  template <typename T>
381  void rotmat(SimpleMatrix<T,4,4> *out_mat,
382  SimpleMatrix<T,4,4> *out_inv,
383  T x, T y, T z,
384  T px, T py, T pz,
385  T theta) {
386  T c = cos(theta);
387  T s = sin(theta);
388  T oneMcos = 1 - c;
389  T x2 = x*x;
390  T y2 = y*y;
391  T z2 = z*z;
392  T fac[4][4] = {
393  {x2 + c*(y2 + z2), x*y*oneMcos, x*z*oneMcos, (px*(y2 + z2) - x*(py*y + pz*z))*oneMcos},
394  {x*y*oneMcos, y2 + c*(x2 + z2), y*z*oneMcos, (py*(x2 + z2) - y*(px*x + pz*z))*oneMcos},
395  {x*z*oneMcos, y*z*oneMcos, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*oneMcos},
396  {0, 0, 0, 1}
397  };
398  T sterms[3][4] = {
399  {0, -z*s, y*s, (py*z - pz*y)*s},
400  {z*s, 0, -x*s, (pz*x - px*z)*s},
401  {-y*s, x*s, 0, (px*y - py*x)*s}
402  };
403 
404  // pre-fill output matrices.
405  T *from = fac[0];
406  T *sfrom = sterms[0];
407  T *mat = out_mat->begin();
408  T *inv = out_inv->begin();
409  std::copy(from, from+16, mat);
410  std::copy(from, from+16, inv);
411 
412  // add sine terms in.
413  for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
414  *mat += *sfrom;
415  *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
416  }
417  }

564  template <typename T, index_t N, StoragePolicy P>
565  AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N,P> &mat) {
566  SimpleMatrix<T,N,N> m_inv;
567  AffineTransform<T,N> xfnew;
568 
569  if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
570  throw NonsquareMatrixException(mat.rows(), mat.cols());
571  }
572 
573  inv(&m_inv, mat);
574  MatrixRegion region(MatrixCoord::zeros, MatrixCoord(mat.rows()));
575  std::copy( mat.begin(), mat.end(), xfnew.mat.region_begin(region));
576  std::copy(m_inv.begin(), m_inv.end(), xfnew.inv.region_begin(region));
577  return xfnew;
578  }
