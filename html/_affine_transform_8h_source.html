168  const Vec<T,N> applyVector(const Vec<T,N> &v) const {
169  Vec<T,N> o;
170  for (index_t r = 0; r < N; r++) {
171  for (index_t c = 0; c < N; c++) {
172  o[r] += mat(r,c) * v[c];
173  }
174  }

182  const Vec<T,N> applyNormal(const Vec<T,N> &n) const {
183  // normal matrix = txpose of inverse.

185  for (index_t r = 0; r < N; r++) {
186  for (index_t c = 0; c < N; c++) {

257  template <typename T>
258  void rotmat(SimpleMatrix<T,4,4> *into, T x, T y, T z, T theta) {
259  T c = std::cos(theta);
260  T s = std::sin(theta);
261  T c_1 = 1 - c;
262  T m[4][4] = {
263  {c + c_1*x*x, c_1*x*y - s*z, c_1*x*z + s*y, 0},
264  {c_1*y*x + s*z, c + c_1*y*y, c_1*y*z - s*x, 0},
265  {c_1*z*x - s*y, c_1*z*y + s*x, c + c_1*z*z, 0},
266  {0, 0, 0, 1}
267  };
268  T *from = m[0];
269  std::copy(from, from+16, into->begin());
270  }

278  // rotation from a quaternion
279  template <typename T>
280  void rotmat(SimpleMatrix<T,4,4> *into, const Quat<T> &q) {
281  T x = q.x; //for convenience
282  T y = q.y;
283  T z = q.z;
284  T w = q.w;
285  T x2 = x*x; // for speed
286  T y2 = y*y; // ...
287  T z2 = z*z;
288  T w2 = w*w;
289  T twox = 2*x;
290  T twoy = 2*y;
291  T twoz = 2*z;
292  T xy = twox * y;
293  T xz = twox * z;
294  T xw = twox * w;
295  T yz = twoy * z;
296  T yw = twoy * w;
297  T zw = twoz * w;
298  T m[4][4] = {
299  {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
300  {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
301  {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
302  {0, 0, 0, 1}
303  };
304  T *from = m[0];
305  std::copy(from, from+16, into->begin());
306  }

308  // rotation about an arbitrary point
309  template <typename T>
310  void rotmat(SimpleMatrix<T,4,4> *into,
311  T x, T y, T z,
312  T px, T py, T pz,
313  T theta) {
314  T c = std::cos(theta);
315  T s = std::sin(theta);
316  T c_1 = 1 - c;
317  T x2 = x*x;
318  T y2 = y*y;
319  T z2 = z*z;
320 
321  const T q0 = diff_of_products(py, z, pz, y);
322  const T q1 = diff_of_products(pz, x, px, z);
323  const T q2 = diff_of_products(px, y, py, x);
324  const T s0 = sum_of_products( py, y, pz, z);
325  const T s1 = sum_of_products( px, x, pz, z);
326  const T s2 = sum_of_products( px, x, py, y);
327  const T t0 = diff_of_products(px, y2 + z2, x, s0);
328  const T t1 = diff_of_products(py, x2 + z2, y, s1);
329  const T t2 = diff_of_products(pz, x2 + y2, z, s2);
330  const T u0 = sum_of_products( t0, c_1, q0, s);
331  const T u1 = sum_of_products( t1, c_1, q1, s);
332  const T u2 = sum_of_products( t2, c_1, q2, s);
333 
334  T m[4][4] = {
335  {x2 + c*(y2 + z2), x*y*c_1 - z*s, x*z*c_1 + y*s, u0},
336  {x*y*c_1 + z*s, y2 + c*(x2 + z2), y*z*c_1 - x*s, u1},
337  {x*z*c_1 - y*s, y*z*c_1 + x*s, z2 + c*(x2 + y2), u2},
338  {0, 0, 0, 1}
339  };
340  T *from = m[0];
341  std::copy(from, from+16, into->begin());
342  }

353  // rotation aligning `dir` with `alignWith`.
354  // modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
355  template <typename T>
356  inline void rotmat_direction_align(SimpleMatrix<T,4,4> *into,
357  const Vec<T,3> &dir,
358  const Vec<T,3> &alignWith) {
359  const Vec<T,3> v = dir ^ alignWith;
360  const T c = dir.dot(alignWith);
361 
362  // handle poles
363  if (std::abs(c) == 1) {
364  into->set_identity();
365  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
366  return;
367  }
368 
369  const T k = (1 - c) / (1 - c * c);
370 
371  T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
372  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
373  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
374  0, 0, 0, 1 };
375 
376  std::copy(m, m + 16, into->begin());
377  }

380  template <typename T>
381  inline void rotmat_direction_align(SimpleMatrix<T,3,3> *into,
382  const Vec<T,3> &dir,
383  const Vec<T,3> &alignWith) {
384  const Vec<T,3> v = dir ^ alignWith;
385  const T c = dir.dot(alignWith);
386 
387  // handle poles
388  if (std::abs(c) == 1) {
389  into->set_identity();
390  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
391  return;
392  }
393 
394  const T k = (1 - c) / (1 - c * c);
395 
396  T m[9] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y,
397  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,
398  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c};

404  // rotation about an arbitrary point (with inverse)
405  template <typename T>
406  void rotmat(SimpleMatrix<T,4,4> *out_mat,
407  SimpleMatrix<T,4,4> *out_inv,
408  T x, T y, T z,
409  T px, T py, T pz,
410  T theta) {
411  T c = std::cos(theta);
412  T s = std::sin(theta);
413  T c_1 = 1 - c;
414  T x2 = x*x;
415  T y2 = y*y;
416  T z2 = z*z;
417  T fac[4][4] = {
418  {x2 + c*(y2 + z2), x*y*c_1, x*z*c_1, (px*(y2 + z2) - x*(py*y + pz*z))*c_1},
419  {x*y*c_1, y2 + c*(x2 + z2), y*z*c_1, (py*(x2 + z2) - y*(px*x + pz*z))*c_1},
420  {x*z*c_1, y*z*c_1, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*c_1},
421  {0, 0, 0, 1}
422  };
423  T sterms[3][4] = {
424  {0, -z*s, y*s, (py*z - pz*y)*s},
425  {z*s, 0, -x*s, (pz*x - px*z)*s},
426  {-y*s, x*s, 0, (px*y - py*x)*s}
427  };
428 
429  // pre-fill output matrices.
430  T *from = fac[0];
431  T *sfrom = sterms[0];
432  T *mat = out_mat->begin();
433  T *inv = out_inv->begin();
434  std::copy(from, from+16, mat);
435  std::copy(from, from+16, inv);
436 
437  // add sine terms in.
438  for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
439  *mat += *sfrom;
440  *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
441  }
442  }

589  template <typename T, index_t N, MatrixLayout Lyt, StoragePolicy P>
590  AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N,Lyt,P> &mat) {
591  SimpleMatrix<T,N,N> m_inv;
592  AffineTransform<T,N> xfnew;
593 
594  if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
595  throw NonsquareMatrixException(mat.rows(), mat.cols());
596  }
597 
598  inv(&m_inv, mat);
599  Rect<index_t,2> region(Vec<index_t,2>::zeros, Vec<index_t,2>(mat.rows()));
600  std::copy( mat.begin(), mat.end(), xfnew.mat.region_begin(region));
601  std::copy(m_inv.begin(), m_inv.end(), xfnew.inv.region_begin(region));
602  return xfnew;
603  }
