31 // todo: We've picked a bad convention. xf3 * xf2 * xf1 * v is counterintuitive.
32 // We should have v * xf1 * xf2 * xf3, with v as a row vector; then
33 // the operations read in order from left to right.
34 // This also means that *= and /= operators will preserve ordering, and
35 // things like `v / xf` will be more consistent.
36 // It's also consistent with left-to-right function application:
37 // thing.apply(xf1).apply(xf2).apply(xf3) aligns with xf1 * xf2 * xf3.
38 // It also means that basis vectors will be rows, which means an array
39 // of basis vectors can be directly cast to a matrix, without copying,
40 // transposing, or tricks with layout templates.
41 // >> Can we change or template this choice so that at the very least,
42 // the user has the capability to choose their desired convention?

301 template <typename T>
302 void rotmat(SimpleMatrix<T,4,4>* into, T x, T y, T z, T theta) {
303  T c = std::cos(theta);
304  T s = std::sin(theta);
305  T c_1 = 1 - c;
306  T m[4][4] = {
307  {c + c_1*x*x, c_1*x*y - s*z, c_1*x*z + s*y, 0},
308  {c_1*y*x + s*z, c + c_1*y*y, c_1*y*z - s*x, 0},
309  {c_1*z*x - s*y, c_1*z*y + s*x, c + c_1*z*z, 0},
310  {0, 0, 0, 1}
311  };
312  T *from = m[0];
313  std::copy(from, from+16, into->begin());
314 }

323 template <typename T>
324 void rotmat(SimpleMatrix<T,4,4>* into, const Quat<T>& q) {
325  T x = q.x; //for convenience
326  T y = q.y;
327  T z = q.z;
328  T w = q.w;
329  T x2 = x*x; // for speed
330  T y2 = y*y; // ...
331  T z2 = z*z;
332  T w2 = w*w;
333  T twox = 2*x;
334  T twoy = 2*y;
335  T twoz = 2*z;
336  T xy = twox * y;
337  T xz = twox * z;
338  T xw = twox * w;
339  T yz = twoy * z;
340  T yw = twoy * w;
341  T zw = twoz * w;
342  T m[4][4] = {
343  {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
344  {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
345  {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
346  {0, 0, 0, 1}
347  };
348  T *from = m[0];
349  std::copy(from, from+16, into->begin());
350 }

353 template <typename T>
354 void rotmat(
355  SimpleMatrix<T,4,4>* into,
356  T x, T y, T z,
357  T px, T py, T pz,
358  T theta)
359 {
360  T c = std::cos(theta);
361  T s = std::sin(theta);
362  T c_1 = 1 - c;
363  T x2 = x*x;
364  T y2 = y*y;
365  T z2 = z*z;
366 
367  const T q0 = diff_of_products(py, z, pz, y);
368  const T q1 = diff_of_products(pz, x, px, z);
369  const T q2 = diff_of_products(px, y, py, x);
370  const T s0 = sum_of_products( py, y, pz, z);
371  const T s1 = sum_of_products( px, x, pz, z);
372  const T s2 = sum_of_products( px, x, py, y);
373  const T t0 = diff_of_products(px, y2 + z2, x, s0);
374  const T t1 = diff_of_products(py, x2 + z2, y, s1);
375  const T t2 = diff_of_products(pz, x2 + y2, z, s2);
376  const T u0 = sum_of_products( t0, c_1, q0, s);
377  const T u1 = sum_of_products( t1, c_1, q1, s);
378  const T u2 = sum_of_products( t2, c_1, q2, s);
379 
380  T m[4][4] = {
381  {x2 + c*(y2 + z2), x*y*c_1 - z*s, x*z*c_1 + y*s, u0},
382  {x*y*c_1 + z*s, y2 + c*(x2 + z2), y*z*c_1 - x*s, u1},
383  {x*z*c_1 - y*s, y*z*c_1 + x*s, z2 + c*(x2 + y2), u2},
384  {0, 0, 0, 1}
385  };
386  T *from = m[0];
387  std::copy(from, from+16, into->begin());
388 }

401 // rotation aligning `dir` with `alignWith`.
402 // modified from http://www.iquilezles.org/www/articles/noacos/noacos.htm
403 template <typename T>
404 inline void rotmat_direction_align(
405  SimpleMatrix<T,4,4>* into,
406  const Vec<T,3>& dir,
407  const Vec<T,3>& alignWith)
408 {
409  const Vec<T,3> v = dir ^ alignWith;
410  const T c = dir.dot(alignWith);
411 
412  // handle poles
413  if (std::abs(c) == 1) {
414  into->set_identity();
415  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
416  return;
417  }
418 
419  const T k = (1 - c) / (1 - c * c);
420 
421  T m[16] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y, 0,
422  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x, 0,
423  v.x*v.z*k - v.y, v.y*v.z*k + v.x, v.z*v.z*k + c, 0,
424  0, 0, 0, 1 };
425 
426  std::copy(m, m + 16, into->begin());
427 }

431 inline void rotmat_direction_align(
432  SimpleMatrix<T,3,3>* into,
433  const Vec<T,3>& dir,
434  const Vec<T,3>& alignWith)
435 {
436  const Vec<T,3> v = dir ^ alignWith;
437  const T c = dir.dot(alignWith);
438 
439  // handle poles
440  if (std::abs(c) == 1) {
441  into->set_identity();
442  if (c < 0) { into->set(0,0,-1); into->set(2,2,-1); }
443  return;
444  }
445

447 
448  T m[9] = { v.x*v.x*k + c, v.y*v.x*k - v.z, v.z*v.x*k + v.y,
449  v.x*v.y*k + v.z, v.y*v.y*k + c, v.z*v.y*k - v.x,

456 // rotation about an arbitrary point (with inverse)
457 template <typename T>
458 void rotmat(
459  SimpleMatrix<T,4,4> *out_mat,
460  SimpleMatrix<T,4,4> *out_inv,
461  T x, T y, T z,
462  T px, T py, T pz,
463  T theta)
464 {
465  T c = std::cos(theta);
466  T s = std::sin(theta);
467  T c_1 = 1 - c;
468  T x2 = x*x;
469  T y2 = y*y;
470  T z2 = z*z;
471  T xyc = x*y*c_1;
472  T xzc = x*z*c_1;
473  T yzc = y*z*c_1;
474  T da = x2 + c*(y2 + z2);
475  T db = y2 + c*(x2 + z2);
476  T dc = z2 + c*(x2 + y2);
477 
478  T fac[4][4] = {
479  {da, xyc, xzc, (px*(y2 + z2) - x*(py*y + pz*z))*c_1},
480  {xyc, db, yzc, (py*(x2 + z2) - y*(px*x + pz*z))*c_1},
481  {xzc, yzc, dc, (pz*(x2 + y2) - z*(px*x + py*y))*c_1},
482  {0, 0, 0, 1}
483  };
484  T sterms[3][4] = {
485  {0, -z*s, y*s, (py*z - pz*y)*s},
486  {z*s, 0, -x*s, (pz*x - px*z)*s},
487  {-y*s, x*s, 0, (px*y - py*x)*s}
488  };
489 
490  // pre-fill output matrices.
491  T* from = fac[0];
492  T* sfrom = sterms[0];
493  T* mat = out_mat->begin();
494  T* inv = out_inv->begin();
495  std::copy(from, from + 16, mat);
496  std::copy(from, from + 16, inv);
497 
498  // add sine terms in.
499  for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
500  *mat += *sfrom;
501  *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
502  }
503 }

658 template <typename T, index_t N, MatrixLayout Lyt, StoragePolicy P>
659 AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N,Lyt,P>& mat) {
660  SimpleMatrix<T,N,N> m_inv;
661  AffineTransform<T,N> xfnew;
662 
663  if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
664  throw NonsquareMatrixException(mat.rows(), mat.cols());
665  }
666 
667  // calculate the inverse
668  inv(&m_inv, mat);
669 
670  // copy the sub-matrices
671  for (index_t r = 0; r < mat.rows(); ++r) {
672  for (index_t c = 0; c < mat.cols(); ++c) {
673  xfnew.inv(r,c) = m_inv(r,c);
674  xfnew.mat(r,c) = mat(r,c);
675  }
676  }
677 
678  return xfnew;
679 }
