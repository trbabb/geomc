311 bool operator==(const Tree<NodeItem, LeafItem>& other) const {
312 if (this == &other) return true;
313 if (item_count() != other.item_count()) return false;
314 if (_items != other._items) return false;
315
316 // because the references will point to different memory,
317 // we can't compare raw data. we have to examine the structure of the tree.
318 auto a_n = _nodes.begin();
319 auto b_n = other._nodes.begin();
320 for (; a_n != _nodes.end() and b_n != other._nodes.end(); ++a_n, ++b_n) {
321 if (a_n->n_items != b_n->n_items) return false;
322 if (a_n->n_children != b_n->n_children) return false;
323 if (a_n->data != b_n->data) return false;
324 }
325
326 // must have same total number of nodes; not just a matching prefix subtree.
327 // we don't check item_count() beforehand because it may not be stored,
328 // and will thus doubly incur the linear cost we have pay above.
329 if (not (a_n == _nodes.end() and b_n == other._nodes.end())) return false;
330
331 return true;
332 }

353 // called when a leaf node adds or removes one or more of its edge items.
354 // the edge items for the anscestors potentially need to be updated to
355 // include/exclude the affected items.
356 void update_boundary_items(
357 NodeRef leaf,
358 ItemRef new_first,
359 ItemRef new_last,
360 index_t delta_items) {
361
362 NodeRef prev_child = _nodes.end();
363 NodeRef n = leaf;
364
365 ItemRef old_first = n->items_first;
366 ItemRef old_last = n->items_last;
367
368 // update ancestors' item count and boundary items
369 while (n != _nodes.end()) {
370 n->n_items += delta_items;
371
372 if (n->n_items == 0) {
373 n->items_first = n->items_last = _items.end();
374 } else {
375 // If we ascended from an edge node, that node certainly
376 // controls (or used to control) our edge item.
377 // But it might have also controlled our edge item if the
378 // edge sibling doesn't have any items. In this case,
379 // its previous edge item will also be ours.
380 if (n->child_first == prev_child or
381 n->items_first == old_first) {
382 n->items_first = new_first;
383 }
384 if (n->child_last == prev_child or
385 n->items_last == old_last) {
386 n->items_last = new_last;
387 }
388 }
389
390 prev_child = n;
391 n = n->parent;
392 }
393 }

543 void next_candidate() {

785 template <bool BoundingFn(const NodeItem&, const LeafItem&)>
786 self_t find_parent(const const_item_iterator& i) const {
787 NodeRef cur_root = _root;
788 NodeRef subtree_last = node_successor(_root);
789 while (cur_root != subtree_last) {
790 if (BoundingFn(cur_root->data, *i)) {
791 // cur_root may contain `i`
792 if (cur_root->n_children > 0) {
793 // recurse
794 cur_root = cur_root->child_first;
795 } else {
796 // look for item
797 ItemRef end_item = cur_root->items_last; ++end_item;
798 for (ItemRef j = cur_root->items_first; j != end_item; ++j) {
799 // found it
800 if (j == i) return self_t(cur_root, _storage);
801 }
802 // our current node is a leaf, so there is no subtree to jump over.
803 ++cur_root;
804 }
805 } else {
806 // check next sibling
807 cur_root = node_successor(cur_root);
808 }
809 }
810 return this->end();;
811 }

848 // compute the position of the node that follows `node`'s subtree in
849 // "sibling-first" order. the successor to `node` occurs just before
850 // the next subtree to the right. an `O(k * log(n))` operation, on `n`
851 // the size of the tree, and `k` the arity. this is different from
852 // `++node` because `++node` may descend into the current node's
853 // subtree; we want to exactly hop over it.
854 NodeRef node_successor(const NodeRef& node) const {
855 NodeRef ancestor = node->parent;
856 NodeRef child = node;
857 // if there is no subtree to the right,
858 // this is the conceptual successor:
859 NodeRef successor = _storage->_nodes.end();
860
861 while (ancestor != _storage->_nodes.end()) {
862 NodeRef off_end_child = ancestor->child_last;
863 if (ancestor->n_children > 0) ++off_end_child;
864 // look for a populated node to the right of us
865 for (NodeRef c = child; c != off_end_child; ++c) {
866 if (c->n_children > 0) {
867 successor = c->child_first;
868 break;
869 }
870 }
871 if (successor != _storage->_nodes.end()) {
872 // found the successor
873 break;
874 } else {
875 // ascend the tree
876 child = ancestor;
877 ancestor = ancestor->parent;
878 }
879 }
880 return successor;
881 }

883
884 // compute the first item that occurs after the last item in the subtree
885 // rooted at `node`, including if that subtree is empty.
886 ItemRef item_successor(NodeRef node) const {
887 for (NodeRef parent = node->parent;
888 parent != _storage->_nodes.end();
889 node = parent, parent = node->parent) {
890 if (parent->n_children != 0) {
891 NodeRef end_c = parent->child_last; ++end_c;
892 for (NodeRef c = node; c != end_c; ++c) {
893 if (c->n_items > 0) {
894 return c->items_first;
895 }
896 }
897 }
898 }
899 return this->_storage->_items.end();
900 }

1003 template <typename ... Args>
1004 self_t insert_child_node(
1005 const self_t& insert_before,
1006 Args&& ... args) const {
1007 NodeRef p = this->_root;
1008 NodeRef n = insert_before._root;
1009 bool parent_childless = p->n_children == 0;
1010
1011 // check invalid cases
1012 if (n == this->_storage->_nodes.begin()) return this->end();
1013 if (n->parent != p and n != this->end()._root) return this->end();
1014
1015 // set the insert point for empty nodes
1016 if (parent_childless) {
1017 n = this->node_successor(p);
1018 }

1402 template <typename Func, typename P, typename ... Args>
1403 self_t split(Func compare, P pivot, Args&& ... args) const {
1404 if (this->_root == this->_storage->_nodes.begin()
1405 or this->node_count() > 0
1406 or this->item_count() < 2) {
1407 return this->end();
1408 }
1409
1410 self_t new_node = this->parent().insert_child_node(*this, args...);
1411 NodeRef hi = this->_root;
1412 NodeRef lo = new_node._root;
1413
1414 // todo: if we had splice operations on list<>, we could avoid
1415 // invalidating the item iterators.
1416
1417 // traverse `node` looking for items to move to the new `lo` node.
1418 // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
1419 // boundary between `lo` and `hi` and swapping the contents of the iterators.
1420 // we traverse backwards so that our swapping does not unsort our items.
1421
1422 for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
1423 if (compare(*i, pivot) < 0) {
1424 // move the item to the lower node
1425
1426 // grab one item from the beginning of `hi`.
1427 // swap `*i` and the stolen item, so that the
1428 // tested item lies in the low node where it belongs,
1429 // and the stolen item lies under `i`. we'll check it
1430 // on the next iteration.
1431
1432 if (lo->n_items == 0) {
1433 lo->items_first = lo->items_last = hi->items_first;
1434 } else {
1435 lo->items_last = hi->items_first;
1436 }
1437
1438 ++(lo->n_items);
1439 --(hi->n_items);
1440 ++(hi->items_first);
1441 std::swap(*i, *(lo->items_last));
1442
1443 // we do not decrement i, because we've just moved
1444 // an untested item underneath it. we should test
1445 // it first before moving on to the next one.
1446 } else {
1447 // `*i` is in the right place.
1448 --i;
1449 }
1450 }
1451
1452 // if the `hi` node was emptied, fix up its item boundaries.
1453 if (hi->n_items == 0) {
1454 hi->items_first = hi->items_last = this->_storage->_items.end();
1455 }
1456
1457 // we didn't delete/add/move any LeafItems (only their contents),
1458 // so we don't need to update the boundary items.
1459
1460 return new_node;
1461 }

1480 item_iterator erase(const item_iterator& item, bool* success=nullptr) const {
1481 item_iterator ret = this->items_end();
1482 bool ok = false;
1483 NodeRef n = this->_root;
1484
1485 if (n->n_children == 0) {
1486 // verify that the item actually belongs to us.
1487 // (otherwise hell could break loose).
1488 // this is a linear check; if we *knew* we were the parent, the
1489 // entire operation would be log(n). laaaaame.
1490 bool found = false;
1491 for (item_iterator i = n->items_begin();
1492 i != n->items_end();
1493 ++i) {
1494 if (i == item) {
1495 found = true;
1496 break;
1497 }
1498 }
1499
1500 if (found) {
1501 ItemRef next_item = this->_storage->_items.erase(item);
1502 ItemRef prev_item = next_item; --prev_item;
1503
1504 // figure out the new begin / end items
1505 ItemRef new_begin, new_end;
1506 if (item == n->items_first) {
1507 new_begin = next_item;
1508 } else {
1509 new_begin = n->items_first;
1510 }
1511 if (item == n->items_last) {
1512 new_end = prev_item;
1513 } else {
1514 new_end = prev_item;
1515 }
1516
1517 this->_storage->update_boundary_items(
1518 n,
1519 new_begin,
1520 new_end,
1521 -1);
1522
1523 ok = true;
1524 ret = next_item;
1525 }
1526 }

1606 void flatten() const {
1607 if (this->node_count() == 0) return; // nothing to do.
1608
1609 // because of sibling-first order,
1610 // this is exactly all the descendents of `_root`:
1611 this->_storage->_nodes.erase(
1612 this->_root->child_first,
1613 this->node_successor(this->_root));
1614
1615 this->_root->n_children = 0;
1616 this->_root->child_first =
1617 this->_root->child_last =
1618 this->_storage->_nodes.end();

1625protected:
1626
1627 // use all the child counts to recompute the first/last
1628 // references in internal nodes. useful when taking ownership
1629 // of another tree's internal node list.
1630 // preconditions:
1631 // - first item AND child pointers of this node are correct
1632 // - the descendents of this subtree have correct child and item counts
1633 // - the nodes belonging to this subtree exist and are correctly ordered
1634 // and placed in the tree
1635 // - item boundaries of all ancestors are already correct
1636 // - item counts of all ancestors are already correct
1637 // This should be an effective no-op if the subtree is already owned
1638 // (though it may still take O(n) to execute).
1639 void recalculate_references() const {
1640 NodeRef begin_child = this->_root->child_first;
1641 // next unassigned leaf item:
1642 ItemRef begin_items = this->_root->items_first;
1643 // last / most recent assigned leaf item in the list:
1644 ItemRef last_item = begin_items; --last_item;
1645 NodeRef end_child = this->node_successor(this->_root);
1646
1647 // in essence we "build" the tree by simply taking ownership of
1648 // the nodes/items that already exist. we walk over the nodes in
1649 // tree order, which means that child nodes are taken in order too.
1650 for (NodeRef node = this->_root; node != end_child; ) {
1651 node->child_first = node->child_last =
1652 (node->n_children > 0) ? begin_child : this->_storage->_nodes.end();
1653 node->items_first = node->items_last =
1654 (node->n_items > 0) ? begin_items : this->_storage->_items.end();
1655
1656 if (node->n_children > 0) {
1657 node->child_first->parent = node;
1658
1659 // acquire direct children
1660 for (index_t c = 0; c < node->n_children - 1; ++c) {
1661 ++(node->child_last);
1662 node->child_last->parent = node;
1663 }
1664
1665 // move the node "free space"
1666 begin_child = node->child_last; ++begin_child;
1667
1668 // populate first child's subtree
1669 node = node->child_first;
1670 } else {
1671 if (node->n_items > 0) {
1672 for (index_t i = 0; i < node->n_items - 1; ++i) {
1673 ++(node->items_last);
1674 }
1675 // move the item "free space"
1676 begin_items = last_item = node->items_last; ++begin_items;
1677 }
1678
1679 // fixup the items_end along the right edge of this subtree.
1680 NodeRef descendent = node;
1681 for (NodeRef ancestor = node->parent;
1682 ancestor != this->_root;
1683 descendent = ancestor, ancestor = ancestor->parent) {
1684 if (ancestor->n_items > 0) ancestor->items_last = last_item;
1685 }
1686
1687 // done populating this tree. populate the next sibling tree.
1688 node = this->node_successor(node);
1689 }
1690 }
1691 }
