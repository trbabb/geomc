667  template <bool BoundingFn(const NodeItem&, const LeafItem&)>
668  self_t find_parent(const const_item_iterator& i) const {
669  NodeRef cur_root = _root;
670  NodeRef subtree_last = node_successor(_root);
671  while (cur_root != subtree_last) {
672  if (BoundingFn(cur_root->data, *i)) {
673  // cur_root may contain `i`
674  if (cur_root->n_children > 0) {
675  // recurse
676  cur_root = cur_root->child_first;
677  } else {
678  // look for item
679  ItemRef end_item = cur_root->items_last; ++end_item;
680  for (ItemRef j = cur_root->items_first; j != end_item; ++j) {
681  // found it
682  if (j == i) return self_t(cur_root, _storage);
683  }
684  // our current node is a leaf, so there is no subtree to jump over.
685  ++cur_root;
686  }
687  } else {
688  // check next sibling
689  cur_root = node_successor(cur_root);
690  }
691  }
692  return self_t(_storage->_nodes.end(), _storage);
693  }

697  // compute the position of the node that follows `node`'s subtree in
698  // "sibling-first" order. the successor to `node` occurs just before
699  // the next subtree to the right. an `O(k * log(n))` operation, on `n`
700  // the size of the tree, and `k` the arity. this is different from
701  // `++node` because `++node` may descend into the current node's
702  // subtree; we want to exactly hop over it.
703  NodeRef node_successor(const NodeRef& node) const {
704  NodeRef ancestor = node->parent;
705  NodeRef child = node;
706  // if there is no subtree to the right,
707  // this is the conceptual successor:
708  NodeRef successor = _storage->_nodes.end();
709 
710  while (ancestor != _storage->_nodes.end()) {
711  NodeRef off_end_child = ancestor->child_last;
712  if (ancestor->n_children > 0) ++off_end_child;
713  // look for a populated node to the right of us
714  for (NodeRef c = child; c != off_end_child; ++c) {
715  if (c->n_children > 0) {
716  successor = c->child_first;
717  break;
718  }
719  }
720  if (successor != _storage->_nodes.end()) {
721  // found the successor
722  break;
723  } else {
724  // ascend the tree
725  child = ancestor;
726  ancestor = ancestor->parent;
727  }
728  }
729  return successor;
730  }

733  // compute the first item that occurs after the last item in the subtree

735  ItemRef item_successor(NodeRef node) const {
736  for (NodeRef parent = node->parent;
737  parent != _storage->_nodes.end();

853  bool parent_childless = p->n_children == 0;
854

856  if (n == this->_storage->_nodes.begin()) return this->end();
857  if (n->parent != p and n != this->end()._root) return this->end();
858 
859  // set the insert point for empty nodes
860  if (parent_childless) {
861  n = this->node_successor(p);

872  // (every item must belong to a leaf node, and the parent
873  // has just become an internal node).

876  new_node->items_last = p->items_last;

882  p->child_last = new_node;
883  }
884  if (insert_before._root == p->child_first) {
885  p->child_first = new_node;

1245  template <typename Func, typename P, typename ... Args>
1246  self_t split(Func compare, P pivot, Args&& ... args) const {
1247  if (this->_root == this->_storage->_nodes.begin() or this->node_count() > 0) {
1248  return this->end();
1249  }
1250 
1251  self_t new_node = this->parent().insert_child_node(*this, args...);
1252  NodeRef hi = this->_root;
1253  NodeRef lo = new_node._root;
1254 
1255  // todo: if we had splice operations on list<>, we could avoid
1256  // invalidating the item iterators.
1257 
1258  // traverse `node` looking for items to move to the new `lo` node.
1259  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
1260  // boundary between `lo` and `hi` and swapping the contents of the iterators.
1261  // we traverse backwards so that our swapping does not unsort our items.
1262 
1263  // corner case: hi is / becomes empty
1264  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
1265  if (compare(*i, pivot) < 0) {
1266  // move the item to the lower node
1267 
1268  // grab one item from the beginning of `hi`.
1269  // swap `*i` and the stolen item, so that the
1270  // tested item lies in the low node where it belongs,
1271  // and the stolen item lies under `i`. we'll check it
1272  // on the next iteration.
1273 
1274  if (lo->n_items == 0) {
1275  lo->items_first = lo->items_last = hi->items_first;
1276  } else {
1277  lo->items_last = hi->items_first;
1278  }
1279 
1280  ++(lo->n_items);
1281  --(hi->n_items);
1282  ++(hi->items_first);
1283  std::swap(*i, *(lo->items_last));
1284 
1285  // we do not decrement i, because we've just moved
1286  // an untested item underneath it. we should test
1287  // it first before moving on to the next one.
1288  } else {
1289  // `*i` is in the right place.
1290  --i;
1291  }
1292  }
1293 
1294  // if the `hi` node was emptied, fix up its item boundaries.
1295  if (hi->n_items == 0) {
1296  hi->items_first = hi->items_last = this->_storage->_items.end();
1297  }
1298 
1299  // we didn't delete/add any LeafItems, so we don't need to
1300  // update the boundary items.
1301 
1302  return new_node;
1303  }

1471 
1472  // use all the child counts to recompute the first/last
1473  // references in internal nodes. useful when taking ownership
1474  // of another tree's internal node list.
1475  // preconditions:
1476  // - first item AND child pointers of this node are correct
1477  // - the descendents of this subtree have correct child and item counts
1478  // - the nodes belonging to this subtree exist and are correctly ordered
1479  // and placed in the tree
1480  // - item boundaries of all ancestors are already correct
1481  // - item counts of all ancestors are already correct
1482  // This should be an effective no-op if the subtree is already owned
1483  // (though it may still take O(n) to execute).
1484  void recalculate_references() const {
1485  NodeRef begin_child = this->_root->child_first;
1486  // next unassigned leaf item:
1487  ItemRef begin_items = this->_root->items_first;
1488  // last / most recent assigned leaf item in the list:
1489  ItemRef last_item = begin_items; --last_item;
1490  NodeRef end_child = this->node_successor(this->_root);
1491 
1492  // in essence we "build" the tree by simply taking ownership of
1493  // the nodes/items that already exist. we walk over the nodes in
1494  // tree order, which means that child nodes are taken in order too.
1495  for (NodeRef node = this->_root; node != end_child; ) {
1496  node->child_first = node->child_last =
1497  (node->n_children > 0) ? begin_child : this->_storage->_nodes.end();
1498  node->items_first = node->items_last =
1499  (node->n_items > 0) ? begin_items : this->_storage->_items.end();
1500 
1501  if (node->n_children > 0) {
1502  node->child_first->parent = node;
1503 
1504  // acquire direct children
1505  for (index_t c = 0; c < node->n_children - 1; ++c) {
1506  ++(node->child_last);
1507  node->child_last->parent = node;
1508  }
1509 
1510  // move the node "free space"
1511  begin_child = node->child_last; ++begin_child;
1512 
1513  // populate first child's subtree
1514  node = node->child_first;
1515  } else {
1516  if (node->n_items > 0) {
1517  for (index_t i = 0; i < node->n_items - 1; ++i) {
1518  ++(node->items_last);
1519  }
1520  // move the item "free space"
1521  begin_items = last_item = node->items_last; ++begin_items;
1522  }
1523 
1524  // fixup the items_end along the right edge of this subtree.
1525  NodeRef descendent = node;
1526  for (NodeRef ancestor = node->parent;
1527  ancestor != this->_root;
1528  descendent = ancestor, ancestor = ancestor->parent) {
1529  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1530  }
1531

1533  node = this->node_successor(node);
