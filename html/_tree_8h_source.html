474  item_iterator insert(const node_iterator& parent, const LeafItem& obj) { // todo: move semantics?
475  if (parent.nodes() > 0) return items.end();
476 
477  // this is a bit complex and special case-y, but fundamentally it is O(log(n)).
478  // we will walk the tree vertically from the insertion node at least once and at most twice.
479  // most of this complexity has to do with handling the case where the new item becomes
480  // the new first_item or last_item of its ancestor(s). furthermore, the bulk of this is only
481  // paid when the first item is added to an empty node.
482 
483  ItemRef new_item;
484  NodeRef ancestor = parent.node;
485 
486  if (parent.items() > 0) {
487  // basic/common case: node already has items.
488  ItemRef old_item = ancestor->items_first;
489  ItemRef insert_pos = old_item; ++insert_pos;
490  new_item = m_items.insert(insert_pos, obj);
491 
492  // walk up the tree updating the boundary (if necessary) and the item count.
493  while (ancestor != m_nodes.end()) {
494  ancestor->n_items++;
495  if (ancestor->items_last == old_item) {
496  ancestor->items_last = new_item;
497  }
498  ancestor = ancestor->parent;
499  }
500  } else {
501  // adding the first item to an empty node.
502  // find the first ancestor with childen
503  NodeRef cur_node = parent.node;
504  while (cur_node->n_items == 0 and cur_node->parent != m_nodes.end()) {
505  ancestor = cur_node;
506  cur_node = cur_node->parent;
507  }
508 
509  // find a populated subtree adjacent to the inserted item's ancestor.
510  // to ensure that items are contiguous in a tree, we must place the new item between them.
511  NodeRef populated_left = m_nodes.end();
512  NodeRef populated_right = m_nodes.end();
513  bool hit_child = false;
514  // walk the children of cur_node looking for a populated subtree.
515  for (NodeRef sibling = cur_node->child_first; true; ++sibling) {
516  if (sibling == ancestor) {
517  hit_child = true; // child abuse
518  continue;
519  }
520  if (sibling->n_items > 0) {
521  // found a populated subtree.
522  if (not hit_child) {
523  // it's on the left side of the ancestor.
524  populated_left = sibling;
525  } else {
526  // it's on the right side of the ancestor.
527  populated_right = sibling;
528  break; // can't discover any new info now.
529  }
530  }
531  if (sibling == cur_node->child_last) break;
532  }
533 
534  // place the new item between a populated subtree and the current one.
535  bool empty_root = false;
536  if (populated_left != m_nodes.end()) {
537  // populated subtree is to the left of us; place new item to the right of it
538  ItemRef tmp = populated_left->child_last; ++tmp;
539  new_item = m_items.insert(tmp, obj);
540  } else {
541  if (populated_right != m_nodes.end()) {
542  // populated subtree is to the right of us; place new item to the left of it
543  new_item = m_items.insert(populated_right->child_first, obj);
544  } else {
545  // empty root node.
546  new_item = m_items.insert(m_items.end(), obj);
547  cur_node->items_first = cur_node->items_last = new_item;
548  empty_root = true;
549  }
550  }
551 
552  // update the boundaries of the parent subtree, if necessary.
553  // we may have inserted an item on the "outside" of the parent's first/last child.
554  // (because we have at least one sibling to the left or right of us with items,
555  // this can only happen on one of the two boundaries).
556  if ((populated_left == m_nodes.end() or populated_right == m_nodes.end()) and not empty_root) {
557  bool left_edge = populated_left == m_nodes.end(); // was it the left edge that was empty?
558  ItemRef old_edge = (left_edge) ? (cur_node->items_first) : (cur_node->items_last);
559  while (cur_node != m_nodes.end()) {
560  NodeRef& update = (left_edge) ? (cur_node->items_first) : (cur_node->items_last);
561  if (update == old_edge) {
562  update = new_item;
563  } else {
564  break;
565  }
566  cur_node = cur_node->parent;
567  }
568  }
569 
570  // walk upwards from the item's parent.
571  // - update the item counts
572  // - set the boundary items for all newly-nonempty ancestors of the placed item
573  ancestor = parent.node;
574  while (ancestor != m_nodes.end()) {
575  ancestor->n_items++;
576  if (ancestor->items_first == m_items.end()) {
577  ancestor->items_first = ancestor->items_last = new_item;
578  }
579  ancestor = ancestor->parent;
580  }
581  }
582 
583  return item_iterator(new_item, parent.node);
584  }

708  template <typename Func, typename P>
709  node_iterator split(node_iterator& node, Func compare, P pivot, NodeItem& obj) {
710  if (node == m_nodes.begin() or node.nodes() > 0) return m_nodes.end();
711 
712  node_iterator new_node = insert_sibling_node(node, obj);
713  NodeRef hi = node.node;
714  NodeRef lo = new_node.node;
715 
716  // traverse `node` looking for items to move to the new `lo` node.
717  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
718  // boundary between `lo` and `hi` and swapping the contents of the iterators.
719  // we traverse backwards so that our swapping does not unsort our items.
720  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
721  if (compare(*i, pivot) < 0) {
722  // move the item to the lower node
723 
724  // grab one item from the beginning of `hi`.
725  // swap `*i` and the stolen item, so that the
726  // tested item lies in the low node where it belongs,
727  // and the stolen item lies under `i`. we'll check it
728  // on the next iteration.
729 
730  if (lo->n_items == 0) {
731  lo->items_first = lo->items_last = hi->items_first;
732  } else {
733  lo->items_last = hi->items_first;
734  }
735 
736  ++lo->n_items;
737  --hi->n_items;
738  ++hi->items_first;
739  std::swap(*i, *(lo->items_last));
740 
741  // we do not decrement i, because we've just moved
742  // an untested item underneath it. we should test
743  // it first before moving on to the next one.
744  } else {
745  // `*i` is in the right place.
746  --i;
747  }
748  }
749  }

955  // the successor to `node` occurs just before the next subtree to the right.

959  NodeRef child = node;

968  successor = child->child_first;

979  }
