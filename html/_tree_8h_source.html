105  // internally, nodes are kept in "sibling-first" order:
106  // - the successor of a node is its next sibling, if there is one,
107  // - otherwise the successor of a node is the first child of
108  // the first sibling with a child, if there is one,
109  // - otherwise the successor is the first child of the first
110  // subtree above and to the right, if there is one,
111  // - otherwise the successor is the end node.
112  // as such, all direct children of a node are contiguous, and come
113  // after (though not necessarily *immediately* after) their parent.
114  // all the descendents of a leftward child come before all the
115  // descendents of the child to its right. Therefore, any given subtree
116  // (excluding its root) is contiguous. It is this recursive pattern:
117 
118  // [node] ... [c1 c2 c3 c4 ...]
119  // [children of c1] [descendents of c1...]
120  // [children of c2] [descendents of c2...]
121  // ...

446  // compute the position of the node that follows `node`'s subtree in "sibling-first" order.
447  // the successor to `node` occurs just before the next subtree to the right.
448  // an `O(k * log(n))` operation, on `n` the size of the tree, and `k` the arity.
449  // this is different from `++node` because `++node` may descend into the
450  // current node's subtree; we want to exactly hop over it.
451  NodeRef node_successor(const NodeRef& node) const {
452  NodeRef ancestor = node->parent;
453  NodeRef child = node;
454  NodeRef successor = nodes_end()._root; // if there is no subtree to the right,
455  // this is the conceptual successor.
456  while (ancestor != _storage->_nodes.end()) {
457  NodeRef off_end_child = ancestor->child_last;
458  if (ancestor->n_children > 0) ++off_end_child;
459  // look for a populated node to the right of us
460  while (++child != off_end_child) {
461  if (child->n_children > 0) {
462  successor = child->child_first;
463  break;
464  }
465  }
466  if (successor != _storage->_nodes.end()) {
467  // found the successor
468  break;
469  } else {
470  // ascend the tree
471  child = ancestor;
472  ancestor = ancestor->parent;
473  }
474  }
475  return successor;
476  }

479  // compute the first item that occurs after the last item in the subtree
480  // rooted at `node`, including if that subtree is empty.
481  ItemRef item_successor(NodeRef node) const {
482  for (NodeRef parent = node->parent;
483  parent != _storage->_nodes.end();
484  node = parent, parent = node->parent) {
485  NodeRef end_c = parent->child_last;
486  if (parent->n_children != 0) ++end_c;
487  for (; node != end_c; ++node) {
488  if (node->n_items > 0) {
489  return node->items_first;
490  }
491  }
492  }
493  return items_end();
494  }

879  template <typename Func, typename P>
880  self_t split(Func compare, P pivot, NodeItem& obj) {
881  if (_root == _storage->_nodes.begin() or node_count() > 0) {
882  return self_t(_storage->_nodes.end(), _storage);
883  }
884 
885  self_t new_node = insert_child_node(_root->parent, _root, obj);
886  NodeRef hi = _root;
887  NodeRef lo = new_node._root;
888 
889  // todo: if we had splice operations on list<>, we could avoid
890  // invalidating the item iterators.
891 
892  // traverse `node` looking for items to move to the new `lo` node.
893  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
894  // boundary between `lo` and `hi` and swapping the contents of the iterators.
895  // we traverse backwards so that our swapping does not unsort our items.
896  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
897  if (compare(*i, pivot) < 0) {
898  // move the item to the lower node
899 
900  // grab one item from the beginning of `hi`.
901  // swap `*i` and the stolen item, so that the
902  // tested item lies in the low node where it belongs,
903  // and the stolen item lies under `i`. we'll check it
904  // on the next iteration.
905 
906  if (lo->n_items == 0) {
907  lo->items_first = lo->items_last = hi->items_first;
908  } else {
909  lo->items_last = hi->items_first;
910  }
911 
912  ++lo->n_items;
913  --hi->n_items;
914  ++hi->items_first;
915  std::swap(*i, *(lo->items_last));
916 
917  // we do not decrement i, because we've just moved
918  // an untested item underneath it. we should test
919  // it first before moving on to the next one.
920  } else {
921  // `*i` is in the right place.
922  --i;
923  }
924  }
925  return new_node;
926  }

1089  void recalculate_references() {

1092  ItemRef begin_items = _storage->_items.begin(); // next unassigned leaf item
1093  ItemRef last_item = _storage->_items.end(); // last / most recent assigned leaf item in the list
1094

1096  // the nodes/items that already exist. we walk over the nodes in
1097  // tree order, which means that child nodes are taken in order too.
1098  // this spares us recursion (i.e. no possibility of stack-blowing)
1099  // and some logic keeping track of descent/ascent.

1102  nod->items_first = nod->items_last = (nod->n_items > 0) ? begin_items : _storage->_items.end();

1105  nod->child_first->parent = nod;
1106

1108  for (index_t c = 0; c < nod->n_children - 1; ++c) {
1109  ++nod->child_last;
1110  nod->child_last->parent = nod;
1111  }
1112

1114  begin_child = nod->child_last; ++begin_child;
1115  } else {

1117  for (index_t i = 0; i < nod->n_items - 1; ++i) {

1119  }

1122  }
