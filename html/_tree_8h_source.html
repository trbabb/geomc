790  template <typename Func, typename P>
791  node_iterator split(node_iterator& node, Func compare, P pivot, NodeItem& obj) {
792  if (node == m_nodes.begin() or node.node_count() > 0) return m_nodes.end();
793 
794  node_iterator new_node = insert_sibling_node(node, obj);
795  NodeRef hi = node.node;
796  NodeRef lo = new_node.node;
797 
798  // traverse `node` looking for items to move to the new `lo` node.
799  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
800  // boundary between `lo` and `hi` and swapping the contents of the iterators.
801  // we traverse backwards so that our swapping does not unsort our items.
802  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
803  if (compare(*i, pivot) < 0) {
804  // move the item to the lower node
805 
806  // grab one item from the beginning of `hi`.
807  // swap `*i` and the stolen item, so that the
808  // tested item lies in the low node where it belongs,
809  // and the stolen item lies under `i`. we'll check it
810  // on the next iteration.
811 
812  if (lo->n_items == 0) {
813  lo->items_first = lo->items_last = hi->items_first;
814  } else {
815  lo->items_last = hi->items_first;
816  }
817 
818  ++lo->n_items;
819  --hi->n_items;
820  ++hi->items_first;
821  std::swap(*i, *(lo->items_last));
822 
823  // we do not decrement i, because we've just moved
824  // an untested item underneath it. we should test
825  // it first before moving on to the next one.
826  } else {
827  // `*i` is in the right place.
828  --i;
829  }
830  }
831  }

980  bool operator==(const Tree<NodeItem, LeafItem>& other) const {
981  if (this == &other) return true;
982  if (item_count() != other.item_count()) return false;
983  if (m_items != other.m_items) return false;
984 
985  // because the references will point to different memory,
986  // we can't compare raw data. we have to examine the structure of the tree.
987  NodeRef a_n = m_nodes.begin();
988  NodeRef b_n = other.m_nodes.begin();
989  for (; a_n != m_nodes.end() and b_n != other.m_nodes.end(); ++a_n, ++b_n) {
990  if (a_n->n_items != b_n->n_items) return false;
991  if (a_n->n_children != b_n->n_children) return false;
992  if (a_n->data != b_n->data) return false;
993  }
994

1037  // use all the child counts to recompute the first/last
1038  // references in internal nodes. useful when taking ownership
1039  // of another tree's internal node list.
1040  void recalculate_references() {
1041  NodeRef nod = m_nodes.begin();
1042  NodeRef begin_child = nod; ++begin_child;
1043  ItemRef begin_items = m_items.begin(); // next unassigned leaf item
1044  ItemRef last_item = m_items.end(); // last / most recent assigned leaf item in the list
1045 
1046  // in essence we "build" the tree by simply taking ownership of
1047  // the nodes/items that already exist. we walk over the nodes in
1048  // tree order, which means that child nodes are taken in order too.
1049  // this spares us recursion (i.e. no possibility of stack-blowing)
1050  // and some logic keeping track of descent/ascent.
1051  for (; nod != m_nodes.end(); ++nod) {
1052  nod->child_first = nod->child_last = (nod->n_children > 0) ? begin_child : m_nodes.end();
1053  nod->items_first = nod->items_last = (nod->n_items > 0) ? begin_items : m_items.end();
1054 
1055  if (nod->n_children > 0) {
1056  nod->child_first->parent = nod;
1057 
1058  // acquire direct children
1059  for (index_t c = 0; c < nod->n_children - 1; ++c) {
1060  ++nod->child_last;
1061  nod->child_last->parent = nod;
1062  }
1063 
1064  // move the node "free space"
1065  begin_child = nod->child_last; ++begin_child;
1066  } else {
1067  if (nod->n_items > 0) {
1068  for (index_t i = 0; i < nod->n_items - 1; ++i) {
1069  ++nod->items_last;
1070  }
1071  // move the item "free space"
1072  begin_items = last_item = nod->items_last; ++begin_items;
1073  }
1074 
1075  // fixup the items_end along the right edge of this subtree.
1076  NodeRef descendent = nod;
1077  for (NodeRef ancestor = nod->parent;
1078  ancestor != m_nodes.end();
1079  descendent = ancestor, ancestor = ancestor->parent) {
1080  if (ancestor->child_last == descendent) {
1081  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1082  } else {
1083  break;
1084  }
1085  }
1086  }
1087  }
1088  }

1091  // find the position of the node that follows `node` in list ("breadth-first") order.
1092  // the successor to `node` occurs just before the next subtree to the right.
1093  // an `O(k * log(n))` operation, on `n` the size of the tree, and `k` the arity.
1094  // this is different from `++node` because `++node` may descend into the
1095  // current node's subtree; we want to exactly hop over it.
1096  NodeRef node_successor(const NodeRef& node) const {
1097  NodeRef ancestor = node->parent;
1098  NodeRef child = node;
1099  NodeRef successor = nodes_end().node; // if there is no subtree to the right,
1100  // this is the conceptual successor.
1101 
1102  while (ancestor != m_nodes.end()) {
1103  NodeRef off_end_child = ancestor->child_last; ++off_end_child;
1104  // look for a populated node to the right of us
1105  while (++child != off_end_child) {
1106  if (child->n_children > 0) {
1107  successor = child->child_first;
1108  break;
1109  }
1110  }
1111  if (successor != m_nodes.end()) {
1112  // found the successor
1113  break;
1114  } else {
1115  // ascend the tree
1116  child = ancestor;
1117  ancestor = ancestor->parent;
1118  }
1119  }
1120  return successor;
1121  }

1124  // find the first item that occurs after where the last item owned by
1125  // the given node would go, in tree order.
1126  ItemRef item_successor(NodeRef node) const {
1127  for (NodeRef parent = node->parent;
1128  parent != m_nodes.end();
1129  node = parent, parent = node->parent) {
1130  NodeRef end_c = parent->child_last; ++end_c;
1131  for (; node != end_c; ++node) {
1132  if (node->n_items > 0) {
1133  return node->items_first;
1134  }
1135  }
1136  }
1137  return items_end().item;
