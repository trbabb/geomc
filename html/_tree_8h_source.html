786  template <bool BoundingFn(const NodeItem&, const LeafItem&)>
787  self_t find_parent(const const_item_iterator& i) const {
788  NodeRef cur_root = _root;
789  NodeRef subtree_last = node_successor(_root);
790  while (cur_root != subtree_last) {
791  if (BoundingFn(cur_root->data, *i)) {
792  // cur_root may contain `i`
793  if (cur_root->n_children > 0) {
794  // recurse
795  cur_root = cur_root->child_first;
796  } else {
797  // look for item
798  ItemRef end_item = cur_root->items_last; ++end_item;
799  for (ItemRef j = cur_root->items_first; j != end_item; ++j) {
800  // found it
801  if (j == i) return self_t(cur_root, _storage);
802  }
803  // our current node is a leaf, so there is no subtree to jump over.
804  ++cur_root;
805  }
806  } else {
807  // check next sibling
808  cur_root = node_successor(cur_root);

849  // compute the position of the node that follows `node`'s subtree in
850  // "sibling-first" order. the successor to `node` occurs just before
851  // the next subtree to the right. an `O(k * log(n))` operation, on `n`
852  // the size of the tree, and `k` the arity. this is different from
853  // `++node` because `++node` may descend into the current node's
854  // subtree; we want to exactly hop over it.
855  NodeRef node_successor(const NodeRef& node) const {
856  NodeRef ancestor = node->parent;
857  NodeRef child = node;
858  // if there is no subtree to the right,
859  // this is the conceptual successor:
860  NodeRef successor = _storage->_nodes.end();
861 
862  while (ancestor != _storage->_nodes.end()) {
863  NodeRef off_end_child = ancestor->child_last;
864  if (ancestor->n_children > 0) ++off_end_child;
865  // look for a populated node to the right of us
866  for (NodeRef c = child; c != off_end_child; ++c) {
867  if (c->n_children > 0) {
868  successor = c->child_first;
869  break;
870  }
871  }
872  if (successor != _storage->_nodes.end()) {
873  // found the successor
874  break;
875  } else {
876  // ascend the tree
877  child = ancestor;
878  ancestor = ancestor->parent;
879  }
880  }
881  return successor;
882  }

885  // compute the first item that occurs after the last item in the subtree
886  // rooted at `node`, including if that subtree is empty.
887  ItemRef item_successor(NodeRef node) const {
888  for (NodeRef parent = node->parent;
889  parent != _storage->_nodes.end();
890  node = parent, parent = node->parent) {
891  if (parent->n_children != 0) {
892  NodeRef end_c = parent->child_last; ++end_c;
893  for (NodeRef c = node; c != end_c; ++c) {
894  if (c->n_items > 0) {
895  return c->items_first;
896  }
897  }
898  }

1403  template <typename Func, typename P, typename ... Args>
1404  self_t split(Func compare, P pivot, Args&& ... args) const {
1405  if (this->_root == this->_storage->_nodes.begin()
1406  or this->node_count() > 0
1407  or this->item_count() < 2) {
1408  return this->end();
1409  }
1410 
1411  self_t new_node = this->parent().insert_child_node(*this, args...);
1412  NodeRef hi = this->_root;
1413  NodeRef lo = new_node._root;
1414 
1415  // todo: if we had splice operations on list<>, we could avoid
1416  // invalidating the item iterators.
1417 
1418  // traverse `node` looking for items to move to the new `lo` node.
1419  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
1420  // boundary between `lo` and `hi` and swapping the contents of the iterators.
1421  // we traverse backwards so that our swapping does not unsort our items.
1422 
1423  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
1424  if (compare(*i, pivot) < 0) {
1425  // move the item to the lower node
1426 
1427  // grab one item from the beginning of `hi`.
1428  // swap `*i` and the stolen item, so that the
1429  // tested item lies in the low node where it belongs,
1430  // and the stolen item lies under `i`. we'll check it
1431  // on the next iteration.
1432 
1433  if (lo->n_items == 0) {
1434  lo->items_first = lo->items_last = hi->items_first;
1435  } else {
1436  lo->items_last = hi->items_first;
1437  }
1438 
1439  ++(lo->n_items);
1440  --(hi->n_items);
1441  ++(hi->items_first);
1442  std::swap(*i, *(lo->items_last));
1443 
1444  // we do not decrement i, because we've just moved
1445  // an untested item underneath it. we should test
1446  // it first before moving on to the next one.
1447  } else {
1448  // `*i` is in the right place.
1449  --i;
1450  }
1451  }
1452 
1453  // if the `hi` node was emptied, fix up its item boundaries.
1454  if (hi->n_items == 0) {
1455  hi->items_first = hi->items_last = this->_storage->_items.end();
1456  }
1457 
1458  // we didn't delete/add/move any LeafItems (only their contents),
1459  // so we don't need to update the boundary items.
1460 
1461  return new_node;
1462  }

1627 
1628  // use all the child counts to recompute the first/last
1629  // references in internal nodes. useful when taking ownership
1630  // of another tree's internal node list.
1631  // preconditions:
1632  // - first item AND child pointers of this node are correct
1633  // - the descendents of this subtree have correct child and item counts
1634  // - the nodes belonging to this subtree exist and are correctly ordered
1635  // and placed in the tree
1636  // - item boundaries of all ancestors are already correct
1637  // - item counts of all ancestors are already correct
1638  // This should be an effective no-op if the subtree is already owned
1639  // (though it may still take O(n) to execute).
1640  void recalculate_references() const {
1641  NodeRef begin_child = this->_root->child_first;
1642  // next unassigned leaf item:
1643  ItemRef begin_items = this->_root->items_first;
1644  // last / most recent assigned leaf item in the list:
1645  ItemRef last_item = begin_items; --last_item;
1646  NodeRef end_child = this->node_successor(this->_root);
1647 
1648  // in essence we "build" the tree by simply taking ownership of
1649  // the nodes/items that already exist. we walk over the nodes in
1650  // tree order, which means that child nodes are taken in order too.
1651  for (NodeRef node = this->_root; node != end_child; ) {
1652  node->child_first = node->child_last =
1653  (node->n_children > 0) ? begin_child : this->_storage->_nodes.end();
1654  node->items_first = node->items_last =
1655  (node->n_items > 0) ? begin_items : this->_storage->_items.end();
1656 
1657  if (node->n_children > 0) {
1658  node->child_first->parent = node;
1659 
1660  // acquire direct children
1661  for (index_t c = 0; c < node->n_children - 1; ++c) {
1662  ++(node->child_last);
1663  node->child_last->parent = node;
1664  }
1665 
1666  // move the node "free space"
1667  begin_child = node->child_last; ++begin_child;
1668 
1669  // populate first child's subtree
1670  node = node->child_first;
1671  } else {
1672  if (node->n_items > 0) {
1673  for (index_t i = 0; i < node->n_items - 1; ++i) {
1674  ++(node->items_last);
1675  }
1676  // move the item "free space"
1677  begin_items = last_item = node->items_last; ++begin_items;
1678  }
1679 
1680  // fixup the items_end along the right edge of this subtree.
1681  NodeRef descendent = node;
1682  for (NodeRef ancestor = node->parent;
1683  ancestor != this->_root;
1684  descendent = ancestor, ancestor = ancestor->parent) {
1685  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1686  }
1687 
1688  // done populating this tree. populate the next sibling tree.
1689  node = this->node_successor(node);
1690  }
1691  }
1692  }
