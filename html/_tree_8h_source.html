55  // internally, nodes are kept in "sibling-first" order:
56  // - visit a node
57  // - visit its next sibling, if there is one
58  // - visit its first child
59  // as such, all direct children of a node are contiguous, and come
60  // after (though not necessarily *immediately* after) their parent.
61  // all the descendents of a leftward child come before all the
62  // descendents of the child to its right. Therefore, any given subtree
63  // (excluding its root) is contiguous. It is this recursive pattern:
64 
65  // [node] ... [c1 c2 c3 c4 ...]
66  // [children of c1] [descendents of c1...]
67  // [children of c2] [descendents of c2...]
68  // ...

929  node_iterator split(node_iterator& node, Func compare, P pivot, NodeItem& obj) {
930  if (node == m_nodes.begin() or node.node_count() > 0) return m_nodes.end();
931 
932  node_iterator new_node = insert_child_node(node.node->parent, node, obj);
933  NodeRef hi = node.node;
934  NodeRef lo = new_node.node;
935 
936  // xxx: can the formulation of this be flipped so that
937  // the new node is higher (rightward / more recent),
938  // rather than lower? it should act more like an `append`.
939 
940  // traverse `node` looking for items to move to the new `lo` node.
941  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
942  // boundary between `lo` and `hi` and swapping the contents of the iterators.
943  // we traverse backwards so that our swapping does not unsort our items.
944  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
945  if (compare(*i, pivot) < 0) {
946  // move the item to the lower node
947 
948  // grab one item from the beginning of `hi`.
949  // swap `*i` and the stolen item, so that the
950  // tested item lies in the low node where it belongs,
951  // and the stolen item lies under `i`. we'll check it
952  // on the next iteration.
953 
954  if (lo->n_items == 0) {
955  lo->items_first = lo->items_last = hi->items_first;
956  } else {
957  lo->items_last = hi->items_first;
958  }
959 
960  ++lo->n_items;
961  --hi->n_items;
962  ++hi->items_first;
963  std::swap(*i, *(lo->items_last));
964 
965  // we do not decrement i, because we've just moved
966  // an untested item underneath it. we should test
967  // it first before moving on to the next one.
968  } else {
969  // `*i` is in the right place.
970  --i;
971  }
972  }

1186  // use all the child counts to recompute the first/last
1187  // references in internal nodes. useful when taking ownership
1188  // of another tree's internal node list.
1189  void recalculate_references() {
1190  NodeRef nod = m_nodes.begin();
1191  NodeRef begin_child = nod; ++begin_child;
1192  ItemRef begin_items = m_items.begin(); // next unassigned leaf item
1193  ItemRef last_item = m_items.end(); // last / most recent assigned leaf item in the list
1194 
1195  // in essence we "build" the tree by simply taking ownership of
1196  // the nodes/items that already exist. we walk over the nodes in
1197  // tree order, which means that child nodes are taken in order too.
1198  // this spares us recursion (i.e. no possibility of stack-blowing)
1199  // and some logic keeping track of descent/ascent.
1200  for (; nod != m_nodes.end(); ++nod) {
1201  nod->child_first = nod->child_last = (nod->n_children > 0) ? begin_child : m_nodes.end();
1202  nod->items_first = nod->items_last = (nod->n_items > 0) ? begin_items : m_items.end();
1203 
1204  if (nod->n_children > 0) {
1205  nod->child_first->parent = nod;
1206 
1207  // acquire direct children
1208  for (index_t c = 0; c < nod->n_children - 1; ++c) {
1209  ++nod->child_last;
1210  nod->child_last->parent = nod;
1211  }
1212 
1213  // move the node "free space"
1214  begin_child = nod->child_last; ++begin_child;
1215  } else {
1216  if (nod->n_items > 0) {
1217  for (index_t i = 0; i < nod->n_items - 1; ++i) {
1218  ++nod->items_last;
1219  }
1220  // move the item "free space"
1221  begin_items = last_item = nod->items_last; ++begin_items;
1222  }
1223

1225  NodeRef descendent = nod;
1226  for (NodeRef ancestor = nod->parent;
1227  ancestor != m_nodes.end();
1228  descendent = ancestor, ancestor = ancestor->parent) {
1229  if (ancestor->child_last == descendent) {
1230  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1231  } else {
1232  break;
1233  }
1234  }
1235  }
1236  }
1237  }

1240  // find the position of the node that follows `node` in list ("sibling-first") order.
1241  // the successor to `node` occurs just before the next subtree to the right.
1242  // an `O(k * log(n))` operation, on `n` the size of the tree, and `k` the arity.
1243  // this is different from `++node` because `++node` may descend into the
1244  // current node's subtree; we want to exactly hop over it.
1245  NodeRef node_successor(const NodeRef& node) const {
1246  NodeRef ancestor = node->parent;
1247  NodeRef child = node;

1249  // this is the conceptual successor.
1250  while (ancestor != m_nodes.end()) {
1251  NodeRef off_end_child = ancestor->child_last;
1252  if (ancestor->n_children > 0) ++off_end_child;
1253  // look for a populated node to the right of us
1254  while (++child != off_end_child) {
1255  if (child->n_children > 0) {
1256  successor = child->child_first;
1257  break;
1258  }
1259  }
1260  if (successor != m_nodes.end()) {
1261  // found the successor
1262  break;
1263  } else {
1264  // ascend the tree
1265  child = ancestor;
1266  ancestor = ancestor->parent;
1267  }
1268  }
1269  return successor;
1270  }
