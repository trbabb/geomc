620  // compute the position of the node that follows `node`'s subtree in
621  // "sibling-first" order. the successor to `node` occurs just before
622  // the next subtree to the right. an `O(k * log(n))` operation, on `n`
623  // the size of the tree, and `k` the arity. this is different from
624  // `++node` because `++node` may descend into the current node's
625  // subtree; we want to exactly hop over it.
626  NodeRef node_successor(const NodeRef& node) const {
627  NodeRef ancestor = node->parent;
628  NodeRef child = node;
629  // if there is no subtree to the right,
630  // this is the conceptual successor:
631  NodeRef successor = _storage->_nodes.end();
632 
633  while (ancestor != _storage->_nodes.end()) {
634  NodeRef off_end_child = ancestor->child_last;
635  if (ancestor->n_children > 0) ++off_end_child;
636  // look for a populated node to the right of us
637  for (NodeRef c = child; c != off_end_child; ++c) {
638  if (c->n_children > 0) {
639  successor = c->child_first;
640  break;
641  }
642  }
643  if (successor != _storage->_nodes.end()) {
644  // found the successor
645  break;
646  } else {
647  // ascend the tree
648  child = ancestor;
649  ancestor = ancestor->parent;
650  }
651  }
652  return successor;
653  }

773  const self_t& insert_before,
774  Args&& ... args) const {
775  NodeRef p = this->_root;
776  NodeRef n = insert_before._root;
777  bool parent_childless = p->n_children == 0;
778 
779  // check invalid cases
780  if (n == this->_storage->_nodes.begin()) return this->end();
781  if (n->parent != p and n != this->end()._root) return this->end();
782 
783  // set the insert point for empty nodes
784  if (parent_childless) {
785  n = this->node_successor(p);
786  }
787 
788  // remember what our off-end node was
789  NodeRef prev_end_node = this->end()._root;
790 
791  // create the new node
792  NodeRef new_node = this->_storage->_nodes.insert(
793  n, typename base_t::Node(this->_storage, p, args...));
794 
795  // take ownership of the parent's items, if we are its first child
796  // (every item must belong to a leaf node, and the parent
797  // has just become an internal node).
798  if (parent_childless) {
799  new_node->items_first = p->items_first;
800  new_node->items_last = p->items_last;
801  new_node->n_items = p->n_items;
802  }
803 
804  // adjust child endpoints if necessary
805  if (insert_before._root == prev_end_node) {
806  p->child_last = new_node;
807  }
808  if (insert_before._root == p->child_first) {
809  p->child_first = new_node;
810  }
811  p->n_children++;
812

1167  template <typename Func, typename P, typename ... Args>
1168  self_t split(Func compare, P pivot, Args&& ... args) const {
1169  if (this->_root == this->_storage->_nodes.begin() or this->node_count() > 0) {
1170  return this->end();
1171  }
1172 
1173  self_t new_node = this->parent().insert_child_node(*this, args...);
1174  NodeRef hi = this->_root;
1175  NodeRef lo = new_node._root;
1176 
1177  // todo: if we had splice operations on list<>, we could avoid
1178  // invalidating the item iterators.
1179 
1180  // traverse `node` looking for items to move to the new `lo` node.
1181  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
1182  // boundary between `lo` and `hi` and swapping the contents of the iterators.
1183  // we traverse backwards so that our swapping does not unsort our items.
1184 
1185  // corner case: hi is / becomes empty
1186  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
1187  if (compare(*i, pivot) < 0) {
1188  // move the item to the lower node
1189 
1190  // grab one item from the beginning of `hi`.
1191  // swap `*i` and the stolen item, so that the
1192  // tested item lies in the low node where it belongs,
1193  // and the stolen item lies under `i`. we'll check it
1194  // on the next iteration.
1195 
1196  if (lo->n_items == 0) {
1197  lo->items_first = lo->items_last = hi->items_first;
1198  } else {
1199  lo->items_last = hi->items_first;
1200  }
1201 
1202  ++(lo->n_items);
1203  --(hi->n_items);
1204  ++(hi->items_first);
1205  std::swap(*i, *(lo->items_last));
1206 
1207  // we do not decrement i, because we've just moved
1208  // an untested item underneath it. we should test
1209  // it first before moving on to the next one.
1210  } else {
1211  // `*i` is in the right place.
1212  --i;
1213  }
1214  }
1215 
1216  // if the `hi` node was emptied, fix up its item boundaries.
1217  if (hi->n_items == 0) {
1218  hi->items_first = hi->items_last = this->_storage->_items.end();
1219  }
1220 
1221  // we didn't delete/add any LeafItems, so we don't need to
1222  // update the boundary items.
1223 
1224  return new_node;
1225  }

1392 protected:
1393 
1394  // use all the child counts to recompute the first/last
1395  // references in internal nodes. useful when taking ownership
1396  // of another tree's internal node list.
1397  // preconditions:
1398  // - first item AND child pointers of this node are correct
1399  // - the descendents of this subtree have correct child and item counts
1400  // - the nodes belonging to this subtree exist and are correctly ordered
1401  // and placed in the tree
1402  // - item boundaries of all ancestors are already correct
1403  // - item counts of all ancestors are already correct
1404  // This should be an effective no-op if the subtree is already owned
1405  // (though it may still take O(n) to execute).
1406  void recalculate_references() const {
1407  NodeRef begin_child = this->_root->child_first;
1408  // next unassigned leaf item:
1409  ItemRef begin_items = this->_root->items_first;
1410  // last / most recent assigned leaf item in the list:
1411  ItemRef last_item = begin_items; --last_item;
1412  NodeRef end_child = this->node_successor(this->_root);
1413 
1414  // in essence we "build" the tree by simply taking ownership of
1415  // the nodes/items that already exist. we walk over the nodes in
1416  // tree order, which means that child nodes are taken in order too.
1417  for (NodeRef node = this->_root; node != end_child; ) {
1418  node->child_first = node->child_last =
1419  (node->n_children > 0) ? begin_child : this->_storage->_nodes.end();
1420  node->items_first = node->items_last =
1421  (node->n_items > 0) ? begin_items : this->_storage->_items.end();
1422 
1423  if (node->n_children > 0) {
1424  node->child_first->parent = node;
1425 
1426  // acquire direct children
1427  for (index_t c = 0; c < node->n_children - 1; ++c) {
1428  ++(node->child_last);
1429  node->child_last->parent = node;
1430  }
1431 
1432  // move the node "free space"
1433  begin_child = node->child_last; ++begin_child;
1434 
1435  // populate first child's subtree
1436  node = node->child_first;
1437  } else {
1438  if (node->n_items > 0) {
1439  for (index_t i = 0; i < node->n_items - 1; ++i) {
1440  ++(node->items_last);
1441  }
1442  // move the item "free space"
1443  begin_items = last_item = node->items_last; ++begin_items;
1444  }
1445 
1446  // fixup the items_end along the right edge of this subtree.
1447  NodeRef descendent = node;
1448  for (NodeRef ancestor = node->parent;
1449  ancestor != this->_root;
1450  descendent = ancestor, ancestor = ancestor->parent) {
1451  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1452  }
1453 
1454  // done populating this tree. populate the next sibling tree.
1455  node = this->node_successor(node);
1456  }
1457  }
1458  }
