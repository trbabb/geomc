285 sample_t sample(const coord_t &pt, EdgeBehavior edge=EDGE_CLAMP, Interpolation interp=INTERP_LINEAR) const {
286 grid_t gridPt;
287 pt = toGridSpace(pt);
288
289 switch (interp) {
290 case INTERP_NEAREST:
291 gridPt = grid_t(pt + coord_t(0.5));
292 return sample_discrete(gridPt, edge);
293 case INTERP_LINEAR:
294 {
295 sample_t buf_l[1<<M];
296 gridPt = (grid_t)pt;
297 coord_t s = pt - ((coord_t)gridPt);
298
299 // copy surrounding 2^M sample pts into a contiguous buffer
300 copy(buf_l, Rect<index_t,M>(gridPt, gridPt + grid_t(2)), edge);
301
302 return interp_linear(s.begin(), buf_l, M);
303 }
304 case INTERP_CUBIC:
305 {
306 sample_t buf_c[1<<(2*M)];
307 gridPt = (grid_t)pt;
308 coord_t s = pt - ((coord_t)gridPt);
309
310 // copy surrounding 4^M sample pts into a contiguous buffer
311 copy(buf_c, Rect<index_t,M>(gridPt - grid_t(1), gridPt + grid_t(3)), edge);
312
313 return interp_cubic(s.begin(), buf_c, M);
314 }
315 default:
316 return m_abyss;
317 }
318 }

327 template <EdgeBehavior Edge>
328 sample_t sample_discrete(const grid_t &pt) const {
329 index_t offs;
330 switch (Edge) {
331 case EDGE_CONSTANT:
332 if (contains_gridpt(pt)) {
333 offs = this->template index<EDGE_CONSTANT>(pt);
334 } else {
335 return m_abyss;
336 }
337 break;
338 default:
339 offs = this->template index<Edge>(pt);
340 break;
341 }
342 return PointType<O,N>::from_ptr(m_data.get() + offs);
343 }

352 sample_t sample_discrete(const grid_t &pt, EdgeBehavior edge) const {
353 index_t offs;
354
355 switch (edge) {
356 case EDGE_CONSTANT:
357 if (!contains_gridpt(pt)) return m_abyss;
358 offs = this->template index<EDGE_CONSTANT>(pt);
359 break;
360 case EDGE_CLAMP:
361 offs = this->template index<EDGE_CLAMP>(pt);
362 break;
363 case EDGE_PERIODIC:
364 offs = this->template index<EDGE_PERIODIC>(pt);
365 break;
366 case EDGE_MIRROR:
367 offs = this->template index<EDGE_MIRROR>(pt);
368 break;
369 default:
370 return m_abyss;
371 }
372
373 return sample_t(m_data.get() + offs);
374 }

380 bool contains_gridpt(const grid_t &idx) const {
381 typedef PointType<index_t,M> grid_info;
382
383 const index_t *p_i = grid_info::iterator(idx);
384 const index_t *e_i = grid_info::iterator(m_extent);
385 for (index_t i = 0; i < M; i++) {
386 index_t c = p_i[i];
387 if (c < 0 or c >= e_i[i]) {
388 return false;
389 }
390 }
391 return true;
392 }
