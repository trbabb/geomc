286  sample_t sample(const coord_t &pt, EdgeBehavior edge=EDGE_CLAMP, Interpolation interp=INTERP_LINEAR) const {
287  grid_t gridPt;
288  pt = toGridSpace(pt);
289 
290  switch (interp) {
291  case INTERP_NEAREST:
292  gridPt = grid_t(pt + coord_t(0.5));
293  return sample_discrete(gridPt, edge);
294  case INTERP_LINEAR:
295  {
296  sample_t buf_l[1<<M];
297  gridPt = (grid_t)pt;
298  coord_t s = pt - ((coord_t)gridPt);
299 
300  // copy surrounding 2^M sample pts into a contiguous buffer
301  copy(buf_l, Rect<index_t,M>(gridPt, gridPt + grid_t(2)), edge);
302 
303  return interp_linear(s.begin(), buf_l, M);
304  }
305  case INTERP_CUBIC:
306  {
307  sample_t buf_c[1<<(2*M)];
308  gridPt = (grid_t)pt;
309  coord_t s = pt - ((coord_t)gridPt);
310 
311  // copy surrounding 4^M sample pts into a contiguous buffer
312  copy(buf_c, Rect<index_t,M>(gridPt - grid_t(1), gridPt + grid_t(3)), edge);
313 
314  return interp_cubic(s.begin(), buf_c, M);
315  }
316  default:
317  return m_abyss;
318  }
319  }

329  sample_t sample_discrete(const grid_t &pt) const {
330  index_t offs;
331  switch (Edge) {
332  case EDGE_CONSTANT:
333  if (contains_gridpt(pt)) {
334  offs = this->template index<EDGE_CONSTANT>(pt);

336  return m_abyss;
337  }

342  }

353  sample_t sample_discrete(const grid_t &pt, EdgeBehavior edge) const {
354  index_t offs;
355 
356  switch (edge) {
357  case EDGE_CONSTANT:
358  if (!contains_gridpt(pt)) return m_abyss;
359  offs = this->template index<EDGE_CONSTANT>(pt);

372  }

381  bool contains_gridpt(const grid_t &idx) const {
382  typedef PointType<index_t,M> grid_info;
383 
384  const index_t *p_i = grid_info::iterator(idx);
385  const index_t *e_i = grid_info::iterator(m_extent);
386  for (index_t i = 0; i < M; i++) {
387  index_t c = p_i[i];
388  if (c < 0 or c >= e_i[i]) {
389  return false;
390  }
391  }
392  return true;
393  }
