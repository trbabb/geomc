51 template <typename Generator>
52 float operator()(Generator& g) const {
53 union FPBox {
54 float f;
55 uint32_t i;
56 };
57
58 // without IEEE754, our bit twiddling will not work.
59 // generally this will only fail for "exotic" systems:
60 static_assert(std::numeric_limits<float>::is_iec559, "floating point must be ieee754");
61
62 auto randbits = std::uniform_int_distribution<uint64_t>();
63 constexpr size_t mantissa_bits = 23;
64 size_t scale_bits = 64 - mantissa_bits;
65 uint64_t r = randbits(g);
66 int32_t exp = 126;
67 uint32_t mant = r >> scale_bits;
68 uint32_t z;
69 while (true) {
70 z = std::countr_zero(r);
71 exp = std::max<int32_t>(exp - z, 0);
72 if (z < scale_bits or exp == 0) [[likely]] break;
73 scale_bits = 64;
74 r = randbits(g);
75 }
76 if (mant == 0) [[unlikely]] {
77 // avoid biasing the result toward zero. above, it is not
78 // possible to generate 1.0. with probability 1/2, bump the exponent.
79 exp += ((r >> (scale_bits - 1)) & 1);
80 }
81 FPBox ans;
82 ans.i = (exp << mantissa_bits) | mant; // combine exp and mantissa
83 return _range.remap(ans.f);
84 }
85
86 void reset() {}
87 Rect<float,1> param() const { return _range; }
88 void param(Rect<float,1> range) { _range = range; }
89
90 float min() const { return _range.lo; }
91 float max() const { return _range.hi; }
92
93 float a() const { return _range.lo; }
94 float b() const { return _range.hi; }

97 return _range == other._range;
98 }
