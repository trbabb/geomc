24// helper function evaluates Legendre polynomials in batch.
25// makes use of recurrent definition of legendre polys to save re-execs.
26// evaluate a column m and -m of Legendre polynomials and stuff them
27// into the appropriate place in the SphericalHarmonics' coeff table
28template <typename T, index_t N>
29inline void legendre(SphericalHarmonics<T,N> *sh, index_t m, T x) {
30 index_t l = sh->bands();
31 T pmm0 = 1;
32 if (m > 0) {
33 // compute double factorial of 2m - 1
34 // fold in the (-1)^m
35 for (index_t i = 1; i <= m; i++) {
36 pmm0 *= -(i * 2 - 1);
37 }
38 pmm0 *= std::pow(1 - x * x, m / 2.0);
39 }
40 sh->coeff(m,m) = sh->coeff(m,-m) = pmm0;
41 if (m == l - 1) return; // there is no higher band
42 T pmm1 = sh->coeff(m+1,m) = sh->coeff(m+1,-m) = x * (2 * m + 1) * pmm0;
43 if (m == l - 2) return; // there is no higher band
44 for (index_t l_i = m + 2; l_i < l; l_i++) {
45 T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) / (l_i - m);
46 pmm0 = pmm1;
47 pmm1 = sh->coeff(l_i,m) = sh->coeff(l_i,-m) = pmi;
48 }
49}

59template <typename T>
60inline T legendre(index_t l, index_t m, T x) {
61 T pmm0 = 1;
62 if (m > 0) {
63 // compute double factorial of 2m - 1
64 // fold in the (-1)^m
65 for (index_t i = 1; i <= m; i++) {
66 pmm0 *= -(i * 2 - 1);
67 }
68 pmm0 *= std::pow(1 - x * x, m / 2.0);
69 }
70 if (l == m) return pmm0;
71 T pmm1 = x * (2 * m + 1) * pmm0;
72 if (l == m + 1) return pmm1;
73 for (index_t i = m + 2; i <= l; i++) {
74 T pmi = (x * (2 * i - 1) * pmm1 - (i + m - 1) * pmm0) / (i - m);
75 pmm0 = pmm1;
76 pmm1 = pmi;
77 }
78 return pmm1;
79}

89template <typename T>
90inline T legendre(index_t n, T x) {
91 if (n <= 0) return 1;
92 T p_0 = 1; // P(i - 1, x)
93 T p_1 = x; // P(i, x)
94 T p_i = x; // P(i + 1, x)
95 for (index_t i = 1; i < n; i++) {
96 p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
97 p_0 = p_1;
98 p_1 = p_i;
99 }
100 return p_i;
101}

111template <typename T>
112inline T legendre_integral(index_t n, T x) {
113 if (n <= 0) return x + 1;
114 T p_0 = 1; // P(i - 1, x)
115 T p_1 = x; // P(i, x)
116 T p_i = x; // P(i + 1, x)
117 for (index_t i = 1; i < n; i++) {
118 p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
119 p_0 = p_1;
120 p_1 = p_i;
121 }
122 p_i = ((2 * n + 1) * x * p_1 - n * p_0) / ((T)n + 1);
123 return (p_i - p_0) / (2 * n + 1);
124}

134template <typename T>
135inline T chebyshev(index_t kind, index_t n, T x) {
136 if (n <= 0) return 1;
137 T t_0 = 1;
138 T t_1 = kind * x; // first kind := x; second kind := 2x
139 T t_i = t_1;
140 for (index_t i = 1; i < n; i++) {
141 t_i = 2 * x * t_1 - t_0;
142 t_0 = t_1;
143 t_1 = t_i;
144 }
145 return t_i;
146}
