81template <typename Md, typename Ms>
82void mtxcopy(Md* into, const Ms& src,
83 typename std::enable_if<
84 detail::LinalgDimensionMatch<Md,Ms>::val and
85 (detail::_ImplVecOrient<Md,Ms>::orient != detail::ORIENT_VEC_UNKNOWN),
86 int>::type dummy=0) {
87#ifdef GEOMC_MTX_CHECK_DIMS
88 typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecOrient<Md,Ms>::orient> D;
89 typedef detail::_ImplMtxAdaptor<Ms, detail::_ImplVecOrient<Md,Ms>::orient> S;
90
91 // runtime dimension match check
92 if ((D::ROWDIM * S::ROWDIM == DYNAMIC_DIM and D::rows(*into) != S::rows(src)) or
93 (D::COLDIM * S::COLDIM == DYNAMIC_DIM and D::cols(*into) != S::cols(src))) {
94 throw DimensionMismatchException(D::rows(*into), D::cols(*into), S::rows(src), S::cols(src));
95 }
96#endif
97 // todo: check mem aliasing?
98 detail::_mtxcopy(into, src);
99}

102// unknown vector orientation case (dynamic matrix <-> vector)
103template <typename Md, typename Ms>
104void mtxcopy(Md* into, const Ms& src,
105 typename std::enable_if<
106 detail::_ImplVecOrient<Md,Ms>::orient == detail::ORIENT_VEC_UNKNOWN,
107 int>::type dummy=0) {
108#ifdef GEOMC_MTX_CHECK_DIMS
109 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
110 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
111 typedef detail::_ImplMtxAdaptor<Ms, detail::ORIENT_VEC_COL> Sc;
112 typedef detail::_ImplMtxAdaptor<Ms, detail::ORIENT_VEC_ROW> Sr;
113
114 // if a dimension match can be made with either orientation, proceed with the copy
115 if ((Dc::rows(*into) == Sc::rows(src) and Dc::cols(*into) == Sc::cols(src)) or
116 (Dr::rows(*into) == Sr::rows(src) and Dr::cols(*into) == Sr::cols(src))) {
117 detail::_mtxcopy(into, src);
118 } else {
119 throw DimensionMismatchException(Dc::rows(*into), Dc::cols(*into), Sc::rows(src), Sc::cols(src));
120 }
121#else
122 detail::_mtxcopy(into, src);
123#endif
124}
