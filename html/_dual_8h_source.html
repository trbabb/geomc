368  template <typename T>
369  inline geom::Dual<T> pow(const geom::Dual<T> &base, const geom::Dual<T> &xp) {
370  // here we use the chain rule for partial derivatives:
371  // f(x,y) = x ^ y
372  // x = g(t); y = h(t)
373  // then:
374  // df / dt = (df / dx) * (dx / dt) + (df / dy) * (dy / dt)
375  // = (df / dx) * g`(t) + (df / dy) * h`(t)
376  // and:
377  // df / dx = d/dx (x ^ y) = y * x ^ (y - 1)
378  // df / dy = d/dy (x ^ y) = log(x) * (x ^ y)
379  T a_c = pow(base.x, xp.x);
380  return geom::Dual<T>(
381  a_c,
382  base.dx * xp.x * a_c / xp.x +
383  xp.dx * a_c * log(base.x));
384 
385  }

431  inline geom::Dual<T> abs(const geom::Dual<T> &d) {
432  bool neg = d.x < 0;
433  T dx;
434 
435  if (d.x == 0) {
436 #if DUAL_DISCONTINUITY_NAN
437  dx = numeric_limits<T>::quiet_NaN();
438 #elif DUAL_DISCONTINUITY_AVERAGE
439  dx = 0;
440 #elif DUAL_DISCONTINUITY_LEFT
441  dx = d.dx > 0 ? -d.dx : d.dx;

447  }

489  template <typename T>
490  inline geom::Dual<T> min(const geom::Dual<T> &d1, const geom::Dual<T> &d2) {
491  bool lt = d1.x <= d2.x;
492  T x = lt ? d1.x : d2.x;
493  T dx;
494  if (d1.x == d2.x) {
495 #if DUAL_DISCONTINUITY_NAN
496  dx = numeric_limits<T>::quiet_NaN();
497 #elif DUAL_DISCONTINUITY_AVERAGE
498  dx = (d1.dx + d2.dx) / 2;
499 #elif DUAL_DISCONTINUITY_LEFT
500  dx = (d1.dx < d2.dx) ? d1.dx : d2.dx;
501 #elif DUAL_DISCONTINUITY_RIGHT
502  dx = (d1.dx > d2.dx) ? d1.dx : d2.dx;
503 #endif
504  } else {
505  dx = lt ? d1.dx : d2.dx;
506  }
507  return geom::Dual<T>(x,dx);
