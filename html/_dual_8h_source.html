38enum class DiscontinuityPolicy {
40 NaN,
42 Average,
44 Left,
46 Right,
51 Inf

56template <typename T, DiscontinuityPolicy P>
57struct Discontinuity {};
58
59template <typename T>
60struct Discontinuity<T, DiscontinuityPolicy::NaN> {
61 static inline T resolve(T a, T b) {
62 return std::numeric_limits<T>::quiet_NaN();
63 }
64};
65
66template <typename T>
67struct Discontinuity<T, DiscontinuityPolicy::Left> {
68 static inline T resolve(T a, T b) {
69 return a;
70 }
71};
72
73template <typename T>
74struct Discontinuity<T, DiscontinuityPolicy::Right> {
75 static inline T resolve(T a, T b) {
76 return b;
77 }
78};
79
80template <typename T>
81struct Discontinuity<T, DiscontinuityPolicy::Average> {
82 static inline T resolve(T a, T b) {
83 return (a + b) / (T)2;
84 }
85};
86
87template <typename T>
88struct Discontinuity<T, DiscontinuityPolicy::Inf> {
89 static inline T resolve(T a, T b) {
90 const T inf = std::numeric_limits<T>::infinity();
91 return (b == a) ? a : (b > a ? inf : -inf);
92 }
93};

474 template <typename T, geom::DiscontinuityPolicy P>
475 inline geom::Dual<T,P> pow(const geom::Dual<T,P> &base, const geom::Dual<T,P> &xp) {
476 // here we use the chain rule for partial derivatives:
477 // f(x,y) = x ^ y
478 // x = g(t); y = h(t)
479 // then:
480 // df / dt = (df / dx) * (dx / dt) + (df / dy) * (dy / dt)
481 // = (df / dx) * g`(t) + (df / dy) * h`(t)
482 // and:
483 // df / dx = d/dx (x ^ y) = y * x ^ (y - 1)
484 // df / dy = d/dy (x ^ y) = log(x) * (x ^ y)
485 T a_c = pow(base.x, xp.x);
486 return geom::Dual<T,P>(
487 a_c,
488 base.dx * xp.x * a_c / xp.x +
489 xp.dx * a_c * log(base.x));
490
491 }

539 template <typename T, geom::DiscontinuityPolicy P>
540 inline geom::Dual<T,P> abs(const geom::Dual<T,P> &d) {
541 bool neg = d.x < 0;
542 T dx;
543
544 if (d.x == 0) {
545 dx = geom::Dual<T,P>::discontinuity_policy::resolve(-d.dx, d.dx);
546 } else {
547 dx = neg ? -d.dx: d.dx;
548 }
549
550 return geom::Dual<T,P>(neg ? -d.x : d.x, dx);
551 }

558 template <typename T, geom::DiscontinuityPolicy P>
559 inline geom::Dual<T,P> ceil(const geom::Dual<T,P> &d) {
560 T x = ceil(d.x);
561 T dx;
562 if (P == geom::DiscontinuityPolicy::Inf and x == d.x) {
563 const T inf = std::numeric_limits<T>::infinity();
564 // jump is in the same direction that the function is changing
565 dx = std::copysign(inf, d.dx);
566 } else if (P == geom::DiscontinuityPolicy::NaN and x == d.x) {
567 dx = std::numeric_limits<T>::quiet_NaN();
568 } else {
569 dx = 0;
570 }
571 return geom::Dual<T,P>(x, dx);
572 }

600 template <typename T, geom::DiscontinuityPolicy P>
601 inline geom::Dual<T,P> min(const geom::Dual<T,P> &d1, const geom::Dual<T,P> &d2) {
602 bool lt = d1.x <= d2.x;
603 T x = lt ? d1.x : d2.x;
604 T dx;
605 if (d1.x == d2.x) {
606 dx = geom::Dual<T,P>::discontinuity_policy::resolve(
607 // faster increasing function wins min() on the left
608 std::max(d1.dx, d2.dx),
609 std::min(d1.dx, d2.dx));
610 } else {
611 dx = lt ? d1.dx : d2.dx;
612 }
613 return geom::Dual<T,P>(x,dx);
614 }

621 template <typename T, geom::DiscontinuityPolicy P>
622 inline geom::Dual<T,P> max(const geom::Dual<T,P> &d1, const geom::Dual<T,P> &d2) {
623 bool gt = d1.x >= d2.x;
624 T x = gt ? d1.x : d2.x;
625 T dx;
626 if (d1.x == d2.x) {
627 dx = geom::Dual<T,P>::discontinuity_policy::resolve(
628 // slower increasing function wins max() on the left
629 std::min(d1.dx, d2.dx),
630 std::max(d1.dx, d2.dx));
631 } else {
632 dx = gt ? d1.dx : d2.dx;
633 }
634 return geom::Dual<T>(x,dx);
635 }
