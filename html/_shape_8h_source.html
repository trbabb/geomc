97 public:
98 typedef typename PointType<T,N>::point_t point_t;
99
112 point_t convex_support(point_t d) const {
113 static_assert(&Derived::convex_support != &Convex<T,N,Derived>::convex_support,
114 "Derived class must provide a convex_support() implementation");
115 Derived* self = (Derived*) this;
116 return self->convex_support(d);

124 template <typename Shape>
125 bool intersects(const Convex<T,N,Shape>& other) const {
126 return gjk_intersect(
127 as_any_convex(*this),
128 as_any_convex(other));
129 }
130
137 Rect<T,N> bounds() const {
138 Rect<T,N> b;
139 T* lo = PointType<T,N>::iterator(b.lo);
140 T* hi = PointType<T,N>::iterator(b.hi);
141 for (index_t i = 0; i < N; ++i) {
142 point_t axis, x;
143 const Derived* self = (const Derived*) this;
144 PointType<T,N>::iterator(axis)[i] = 1;
145 x = self->convex_support(axis);
146 hi[i] = PointType<T,N>::iterator(x)[i];
147 PointType<T,N>::iterator(axis)[i] = -1;
148 x = self->convex_support(axis);
149 lo[i] = PointType<T,N>::iterator(x)[i];
150 }
151 return b;
152 }

230template <typename T, index_t N, typename Derived>
231class Projectable: public SdfEvaluable<T,N,Derived> {
232public:
233
239 Vec<T,N> project(Vec<T,N> p) const {
240 static_assert(&Derived::project != &Projectable<T,N,Derived>::project,
241 "Derived class must provide a project() implementation");
242 Derived* self = (Derived*) this;
243 return self->project(p);
244 }
245
246 inline T sdf(Vec<T,N> p) const {
247 static_assert(
248 &Derived::sdf != &Projectable<T,N,Derived>::sdf or
249 &Derived::contains != &SdfEvaluable<T,N,Derived>::contains,
250 "Derived class must override at least one of `sdf()` or `contains()`");
251 Derived* self = (Derived*) this;
252 return (self->contains(p) ? -1 : 1) * self->project(p).dist(p);
253 }
254
263 inline Vec<T,N> normal(Vec<T,N> p) const {
264 Derived* self = (Derived*) this;
265 Vec<T,N> n = (p - self->project(p)).unit();
266 return (self->contains(p) ? -1 : 1) * n;
267 }
268
275 inline Vec<T,N> clip(Vec<T,N> p) const {
276 Derived* self = (Derived*) this;
277 return self->contains(p) ? p : project(p);
278 }
