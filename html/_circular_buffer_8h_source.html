30  typedef typename std::aligned_storage<sizeof(T), alignof(T)>::type storage_t;

32  // `head + size` scheme chosen over `head + tail` to avoid
33  // ambiguity of "full vs. empty buffer"
34 
35  // We usage storage_t both to admit objects with no default constructor,
36  // and also to avoid unnecessary constructions/destructions of items
37  // in the empty area of the buffer.
38 
39  // We manage our own static/dynamic buffer instead of delegating to SmallStorage
40  // because we need to manage the liveness of the elements in the buffer.
41  // A SmallStorage<storage_t> won't properly call move, copy, or assignment
42  // operators because storage_t hides the overlying class, and of course
43  // SmallStorage does not know which items are live.

95  CircularBuffer(CircularBuffer<T,N>&& other):
96  _data((other._capacity > N) ? other._data : _buf),
97  _capacity(other._capacity),
98  _head((other._capacity > N) ? other._head : 0),

101  // move-construct individual items.
102  for (index_t i = 0; i < _size; ++i) {
103  T* p = get() + i;
104  new (p) T(std::move(other[i]));
105  other[i].~T();
106  }
107  } else {
108  // we took ownership of `other`'s allocated array.
109  // turn `other` back into a static buffer.
110  other._capacity = N;
111  other._data = other._buf;
112  }
113  other._size = 0;
114  other._head = 0;

333  void ensure_capacity(index_t new_size) {
334  if (new_size > _capacity) {
