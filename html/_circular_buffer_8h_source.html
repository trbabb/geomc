34 typedef typename std::aligned_storage<sizeof(T), alignof(T)>::type storage_t;
35
36 // `head + size` scheme chosen over `head + tail` to avoid
37 // ambiguity of "full vs. empty buffer"
38
39 // We usage storage_t both to admit objects with no default constructor,
40 // and also to avoid unnecessary constructions/destructions of items
41 // in the empty area of the buffer.
42
43 // We manage our own static/dynamic buffer instead of delegating to SmallStorage
44 // because we need to manage the liveness of the elements in the buffer.
45 // A SmallStorage<storage_t> won't properly call move, copy, or assignment
46 // operators because storage_t hides the overlying class, and of course
47 // SmallStorage does not know which items are live.

102 CircularBuffer(CircularBuffer<T,N>&& other):
103 _data((other._capacity > N) ? other._data : _buf),
104 _capacity(other._capacity),
105 _head((other._capacity > N) ? other._head : 0),
106 _size(other._size) {
107 if (other._capacity <= N) {
108 // move-construct individual items.
109 for (index_t i = 0; i < _size; ++i) {
110 T* p = get() + i;
111 new (p) T(std::move(other[i]));
112 other[i].~T();
113 }
114 } else {
115 // we took ownership of `other`'s allocated array.
116 // turn `other` back into a static buffer.
117 other._capacity = N;
118 other._data = other._buf;
119 }
120 other._size = 0;
121 other._head = 0;
122 }

125 virtual ~CircularBuffer() {
126 for (index_t i = 0; i < size(); ++i) {
127 item(i)->~T();
128 }
129 if (_capacity > N and _capacity > 0) {
130 delete [] _data;
131 _capacity = 0;
132 _data = nullptr;
133 }
134 _size = 0;
135 }

138 CircularBuffer& operator=(const CircularBuffer& other) {
139 index_t old_size = size();
140 index_t new_size = other.size();
141 index_t n = std::min(old_size, new_size);
142 reserve(other._size);
143 // delegate assignment to T
144 for (index_t i = 0; i < n; ++i) {
145 this->operator[](i) = other[i];
146 }
147 // copy-construct new items
148 for (index_t i = n; i < new_size; ++i) {
149 T* p = item(i);
150 new (p) T(other[i]);
151 }
152 // destroy extra items
153 for (index_t i = n; i < old_size; ++i) {
154 T* p = item(i);
155 p->~T();
156 }
157 _size = other._size;
158
159 return *this;
160 }

163 CircularBuffer& operator=(CircularBuffer&& other) {
164 // destroy all our stuff
165 for (index_t i = 0; i < _size; ++i) {
166 item(i)->~T();
167 }
168 if (other._capacity > N) {
169 // `other` has a dynamic array, just take ownership of it
170 if (_capacity > N) {
171 // delete our own dynamic array
172 delete [] _data;
173 }
174 _data = other._data;
175 _capacity = other._capacity;
176 _head = other._head;
177 // turn `other` back into a static buffer
178 other._data = other._buf;
179 other._capacity = N;
180 } else {
181 // `other`'s array cannot be moved; copy items instead.
182 // we do not need to reserve(), obviously, because _size < N.
183 // if we have space for more than N, that's fine too.
184 _head = 0;
185 for (index_t i = 0; i < other._size; ++i) {
186 T* p = get() + i;
187 // move-construct a new item in our own buffer
188 new (p) T(std::move(other.operator[](i)));
189 // destroy its source
190 other.operator[](i).~T();
191 }
192 }
193
194 _size = other._size;
195 other._size = 0;
196 other._head = 0;
197
198 return *this;
199 }

345 void reserve(index_t new_cap) {
346 if (new_cap > _capacity) {
347 index_t resize_to = std::max(_capacity * 2, new_cap);
348 storage_t* array = new storage_t[resize_to];
349
350 // move the items into the new array
351 for (index_t i = 0; i < _size; ++i) {
352 T* p = reinterpret_cast<T*>(array + i);
353 // move-construct
354 new (p) T(std::move(*item(i)));
355 // destroy the source
356 item(i)->~T();
357 }
358
359 if (_capacity > N) {
360 delete [] _data;
361 }
362
363 _data = array;
364 _head = 0;
365 _capacity = resize_to;
366 }
367 }
