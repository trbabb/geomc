100 as_any_convex(other)

102 }

104 Rect<T,N> bounds() const requires BoundedObject<Shape> {
105 return shape.bounds().dilated(dilation);
106 }
107
108 Vec<T,N> normal(Vec<T,N> p) const requires ProjectableObject<Shape> {
109 // we are dilating the shape along the normal, so the normal is unchanged
110 return shape.normal(p);
111 }
112
114 Vec<T,N> project(Vec<T,N> p) const
115 requires ProjectableObject<Shape> and ConvexObject<Shape>
116 {
117 Vec<T,N> p_proj = shape.project(p);
118 Vec<T,N> dp = shape.normal(p);
119 if (shape.contains(p)) {
120 // dilated projection is further away from the point
121 return p_proj + dilation * dp;
122 } else {
123 // dilated projection is more toward point
124 return p_proj - dilation * dp;
125 }
126 }
127
128 Vec<T,N> clip(Vec<T,N> p) const requires ProjectableObject<Shape> {
129 return contains(p) ? project(p) : p;
130 }
