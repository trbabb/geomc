51template <typename Shape>
52class Dilated:
53 public Convex <typename Shape::elem_t, Shape::N, Dilated<Shape>>,
54 public Projectable<typename Shape::elem_t, Shape::N, Dilated<Shape>>
55{
56public:
58 typedef typename Shape::elem_t T;
60 static constexpr size_t N = Shape::N;
61
63 Shape shape;
65 T dilation;
66
68 Dilated():dilation((T)0) {}
69
71 Dilated(const Shape& s):shape(s) {}
72
74 Dilated(const Shape& s, T dilation):shape(s), dilation(dilation) {}
75
76 bool operator==(const Dilated& other) const {
77 return shape == other.shape && dilation == other.dilation;
78 }
79
81 bool contains(Vec<T,N> p) const {
82 return sdf(p) < 0;
83 }
84
86 T sdf(Vec<T,N> p) const {
87 return shape.sdf(p) - dilation;
88 }
89
90 Vec<T,N> normal(Vec<T,N> p) const {
91 // we are dilating the shape along the normal, so the normal is unchanged
92 return shape.normal(p);
93 }
94
95 Vec<T,N> convex_support(Vec<T,N> d) const {
96 Vec<T,N> p = shape.convex_support(d);
97 return p + d.unit() * dilation;
98 }
99
100 Rect<T,N> bounds() const {
101 return shape.bounds().dilated(dilation);
102 }
103
105 Vec<T,N> project(Vec<T,N> p) const {
106 Vec<T,N> p_proj = shape.project(p);
107 Vec<T,N> dp = (p_proj - p).unit();
108 if (shape.contains(p)) {
109 // dilated projection is further away from the point
110 return p_proj + dilation * dp;
111 } else {
112 // dilated projection is more toward point
113 return p_proj - dilation * dp;
114 }
115 }
116
117 // todo: trace()
118};
