51class Dilated:

55public:

76 bool contains(Vec<T,N> p) const {
77 return sdf(p) < 0;
78 }
79
81 inline T sdf(Vec<T,N> p) const {
82 return shape.sdf(p) - dilation;
83 }
84
85 Vec<T,N> convex_support(Vec<T,N> d) const {
86 Vec<T,N> p = shape.convex_support(d);
87 return p + d.unit() * dilation;
88 }
89
90 Rect<T,N> bounds() const {
91 return shape.bounds().dilated(dilation);
92 }
93
95 Vec<T,N> project(Vec<T,N> p) const {
96 Vec<T,N> p_proj = shape.project(p);
97 Vec<T,N> dp = (p_proj - p).unit();
98 if (shape.contains(p)) {
99 // dilated projection is further away from the point
100 return p_proj + dilation * dp;
101 } else {
102 // dilated projection is more toward point
103 return p_proj - dilation * dp;
104 }
105 }
