51template <typename Shape>
52class Dilated:
53 public Convex <typename Shape::elem_t, Shape::N, Dilated<Shape>>,
54 public Projectable<typename Shape::elem_t, Shape::N, Dilated<Shape>>
55{
56public:
58 typedef typename Shape::elem_t T;
60 static constexpr size_t N = Shape::N;
61
63 Shape shape;
65 T dilation;
66
68 Dilated():dilation((T)0) {}
69
71 Dilated(const Shape& s):shape(s) {}
72
74 Dilated(const Shape& s, T dilation):
75 shape(s),
76 dilation(std::max<T>(dilation, 0)) {}
77
78 static constexpr bool admits_cusps() { return false; }
79
80 bool operator==(const Dilated& other) const {
81 return shape == other.shape && dilation == other.dilation;
82 }
83
85 bool contains(Vec<T,N> p) const requires RegionObject<Shape> {
86 return sdf(p) < 0;
87 }
88
90 T sdf(Vec<T,N> p) const requires SdfObject<Shape> {
91 return shape.sdf(p) - dilation;
92 }
93
94 Vec<T,N> normal(Vec<T,N> p) const requires ProjectableObject<Shape> {
95 // we are dilating the shape along the normal, so the normal is unchanged
96 return shape.normal(p);
97 }
98
99 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
100 Vec<T,N> p = shape.convex_support(d);
101 return p + d.unit() * dilation;
102 }
103
104 Rect<T,N> bounds() const requires BoundedObject<Shape> {
105 return shape.bounds().dilated(dilation);
106 }
107
109 Vec<T,N> project(Vec<T,N> p) const
110 requires ProjectableObject<Shape> and ConvexObject<Shape>
111 {
112 Vec<T,N> p_proj = shape.project(p);
113 Vec<T,N> dp = shape.normal(p);
114 if (shape.contains(p)) {
115 // dilated projection is further away from the point
116 return p_proj + dilation * dp;
117 } else {
118 // dilated projection is more toward point
119 return p_proj - dilation * dp;
120 }
121 }
