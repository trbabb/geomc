1 /*
2  * SimpleMatrix.h
3  *
4  * This involves a gambit very similar to the VecBase scenario:
5  * We have a class which is essentially unchanging in its body, but
6  * needs to have permuted constructors based on a template parameter.
7  * Therefore we put the entire body in a base class, and then make a thin
8  * derived class with partial specialization, and design the constructor
9  * interface as we please for each specialization.
10  *
11  * This class structure is pretty darn ugly, so we hide it from the user
12  * for the purposes of Doxygen by simply pretending the (hidden) common base class
13  * is the user-facing one.
14  *
15  * Created on: Apr 18, 2013
16  * Author: tbabb
17  */

389  void transpose() {
390  if (M == N or rows() == cols()) {
391  for (index_t r = 0; r < rows(); r++) {
392  for (index_t c = 0; c < r; c++) {
393  std::swap(get(r,c), get(c,r));
394  }
395  }
396  } else if (M == 0 and N == 0) {
397  // todo: someday: there are nonsquare matrix in-place transpose algorithms
398 
399  // here we use stack memory for matrices 6x6 and smaller:
400  SmallStorage<T,36> buf(rows() * cols());
401  WrapperMatrix<T,N,M> tmp(buf.get(), cols(), rows());
402 
403  // transpose
404  for (index_t r = 0; r < rows(); ++r) {
405  for (index_t c = 0; c < cols(); ++c) {
406  tmp.set(c, r, get(r,c));
407  }
408  }
409  // copy back and adjust our dimensions:
410  std::copy(tmp.begin(), tmp.end(), begin());
411  Dimension<M>::set(n_rows, tmp.rows());
412  Dimension<N>::set(n_cols, tmp.cols());
413  }
414  }
