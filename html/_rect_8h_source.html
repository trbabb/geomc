647  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
648  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
649  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
650  index_t near_axis = 0;
651  index_t far_axis = 0;
652  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
653  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
654  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
655 
656  // test ray intersection with an infinite slab along each axis
657  // box intersections are the farthest near hit and nearest far hit
658  for (index_t axis = 0; axis < N; axis++) {
659  if (r.direction[axis] == 0) {
660  // ray direction tangent to test planes, no intersection along this axis
661  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
662  // origin outside of test slab; miss
663  return hit;
664  }
665  } else {
666  // coordinate of hit, along tested axis
667  T c1 = maxs[axis];
668  T c2 = mins[axis];
669  // ray multiple of hit
670  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
671  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
672 
673  // order s1, s2 to (near, far) (aka front, back)
674  if (s2 < s1) {
675  std::swap(s1,s2);
676  std::swap(c1,c2);
677  }
678  if (s1 > near_root) {
679  near_root = s1;
680  near_axis = axis;
681  near_coord = c1;
682  }
683  if (s2 < far_root) {
684  far_root = s2;
685  far_axis = axis;
686  far_coord = c2;
687  }
688  }
689  }
690 
691  if (near_root > far_root) {
692  // miss
693  return hit;
694  } else if (near_root > 0 and (sides & HIT_FRONT)) {
695  // hit front
696  hit.p = r * near_root;
697  hit.p[near_axis] = near_coord;
698  hit.s = near_root;
699  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
700  hit.side = HIT_FRONT;
701  hit.hit = true;
702  } else if (far_root > 0 and (sides & HIT_BACK)) {
703  // hit back
704  hit.p = r * far_root;
705  hit.p[far_axis] = far_coord;
706  hit.s = far_root;
707  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
708  hit.side = HIT_BACK;
709  hit.hit = true;
710  }
711  // if encountered cases above; hit.
712  // else miss (box entirely behind ray origin)
713  return hit;
714  }
