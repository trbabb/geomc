19// - union of rect with a point should thereafter contain the point

418 Rect<T,N> operator-(const Rect<T,N>& other) {
419 // todo: instead of returning empty() when an interval is split in two,
420 // choose the larger
421 // todo: make an exclude_signed() and handle empty-interval case,
422 // and make it behave sensibly? e.g.:
423 // • a - (-b) = a + b
424 // • (-a) - b = -(a + b)
425 // • (-a) - (-b) = -(a - b)
426 Rect<T,N> out;
427 for (index_t i = 0; i < N; ++i) {
428 T b_lo = coord(other.lo, i);
429 T a_lo = coord(lo, i);
430 T a_hi = coord(hi, i);
431 T b_hi = coord(other.hi, i);
432 if (b_lo > b_hi) continue;
433 bool pick_lo = b_lo < a_lo;
434 bool pick_hi = b_hi > a_hi;
435 coord(lo, i) = pick_lo ? a_hi : std::min(a_hi, b_lo);
436 coord(hi, i) = pick_hi ? a_lo : std::max(b_hi, a_lo);
437 }
438 return out;
439 }

745 Rect<T,N> fitted_to(Rect<T,N> other) const requires (not std::integral<T>) {
746 T max_ratio = 0;
747 point_t d0 = dimensions();
748 point_t d1 = other.dimensions();
749 // find the dimension which constrains the final size:
750 for (index_t i = 0; i < N; ++i) {
751 // do not try to "expand" any dimensions which are empty:
752 auto ratio = (coord(d0, i) == 0) ? 0 : (coord(d1, i) / coord(d0, i));
753 max_ratio = std::max(max_ratio, std::abs(ratio));
754 }
755 return Rect<T,N>::from_center(other.center(), d0 * max_ratio);
756 }

811 point_t face_areas() const {
812 point_t axial_areas;
813 T prod = 1;
814 point_t dims = dimensions();
815 // include all the extents preceding this axis (but not the axis itself)
816 for (index_t i = 0; i < N - 1; ++i) {
817 coord(axial_areas, i) = prod;
818 prod *= coord(dims, i);
819 }
820 // include all the extents following this axis (but not the axis itself)
821 prod = 1;
822 for (index_t i = N - 1; i >= 0; --i) {
823 coord(axial_areas, i) *= prod;
824 prod *= coord(dims, i);
825 }
826 return axial_areas;
827 }

870 point_t project(point_t p) const {
871 if (contains(p)) {
872 // project to the closest face;
873 // i.e. find the axis along which p is closest to a boundary and
874 // snap that coordinate to the boundary.
875 index_t best_axis = 0;
876 T winning_coord = 0;
877 T best_distance = std::numeric_limits<T>::max();
878 for (index_t i = 0; i < N; ++i) {
879 T p_i = coord(p, i);
880 T lo_i = coord(lo, i);
881 T hi_i = coord(hi, i);
882 T to_lo = std::abs(p_i - lo_i);
883 T to_hi = std::abs(p_i - hi_i);
884 T dist = std::min(to_lo, to_hi);
885 if (dist < best_distance) {
886 winning_coord = to_lo < to_hi ? lo_i : hi_i;
887 best_axis = i;
888 best_distance = dist;
889 }
890 }
891 p[best_axis] = winning_coord;
892 return p;
893 } else {
894 return clip(p);
895 }
896 }

899 point_t normal(point_t p) const {
900 // compute the outward-pointing normal from `p`.
901 // we don't just do `p - project(p)`, because that behaves degenerately near
902 // the surface of the range (where we are most likely to care about the normal).
903
904 // nearest point in the range to `p`
905 point_t clipped;
906 // how many axes are outside the bound?
907 index_t outside = 0;
908 // which axis' plane is closest to p?
909 index_t nearest_axis;
910 // what is the distance to the nearest bounding plane?
911 T nearest_plane = std::numeric_limits<T>::max();
912 // is the nearest plane a lower (-1) or upper (+1) bound?
913 index_t nearest_sign;
914 // the last axis we encountered that we were not contained by
915 index_t exterior_axis;
916 // is the nearest exterior face a lower bound (-1) or upper bound (+1)?
917 index_t exterior_sign;
918
919 for (index_t i = 0; i < N; ++i) {
920 T x = coord(p, i);
921 T lo_i = coord(lo, i);
922 T hi_i = coord(hi, i);
923 coord(clipped, i) = std::min(hi_i, std::max(lo_i, x));
924 bool is_out = x < lo_i or x > hi_i;
925 outside += is_out;
926 // compute distances to the two bounding planes
927 T d0 = std::abs(x - coord(lo, i));
928 T d1 = std::abs(x - coord(hi, i));
929 T d_min = std::min(d0, d1);
930 if (d_min < nearest_plane) {
931 nearest_axis = i;
932 nearest_plane = d_min;
933 nearest_sign = d0 < d1 ? -1 : 1;
934 }
935 if (is_out) {
936 exterior_axis = i;
937 exterior_sign = d0 < d1 ? -1 : 1;
938 }
939 }
940 if (outside == 0) {
941 // we're contained by the range. return a normal pointing toward the nearest face.
942 point_t n;
943 coord(n, nearest_axis) = nearest_sign;
944 return n;
945 } else if (outside == 1) {
946 // we're outside the range, but project to an axial face. return its normal
947 point_t n;
948 coord(n, exterior_axis) = exterior_sign;
949 return n;
950 } else {
951 // the nearest point is on a corner / edge.
952 // draw a vector from that point to `p` and unitize.
953 // this still suffers numerical instability near the surface, but it's
954 // near a degenerate feature, so there's not really anything we can do.
955 return (p - clipped).unit();
956 }
957 }

1037 std::pair<point_t,bool> contact_vector(const Rect<T,N>& other) const {
1038 point_t v;
1039 bool overlap[N];
1040 bool all_overlap = true;
1041 index_t shortest_axis;
1042 index_t shortest_val = std::numeric_limits<T>::max();
1043 for (index_t i = 0; i < N; ++i) {
1044 auto r0 = this->axis(i);
1045 auto r1 = other.axis(i);
1046 T d0 = r1.lo - r0.hi;
1047 T d1 = r1.hi - r0.lo;
1048 overlap[i] = r0.intersects(r1);
1049 all_overlap = all_overlap and overlap[i];
1050 auto s = std::abs(d0) < std::abs(d1) ? d0 : d1;
1051 coord(v, i) = s;
1052 if (std::abs(s) < std::abs(shortest_val)) {
1053 shortest_val = s;
1054 shortest_axis = i;
1055 }
1056 }
1057 if (all_overlap) {
1058 // if the two rects overlap, move along the axis that is closest to the
1059 // surface. once those ranges are disjoint, it doesn't matter that the
1060 // other axes overlap. (and separating other axes will just increase the
1061 // displacement distance)
1062 v = {};
1063 coord(v, shortest_axis) = shortest_val;
1064 } else {
1065 // if the two rects don't overlap, then close the gap on all
1066 // non-overlapping axes
1067 for (index_t i = 0; i < N; ++i) {
1068 if (overlap[i]) {
1069 // overlap on this axis.
1070 coord(v, i) = 0;
1071 }
1072 }
1073 }
1074 return {v, all_overlap};
1075 }

1087 Rect<T,1> intersect(const Ray<T,N>& r) const {
1088 Rect<T,1> interval = Rect<T,1>::full;
1089 for (index_t axis = 0; axis < N; ++axis) {
1090 T dx = coord(r.direction, axis);
1091 T o_i = coord(r.origin, axis);
1092 T lo_i = coord(lo, axis);
1093 T hi_i = coord(hi, axis);
1094 if (dx == 0) {
1095 // ray direction is tangent to the test planes; no intersection on this axis.
1096 if (o_i < lo_i or o_i > hi_i) {
1097 // origin outside the test slab; miss
1098 return Rect<T,1>();
1099 }
1100 } else {
1101 interval &= Rect<T,1>::from_corners(
1102 (hi_i - o_i) / dx,
1103 (lo_i - o_i) / dx);
1104 }
1105 }
1106 return interval;
1107 }
