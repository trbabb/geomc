676  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
677  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
678  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
679  index_t near_axis = 0;
680  index_t far_axis = 0;
681  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
682  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
683  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
684 
685  // test ray intersection with an infinite slab along each axis
686  // box intersections are the farthest near hit and nearest far hit
687  for (index_t axis = 0; axis < N; axis++) {
688  if (r.direction[axis] == 0) {
689  // ray direction tangent to test planes, no intersection along this axis
690  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
691  // origin outside of test slab; miss
692  return hit;
693  }
694  } else {
695  // coordinate of hit, along tested axis
696  T c1 = maxs[axis];
697  T c2 = mins[axis];
698  // ray multiple of hit
699  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
700  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
701 
702  // order s1, s2 to (near, far) (aka front, back)
703  if (s2 < s1) {
704  std::swap(s1,s2);
705  std::swap(c1,c2);
706  }
707  if (s1 > near_root) {
708  near_root = s1;
709  near_axis = axis;
710  near_coord = c1;
711  }
712  if (s2 < far_root) {
713  far_root = s2;
714  far_axis = axis;
715  far_coord = c2;
716  }
717  }
718  }
719 
720  if (near_root > far_root) {
721  // miss
722  return hit;
723  } else if (near_root > 0 and (sides & HIT_FRONT)) {
724  // hit front
725  hit.p = r * near_root;
726  hit.p[near_axis] = near_coord;
727  hit.s = near_root;
728  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
729  hit.side = HIT_FRONT;
730  hit.hit = true;
731  } else if (far_root > 0 and (sides & HIT_BACK)) {
732  // hit back
733  hit.p = r * far_root;
734  hit.p[far_axis] = far_coord;
735  hit.s = far_root;
736  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
737  hit.side = HIT_BACK;
738  hit.hit = true;
739  }
740  // if encountered cases above; hit.
741  // else miss (box entirely behind ray origin)
742  return hit;
743  }
