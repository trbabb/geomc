420 Rect<T,N> operator-(const Rect<T,N>& other) {
421 // todo: instead of returning empty() when an interval is split in two,
422 // choose the larger
423 // todo: make an exclude_signed() and handle empty-interval case,
424 // and make it behave sensibly? e.g.:
425 // a - (-b) = a + b
426 // (-a) - b = -(a + b)
427 // (-a) - (-b) = -(a - b)
428 Rect<T,N> out;
429 for (index_t i = 0; i < N; ++i) {
430 T b_lo = ptype::iterator(other.lo)[i];
431 T a_lo = ptype::iterator(lo)[i];
432 T a_hi = ptype::iterator(hi)[i];
433 T b_hi = ptype::iterator(other.hi)[i];
434 if (b_lo > b_hi) continue;
435 bool pick_lo = b_lo < a_lo;
436 bool pick_hi = b_hi > a_hi;
437 ptype::iterator(lo)[i] = pick_lo ? a_hi : std::min(a_hi, b_lo);
438 ptype::iterator(hi)[i] = pick_hi ? a_lo : std::max(b_hi, a_lo);
439 }
440 return out;

754 Rect<T,N> fitted_to(Rect<T,N> other) const
755#if __cplusplus >= 202002L
756 requires (not std::integral<T>)
757#endif
758 {
759 static_assert(
760 not std::is_integral<T>::value,
761 "fitted_to() is not available for integral types"
762 );
763 T max_ratio = 0;
764 point_t d0 = dimensions();
765 point_t d1 = other.dimensions();
766 auto d0p = ptype::iterator(d0);
767 auto d1p = ptype::iterator(d1);
768 // find the dimension which constrains the final size:
769 for (index_t i = 0; i < N; ++i) {
770 // do not try to "expand" any dimensions which are empty:
771 auto ratio = (d0p[i] == 0) ? 0 : (d1p[i] / d0p[i]);
772 max_ratio = std::max(max_ratio, std::abs(ratio));
773 }
774 return Rect<T,N>::from_center(other.center(), d0 * max_ratio);
775 }

857 point_t project(point_t p) const {
858 if (contains(p)) {
859 // project to the closest face;
860 // i.e. find the axis along which p is closest to a boundary and
861 // snap that coordinate to the boundary.
862 index_t best_axis = 0;
863 T winning_coord = 0;
864 T best_distance = std::numeric_limits<T>::max();
865 for (index_t i = 0; i < N; ++i) {
866 T p_i = ptype::iterator(p) [i];
867 T lo_i = ptype::iterator(lo)[i];
868 T hi_i = ptype::iterator(hi)[i];
869 T to_lo = std::abs(p_i - lo_i);
870 T to_hi = std::abs(p_i - hi_i);
871 T dist = std::min(to_lo, to_hi);
872 if (dist < best_distance) {
873 winning_coord = to_lo < to_hi ? lo_i : hi_i;
874 best_axis = i;
875 best_distance = dist;
876 }
877 }
878 p[best_axis] = winning_coord;
879 return p;
880 } else {
881 return clip(p);
882 }
883 }

932 std::pair<point_t,bool> contact_vector(const Rect<T,N>& other) const {
933 point_t v;
934 bool overlap[N];
935 bool all_overlap = true;
936 index_t shortest_axis;
937 index_t shortest_val = std::numeric_limits<T>::max();
938 for (index_t i = 0; i < N; ++i) {
939 auto r0 = this->axis(i);
940 auto r1 = other.axis(i);
941 T d0 = r1.lo - r0.hi;
942 T d1 = r1.hi - r0.lo;
943 overlap[i] = r0.intersects(r1);
944 all_overlap = all_overlap and overlap[i];
945 auto s = std::abs(d0) < std::abs(d1) ? d0 : d1;
946 ptype::iterator(v)[i] = s;
947 if (std::abs(s) < std::abs(shortest_val)) {
948 shortest_val = s;
949 shortest_axis = i;
950 }
951 }
952 if (all_overlap) {
953 // if the two rects overlap, move along the axis that is closest to the
954 // surface. once those ranges are disjoint, it doesn't matter that the
955 // other axes overlap. (and separating other axes will just increase the
956 // displacement distance)
957 v = {};
958 ptype::iterator(v)[shortest_axis] = shortest_val;
959 } else {
960 // if the two rects don't overlap, then close the gap on all
961 // non-overlapping axes
962 for (index_t i = 0; i < N; ++i) {
963 if (overlap[i]) {
964 // overlap on this axis.
965 ptype::iterator(v)[i] = 0;
966 }
967 }
968 }
969 return {v, all_overlap};
970 }

982 Rect<T,1> intersect(const Ray<T,N>& r) const {
983 Rect<T,1> interval = Rect<T,1>::full;
984 for (index_t axis = 0; axis < N; ++axis) {
985 T dx = ptype::iterator(r.direction)[axis];
986 T o_i = ptype::iterator(r.origin)[axis];
987 T lo_i = ptype::iterator(lo)[axis];
988 T hi_i = ptype::iterator(hi)[axis];
989 if (dx == 0) {
990 // ray direction is tangent to the test planes; no intersection on this axis.
991 if (o_i < lo_i or o_i > hi_i) {
992 // origin outside the test slab; miss
993 return Rect<T,1>();
994 }
995 } else {
996 interval &= Rect<T,1>::spanning_corners(
997 (hi_i - o_i) / dx,
998 (lo_i - o_i) / dx);
999 }
1000 }
1001 return interval;
1002 }
