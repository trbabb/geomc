416 Rect<T,N> operator-(const Rect<T,N>& other) {
417 // todo: instead of returning empty() when an interval is split in two,
418 // choose the larger
419 // todo: make an exclude_signed() and handle empty-interval case,
420 // and make it behave sensibly? e.g.:
421 // • a - (-b) = a + b
422 // • (-a) - b = -(a + b)
423 // • (-a) - (-b) = -(a - b)
424 Rect<T,N> out;
425 for (index_t i = 0; i < N; ++i) {
426 T b_lo = coord(other.lo, i);
427 T a_lo = coord(lo, i);
428 T a_hi = coord(hi, i);
429 T b_hi = coord(other.hi, i);
430 if (b_lo > b_hi) continue;
431 bool pick_lo = b_lo < a_lo;
432 bool pick_hi = b_hi > a_hi;
433 coord(lo, i) = pick_lo ? a_hi : std::min(a_hi, b_lo);
434 coord(hi, i) = pick_hi ? a_lo : std::max(b_hi, a_lo);
435 }
436 return out;

784 Rect<T,N> fitted_to(Rect<T,N> other) const requires (not std::integral<T>) {
785 T max_ratio = 0;
786 point_t d0 = dimensions();
787 point_t d1 = other.dimensions();
788 // find the dimension which constrains the final size:
789 for (index_t i = 0; i < N; ++i) {
790 // do not try to "expand" any dimensions which are empty:
791 auto ratio = (coord(d0, i) == 0) ? 0 : (coord(d1, i) / coord(d0, i));
792 max_ratio = std::max(max_ratio, std::abs(ratio));
793 }
794 return Rect<T,N>::from_center(other.center(), d0 * max_ratio);
795 }

850 point_t face_areas() const {
851 point_t axial_areas;
852 T prod = 1;
853 point_t dims = dimensions();
854 // include all the extents preceding this axis (but not the axis itself)
855 for (index_t i = 0; i < N - 1; ++i) {
856 coord(axial_areas, i) = prod;
857 prod *= coord(dims, i);
858 }
859 // include all the extents following this axis (but not the axis itself)
860 prod = 1;
861 for (index_t i = N - 1; i >= 0; --i) {
862 coord(axial_areas, i) *= prod;
863 prod *= coord(dims, i);
864 }
865 return axial_areas;
866 }

909 point_t project(point_t p) const {
910 if (contains(p)) {
911 // project to the closest face;
912 // i.e. find the axis along which p is closest to a boundary and
913 // snap that coordinate to the boundary.
914 index_t best_axis = 0;
915 T winning_coord = 0;
916 T best_distance = std::numeric_limits<T>::max();
917 for (index_t i = 0; i < N; ++i) {
918 T p_i = coord(p, i);
919 T lo_i = coord(lo, i);
920 T hi_i = coord(hi, i);
921 T to_lo = std::abs(p_i - lo_i);
922 T to_hi = std::abs(p_i - hi_i);
923 T dist = std::min(to_lo, to_hi);
924 if (dist < best_distance) {
925 winning_coord = to_lo < to_hi ? lo_i : hi_i;
926 best_axis = i;
927 best_distance = dist;
928 }
929 }
930 p[best_axis] = winning_coord;
931 return p;
932 } else {
933 return clip(p);
934 }
935 }

938 point_t normal(point_t p) const {
939 // compute the outward-pointing normal from `p`.
940 // we don't just do `p - project(p)`, because that behaves degenerately near
941 // the surface of the range (where we are most likely to care about the normal).
942
943 // nearest point in the range to `p`
944 point_t clipped;
945 // how many axes are outside the bound?
946 index_t outside = 0;
947 // which axis' plane is closest to p?
948 index_t nearest_axis;
949 // what is the distance to the nearest bounding plane?
950 T nearest_plane = std::numeric_limits<T>::max();
951 // is the nearest plane a lower (-1) or upper (+1) bound?
952 index_t nearest_sign;
953 // the last axis we encountered that we were not contained by
954 index_t exterior_axis;
955 // is the nearest exterior face a lower bound (-1) or upper bound (+1)?
956 index_t exterior_sign;
957
958 for (index_t i = 0; i < N; ++i) {
959 T x = coord(p, i);
960 T lo_i = coord(lo, i);
961 T hi_i = coord(hi, i);
962 coord(clipped, i) = std::min(hi_i, std::max(lo_i, x));
963 bool is_out = x < lo_i or x > hi_i;
964 outside += is_out;
965 // compute distances to the two bounding planes
966 T d0 = std::abs(x - coord(lo, i));
967 T d1 = std::abs(x - coord(hi, i));
968 T d_min = std::min(d0, d1);
969 if (d_min < nearest_plane) {
970 nearest_axis = i;
971 nearest_plane = d_min;
972 nearest_sign = d0 < d1 ? -1 : 1;
973 }
974 if (is_out) {
975 exterior_axis = i;
976 exterior_sign = d0 < d1 ? -1 : 1;
977 }
978 }
979 if (outside == 0) {
980 // we're contained by the range. return a normal pointing toward the nearest face.
981 point_t n;
982 coord(n, nearest_axis) = nearest_sign;
983 return n;
984 } else if (outside == 1) {
985 // we're outside the range, but project to an axial face. return its normal
986 point_t n;
987 coord(n, exterior_axis) = exterior_sign;
988 return n;
989 } else {
990 // the nearest point is on a corner / edge.
991 // draw a vector from that point to `p` and unitize.
992 // this still suffers numerical instability near the surface, but it's
993 // near a degenerate feature, so there's not really anything we can do.
994 return (p - clipped).unit();
995 }
996 }

1076 std::pair<point_t,bool> contact_vector(const Rect<T,N>& other) const {
1077 point_t v;
1078 bool overlap[N];
1079 bool all_overlap = true;
1080 index_t shortest_axis;
1081 index_t shortest_val = std::numeric_limits<T>::max();
1082 for (index_t i = 0; i < N; ++i) {
1083 auto r0 = this->axis(i);
1084 auto r1 = other.axis(i);
1085 T d0 = r1.lo - r0.hi;
1086 T d1 = r1.hi - r0.lo;
1087 overlap[i] = r0.intersects(r1);
1088 all_overlap = all_overlap and overlap[i];
1089 auto s = std::abs(d0) < std::abs(d1) ? d0 : d1;
1090 coord(v, i) = s;
1091 if (std::abs(s) < std::abs(shortest_val)) {
1092 shortest_val = s;
1093 shortest_axis = i;
1094 }
1095 }
1096 if (all_overlap) {
1097 // if the two rects overlap, move along the axis that is closest to the
1098 // surface. once those ranges are disjoint, it doesn't matter that the
1099 // other axes overlap. (and separating other axes will just increase the
1100 // displacement distance)
1101 v = {};
1102 coord(v, shortest_axis) = shortest_val;
1103 } else {
1104 // if the two rects don't overlap, then close the gap on all
1105 // non-overlapping axes
1106 for (index_t i = 0; i < N; ++i) {
1107 if (overlap[i]) {
1108 // overlap on this axis.
1109 coord(v, i) = 0;
1110 }
1111 }
1112 }
1113 return {v, all_overlap};
1114 }

1126 Rect<T,1> intersect(const Ray<T,N>& r) const {
1127 Rect<T,1> interval = Rect<T,1>::full;
1128 for (index_t axis = 0; axis < N; ++axis) {
1129 T dx = coord(r.direction, axis);
1130 T o_i = coord(r.origin, axis);
1131 T lo_i = coord(lo, axis);
1132 T hi_i = coord(hi, axis);
1133 if (dx == 0) {
1134 // ray direction is tangent to the test planes; no intersection on this axis.
1135 if (o_i < lo_i or o_i > hi_i) {
1136 // origin outside the test slab; miss
1137 return Rect<T,1>();
1138 }
1139 } else {
1140 interval &= Rect<T,1>::from_corners(
1141 (hi_i - o_i) / dx,
1142 (lo_i - o_i) / dx);
1143 }
1144 }
1145 return interval;
1146 }
