23// all boundaries are inclusive. this is inconsistent with half-open interval convention,
24// but:
25// - union of rect with a point should thereafter contain the point
26// - with a half-open interval, we'd need to "increment" the upper bound
27// to contain p
28// - which is slow and ugly
29// - the upper and lower bounds now mean different things, and
30// that means you need to increment and decrement
31// Sometimes And Not Other Times.
32// - compare to: "both endpoints are symmetrical and the same logic applies everywhere"
33// - the biggest downside: ownership of points on the faces of abutting Rects
34// is ambiguous (for Rects with real-valued coordinates).

370 inline Rect<T,N> operator-(const Rect<T,N>& other) {
371 // todo: instead of returning empty() when an interval is split in two,
372 // choose the larger
373 // todo: make an exclude_signed() and handle empty-interval case,
374 // and make it behave sensibly? e.g.:
375 // • a - (-b) = a + b
376 // • (-a) - b = -(a + b)
377 // • (-a) - (-b) = -(a - b)
378 Rect<T,N> out;
379 for (index_t i = 0; i < N; ++i) {
380 T b_lo = ptype::iterator(other.lo)[i];
381 T a_lo = ptype::iterator(lo)[i];
382 T a_hi = ptype::iterator(hi)[i];
383 T b_hi = ptype::iterator(other.hi)[i];
384 if (b_lo > b_hi) continue;
385 bool pick_lo = b_lo < a_lo;
386 bool pick_hi = b_hi > a_hi;
387 ptype::iterator(lo)[i] = pick_lo ? a_hi : std::min(a_hi, b_lo);
388 ptype::iterator(hi)[i] = pick_hi ? a_lo : std::max(b_hi, a_lo);
389 }
390 return out;
391 }

721 inline point_t project(point_t p) const {
722 if (contains(p)) {
723 // project to the closest face;
724 // i.e. find the axis along which p is closest to a boundary and
725 // snap that coordinate to the boundary.
726 index_t best_axis = 0;
727 T winning_coord = 0;
728 T best_distance = std::numeric_limits<T>::max();
729 for (index_t i = 0; i < N; ++i) {
730 T p_i = ptype::iterator(p) [i];
731 T lo_i = ptype::iterator(lo)[i];
732 T hi_i = ptype::iterator(hi)[i];
733 T to_lo = std::abs(p_i - lo_i);
734 T to_hi = std::abs(p_i - hi_i);
735 T dist = std::min(to_lo, to_hi);
736 if (dist < best_distance) {
737 winning_coord = to_lo < to_hi ? lo_i : hi_i;
738 best_axis = i;
739 best_distance = dist;
740 }
741 }
742 p[best_axis] = winning_coord;
743 return p;
744 } else {
745 return clip(p);
746 }
747 }

796 std::pair<point_t,bool> contact_vector(const Rect<T,N>& other) const {
797 point_t v;
798 bool overlap[N];
799 bool all_overlap = true;
800 index_t shortest_axis;
801 index_t shortest_val = std::numeric_limits<T>::max();
802 for (index_t i = 0; i < N; ++i) {
803 auto r0 = this->axis(i);
804 auto r1 = other.axis(i);
805 T d0 = r1.lo - r0.hi;
806 T d1 = r1.hi - r0.lo;
807 overlap[i] = r0.intersects(r1);
808 all_overlap = all_overlap and overlap[i];
809 auto s = std::abs(d0) < std::abs(d1) ? d0 : d1;
810 ptype::iterator(v)[i] = s;
811 if (std::abs(s) < std::abs(shortest_val)) {
812 shortest_val = s;
813 shortest_axis = i;
814 }
815 }
816 if (all_overlap) {
817 // if the two rects overlap, move along the axis that is closest to the
818 // surface. once those ranges are disjoint, it doesn't matter that the
819 // other axes overlap. (and separating other axes will just increase the
820 // displacement distance)
821 v = {};
822 ptype::iterator(v)[shortest_axis] = shortest_val;
823 } else {
824 // if the two rects don't overlap, then close the gap on all
825 // non-overlapping axes
826 for (index_t i = 0; i < N; ++i) {
827 if (overlap[i]) {
828 // overlap on this axis.
829 ptype::iterator(v)[i] = 0;
830 }
831 }
832 }
833 return {v, all_overlap};
834 }

846 Rect<T,1> intersect(const Ray<T,N>& r) const {
847 Rect<T,1> interval = Rect<T,1>::full;
848 for (index_t axis = 0; axis < N; ++axis) {
849 T dx = ptype::iterator(r.direction)[axis];
850 T o_i = ptype::iterator(r.origin)[axis];
851 T lo_i = ptype::iterator(lo)[axis];
852 T hi_i = ptype::iterator(hi)[axis];
853 if (dx == 0) {
854 // ray direction is tangent to the test planes; no intersection on this axis.
855 if (o_i < lo_i or o_i > hi_i) {
856 // origin outside the test slab; miss
857 return Rect<T,1>();
858 }
859 } else {
860 interval &= Rect<T,1>::spanning_corners(
861 (hi_i - o_i) / dx,
862 (lo_i - o_i) / dx);
863 }
864 }
865 return interval;
866 }
