430 Rect<T,N> operator-(const Rect<T,N>& other) {
431 // todo: instead of returning empty() when an interval is split in two,
432 // choose the larger
433 // todo: make an exclude_signed() and handle empty-interval case,
434 // and make it behave sensibly? e.g.:
435 // a - (-b) = a + b
436 // (-a) - b = -(a + b)
437 // (-a) - (-b) = -(a - b)
438 Rect<T,N> out;
439 for (index_t i = 0; i < N; ++i) {
440 T b_lo = ptype::iterator(other.lo)[i];
441 T a_lo = ptype::iterator(lo)[i];
442 T a_hi = ptype::iterator(hi)[i];
443 T b_hi = ptype::iterator(other.hi)[i];
444 if (b_lo > b_hi) continue;
445 bool pick_lo = b_lo < a_lo;
446 bool pick_hi = b_hi > a_hi;
447 ptype::iterator(lo)[i] = pick_lo ? a_hi : std::min(a_hi, b_lo);
448 ptype::iterator(hi)[i] = pick_hi ? a_lo : std::max(b_hi, a_lo);
449 }
450 return out;
451 }

764 Rect<T,N> fitted_to(Rect<T,N> other) const
765#if __cplusplus >= 202002L
766 requires (not std::integral<T>)
767#endif
768 {
769 static_assert(
770 not std::is_integral<T>::value,
771 "fitted_to() is not available for integral types"
772 );
773 T max_ratio = 0;
774 point_t d0 = dimensions();
775 point_t d1 = other.dimensions();
776 auto d0p = ptype::iterator(d0);
777 auto d1p = ptype::iterator(d1);
778 // find the dimension which constrains the final size:
779 for (index_t i = 0; i < N; ++i) {
780 // do not try to "expand" any dimensions which are empty:
781 auto ratio = (d0p[i] == 0) ? 0 : (d1p[i] / d0p[i]);
782 max_ratio = std::max(max_ratio, std::abs(ratio));
783 }
784 return Rect<T,N>::from_center(other.center(), d0 * max_ratio);
785 }

840 point_t face_areas() const {
841 point_t axial_areas;
842 T prod = 1;
843 point_t dims = dimensions();
844 // include all the extents preceding this axis (but not the axis itself)
845 for (index_t i = 0; i < N - 1; ++i) {
846 ptype::iterator(axial_areas)[i] = prod;
847 prod *= ptype::iterator(dims)[i];
848 }
849 // include all the extents following this axis (but not the axis itself)
850 prod = 1;
851 for (index_t i = N - 1; i >= 0; --i) {
852 ptype::iterator(axial_areas)[i] *= prod;
853 prod *= ptype::iterator(dims)[i];
854 }
855 return axial_areas;
856 }

899 point_t project(point_t p) const {
900 if (contains(p)) {
901 // project to the closest face;
902 // i.e. find the axis along which p is closest to a boundary and
903 // snap that coordinate to the boundary.
904 index_t best_axis = 0;
905 T winning_coord = 0;
906 T best_distance = std::numeric_limits<T>::max();
907 for (index_t i = 0; i < N; ++i) {
908 T p_i = ptype::iterator(p) [i];
909 T lo_i = ptype::iterator(lo)[i];
910 T hi_i = ptype::iterator(hi)[i];
911 T to_lo = std::abs(p_i - lo_i);
912 T to_hi = std::abs(p_i - hi_i);
913 T dist = std::min(to_lo, to_hi);
914 if (dist < best_distance) {
915 winning_coord = to_lo < to_hi ? lo_i : hi_i;
916 best_axis = i;
917 best_distance = dist;
918 }
919 }
920 p[best_axis] = winning_coord;
921 return p;
922 } else {
923 return clip(p);
924 }

928 point_t normal(point_t p) const {
929 // compute the outward-pointing normal from `p`.
930 // we don't just do `p - project(p)`, because that behaves degenerately near
931 // the surface of the range (where we are most likely to care about the normal).
932
933 // nearest point in the range to `p`
934 point_t clipped;
935 // how many axes are outside the bound?
936 index_t outside = 0;
937 // which axis' plane is closest to p?
938 index_t nearest_axis;
939 // what is the distance to the nearest bounding plane?
940 T nearest_plane = std::numeric_limits<T>::max();
941 // is the nearest plane a lower (-1) or upper (+1) bound?
942 index_t nearest_sign;
943 // the last axis we encountered that we were not contained by
944 index_t exterior_axis;
945 // is the nearest exterior face a lower bound (-1) or upper bound (+1)?
946 index_t exterior_sign;
947
948 for (index_t i = 0; i < N; ++i) {
949 T x = ptype::iterator(p)[i];
950 T lo_i = ptype::iterator(lo)[i];
951 T hi_i = ptype::iterator(hi)[i];
952 ptype::iterator(clipped)[i] = std::min(hi_i, std::max(lo_i, x));
953 bool is_out = x < lo_i or x > hi_i;
954 outside += is_out;
955 // compute distances to the two bounding planes
956 T d0 = std::abs(x - ptype::iterator(lo)[i]);
957 T d1 = std::abs(x - ptype::iterator(hi)[i]);
958 T d_min = std::min(d0, d1);
959 if (d_min < nearest_plane) {
960 nearest_axis = i;
961 nearest_plane = d_min;
962 nearest_sign = d0 < d1 ? -1 : 1;
963 }
964 if (is_out) {
965 exterior_axis = i;
966 exterior_sign = d0 < d1 ? -1 : 1;
967 }
968 }
969 if (outside == 0) {
970 // we're contained by the range. return a normal pointing toward the nearest face.
971 point_t n;
972 ptype::iterator(n)[nearest_axis] = nearest_sign;
973 return n;
974 } else if (outside == 1) {
975 // we're outside the range, but project to an axial face. return its normal
976 point_t n;
977 ptype::iterator(n)[exterior_axis] = exterior_sign;
978 return n;
979 } else {
980 // the nearest point is on a corner / edge.
981 // draw a vector from that point to `p` and unitize.
982 // this still suffers numerical instability near the surface, but it's
983 // near a degenerate feature, so there's not really anything we can do.
984 return (p - clipped).unit();
985 }
986 }

1066 std::pair<point_t,bool> contact_vector(const Rect<T,N>& other) const {
1067 point_t v;
1068 bool overlap[N];
1069 bool all_overlap = true;
1070 index_t shortest_axis;
1071 index_t shortest_val = std::numeric_limits<T>::max();
1072 for (index_t i = 0; i < N; ++i) {
1073 auto r0 = this->axis(i);
1074 auto r1 = other.axis(i);
1075 T d0 = r1.lo - r0.hi;
1076 T d1 = r1.hi - r0.lo;
1077 overlap[i] = r0.intersects(r1);
1078 all_overlap = all_overlap and overlap[i];
1079 auto s = std::abs(d0) < std::abs(d1) ? d0 : d1;
1080 ptype::iterator(v)[i] = s;
1081 if (std::abs(s) < std::abs(shortest_val)) {
1082 shortest_val = s;
1083 shortest_axis = i;
1084 }
1085 }
1086 if (all_overlap) {
1087 // if the two rects overlap, move along the axis that is closest to the
1088 // surface. once those ranges are disjoint, it doesn't matter that the
1089 // other axes overlap. (and separating other axes will just increase the
1090 // displacement distance)
1091 v = {};
1092 ptype::iterator(v)[shortest_axis] = shortest_val;
1093 } else {
1094 // if the two rects don't overlap, then close the gap on all
1095 // non-overlapping axes
1096 for (index_t i = 0; i < N; ++i) {
1097 if (overlap[i]) {
1098 // overlap on this axis.
1099 ptype::iterator(v)[i] = 0;
1100 }
1101 }
1102 }
1103 return {v, all_overlap};
1104 }

1116 Rect<T,1> intersect(const Ray<T,N>& r) const {
1117 Rect<T,1> interval = Rect<T,1>::full;
1118 for (index_t axis = 0; axis < N; ++axis) {
1119 T dx = ptype::iterator(r.direction)[axis];
1120 T o_i = ptype::iterator(r.origin)[axis];
1121 T lo_i = ptype::iterator(lo)[axis];
1122 T hi_i = ptype::iterator(hi)[axis];
1123 if (dx == 0) {
1124 // ray direction is tangent to the test planes; no intersection on this axis.
1125 if (o_i < lo_i or o_i > hi_i) {
1126 // origin outside the test slab; miss
1127 return Rect<T,1>();
1128 }
1129 } else {
1130 interval &= Rect<T,1>::spanning_corners(
1131 (hi_i - o_i) / dx,
1132 (lo_i - o_i) / dx);
1133 }
1134 }
1135 return interval;
1136 }
