12 // returns 0 if unequal, 1 if same orientation, -1 if opposite orientation
13 // pass a mutable simplex to s1 plz, kthx
14 int simplex_comparison(const index_t* s0, index_t* s1, index_t n) {
15  // reorder s1 to look like s0, keeping track of how many transpositions are made
16  // to determine parity.
17  int swaps = 0;
18  for (index_t i = 0; i < n; i++) {
19  // find s0[i] in s1 and swap positions.
20  for (index_t j = i; j < n; j++) {
21  if (s0[i] == s1[j]) {
22  std::swap(s1[i], s1[j]);
23  swaps++;
24  goto FOUNDIT;
25  }
26  }
27  return 0; // got to end of list without finding current element.
28  FOUNDIT:
29  }
30  return (swaps & 1) ? -1 : 1;
31 }

66 template <typename T, index_t N>
67 bool separation_axis(const Convex<T,N>& shape_a,
68  const Convex<T,N>& shape_b,
69  Vec<T,N>* overlap_axis,
70  double fractional_tolerance = 0.001,
71  index_t iteration_limit = -1) {
72  Simplex<T,N> splex;
73 
74  // assignment below is intended.
75  if (not gjk_intersect(shape_a, shape_b, overlap_axis, &splex)) {
76  return false;
77  }
78 
79  std::vector< Vec<T,N> > verts(splex.pts, splex.pts + splex.n);
80  std::vector< detail::Edge<T,N> > edges;
81  std::vector< detail::Face<T,N> > faces;
82 
83  Vec<T,N> splex_vtx_buf[N-1];
84 
85  // initialize the `faces` vec with those of the GJK simplex.
86  for (index_t i = 0; i <= N; i++) {
87  // choose a vertex to exclude, making a face, and construct it.
88  Vec<T,N> A = verts[i];
89  Face<T,N> face;
90  for (index_t j = 0; j < N; j++) {
91  face.v[j] = (i + j + 1) % N;
92  if (j > 0) splex_vtx_buf[j-1] = verts[face.v[0]] - verts[i];
93  }
94 
95  // correct the winding.
96  Vec<T,N> n = orthogonal(splex_vtx_buf);
97  if (n.dot(A - verts[face.v[0]]) > 0) {
98  // winding is backwards; normal points inwards.
99  // make an arbitrary transposition to flip it.
100  std::swap(face.v[0], face.v[1]);
101  n = -n;
102  }
103  face.n = n.unit();
104  faces.push_back(face);
105  }
106 
107  std::vector< detail::Face<T,N> >::iterator best_face;
108  Vec<T,N> last_proj;
109  bool looped = false;
110 
111  // iterate on the face list, choosing the closest one and expanding it
112  while (index_t k = 0; k < iteration_limit or iteration_limit < 1; k++) {
113 
114  // find the face closest to the origin.
115  T d = std::numeric_limits<T>::max();
116  for (auto f = faces.begin(); f != faces.end(); ++f) {
117  Vec<T,N> v = verts[f->v[0]]; // a vertex on the face
118  T d_this = std::abs(f->n.dot(v));
119  if (d_this < d) {
120  d = d_this;
121  best_face = f;
122  }
123  }
124 
125  // project the origin to the closest face.
126  *overlap_axis = verts[best_face->v[0]].projectOn(best_face->n);
127 
128  // if our estimated closest point is not sufficiently different from
129  // our last estimate, decide that we've converged and quit.
130  if ((looped and fractional_tolerance > 0 and
131  (last_proj - overlap_axis).mag() / overlap_axis.mag() < fractional_tolerance)
132  or *overlap_axis == Vec<T,N>::zeros) {
133  break;
134  }
135 
136  last_proj = *overlap_axis;
137 
138  // look in a direction normal to that face
139  // and find a new point on the minkowski difference.
140  Vec<T,N> minkowski_pt = shape_a.convex_support( best_face->n) -
141  shape_b.convex_support(-best_face->n);
142 
143  // finish if we add a duplicate point (possible for polytopes)
144  // such a point would produce a degenerate face, i.e. we aren't going
145  // to find any more new faces. this is a linear algorithm, but I am
146  // not sure whether a hashset construction/test is actually worth it.
147  for (auto v : verts) if (v == minkowski_pt) break;
148 
149  verts.push_back(minkowski_pt);
150 
151  // now delete all the faces that fall "behind" the new point.
152  for (auto f = faces.begin(); f != faces.end();) {
153  // if this face is below the new point...
154  if (f->n.dot(minkowski_pt - verts[f->pts[0]]) > 0) {
155  // add all the edges of this face to the boundary edge list.
156  // (all the non-internal edges are hole boundary, which must each
157  // become part of a new face).
158  for (index_t i = 0; i < N; i++) {
159  // find a vertex to exclude from the face, making an edge.
160  Edge<T,N> e;
161  for (index_t j = 0; j < N - 1; j++) {
162  e.v[j] = f->pts[(i + j + 1) % N];
163  if (N > 3 and j > 0) {
164  splex_vtx_buf[j-1] = verts[e.v[0]] - verts[e.v[j]];
165  }
166  }
167 
168  // correct the edge winding, if necessary.
169  if (N > 3) {
170  splex_vtx_buf[N-2] = f->n;
171  // the parity here is arbitrary; all that matters is that it's consistent.
172  Vec<T,N> n_edge = orthogonal(splex_vtx_buf);
173  if (n_edge.dot(verts[e.v[0]] - splex_vtx_buf[0]) > 0) {
174  std::swap(e.v[0], e.v[1]);
175  }
176  }
177 
178  // add the edge.
179  // if there exists an edge in the list with opposite winding, they "annihilate"
180  // we should actually never see a collision with equal winding, so if we do, that's bad.
181  Edge<T,N> e_tmp;
182  bool annihilated = false;
183  for (auto e_other = edges.begin(); e_other != edges.end(); ++e_other) {
184  e_tmp = *e_other;
185  if (simplex_comparison(e, e_tmp, N-1) == -1) {
186  edges.erase(e_other);
187  annihilated = true;
188  break;
189  }
190  }
191  if (not annihilated) edges.push_back(e);
192  }
193  // delete the occluded face.
194  f = faces.erase(f);
195  } else {
196  ++f;
197  }
198  }
199 
200  // patch up the hole by constructing new faces,
201  // connecting each hole-adjacent edge to the new vertex.
202  index_t new_vert = verts.size() - 1;
203  for (auto e = edges.begin(); e != edges.end(); ++e) {
204  Face<T,N> f;
205  std::copy(e->v, e->v + N - 1, f.v);
206  f.v[N-1] = new_vert;
207 
208  // compute face normal
209  for (index_t i = 0; i < N - 1; i++) {
210  splex_vtx_buf[i] = verts[face[i]] - minkowski_pt;
211  }
212  f.n = orthogonal(splex_vtx_buf).unit();
213 
214  // correct the face winding, if necessary.
215  if (N > 3) {
216  // we know the origin is inside the polytope, so we will
217  // use that to test whether the normal is pointing the right way.
218  // so here we treat minkowski_pt as a vector pointing from inside
219  // the polytope to a point on the face simplex.
220  if (f.n.dot(minkowski_pt) < 0) {
221  f.n = -f.n;
222  std::swap(f.v[0], f.v[1]);
223  }
224  }
225 
226  faces.push_back(f);
227  }
228 
229  edges.clear();
230  looped = true;
231  }
232 
233  return true;
234 }
