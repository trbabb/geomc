310template <typename Ma, typename Mb>
311inline bool mtx_aliases_storage(const Ma& a, const Mb& b) {
312 static_assert(detail::IsMatrix<Ma>::val and detail::IsMatrix<Mb>::val, "Arguments must both be matrices");
313 // todo: currently assumes no /internal/ aliasing.
314 typename Ma::storagebuffer_t buf_a = a.get_storage_token_buffer();
315 typename Mb::storagebuffer_t buf_b = b.get_storage_token_buffer();
316
317 a.get_storage_tokens(buf_a.get()); // may be dynamically allocated
318 b.get_storage_tokens(buf_b.get()); // but usually static, and usually only one ptr
319 // do any of the tokens have overlapping range?
320 for (index_t i = 0; i < a.storage_token_count(); i++) {
321 for (index_t j = 0; j < b.storage_token_count(); j++) {
322 auto& tka = buf_a[i];
323 auto& tkb = buf_b[j];
324 if (tka.lo < tkb.lo or tka.hi > tkb.hi) return false;
325 }
326 }
327
328 return true;
329}

366// (a x b) * (b x c) -> (a x c)
367/* Implementation:
368 *
369 * This function's main activity is to guarantee dimension safety and
370 * protect against memory aliasing. The actual calculation is performed
371 * by _ImplMatrixMul, for which there are specializations over meaningful
372 * operand types. _ImplMatrixMul also provides a resultant type (usually
373 * a SimpleMatrix) for use when a temporary destination buffer is needed.
374 *
375 * Because Vecs do not have Matrix-like 2D dimension, an _ImplMtxAdaptor
376 * is used to make Vecs interoperable with matrices. The adaptor must be
377 * told whether a vector is expected to be a row vector or a column vector.
378 * Result vectors may be either a row or column vector depending on the
379 * order of the operands, so an _ImplVecMulOrient is used to select the
380 * appropriate orientation for result vectors.
381 */
382template <typename Md, typename Ma, typename Mb>
383typename std::enable_if<
384 detail::MatrixMultipliable<Ma,Mb>::val and
385 detail::_ImplMtxResult<Ma, Mb, Md>::agreement == detail::MTX_RESULT_MATCH,
386 Md&>::type
387mul(Md *into, const Ma &a, const Mb &b) {
388 // adaptor types for each argument:
389 typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
390 typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
391 typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecMulOrient<Ma,Mb,Md>::orient> D;
392 // multiplier implementation:
393 typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
394 typedef typename mult_t::return_t buffer_t;
395
396#ifdef GEOMC_MTX_CHECK_DIMS
397 // do the source matrix dimensions agree?
398 // any compiler worth its salt should eliminate this test for template instantiations with static-dimensioned operands
399 if ((A::COLDIM == DYNAMIC_DIM or B::ROWDIM == DYNAMIC_DIM) and A::cols(a) != B::rows(b)) {
400 throw DimensionMismatchException(A::rows(a), A::cols(a), B::rows(b), B::cols(b));
401 }
402 // does the destination matrix have correct dims?
403 // again, runtime checks are to be avoided.
404 if (((D::ROWDIM == DYNAMIC_DIM or A::ROWDIM == DYNAMIC_DIM) and A::rows(a) != D::rows(*into)) or
405 ((D::COLDIM == DYNAMIC_DIM or B::COLDIM == DYNAMIC_DIM) and B::cols(b) != D::cols(*into))) {
406 throw DimensionMismatchException(D::rows(*into), D::cols(*into), A::rows(a), B::cols(b));
407 }
408#endif
409
410#ifdef GEOMC_MTX_CHECK_ALIASING
411 // allocate a temp destination matrix, in case of storage aliasing
412 if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
413 buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
414 mult_t::mul(&tmp, a, b);
415 detail::_mtxcopy(into, tmp);
416 } else {
417 mult_t::mul(into, a, b);
418 }
419#else
420 mult_t::mul(into, a, b);
421#endif
422
423 return *into;
424}
425
426// vec <- mtx * mtx
427// Special case where matrices have dynamic size and destination is a vector.
428// The desired row/col orientation of the vector cannot be determined at compile
429// time, so special logic is needed to check both configurations.
430template <typename Md, typename Ma, typename Mb>
431typename std::enable_if<
432 detail::MatrixMultipliable<Ma,Mb>::val and
433 detail::_ImplMtxResult<Ma,Mb,Md>::agreement == detail::MTX_RESULT_UNKNOWN,
434 Md&>::type
435mul(Md *into, const Ma &a, const Mb &b) {
436 // adaptors types for arguments:
437 typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
438 typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
439 //multiplier implementation
440 typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
441 typedef typename mult_t::return_t buffer_t;
442
443#ifdef GEOMC_MTX_CHECK_DIMS
444 // trial orientation adaptors:
445 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
446 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
447
448 // if neither configuration yields dimension agreement:
449 if (not ((Dr::rows(*into) == A::rows(a) and Dr::cols(*into) == B::cols(b)) or
450 (Dc::rows(*into) == A::rows(a) and Dc::cols(*into) == B::cols(b))) ) {
451 throw DimensionMismatchException(Dr::rows(*into), Dr::cols(*into), A::rows(a), B::cols(b));
452 }
453#endif
454
455#ifdef GEOMC_MTX_CHECK_ALIASING
456 if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
457 buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
458 mult_t::mul(&tmp, a, b);
459 detail::_mtxcopy(into, tmp);
460 } else {
461 mult_t::mul(into, a, b);
462 }
463#else
464 mult_t::mul(into, a, b);
465#endif
466
467 return *into;
468}

520// allow txpose in-place
521template <typename T, index_t M, index_t N, MatrixLayout Lyt, StoragePolicy P>
522typename std::enable_if<M * N == 0 or M == N, void>::type
523transpose(SimpleMatrix<T,M,N,Lyt,P>* into, const SimpleMatrix<T,M,N,Lyt,P>& m) {
524 if (into == &m and (M * N > 0 or m.rows() == m.cols())) {
525
526 // txpose in place
527 for (index_t r = 0; r < m.rows(); r++) {
528 for (index_t c = 0; c < r; c++) {
529 std::swap((*into)(r,c), (*into)(c,r));
530 }
531 }
532 } else {
533#ifdef GEOMC_MTX_CHECK_DIMS
534 if (M * N == 0 and (m.rows() != into->cols() or m.cols() != into->rows())) {
535 throw DimensionMismatchException(
536 into->rows(), into->cols(),
537 m.cols(), m.rows());
538 }
539#endif
540#ifdef GEOMC_MTX_CHECK_ALIASING
541 if (mtx_aliases_storage(*into, m)) {
542 // matrices alias each other; mutating one will affect the other,
543 // in such a way as to prohibit a clean in-place transpose.
544 // (e.g., matrices are different objects but have shared storage,
545 // and/or they are nonsquare). copy aside the source
546 index_t rows = m.rows();
547 index_t cols = m.cols();
548 UniqueStorage<T,M*N> buf(rows * cols, m.data_begin());
549 detail::MxWrap<T,Lyt == ROW_MAJOR> mx = {buf.get(), rows, cols};
550 for (index_t r = 0; r < rows; ++r) {
551 for (index_t c = 0; c < cols; ++c) {
552 (*into)(c,r) = mx(r,c);
553 }
554 }
555 } else {
556#else
557 {
558#endif
559 // ordinary case: two unrelated matrices; txpose by copying
560 for (index_t r = 0; r < m.rows(); ++r) {
561 for (index_t c = 0; c < m.cols(); ++c) {
562 (*into)(c,r) = m(r,c);
563 }
564 }
565 }
566 }
567}

578template <typename Md, typename Mx>
579typename std::enable_if<
580 detail::IsMatrix<Md>::val and detail::IsMatrix<Mx>::val and
581 (Md::ROWDIM == Mx::COLDIM or
582 Md::ROWDIM * Mx::COLDIM == 0) and
583 (Md::COLDIM == Mx::ROWDIM or
584 Md::COLDIM * Mx::ROWDIM == 0),
585 void
586>::type
587transpose(Md *into, const Mx &m) {
588
589 typedef detail::_ImplMtxTxpose<Mx> txpose_t;
590 typedef typename txpose_t::return_t return_t;
591
592#ifdef GEOMC_MTX_CHECK_DIMS
593 // dimension mismatch?
594 if ((Md::ROWDIM * Mx::COLDIM == 0 and into->rows() != m.cols()) or
595 (Md::COLDIM * Mx::ROWDIM == 0 and into->cols() != m.rows())) {
596 throw DimensionMismatchException(into->rows(), into->cols(), m.cols(), m.rows());
597 }
598#endif
599
600#ifdef GEOMC_MTX_CHECK_ALIASING
601 // memory aliasing?
602 if (mtx_aliases_storage(*into, m)) {
603 return_t buf = detail::_ImplMtxInstance<return_t>::instance(m.cols(), m.rows());
604 txpose_t::transpose(&buf, m);
605 detail::_mtxcopy(into, buf);
606 }
607#endif
608 txpose_t::transpose(into, m);
609}

649template <typename Matrix1, typename Matrix2> bool inv(Matrix1 *into, const Matrix2 &src) {}
650#endif
651template <typename Md, typename Mx>
652bool inv(Md *into, const Mx &src,
653 M_ENABLE_IF(
654 (detail::MatrixDimensionMatch<Md,Mx>::isStaticMatch and
655 (Mx::ROWDIM == Mx::COLDIM or Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM))
656 )) {
657 typedef detail::_ImplMtxInv<Mx> inv_t;
658#ifdef GEOMC_MTX_CHECK_DIMS
659 if (Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM and src.rows() != src.cols()) {
660 throw NonsquareMatrixException(src.rows(), src.cols());
661 }
662 detail::MatrixDimensionMatch<Md,Mx>::check(*into, src);
663#endif
664 // matrix inv() implementations, unlike mul() or txpose(), shall assume
665 // that the destination and source may be aliased. this design decision
666 // was made because several inv() implementations are agnostic about
667 // aliasing (they perform internal copies), making a check unnecessary.
668 // therefore we delegate the aliasing check to those inv()s which care.
669 return inv_t::inv(into, src);
670}
