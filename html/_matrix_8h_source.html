310template <typename Ma, typename Mb>
311inline bool mtx_aliases_storage(const Ma& a, const Mb& b) {
312 static_assert(detail::IsMatrix<Ma>::val and detail::IsMatrix<Mb>::val, "Arguments must both be matrices");
313 // todo: currently assumes no /internal/ aliasing.
314 typename Ma::storagebuffer_t buf_a = a.get_storage_token_buffer();
315 typename Mb::storagebuffer_t buf_b = b.get_storage_token_buffer();
316
317 a.get_storage_tokens(buf_a.get()); // may be dynamically allocated
318 b.get_storage_tokens(buf_b.get()); // but usually static, and usually only one ptr
319 // do any of the tokens have overlapping range?
320 for (index_t i = 0; i < a.storage_token_count(); i++) {
321 for (index_t j = 0; j < b.storage_token_count(); j++) {
322 auto& tka = buf_a[i];
323 auto& tkb = buf_b[j];
324 if (tka.lo < tkb.lo or tka.hi > tkb.hi) return false;
325 }
326 }
327
328 return true;
329}

366// (a x b) * (b x c) -> (a x c)
367/* Implementation:
368 *
369 * This function's main activity is to guarantee dimension safety and
370 * protect against memory aliasing. The actual calculation is performed
371 * by _ImplMatrixMul, for which there are specializations over meaningful
372 * operand types. _ImplMatrixMul also provides a resultant type (usually
373 * a SimpleMatrix) for use when a temporary destination buffer is needed.
374 *
375 * Because Vecs do not have Matrix-like 2D dimension, an _ImplMtxAdaptor
376 * is used to make Vecs interoperable with matrices. The adaptor must be
377 * told whether a vector is expected to be a row vector or a column vector.
378 * Result vectors may be either a row or column vector depending on the
379 * order of the operands, so an _ImplVecMulOrient is used to select the
380 * appropriate orientation for result vectors.
381 */
382template <typename Md, typename Ma, typename Mb>
383typename std::enable_if<
384 detail::MatrixMultipliable<Ma,Mb>::val and
385 detail::_ImplMtxResult<Ma, Mb, Md>::agreement == detail::MTX_RESULT_MATCH,
386 Md&>::type
387mul(Md *into, const Ma &a, const Mb &b) {
388 // adaptor types for each argument:
389 typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
390 typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
391 typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecMulOrient<Ma,Mb,Md>::orient> D;
392 // multiplier implementation:
393 typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
394 typedef typename mult_t::return_t buffer_t;
395
396#ifdef GEOMC_MTX_CHECK_DIMS
397 // do the source matrix dimensions agree?
398 // any compiler worth its salt should eliminate this test for template instantiations with static-dimensioned operands
399 if ((A::COLDIM == DYNAMIC_DIM or B::ROWDIM == DYNAMIC_DIM) and A::cols(a) != B::rows(b)) {
400 throw DimensionMismatchException(A::rows(a), A::cols(a), B::rows(b), B::cols(b));
401 }
402 // does the destination matrix have correct dims?
403 // again, runtime checks are to be avoided.
404 if (((D::ROWDIM == DYNAMIC_DIM or A::ROWDIM == DYNAMIC_DIM) and A::rows(a) != D::rows(*into)) or
405 ((D::COLDIM == DYNAMIC_DIM or B::COLDIM == DYNAMIC_DIM) and B::cols(b) != D::cols(*into))) {
406 throw DimensionMismatchException(D::rows(*into), D::cols(*into), A::rows(a), B::cols(b));
407 }
408#endif
409
410#if GEOMC_MTX_CHECK_ALIASING
411 // allocate a temp destination matrix, in case of storage aliasing
412 if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
413 buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
414 mult_t::mul(&tmp, a, b);
415 detail::_mtxcopy(into, tmp);
416 } else {
417 mult_t::mul(into, a, b);
418 }
419#else
420 mult_t::mul(into, a, b);
421#endif
422
423 return *into;
424}

460// vec <- mtx * mtx
461// Special case where matrices have dynamic size and destination is a vector.
462// The desired row/col orientation of the vector cannot be determined at compile
463// time, so special logic is needed to check both configurations.
464template <typename Md, typename Ma, typename Mb>
465typename std::enable_if<
466 detail::MatrixMultipliable<Ma,Mb>::val and
467 detail::_ImplMtxResult<Ma,Mb,Md>::agreement == detail::MTX_RESULT_UNKNOWN,
468 Md&>::type
469mul(Md *into, const Ma &a, const Mb &b) {
470 // adaptors types for arguments:
471 typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
472 typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
473 //multiplier implementation
474 typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
475 typedef typename mult_t::return_t buffer_t;
476
477#ifdef GEOMC_MTX_CHECK_DIMS
478 // trial orientation adaptors:
479 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
480 typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
481
482 // if neither configuration yields dimension agreement:
483 if (not ((Dr::rows(*into) == A::rows(a) and Dr::cols(*into) == B::cols(b)) or
484 (Dc::rows(*into) == A::rows(a) and Dc::cols(*into) == B::cols(b))) ) {
485 throw DimensionMismatchException(Dr::rows(*into), Dr::cols(*into), A::rows(a), B::cols(b));
486 }
487#endif
488
489#if GEOMC_MTX_CHECK_ALIASING
490 if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
491 buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
492 mult_t::mul(&tmp, a, b);
493 detail::_mtxcopy(into, tmp);
494 } else {
495 mult_t::mul(into, a, b);
496 }
497#else
498 mult_t::mul(into, a, b);
499#endif
500
501 return *into;
502}

585// allow txpose in-place
586template <typename T, index_t M, index_t N, MatrixLayout Lyt, StoragePolicy P>
587typename std::enable_if<M * N == 0 or M == N, void>::type
588transpose(SimpleMatrix<T,M,N,Lyt,P>* into, const SimpleMatrix<T,M,N,Lyt,P>& m) {
589 if (into == &m and (M * N > 0 or m.rows() == m.cols())) {
590
591 // txpose in place
592 for (index_t r = 0; r < m.rows(); r++) {
593 for (index_t c = 0; c < r; c++) {
594 std::swap((*into)(r,c), (*into)(c,r));
595 }
596 }
597 } else {
598#ifdef GEOMC_MTX_CHECK_DIMS
599 if (M * N == 0 and (m.rows() != into->cols() or m.cols() != into->rows())) {
600 throw DimensionMismatchException(
601 into->rows(), into->cols(),
602 m.cols(), m.rows());
603 }
604#endif
605#if GEOMC_MTX_CHECK_ALIASING
606 if (mtx_aliases_storage(*into, m)) {
607 // matrices alias each other; mutating one will affect the other,
608 // in such a way as to prohibit a clean in-place transpose.
609 // (e.g., matrices are different objects but have shared storage,
610 // and/or they are nonsquare). copy aside the source
611 index_t rows = m.rows();
612 index_t cols = m.cols();
613 UniqueStorage<T,M*N> buf(rows * cols, m.data_begin());
614 detail::MxWrap<T,Lyt == ROW_MAJOR> mx = {buf.get(), rows, cols};
615 for (index_t r = 0; r < rows; ++r) {
616 for (index_t c = 0; c < cols; ++c) {
617 (*into)(c,r) = mx(r,c);
618 }
619 }
620 } else {
621#else
622 {
623#endif
624 // ordinary case: two unrelated matrices; txpose by copying
625 for (index_t r = 0; r < m.rows(); ++r) {
626 for (index_t c = 0; c < m.cols(); ++c) {
627 (*into)(c,r) = m(r,c);
628 }
629 }
630 }
631 }
632}

643template <typename Md, typename Mx>
644typename std::enable_if<
645 detail::IsMatrix<Md>::val and detail::IsMatrix<Mx>::val and
646 (Md::ROWDIM == Mx::COLDIM or
647 Md::ROWDIM * Mx::COLDIM == 0) and
648 (Md::COLDIM == Mx::ROWDIM or

653
654 typedef detail::_ImplMtxTxpose<Mx> txpose_t;

657#ifdef GEOMC_MTX_CHECK_DIMS
658 // dimension mismatch?
659 if ((Md::ROWDIM * Mx::COLDIM == 0 and into->rows() != m.cols()) or

667 if (mtx_aliases_storage(*into, m)) {
668 return_t buf = detail::_ImplMtxInstance<return_t>::instance(m.cols(), m.rows());

670 detail::_mtxcopy(into, buf);
671 }
672#endif
673 txpose_t::transpose(into, m);
674}

713#ifdef PARSING_DOXYGEN
714template <typename Matrix1, typename Matrix2> bool inv(Matrix1 *into, const Matrix2 &src) {}
715#endif
716template <typename Md, typename Mx>
717bool inv(Md *into, const Mx &src,
718 M_ENABLE_IF(
719 (detail::MatrixDimensionMatch<Md,Mx>::isStaticMatch and
720 (Mx::ROWDIM == Mx::COLDIM or Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM))
721 )) {
722 typedef detail::_ImplMtxInv<Mx> inv_t;
723#ifdef GEOMC_MTX_CHECK_DIMS
724 if (Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM and src.rows() != src.cols()) {
725 throw NonsquareMatrixException(src.rows(), src.cols());
726 }
727 detail::MatrixDimensionMatch<Md,Mx>::check(*into, src);
728#endif
729 // matrix inv() implementations, unlike mul() or txpose(), shall assume
730 // that the destination and source may be aliased. this design decision
731 // was made because several inv() implementations are agnostic about
732 // aliasing (they perform internal copies), making a check unnecessary.
733 // therefore we delegate the aliasing check to those inv()s which care.
734 return inv_t::inv(into, src);
735}
