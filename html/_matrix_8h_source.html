312 // do two matrices share any storage?
313 template <typename Ma, typename Mb>
314 typename boost::enable_if_c<
315  detail::IsMatrix<Ma>::val and detail::IsMatrix<Mb>::val,
316  bool>::type
317 mtx_aliases_storage(const Ma &a, const Mb &b) {
318  // todo: currently assumes no /internal/ aliasing.
319  typename Ma::storagebuffer_t buf_a = a.getStorageIDBuffer();
320  typename Mb::storagebuffer_t buf_b = b.getStorageIDBuffer();
321 
322  a.getStorageIDs(buf_a.get()); // may be dynamically allocated
323  b.getStorageIDs(buf_b.get()); // but usually static, and usually only one ptr
324  for (index_t i = 0; i < a.getStorageIDCount(); i++) {
325  for (index_t j = 0; j < b.getStorageIDCount(); j++) {
326  if (buf_a.get()[i] == buf_b.get()[j]) return true;
327  }
328  }
329 
330  return false;
331 }

376 // (a x b) * (b x c) -> (a x c)
377 /* Implementation:
378  *
379  * This function's main activity is to guarantee dimension safety and
380  * protect against memory aliasing. The actual calculation is performed
381  * by _ImplMatrixMul, for which there are specializations over meaningful
382  * operand types. _ImplMatrixMul also provides a resultant type (usually
383  * a SimpleMatrix) for use when a temporary destination buffer is needed.
384  *
385  * Because Vecs do not have Matrix-like 2D dimension, an _ImplMtxAdaptor
386  * is used to make Vecs interoperable with matrices. The adaptor must be
387  * told whether a vector is expected to be a row vector or a column vector.
388  * Result vectors may be either a row or column vector depending on the
389  * order of the operands, so an _ImplVecMulOrient is used to select the
390  * appropriate orientation for result vectors.
391  */
392 template <typename Md, typename Ma, typename Mb>
393 typename boost::enable_if_c<
394  detail::MatrixMultipliable<Ma,Mb>::val and
395  detail::_ImplMtxResult<Ma, Mb, Md>::agreement == detail::MTX_RESULT_MATCH,
396  Md&>::type
397 mul(Md *into, const Ma &a, const Mb &b) {
398  // adaptor types for each argument:
399  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
400  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
401  typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecMulOrient<Ma,Mb,Md>::orient> D;
402  // multiplier implementation:
403  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
404  typedef typename mult_t::return_t buffer_t;
405 
406 #ifdef GEOMC_MTX_CHECK_DIMS
407  // do the source matrix dimensions agree?
408  // any compiler worth its salt should eliminate this test for template instantiations with static-dimensioned operands
409  if ((A::COLDIM == DYNAMIC_DIM or B::ROWDIM == DYNAMIC_DIM) and A::cols(a) != B::rows(b)) {
410  throw DimensionMismatchException(A::rows(a), A::cols(a), B::rows(b), B::cols(b));
411  }
412  // does the destination matrix have correct dims?
413  // again, runtime checks are to be avoided.
414  if (((D::ROWDIM == DYNAMIC_DIM or A::ROWDIM == DYNAMIC_DIM) and A::rows(a) != D::rows(*into)) or
415  ((D::COLDIM == DYNAMIC_DIM or B::COLDIM == DYNAMIC_DIM) and B::cols(b) != D::cols(*into))) {
416  throw DimensionMismatchException(D::rows(*into), D::cols(*into), A::rows(a), B::cols(b));
417  }
418 #endif
419 
420 #ifdef GEOMC_MTX_CHECK_ALIASING
421  // allocate a temp destination matrix, in case of storage aliasing
422  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
423  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
424  mult_t::mul(&tmp, a, b);
425  detail::_mtxcopy(into, tmp);
426  } else {
427  mult_t::mul(into, a, b);
428  }
429 #else
430  mult_t::mul(into, a, b);
431 #endif
432 
433  return *into;
434 }

436 // vec <- mtx * mtx
437 // Special case where matrices have dynamic size and destination is a vector.
438 // The desired row/col orientation of the vector cannot be determined at compile
439 // time, so special logic is needed to check both configurations.
440 template <typename Md, typename Ma, typename Mb>
441 typename boost::enable_if_c<
442  detail::MatrixMultipliable<Ma,Mb>::val and
443  detail::_ImplMtxResult<Ma,Mb,Md>::agreement == detail::MTX_RESULT_UNKNOWN,
444  Md&>::type
445 mul(Md *into, const Ma &a, const Mb &b) {
446  // adaptors types for arguments:
447  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
448  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
449  //multiplier implementation
450  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
451  typedef typename mult_t::return_t buffer_t;
452 
453 #ifdef GEOMC_MTX_CHECK_DIMS
454  // trial orientation adaptors:
455  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
456  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
457 
458  // if neither configuration yields dimension agreement:
459  if (not ((Dr::rows(*into) == A::rows(a) and Dr::cols(*into) == B::cols(b)) or
460  (Dc::rows(*into) == A::rows(a) and Dc::cols(*into) == B::cols(b))) ) {
461  throw DimensionMismatchException(Dr::rows(*into), Dr::cols(*into), A::rows(a), B::cols(b));
462  }
463 #endif
464 
465 #ifdef GEOMC_MTX_CHECK_ALIASING
466  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
467  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
468  mult_t::mul(&tmp, a, b);
469  detail::_mtxcopy(into, tmp);
470  } else {
471  mult_t::mul(into, a, b);
472  }
473 #else
474  mult_t::mul(into, a, b);
475 #endif
476 
477  return *into;
478 }
