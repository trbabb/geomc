322 // do two matrices share any storage?
323 template <typename Ma, typename Mb>
324 typename boost::enable_if_c<
325  detail::IsMatrix<Ma>::val and detail::IsMatrix<Mb>::val,
326  bool>::type
327 mtx_aliases_storage(const Ma &a, const Mb &b) {
328  // todo: currently assumes no /internal/ aliasing.
329  typename Ma::storagebuffer_t buf_a = a.getStorageIDBuffer();
330  typename Mb::storagebuffer_t buf_b = b.getStorageIDBuffer();
331 
332  a.getStorageIDs(buf_a.get()); // may be dynamically allocated
333  b.getStorageIDs(buf_b.get()); // but usually static, and usually only one ptr
334  for (index_t i = 0; i < a.getStorageIDCount(); i++) {
335  for (index_t j = 0; j < b.getStorageIDCount(); j++) {
336  if (buf_a.get()[i] == buf_b.get()[j]) return true;
337  }
338  }
339 
340  return false;
341 }

386 // (a x b) * (b x c) -> (a x c)
387 /* Implementation:
388  *
389  * This function's main activity is to guarantee dimension safety and
390  * protect against memory aliasing. The actual calculation is performed
391  * by _ImplMatrixMul, for which there are specializations over meaningful
392  * operand types. _ImplMatrixMul also provides a resultant type (usually
393  * a SimpleMatrix) for use when a temporary destination buffer is needed.
394  *
395  * Because Vecs do not have Matrix-like 2D dimension, an _ImplMtxAdaptor
396  * is used to make Vecs interoperable with matrices. The adaptor must be
397  * told whether a vector is expected to be a row vector or a column vector.
398  * Result vectors may be either a row or column vector depending on the
399  * order of the operands, so an _ImplVecMulOrient is used to select the
400  * appropriate orientation for result vectors.
401  */
402 template <typename Md, typename Ma, typename Mb>
403 typename boost::enable_if_c<
404  detail::MatrixMultipliable<Ma,Mb>::val and
405  detail::_ImplMtxResult<Ma, Mb, Md>::agreement == detail::MTX_RESULT_MATCH,
406  Md&>::type
407 mul(Md *into, const Ma &a, const Mb &b) {
408  // adaptor types for each argument:
409  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
410  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
411  typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecMulOrient<Ma,Mb,Md>::orient> D;
412  // multiplier implementation:
413  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
414  typedef typename mult_t::return_t buffer_t;
415 
416 #ifdef GEOMC_MTX_CHECK_DIMS
417  // do the source matrix dimensions agree?
418  // any compiler worth its salt should eliminate this test for template instantiations with static-dimensioned operands
419  if ((A::COLDIM == DYNAMIC_DIM or B::ROWDIM == DYNAMIC_DIM) and A::cols(a) != B::rows(b)) {
420  throw DimensionMismatchException(A::rows(a), A::cols(a), B::rows(b), B::cols(b));
421  }
422  // does the destination matrix have correct dims?
423  // again, runtime checks are to be avoided.
424  if (((D::ROWDIM == DYNAMIC_DIM or A::ROWDIM == DYNAMIC_DIM) and A::rows(a) != D::rows(*into)) or
425  ((D::COLDIM == DYNAMIC_DIM or B::COLDIM == DYNAMIC_DIM) and B::cols(b) != D::cols(*into))) {
426  throw DimensionMismatchException(D::rows(*into), D::cols(*into), A::rows(a), B::cols(b));
427  }
428 #endif
429 
430 #ifdef GEOMC_MTX_CHECK_ALIASING
431  // allocate a temp destination matrix, in case of storage aliasing
432  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
433  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
434  mult_t::mul(&tmp, a, b);
435  detail::_mtxcopy(into, tmp);
436  } else {
437  mult_t::mul(into, a, b);
438  }
439 #else
440  mult_t::mul(into, a, b);
441 #endif
442 
443  return *into;
444 }

446 // vec <- mtx * mtx
447 // Special case where matrices have dynamic size and destination is a vector.
448 // The desired row/col orientation of the vector cannot be determined at compile
449 // time, so special logic is needed to check both configurations.
450 template <typename Md, typename Ma, typename Mb>
451 typename boost::enable_if_c<
452  detail::MatrixMultipliable<Ma,Mb>::val and
453  detail::_ImplMtxResult<Ma,Mb,Md>::agreement == detail::MTX_RESULT_UNKNOWN,
454  Md&>::type
455 mul(Md *into, const Ma &a, const Mb &b) {
456  // adaptors types for arguments:
457  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
458  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
459  //multiplier implementation
460  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
461  typedef typename mult_t::return_t buffer_t;
462 
463 #ifdef GEOMC_MTX_CHECK_DIMS
464  // trial orientation adaptors:
465  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
466  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
467 
468  // if neither configuration yields dimension agreement:
469  if (not ((Dr::rows(*into) == A::rows(a) and Dr::cols(*into) == B::cols(b)) or
470  (Dc::rows(*into) == A::rows(a) and Dc::cols(*into) == B::cols(b))) ) {
471  throw DimensionMismatchException(Dr::rows(*into), Dr::cols(*into), A::rows(a), B::cols(b));
472  }
473 #endif
474 
475 #ifdef GEOMC_MTX_CHECK_ALIASING
476  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
477  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
478  mult_t::mul(&tmp, a, b);
479  detail::_mtxcopy(into, tmp);
480  } else {
481  mult_t::mul(into, a, b);
482  }
483 #else
484  mult_t::mul(into, a, b);
485 #endif
486 
487  return *into;
488 }

547 template <typename Md, typename Mx>
548 typename boost::enable_if_c<
549  detail::IsMatrix<Md>::val and detail::IsMatrix<Mx>::val and
550  (Md::ROWDIM == Mx::COLDIM or
551  Md::ROWDIM * Mx::COLDIM == 0) and
552  (Md::COLDIM == Mx::ROWDIM or
553  Md::COLDIM * Mx::ROWDIM == 0),
554  void
555 >::type
556 transpose(Md *into, const Mx &m) {
557 
558  typedef detail::_ImplMtxTxpose<Mx> txpose_t;
559  typedef typename txpose_t::return_t return_t;
560 
561 #ifdef GEOMC_MTX_CHECK_DIMS
562  // dimension mismatch?
563  if ((Md::ROWDIM * Mx::COLDIM == 0 and into->rows() != m.cols()) or
564  (Md::COLDIM * Mx::ROWDIM == 0 and into->cols() != m.rows())) {
565  throw DimensionMismatchException(into->rows(), into->cols(), m.cols(), m.rows());

567 #endif

570  // memory aliasing?
571  if (mtx_aliases_storage(*into, m)) {
572  return_t buf = detail::_ImplMtxInstance<return_t>::instance(m.cols(), m.rows());
573  txpose_t::transpose(&buf, m);
574  detail::_mtxcopy(into, buf);
575  }
576 #endif
577  txpose_t::transpose(into, m);
578 }

620 template <typename Md, typename Mx>
621 bool inv(Md *into, const Mx &src,
622  M_ENABLE_IF_C(
623  (detail::MatrixDimensionMatch<Md,Mx>::isStaticMatch and
624  (Mx::ROWDIM == Mx::COLDIM or Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM))
625  )) {
626  typedef detail::_ImplMtxInv<Mx> inv_t;
627 #ifdef GEOMC_MTX_CHECK_DIMS
628  if (Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM and src.rows() != src.cols()) {
629  throw NonsquareMatrixException(src.rows(), src.cols());
630  }
631  detail::MatrixDimensionMatch<Md,Mx>::check(*into, src);
632 #endif
633  // matrix inv() implementations, unlike mul() or txpose(), shall assume
634  // that the destination and source may be aliased. this design decision
635  // was made because several inv() implementations are agnostic about
636  // aliasing (they perform internal copies), making a check unnecessary.
637  // therefore we delegate the aliasing check to those inv()s which care.
638  return inv_t::inv(into, src);
639 }
