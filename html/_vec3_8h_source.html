39 Vec<T,3> rainbow(double s){
40 s = std::fmod(s,1);
41 if (s < 0) s = 1+s;
42 return Vec<T,3>(
43 std::min( std::max( std::max( -6*s+2, 6*s-4 ), 0.0), 1.0),
44 std::max( std::min( std::min( 6*s, -6*s+4 ), 1.0), 0.0),
45 std::max( std::min( std::min( 6*s-2, -6*s+6 ), 1.0), 0.0));

197 Vec<T,3> cross(Vec<T,3> v) const {
198 T newX = diff_of_products(
199 detail::VecBase<T,3>::y, v.z,
200 detail::VecBase<T,3>::z, v.y);
201 T newY = diff_of_products(
202 detail::VecBase<T,3>::z, v.x,
203 detail::VecBase<T,3>::x, v.z);
204 T newZ = diff_of_products(
205 detail::VecBase<T,3>::x, v.y,
206 detail::VecBase<T,3>::y, v.x);
207
208 return Vec<T,3>(newX, newY, newZ);

256 Vec<T,3> rotate(T u, T v, T w, T radians) const {
257 const T& x = detail::VecBase<T,3>::x;
258 const T& y = detail::VecBase<T,3>::y;
259 const T& z = detail::VecBase<T,3>::z;
260 T u2 = u * u;
261 T v2 = v * v;
262 T w2 = w * w; // likely to become an FMA, which will keep precision ↙︎
263 T uxvywz = sum_of_products(u, x, v, y) + w * z;
264 T uvw2 = u2 + v2 + w2;
265 T rsin = std::sqrt(uvw2) * std::sin(radians);
266 T c = std::cos(radians);
267
268 // intermediate sums, using a safe high-precision formula
269 // (expected to be within 5% of "naïve" performance)
270 T p0 = sum_of_products(v, y, w, z);
271 T p1 = sum_of_products(u, x, w, z);
272 T p2 = sum_of_products(u, z, v, y);
273
274 T q0 = diff_of_products(v, z, w, y);
275 T q1 = diff_of_products(w, x, u, z);
276 T q2 = diff_of_products(u, y, v, x);
277
278 T s0 = diff_of_products(x, v2 + w2, u, p0);
279 T s1 = diff_of_products(y, u2 + w2, v, p1);
280 T s2 = diff_of_products(z, u2 + v2, w, p2);
281
282 T t0 = sum_of_products(u, uxvywz, s0, c);
283 T t1 = sum_of_products(v, uxvywz, s1, c);
284 T t2 = sum_of_products(w, uxvywz, s2, c);
285
286 return Vec<T,3>(
287 (t0 + rsin * q0) / uvw2, // <- also likely to become FMA
288 (t1 + rsin * q1) / uvw2,
289 (t2 + rsin * q2) / uvw2);
290 }

303 Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
304 const T& x = detail::VecBase<T,3>::x;
305 const T& y = detail::VecBase<T,3>::y;
306 const T& z = detail::VecBase<T,3>::z;
307
308 T cc = std::cos(radians);
309 T ss = std::sin(radians);
310 T oneMcos = 1 - cc;
311
312 T u2 = u*u;
313 T v2 = v*v;
314 T w2 = w*w;
315 T uxvywz = diff_of_products(u, x, v, y) - (w * z);
316
317 T p0 = sum_of_products(b, v, c, w);
318 T p1 = sum_of_products(a, u, c, w);
319 T p2 = sum_of_products(a, u, b, v);
320 T q0 = diff_of_products(a, v2 + w2, u, p0 - uxvywz);
321 T q1 = diff_of_products(b, u2 + w2, v, p1 - uxvywz);
322 T q2 = diff_of_products(c, u2 + v2, w, p2 - uxvywz);
323
324 return Vec<T,3>(
325 q0 * oneMcos + x * cc + (-c*v + b*w - w*y + v*z) * ss,
326 q1 * oneMcos + y * cc + ( c*u - a*w + w*x - u*z) * ss,
327 q2 * oneMcos + z * cc + (-b*u + a*v - v*x + u*y) * ss);
328 }

333 Vec<T,3> align(const Vec<T,3>& from, const Vec<T,3>& to) {
334 // modified from http://iquilezles.org/www/articles/noacos/noacos.htm
335 const Vec<T,3>& v = *this;
336 const Vec<T,3> q = from.cross(to);
337 const T c = from.dot(to);
338 const T k = 1 / (1 + c);
339 return Vec<T,3>(
340 (q.x * q.x * k + c) * v.x + (q.y * q.x * k - q.z) * v.y + (q.z * q.x * k + q.y) * v.z,
341 (q.x * q.y * k + q.z) * v.x + (q.y * q.y * k + c) * v.y + (q.z * q.y * k - q.x) * v.z,
342 (q.x * q.z * k - q.y) * v.x + (q.y * q.z * k + q.x) * v.y + (q.z * q.z * k + c) * v.z);
343 }
