228  Vec<T,3> rotate(T u, T v, T w, T radians) const {
229  const T& x = detail::VecBase<T,3>::x;
230  const T& y = detail::VecBase<T,3>::y;
231  const T& z = detail::VecBase<T,3>::z;
232  T u2 = u * u;
233  T v2 = v * v;
234  T w2 = w * w; // likely to become an FMA, which will keep precision ↙︎
235  T uxvywz = sum_of_products(u, x, v, y) + w * z;
236  T uvw2 = u2 + v2 + w2;
237  T rsin = std::sqrt(uvw2) * std::sin(radians);
238  T c = std::cos(radians);
239 
240  // intermediate sums, using a safe high-precision formula
241  // (expected to be within 5% of "naïve" performance)
242  T p0 = sum_of_products(v, y, w, z);
243  T p1 = sum_of_products(u, x, w, z);
244  T p2 = sum_of_products(u, z, v, y);
245 
246  T q0 = diff_of_products(v, z, w, y);
247  T q1 = diff_of_products(w, x, u, z);
248  T q2 = diff_of_products(u, y, v, x);
249 
250  T s0 = diff_of_products(x, v2 + w2, u, p0);
251  T s1 = diff_of_products(y, u2 + w2, v, p1);
252  T s2 = diff_of_products(z, u2 + v2, w, p2);
253 
254  T t0 = sum_of_products(u, uxvywz, s0, c);
255  T t1 = sum_of_products(v, uxvywz, s1, c);
256  T t2 = sum_of_products(w, uxvywz, s2, c);
257 
258  return Vec<T,3>(
259  (t0 + rsin * q0) / uvw2, // <- also likely to become FMA
260  (t1 + rsin * q1) / uvw2,
261  (t2 + rsin * q2) / uvw2);
262  }

275  Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
276  const T& x = detail::VecBase<T,3>::x;
277  const T& y = detail::VecBase<T,3>::y;
278  const T& z = detail::VecBase<T,3>::z;
279 
280  T cc = std::cos(radians);
281  T ss = std::sin(radians);
282  T oneMcos = 1 - cc;
283 
284  T u2 = u*u;
285  T v2 = v*v;
286  T w2 = w*w;
287  T uxvywz = diff_of_products(u, x, v, y) - (w * z);
288 
289  T p0 = sum_of_products(b, v, c, w);
290  T p1 = sum_of_products(a, u, c, w);
291  T p2 = sum_of_products(a, u, b, v);
292  T q0 = diff_of_products(a, v2 + w2, u, p0 - uxvywz);
293  T q1 = diff_of_products(b, u2 + w2, v, p1 - uxvywz);
294  T q2 = diff_of_products(c, u2 + v2, w, p2 - uxvywz);
295 
296  return Vec<T,3>(
297  q0 * oneMcos + x * cc + (-c*v + b*w - w*y + v*z) * ss,
298  q1 * oneMcos + y * cc + ( c*u - a*w + w*x - u*z) * ss,
299  q2 * oneMcos + z * cc + (-b*u + a*v - v*x + u*y) * ss);
300  }

305  Vec<T,3> align(const Vec<T,3>& from, const Vec<T,3>& to) {
306  // modified from http://iquilezles.org/www/articles/noacos/noacos.htm
307  const Vec<T,3>& v = *this;
308  const Vec<T,3> q = from.cross(to);
309  const T c = from.dot(to);
310  const T k = 1 / (1 + c);
311  return Vec<T,3>(
312  (q.x * q.x * k + c) * v.x + (q.y * q.x * k - q.z) * v.y + (q.z * q.x * k + q.y) * v.z,
313  (q.x * q.y * k + q.z) * v.x + (q.y * q.y * k + c) * v.y + (q.z * q.y * k - q.x) * v.z,
314  (q.x * q.z * k - q.y) * v.x + (q.y * q.z * k + q.x) * v.y + (q.z * q.z * k + c) * v.z);
315  }
