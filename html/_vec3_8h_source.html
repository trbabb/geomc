41 if (s < 0) s = 1+s;
42 return Vec<T,3>(
43 std::min( std::max( std::max( -6*s+2, 6*s-4 ), 0.0), 1.0),

206 Vec<T,3> cross(Vec<T,3> v) const {
207 T newX = diff_of_products(
208 this->y, v.z,
209 this->z, v.y);
210 T newY = diff_of_products(
211 this->z, v.x,
212 this->x, v.z);
213 T newZ = diff_of_products(
214 this->x, v.y,
215 this->y, v.x);
216
217 return Vec<T,3>(newX, newY, newZ);

246 static Vec<T,3> from_spherical(T r, T elev, T azi) {

264 Vec<T,3> rotate(T u, T v, T w, T radians) const {
265 const T& x = this->x;
266 const T& y = this->y;
267 const T& z = this->z;
268 T u2 = u * u;
269 T v2 = v * v;
270 T w2 = w * w; // likely to become an FMA, which will keep precision ↙︎
271 T uxvywz = sum_of_products(u, x, v, y) + w * z;
272 T uvw2 = u2 + v2 + w2;
273 T rsin = std::sqrt(uvw2) * std::sin(radians);
274 T c = std::cos(radians);
275
276 // intermediate sums, using a safe high-precision formula
277 // (expected to be within 5% of "naïve" performance)
278 T p0 = sum_of_products(v, y, w, z);
279 T p1 = sum_of_products(u, x, w, z);
280 T p2 = sum_of_products(u, z, v, y);
281
282 T q0 = diff_of_products(v, z, w, y);
283 T q1 = diff_of_products(w, x, u, z);
284 T q2 = diff_of_products(u, y, v, x);
285
286 T s0 = diff_of_products(x, v2 + w2, u, p0);
287 T s1 = diff_of_products(y, u2 + w2, v, p1);
288 T s2 = diff_of_products(z, u2 + v2, w, p2);
289
290 T t0 = sum_of_products(u, uxvywz, s0, c);
291 T t1 = sum_of_products(v, uxvywz, s1, c);
292 T t2 = sum_of_products(w, uxvywz, s2, c);
293
294 return Vec<T,3>(
295 (t0 + rsin * q0) / uvw2, // <- also likely to become FMA
296 (t1 + rsin * q1) / uvw2,
297 (t2 + rsin * q2) / uvw2);
298 }

311 Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
312 const T& x = this->x;
313 const T& y = this->y;
314 const T& z = this->z;
315
316 T cc = std::cos(radians);
317 T ss = std::sin(radians);
318 T oneMcos = 1 - cc;
319
320 T u2 = u * u;
321 T v2 = v * v;
322 T w2 = w * w;
323 T uxvywz = diff_of_products(u, x, v, y) - (w * z);
324
325 T p0 = sum_of_products(b, v, c, w);
326 T p1 = sum_of_products(a, u, c, w);
327 T p2 = sum_of_products(a, u, b, v);
328 T q0 = diff_of_products(a, v2 + w2, u, p0 - uxvywz);
329 T q1 = diff_of_products(b, u2 + w2, v, p1 - uxvywz);
330 T q2 = diff_of_products(c, u2 + v2, w, p2 - uxvywz);
331
332 return Vec<T,3>(
333 q0 * oneMcos + x * cc + (-c*v + b*w - w*y + v*z) * ss,
334 q1 * oneMcos + y * cc + ( c*u - a*w + w*x - u*z) * ss,
335 q2 * oneMcos + z * cc + (-b*u + a*v - v*x + u*y) * ss);
336 }

341 Vec<T,3> align(const Vec<T,3>& from, const Vec<T,3>& to) {
342 // modified from http://iquilezles.org/www/articles/noacos/noacos.htm
343 const Vec<T,3>& v = *this;
344 const Vec<T,3> q = from.cross(to);
345 const T c = from.dot(to);
346 const T k = 1 / (1 + c);
347 return Vec<T,3>(
348 (q.x * q.x * k + c) * v.x + (q.y * q.x * k - q.z) * v.y + (q.z * q.x * k + q.y) * v.z,
349 (q.x * q.y * k + q.z) * v.x + (q.y * q.y * k + c) * v.y + (q.z * q.y * k - q.x) * v.z,
350 (q.x * q.z * k - q.y) * v.x + (q.y * q.z * k + q.x) * v.y + (q.z * q.z * k + c) * v.z);
351 }
