109  // create a new, complete simplex from `s`. The final vertex is of `s` is "A".
110  SubSimplex& createFrom(const Simplex<T,N>& s) {
111  n = s.n - 1;
112  for (index_t i = 0; i < s.n; i++) pts[i] = i;
113 
114  // compute spanning and null spaces
115  if (n > 0) {
116  Vec<T,N> A = s[n];
117  Vec<T,N> *spanning_basis = spanning();
118  for (index_t i = 0; i < n; i++) {
119  spanning_basis[i] = s[i] - A;
120  }
121  // (short circuits if n == N)
122  geom::nullspace(spanning_basis, n, basis);
123  }
124  return *this;
125  }

128  // create a new sub-simplex of lower dimension by excluding the vertex with index `excluded`.
129  SubSimplex& createFrom(const SubSimplex& s, index_t excluded) {
130  n = s.n - 1;
131 
132  // copy the source's null basis, leaving a hole for the normal to come
133  std::copy(s.nullspace(), s.nullspace() + N - s.n, nullspace() + 1);
134 
135  // copy the source's spanning basis, skipping the excluded vtx.
136  Vec<T,N>* spanning_basis = spanning();
137  if (excluded < s.n) {
138  index_t j = 0;
139  for (index_t i = 0; i < s.n; i++) {
140  if (i == excluded) continue;
141  pts[j] = s.pts[i];
142  spanning_basis[j] = s.spanning(i);
143  ++j;
144  }
145  pts[n] = s.pts[s.n]; // a pt on the simplex; use A.
146  } else {
147  // refactor the spanning basis to exclude "A"
148  // (only used in contact-finding; ordinary EPA can, by construction,
149  // skip checking sub-simplexes not involving "A").
150  Vec<T,N> B = s.spanning(s.n - 1);
151  for (index_t i = 0; i < n; i++) {
152  spanning_basis[i] = s.spanning(i) - B;
153  pts[i] = s.pts[i];
154  }
155  pts[n] = s.pts[n]; // pts[n] is a point on the simplex, used later.
156  // we are exluding s.pts[s.n] (A), so use B instead.
157  }
158  // `basis` now looks like: [1 blank] [parent null space] [spanning basis]
159 
160  // compute the normal
161  if (n > 0) {
162  // the normal lies orthogonal to the null space of the parent
163  // and orthogonal to the space of the child. the `n-1` tail of `basis`
164  // now contains both.
165  Vec<T,N> normal = orthogonal(basis + 1);
166  // flip the normal if it points "inside"
167  Vec<T,N> v = (excluded == s.n) ? s.spanning(0) : s.spanning(excluded);
168  if (normal.dot(v) > 0) normal = -normal;
169  // extend the null basis.
170  basis[0] = normal;
171  } else {
172  // we are a point; parent simplex is a line; there is only one normal
173  basis[0] = -s.spanning(0);
174  }
175  return *this;
176  }

179  inline Vec<T,N> getNormal() const {
180  return basis[0];
181  }

185  // returns 0 if unequal, 1 if same orientation, -1 if opposite orientation
186  // pass a mutable simplex to s1 plz, kthx
187  int simplex_comparison(const index_t* s0, index_t* s1, index_t n) {
188  // reorder s1 to look like s0, keeping track of how many transpositions are made
189  // to determine parity.
190  int swaps = 0;
191  for (index_t i = 0; i < n; i++) {
192  // find s0[i] in s1 and swap positions.
193  for (index_t j = i; j < n; j++) {
194  if (s0[i] == s1[j]) {
195  std::swap(s1[i], s1[j]);
196  swaps++;
197  goto FOUNDIT;
198  }
199  }
200  return 0; // got to end of list without finding current element.
201  FOUNDIT: continue;
202  }
203  return (swaps & 1) ? -1 : 1;
204  }

365  // never thought I'd have use for template templates.
366  // this allows dj() to accept either a Simplex or a SubSimplex_.
367  // (construction of a sub-simplex from either is syntactically identical).
368  // we can't just make it a single, named template type, because then we can't extract T and N.
369  template <template <typename, index_t> typename SimplexType, typename T, index_t N>
370  T dj(const SimplexType<T,N>& s, index_t j) {
371  if (s.n <= 1) return (T)1;
372  detail::SubSimplex_<T,N> sj(s, j);
373 
374  T sum = (T)0;
375  Vec<T,N> v = sj[0] - s[j];
376  for (index_t i = 0; i < sj.n; ++i) {
377  sum += dj(sj, i) * v.dot(sj[i]);
378  }
379  return sum;
380  }

386  template <typename T, index_t N>
387  Vec<T,N> gjk_johnson_simplex_nearest_origin(detail::Simplex<T,N>* s, bool* degenerate) {
388  *degenerate = false;
389  T sum = (T)0;
390  detail::Simplex<T,N> out;
391  Vec<T,N> p;
392  for (index_t i = 0; i < s->n; ++i) {
393  T d_j = dj(*s, i);
394  if (d_j > 0) out[out.n++] = (*s)[i];
395  p += (*s)[i] * d_j;
396  sum += d_j;
397  }
398  if (sum == 0) *degenerate = true;
399  *s = out;
400  return p / sum;
401  }

404  template <typename T, index_t N>
405  bool gjk_direction_to_origin(detail::SubSimplex<T,N> &boundary, const Vec<T,N> &A, Vec<T,N> *d) {
406  // find the direction to the origin.
407  // assumes a simplex which has already passed containment tests.
408  // project the direction to the origin onto the null space of our simplex.
409  // (the final `else` would be sufficient by itself, but in some cases
410  // we can be clever and save some redundant calculations).
411  index_t null_dim = N - boundary.n;
412  if (null_dim == N) {
413  // single point case; "projection" is trivial.
414  *d = -A;
415  } else if (null_dim == 1) {
416  // the simplex spans a hyperplane; we already have the normal.
417  *d = boundary.getNormal();
418  // if we created the plane by removing a vertex, we already know the
419  // normal is facing toward the origin. otherwise, we need to check
420  // and flip it if necessary.
421  if (d->dot(A) > 0) *d *= -1;
422  // check for degenerate case: origin lies on face
423  // (we do not check this in the other cases because `d` will represent a vector
424  // from the origin to the simplex, and we check this against zero at the end).
425  if (-A.projectOn(*d).mag2() == 0) return true;
426  } else if (null_dim == 0) {
427  // simplex forms a complete basis; no vertexes were eliminated.
428  // in other words, no plane test failed. The origin is inside
429  // this simplex!
430  return true;
431  } else {
432  // explicitly project the direction to the origin onto the null space
433  // of the boundary simplex.
434  if (N != 3) {
435  *d = Vec<T,N>::zeros;
436 
437  // either project onto the null basis directly, or subtract a vector
438  // orthogonal to the null basis from -A, according to whichever
439  // projection is simpler.
440  bool swap_proj = false;
441  index_t proj_n = null_dim;
442  Vec<T,N> *proj_basis = boundary.nullspace();
443 
444  if ( null_dim > boundary.n ) {
445  proj_basis = boundary.spanning();
446  proj_n = boundary.n;
447  swap_proj = true;
448  }
449 
450  // gram-schmidt orthonormalization.
451  for (index_t b = 0; b < proj_n; b++) {
452  for (index_t i = 0; i < b; i++) {
453  proj_basis[b] -= proj_basis[b].projectOn(proj_basis[i]);
454  }
455  *d += -A.projectOn(proj_basis[b]);
456  }
457 
458  if (swap_proj) *d = -A - *d;
459  } else {
460  // cross product is faster/more numerically stable in 3D, but
461  // results are in principle identical.
462 
463  // Vec<T,N> b = A - boundary[0];
464  // Vec<T,N> c = b ^ -A;
465  // *d = c ^ b;
466 
467  // We use orthogonal() instead of ^ because this is a
468  // dimension-agnostic function. orthogonal inlines to ^ anyway.
469  Vec<T,N> vs[3] = {(T)0, -boundary.spanning(0), -A};
470  vs[0] = orthogonal(vs+1);
471  *d = orthogonal(vs);
472  }
473  }
474  // if the distance from the simplex to the origin is zero,
475  // then the simplex "contains" the origin (i.e. the origin lies on it).
476  // we must terminate now, because further simplexes will be degenerate.
477  return d->mag2() == 0;
478  }

481  /* todo:
482  * The only vertex that might be a boundary vertex is A. before we know A is the
483  * boundary vertex, we must check that the origin does not belong to an edge.
484  * There are at most only N edges in the entire simplex. However, there are potentially
485  * an absurd number of triangles, and we will check as many as two times that number of edges.
486  * This is obviously very redundant because the null space of an edge (and uniquely so
487  * for edges only) is not dependent on the parent, since there is only one degree
488  * of freedom. However, it is not clear to me how to avoid redundant checks in a way
489  * that is not O(num children of triangles) anyway. Since a plane test is so damn
490  * cheap, it may not be a big deal. (Also note that each edge needs a complete null basis,
491  * which would have to be constructed for each edge). At best, this might save
492  * us some storage.
493  *
494  * edit: are we simply guaranteed to terminate before we check more than N edges?
495  */
496  template <typename T, index_t N>
497  bool gjk_simplex_nearest_origin(detail::Simplex<T,N> *simplex, Vec<T,N> *d, bool fullcheck=false) {
498  typedef detail::SubSimplex<T,N> SS;
499  // queue for sub-simplexes to check.
500  // (sub-simplexes store their spanning and null bases).
501  // xxx: This is questionable. replace with deque / abstract to cache object.
502  // ...or can we factor to recursion, and use the stack?
503  detail::StaticQueue<SS, 2 * detail::gjk_queue_size_fac(N) + 1> q; // xxx doubled in size :S
504 
505  SS boundary;
506 
507  // enqueue this simplex, and its null basis
508  q.create_back()->createFrom(*simplex);
509 
510  // process the "frontier"
511  while (q.size > 0) {
512  SS &parent = q.peek_front();
513  // determine whether the origin projects onto this simplex, or instead to one of its edges.
514  // edge simplexes must be recursively checked for projection containment and so are enqueued.
515  bool parent_is_boundary = true;
516  index_t ct = (fullcheck and parent.n > 0) ? parent.n + 1 : parent.n;
517  for (index_t i = 0; i < ct; i++) {
518  SS &child = q.create_back()->createFrom(parent, i);
519  Vec<T,N> normal = child.getNormal();
520  Vec<T,N> B = simplex->pts[child.pts[0]]; // a pt on the simplex
521 
522  if (B.dot(normal) < 0) {
523  // the origin is "outside" this sub-simplex, and therefore it
524  // may be (or contain) a boundary simplex. We also know the parent
525  // cannot be the boundary simplex.
526  parent_is_boundary = false;
527  } else {
528  // no need to check the grandchildren.
529  q.destroy_back();
530  }
531  }
532  if (parent_is_boundary) {
533  boundary = parent;
534  break;
535  } else {
536  // this simplex is checked; pop it off the stack.
537  q.destroy_front();
538  }
539  }
540 
541  bool containment = gjk_direction_to_origin(boundary, simplex->pts[boundary.pts[boundary.n]], d);
542 
543  // put the new simplex back into the return variable.
544  // (boundary's indecies are strictly increasing, so this is ok).
545  for (index_t i = 0; i < boundary.n + 1; i++) {
546  (*simplex)[i] = (*simplex)[boundary.pts[i]];
547  }
548  simplex->n = boundary.n + 1;
549 
550  return containment;
551  }

574  template <typename T, index_t N>
575  bool gjk_intersect_old(const Convex<T,N> &shape_a,
576  const Convex<T,N> &shape_b,
577  Vec<T,N> *overlap_axis,
578  detail::Simplex<T,N> *s,
579  bool emit_debug=false) {
580  // choose an arbitrary initial direction.
581  Vec<T,N> initial;
582  if (overlap_axis and *overlap_axis != Vec<T,N>::zeros)
583  initial = *overlap_axis;
584  else
585  initial[0] = 1;
586 
587  Vec<T,N> a = shape_a.convexSupport( initial) -
588  shape_b.convexSupport(-initial);
589  Vec<T,N> d = -a;
590  s->n = 0;
591  s->insert(a);

594  const index_t cutoff = 10 * (1 << (N - 2));
595  index_t i = 0;

598  a = shape_a.convexSupport( d) -
599  shape_b.convexSupport(-d);
600 
601  s->insert(a);
602

604  if (overlap_axis) *overlap_axis = -a.projectOn(d);
605  return false;
606  }
607

609  // the "nearest" simplex containing a projection of the origin is a complete volume,
610  // or the origin lies on its projection
611  if (overlap_axis) *overlap_axis = d;
612  return true;
613  }
614 
615  if (++i > cutoff) { return true; }
616  }
617  }

838 template <typename T, index_t N>
839 void disjoint_separation_axis(const Convex<T,N>& shape_a,
840  const Convex<T,N>& shape_b,
841  Vec<T,N> *overlap_axis,
842  detail::Simplex<T,N>* splex,
843  double fractional_tolerance = 0.001,
844  index_t iteration_limit = -1,
845  bool emit_debug = false) {
846 
847  // final pt may have been a duplicate; if so, remove it.
848  Vec<T,N> A = splex->pts[splex->n - 1];
849  for (index_t i = 0; i < splex->n - 1; i++) {
850  if (splex->pts[i] == A) {
851  splex->remove(splex->n - 1);
852  break;
853  }
854  }

856  // expand the simplex to a volume
857  detail::explode_simplex(shape_a, shape_b, splex);
858 
859  // iteratively search outward from the closest (sub-)simplex.
860  Vec<T,N> last_proj;
861  Vec<T,N> new_proj;
862  bool looped = false;
863  for (index_t j = 0; j != iteration_limit; j++) {
864  Vec<T,N> d;
865 
866  // update `splex` and `d`
867  gjk_simplex_nearest_origin(splex, &d, true);
868

870  Vec<T,N> a = shape_a.convexSupport( d) -
871  shape_b.convexSupport(-d);
872

874  new_proj = ((*splex)[0]).projectOn(d);

877  for (index_t i = 0; i < splex->n; i++) if ((*splex)[i] == a) {

882  // our last estimate (or if the origin is on the current face),
883  // decide that we've converged and quit.
884  if ((looped and fractional_tolerance > 0 and
885  (last_proj - new_proj).mag() / new_proj.mag() < fractional_tolerance)

887  or new_proj == Vec<T,N>::zeros) {
888  break;
889  }
890

893  looped = true;

915  const Convex<T,N>& shape_b,
916  Vec<T,N>* overlap_axis,

934  // simplex must be "exploded" to a volume.

953  face.v[j] = (i + j + 1) % (N + 1);
954  if (j > 0) splex_vtx_buf[j-1] = verts[face.v[0]] - verts[face.v[j]];

959  if (n.dot(A - verts[face.v[0]]) > 0) {

961  // make an arbitrary transposition to flip it.
962  std::swap(face.v[0], face.v[1]);

975  typename std::vector< detail::Face<T,N> >::iterator best_face;
976

982  Vec<T,N> v = verts[f->v[0]]; // a vertex on the face
983  T d_this = std::abs(f->n.dot(v));
984  if (d_this < d) {
985  d = d_this;
986  best_face = f;
987  }
988  }
989

991  for (auto f = faces.begin(); f != faces.end(); ++f) EMIT(*f, f == best_face);
992 
993  // project the origin to the closest face.
994  *overlap_axis = verts[best_face->v[0]].projectOn(best_face->n);
995 
996  // if our estimated closest point is not sufficiently different from
997  // our last estimate (or if the origin is on the current face),
998  // decide that we've converged and quit.
999  if ((looped and fractional_tolerance > 0 and
1000  (last_proj - *overlap_axis).mag() / overlap_axis->mag() < fractional_tolerance)
1001  // origin is exactly on the face:
1002  or *overlap_axis == Vec<T,N>::zeros) {
1003  break;
1004  }
1005 
1006  last_proj = *overlap_axis;
1007

1009  // and find a new point on the minkowski difference.
1010  Vec<T,N> minkowski_pt = shape_a.convexSupport( best_face->n) -
1011  shape_b.convexSupport(-best_face->n);
1012 
1013  // finish if we add a duplicate point (possible for polytopes).
1014  // such a point would produce a degenerate face, i.e. we aren't going
1015  // to find any more new faces. this is a linear algorithm-- but I am
1016  // not sure whether a hashset construction/test is actually worth it.
1017  for (auto v : verts) if (v == minkowski_pt) goto FINISH;
1018 
1019  verts.push_back(minkowski_pt);

1021  // debug: emit newest hull pt
1022  EMIT(minkowski_pt);
1023

1025  for (auto f = faces.begin(); f != faces.end();) {
1026  // if this face is behind the new point...
1027  if (f->n.dot(minkowski_pt - verts[f->v[0]]) > 0) {
1028  // add all the edges of this face to the boundary edge list.
1029  // (all the non-internal edges are hole boundary, which must each
1030  // become part of a new face).
1031  for (index_t i = 0; i < N; i++) {
1032  // find a vertex to exclude from the face, making an edge.
1033  detail::Edge<T,N> e;
1034  for (index_t j = 0; j < N - 1; j++) {
1035  e.v[j] = f->v[(i + j + 1) % N];
1036  }
1037 
1038  // add the edge.
1039  // if there exists an edge in the list with opposite winding, they "annihilate"
1040  // but edges are only ever shared by exactly two faces, so we don't need to check the parity!
1041  detail::Edge<T,N> e_tmp;
1042  bool annihilated = false;
1043  for (auto e_other = edges.begin(); e_other != edges.end(); ++e_other) {
1044  e_tmp = *e_other;
1045  if (std::abs(detail::simplex_comparison(e.v, e_tmp.v, N-1)) == 1) {
1046  edges.erase(e_other);
1047  annihilated = true;
1048  break;
1049  }
1050  }
1051  if (not annihilated) edges.push_back(e);
1052  }
1053  // delete the occluded face.
1054  f = faces.erase(f);
1055  } else {
1056  ++f;
1057  }
1058  }
1059 
1060  // debug
1061  for (auto e : edges) {
1062  EMIT(e, true);
1063  }
1064 
1065  // patch up the hole by constructing new faces,
1066  // connecting each hole-adjacent edge to the new vertex.
1067  index_t new_vert = verts.size() - 1;
1068  for (auto e = edges.begin(); e != edges.end(); ++e) {
1069  detail::Face<T,N> f;
1070  std::copy(e->v, e->v + N - 1, f.v);
1071  f.v[N-1] = new_vert;
1072 
1073  // compute face normal
1074  for (index_t i = 0; i < N - 1; i++) {
1075  splex_vtx_buf[i] = verts[f.v[i]] - minkowski_pt;
1076  }
1077  f.n = orthogonal(splex_vtx_buf).unit();
1078 
1079  // correct the face winding, if necessary.
1080  if (N > 3) {
1081  // we know the origin is inside the polytope, so we will
1082  // use that to test whether the normal is pointing the right way.
1083  // so here we treat minkowski_pt as a vector pointing from inside
1084  // the polytope to a point on the face simplex.
1085  if (f.n.dot(minkowski_pt) < 0) {
1086  f.n = -f.n;
1087  std::swap(f.v[0], f.v[1]);
1088  }
1089  }
1090 
1091  faces.push_back(f);
1092  }
1093  edges.clear();
1094  looped = true;
1095  }
1096 
1097  FINISH:
1098 
1099  if (emit_debug) std::cout << "=\n";
