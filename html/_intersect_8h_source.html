48 bool gjk_intersect(
49  const Convex<T,N>& shape_a,
50  const Convex<T,N>& shape_b,
51  Vec<T,N>* overlap_axis=nullptr,
52  Simplex<T,N>* simplex=nullptr) {
53 
54  // choose an arbitrary initial direction.
55  Vec<T,N> initial;
56  if (overlap_axis and *overlap_axis != Vec<T,N>::zeros) {
57  initial = *overlap_axis;
58  } else {
59  initial[0] = 1;
60  }
61 
62  // initialize with a point on the minkowski volume.
63  Simplex<T,N> s;
64  Vec<T,N> a = shape_a.convexSupport( initial) -
65  shape_b.convexSupport(-initial);
66  Vec<T,N> d = -a;
67  s += a;
68 
69  // maximum number of iterations.
70  // (chosen by empirical measurement, plus a reasonable margin)
71  const index_t cutoff = 10 * (1 << (N - 2));
72  index_t i = 0;
73 
74  // expand the minkowski volume, looking to see
75  // if it can be made to contain the origin.
76  bool isxn = false;
77  while (true) {
78  // new point on the minkowski volume
79  a = shape_a.convexSupport( d) -
80  shape_b.convexSupport(-d);
81 
82  // add it to the simplex
83  s += a;
84 
85  if (a.dot(d) < 0) {
86  // if the new point is not further along the direction
87  // we just tried to look, we've reached the "edge" of the
88  // minkowski volume without enveloping the origin.
89  // return no intersection.
90  isxn = false;
91  break;
92  }
93 
94  // d is the next direction in which to look: from a point on
95  // the minkowski vol, outward toward the origin. because
96  // the projection is orthogonal onto `s`, this is the minimal
97  // distance from the (known) surface to the origin.
98  d = -s.project(Vec<T,N>::zeros, &s);
99 
100  if (s.n - 1 == N) {
101  // the origin was projected into a complete volume.
102  // return containment.
103  isxn = true;
104  break;
105  }
106 
107  if (++i > cutoff) {
108  isxn = true;
109  break;
110  }
111  }
112 
113  if (overlap_axis) *overlap_axis = d;
114  if (simplex) *simplex = s;
115  return isxn;
116 }
