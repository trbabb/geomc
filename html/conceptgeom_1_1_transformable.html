template<typename S, typename Xf>
concept geom::Transformable =
Transform<Xf, typename S::elem_t, S::N> and
requires (S s, Xf xf) {
{ xf * s } -> std::convertible_to<S>;
{ s / xf } -> std::convertible_to<S>;
}
