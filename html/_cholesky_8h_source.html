34 template <typename T, bool RowMajor>
35 bool cholesky(T* m, index_t n) {
36  bool ok = true;
37  detail::MxWrap<T,RowMajor> mx = {m, n, n};
38  for (index_t r = 0; r < n; ++r) {
39  for (index_t c = 0; c <= r; ++c) {
40  T s = mx.elem(r, c);
41  for (index_t k = 0; k < c; ++k) {
42  s -= mx.elem(r, k) * mx.elem(c, k);
43  }
44  if (r == c) {
45  ok = ok and (s > 0);
46  s = std::sqrt(s);
47  } else {
48  s = s / mx.elem(c, c);
49  mx.elem(c, r) = 0;
50  }
51  mx.elem(r, c) = s;
52  }
53  }
54  return ok;
55 }
