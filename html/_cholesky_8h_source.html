37template <typename T, bool RowMajor>
38bool cholesky(T* m, index_t n) {
39 bool ok = true;
40 detail::MxWrap<T,RowMajor> mx = {m, n, n};
41 for (index_t r = 0; r < n; ++r) {
42 for (index_t c = 0; c <= r; ++c) {
43 T s = mx.elem(r, c);
44 for (index_t k = 0; k < c; ++k) {
45 s -= mx.elem(r, k) * mx.elem(c, k);
46 }
47 if (r == c) {
48 ok = ok and (s > 0);
49 s = std::sqrt(s);
50 } else {
51 s = s / mx.elem(c, c);
52 // upper matrix gets set to 0:
53 mx.elem(c, r) = 0;
54 }
55 mx.elem(r, c) = s;
56 }
57 }
58 return ok;
59}
