149 template <typename Xf, typename T, index_t N>
150 concept Transform = requires (Xf xf, Vec<T,N> p) {
151 // transforms can be applied to points
152 { xf * p } -> std::convertible_to<Vec<T,N>>;
153 // inverse transforms can be applied to points
154 { p / xf } -> std::convertible_to<Vec<T,N>>;
155 // transforms can be composed
156 { xf * xf } -> std::convertible_to<Xf>;
157 // transforms have inverses
158 { xf.inverse() } -> std::convertible_to<Xf>;
159 // transforms can operate on directions
160 { xf.apply_direction(p) } -> std::convertible_to<Vec<T,N>>;
161 // inverse transforms can operate on directions
162 { xf.apply_inverse_direction(p) } -> std::convertible_to<Vec<T,N>>;
163 // transforms can in-place compose
164 { xf *= xf } -> std::convertible_to<Xf>;
165 // transforms can in-place apply inverse
166 { xf /= xf } -> std::convertible_to<Xf>;
167 };
168
171 template <typename S, typename Xf>
172 concept Transformable =
173 Transform<Xf, typename S::elem_t, S::N> and
174 requires (S s, Xf xf) {
175 // transformables can be transformed
176 { xf * s } -> std::convertible_to<S>;
177 // transformables can be inverse-transformed
178 { s / xf } -> std::convertible_to<S>;
179 };
