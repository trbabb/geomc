134 template <typename Xf, typename T, index_t N>
135 concept Transform = requires (Xf xf, Vec<T,N> p) {
136 // transforms can be applied to points
137 { xf * p } -> std::convertible_to<Vec<T,N>>;
138 // inverse transforms can be applied to points
139 { p / xf } -> std::convertible_to<Vec<T,N>>;
140 // transforms can be composed
141 { xf * xf } -> std::convertible_to<Xf>;
142 // transforms have inverses
143 { xf.inverse() } -> std::convertible_to<Xf>;
144 // transforms can operate on directions
145 { xf.apply_direction(p) } -> std::convertible_to<Vec<T,N>>;
146 // inverse transforms can operate on directions
147 { xf.apply_inverse_direction(p) } -> std::convertible_to<Vec<T,N>>;
148 // transforms can in-place compose
149 { xf *= xf } -> std::convertible_to<Xf>;
150 // transforms can in-place apply inverse
151 { xf /= xf } -> std::convertible_to<Xf>;
152 };
153
156 template <typename S, typename Xf>
157 concept Transformable =
158 Transform<Xf, typename S::elem_t, S::N> and
159 requires (S s, Xf xf) {
160 // transformables can be transformed
161 { xf * s } -> std::convertible_to<S>;
162 // transformables can be inverse-transformed
163 { s / xf } -> std::convertible_to<S>;
164 };
