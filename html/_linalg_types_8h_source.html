161  template <typename T, index_t N>
162  struct PointType {
163  typedef Vec<T,N> point_t;

165  static inline T* iterator(point_t &p) {
166  return p.begin();

168 
169  static inline const T* iterator(const point_t &p) {
170  return p.begin();

172 
173  static inline point_t from_ptr(T* p) {
174  return point_t(p);
175  }
176

178  return v.template resized<N>();
