70 bool contains(Vec<T,N> p) const {
71 if (not height.contains(p[N-1])) return false;
72 return base.contains(p.template resized<N-1>());
73 }

75 inline T sdf(Vec<T,N> p) const {
76 Vec<T,N-1> p_proj = (Vec<T,N-1>) p.template resized<N-1>();
77 T h = p[N-1];
78 // distance to the base shape within the base shape's plane
79 T sdf_base = base.sdf(p_proj);
80 // sdf of the slab that contains the height range of the shape
81 T sdf_h_slab = std::max(h - height.hi, height.lo - h);
82 if (height.contains((T)h)) {
83 // intersect base extrusion with the height slab
84 return std::max(sdf_base, sdf_h_slab);
85 } else {
86 // nearest point must be a cap
87 return (sdf_base < 0)
88 // point projects orthogonally to the cap
89 ? sdf_h_slab
90 // point is "away" from the edge of the cap, above/below it
91 : std::sqrt(sdf_base * sdf_base + sdf_h_slab * sdf_h_slab);
92 }
93 }
94
98 inline Vec<T,N> project(Vec<T,N> p) const {
99 const T h = p[N-1];
100 const T h_cap = std::abs(h - height.lo) < std::abs(h - height.hi)
101 ? height.lo
102 : height.hi;
103 // point in the plane of the base shape:
104 const Vec<T,N-1> p_base = p.template resized<N-1>();
105 // point projected to the boundary of the base shape:
106 const Vec<T,N-1> p_proj = base.project(p_base);
107 // nearest point on the wall:
108 Vec<T,N> wall_pt = Vec<T,N>(p_proj, height.clip(h));
109 // point on the nearer cap:
110 Vec<T,N> cap_pt = base.contains(p_base)
111 ? Vec<T,N>(p_base, h_cap)
112 : Vec<T,N>(p_proj, h_cap);
113 // pick the closer of the cap or the wall:
114 return p.dist2(wall_pt) < p.dist2(cap_pt) ? wall_pt : cap_pt;
115 }

117 Vec<T,N> convex_support(Vec<T,N> d) const {
118 typedef PointType<T,N-1> Pt;
119
120 Vec<T,N-1> d_ = d.template resized<N-1>();
121
122 if (d_.is_zero()) {
123 // support is normal to the endcap
124 // pick an arbitrary point on the endcap
125 d_[0] = 1;
126 }
127
128 Vec<T,N-1> p0 = base.convex_support(d_);
129
130 // top or bottom face?
131 return Vec<T,N>(p0, (d[N-1] > 0) ? height.hi : height.lo);
132 }
133
134 Rect<T,N> bounds() const {
135 return base.bounds() * height;
136 }
137
139 Rect<T,1> intersect(const Ray<T,N>& r) const {
140 Ray<T,N-1> r_base = r.template resized<N-1>();
141 Rect<T,1> interval;
142 if (r_base.direction.is_zero()) {
143 // ray is pointing directly along extrusion axis
144 if (base.contains(r_base.origin)) {
145 interval = Rect<T,1>::full;
146 }
147 } else {
148 interval = base.intersect(r_base);
149 }
150 T h = r.origin[N-1];
151 T dH = r.direction[N-1];
152 if (dH == 0) {
153 // ray is parallel to the base plane
154 return height.contains(h) ? interval : Rect<T,1>::empty;
155 } else {
156 // trace the height slab and intersect its interval
157 interval &= Rect<T,1>::spanning_corners(
158 (height.lo - h) / dH,
159 (height.hi - h) / dH);
160 return interval;
161 }
162 }
