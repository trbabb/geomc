86 // 4x4 inverse spec
87 template <typename T>
88 struct _ImplMtxInvRaw<T,4>{
89 
90  // this allows us to template _inv() for a particular T /only/ over the
91  // output iterator, saving some code bloat. also keeps our matrix
92  // elements syntactically accessible.
93  struct _buf {
94  T a00, a01, a02, a03,
95  a10, a11, a12, a13,
96  a20, a21, a22, a23,
97  a30, a31, a32, a33;
98  };
99 
100  template <typename Md, typename Mx>
101  static bool inv(Md* into, const Mx& m) {
102  _buf b;
103  std::copy(m.begin(), m.end(), &b.a00);
104  if (_inv(into->begin(), b)) {
105  return true;
106  }
107  return false;
108  }
109 
110  // prevent a little code bloat by templating over iterator type.
111  // this allows all the "contiguous memory" matrices to share the
112  // same inversion function. Would save even more bloat if we could
113  // assign back to <s> inside _inv(), but sadly c++ is dumb
114  // and doesn't allow the array initialization syntax for assignments,
115  // thus mandating two copies instead of one (one inside the function,
116  // and one outside).
117 
118  template <typename iter_out>
119  static bool _inv(iter_out into, const _buf& s) {
120  T s0 = DET2x2(s.a00, s.a01, s.a10, s.a11);
121  T s1 = DET2x2(s.a00, s.a02, s.a10, s.a12);
122  T s2 = DET2x2(s.a00, s.a03, s.a10, s.a13);
123  T s3 = DET2x2(s.a01, s.a02, s.a11, s.a12);
124  T s4 = DET2x2(s.a01, s.a03, s.a11, s.a13);
125  T s5 = DET2x2(s.a02, s.a03, s.a12, s.a13);
126 
127  T c0 = DET2x2(s.a20, s.a21, s.a30, s.a31);
128  T c1 = DET2x2(s.a20, s.a22, s.a30, s.a32);
129  T c2 = DET2x2(s.a20, s.a23, s.a30, s.a33);
130  T c3 = DET2x2(s.a21, s.a22, s.a31, s.a32);
131  T c4 = DET2x2(s.a21, s.a23, s.a31, s.a33);
132  T c5 = DET2x2(s.a22, s.a23, s.a32, s.a33);
133 
134  T det = s0*c5 - s1*c4 + s2*c3 + s3*c2 - s4*c1 + s5*c0;
135 
136  if (det == 0) return false;
137 
138  T inv[4][4] = {
139  {( s.a11*c5 - s.a12*c4 + s.a13*c3) / det,
140  (-s.a01*c5 + s.a02*c4 - s.a03*c3) / det,
141  ( s.a31*s5 - s.a32*s4 + s.a33*s3) / det,
142  (-s.a21*s5 + s.a22*s4 - s.a23*s3) / det},
143 
144  {(-s.a10*c5 + s.a12*c2 - s.a13*c1) / det,
145  ( s.a00*c5 - s.a02*c2 + s.a03*c1) / det,
146  (-s.a30*s5 + s.a32*s2 - s.a33*s1) / det,
147  ( s.a20*s5 - s.a22*s2 + s.a23*s1) / det},
148 
149  {( s.a10*c4 - s.a11*c2 + s.a13*c0) / det,
150  (-s.a00*c4 + s.a01*c2 - s.a03*c0) / det,
151  ( s.a30*s4 - s.a31*s2 + s.a33*s0) / det,
152  (-s.a20*s4 + s.a21*s2 - s.a23*s0) / det},
153 
154  {(-s.a10*c3 + s.a11*c1 - s.a12*c0) / det,
155  ( s.a00*c3 - s.a01*c1 + s.a02*c0) / det,
156  (-s.a30*s3 + s.a31*s1 - s.a32*s0) / det,
157  ( s.a20*s3 - s.a21*s1 + s.a22*s0) / det}
158  };
159 
160  T* inv_ptr = inv[0];
161  std::copy(inv_ptr, inv_ptr+16, into);
162  return true;
163  }
164 };

168 template <typename T>
169 struct _ImplMtxInvRaw<T,3> {
170 
171  struct _buf {
172  T a, b, c,
173  d, e, f,
174  g, h, i;
175  };

178  static bool inv(Md* into, const Mx& m) {
179  _buf b;
180  std::copy(m.begin(), m.end(), &b.a);
181  return _inv(into->begin(), b);
182  }
183 
184  template <typename iter_out>
185  static bool _inv(iter_out into, const _buf& s) {
186  // inverse 3x3 matrix by cramer's rule
187  T p0 = D(s.i, s.e, s.h, s.f);
188  T p1 = D(s.i, s.b, s.h, s.c);
189  T p2 = D(s.f, s.b, s.e, s.c);
190  T det = D(s.a, p0, s.d, p1) + s.g * p2;
191  if (det == 0) return false;
192 
193  T inv[3][3] = {
194  { D(s.i, s.e, s.h, s.f) / det,
195  D(s.h, s.c, s.i, s.b) / det,
196  D(s.f, s.b, s.e, s.c) / det },
197 
198  { D(s.g, s.f, s.i, s.d) / det,
199  D(s.i, s.a, s.g, s.c) / det,
200  D(s.d, s.c, s.f, s.a) / det },
201 
202  { D(s.h, s.d, s.g, s.e) / det,
203  D(s.g, s.b, s.h, s.a) / det,
204  D(s.e, s.a, s.d, s.b) / det }
205  };
206 
207  T* from = &(inv[0][0]);
208  std::copy(from, from + 9, into);
209  return true;
210  }
211 };

228  template <typename iter_out, typename iter_in>
229  static inline bool _inv(iter_out into, iter_in begin, iter_in end) {
230  T a[4];
231  std::copy(begin, end, a);
232  return _inv(into, a[0], a[1], a[2], a[3]);
233  }
234 
235  template <typename iter_out>
236  static bool _inv(iter_out into, T a, T b, T c, T d) {
237  T det = DET2x2(a, b, c, d);
238  if (det == 0) return false;
239  T inv[2][2] = {{ d / det, -b / det},
240  {-c / det, a / det}};
241  T* from = inv[0];
242  std::copy(from, from + 4, into);
243  return true;
244  }

278 template <typename T, index_t M, index_t N>
279 class _ImplMtxInv<geom::DiagMatrix<T,M,N> > {
280 public:
281 
282  typedef geom::DiagMatrix<T, (M != 0 ? M : N), (M != 0 ? M : N)> return_t;
283 
284  template <typename Md>
285  static bool inv(Md *into, const geom::DiagMatrix<T,M,N>& m) {
286  into->set_zero(); // for newly allocated matrices, redundant. oh well.
287  T *p = m.diagonal_begin();
288  for (index_t i = 0; p != m.diagonal_end(); ++i, ++p) {
289  T elem = *p;
290  if (elem == 0) return false;
291  into->set(i,i, 1/elem);
292  }
293  return true;
294  }
