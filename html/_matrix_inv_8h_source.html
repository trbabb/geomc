50template <typename T>
51bool inv2x2(T* out, const T m[4]) {
52 T a = m[0];
53 T b = m[1];
54 T c = m[2];
55 T d = m[3];
56 T det = det2x2(a, b, c, d);
57 if (det == 0) return false;
58 T inv[4] = { d / det, -b / det,
59 -c / det, a / det };
60 std::copy(inv, inv + 4, out);
61 return true;
62}

76template <typename T>
77bool inv3x3(T* out, const T m[9]) {
78 const detail::_m3x3<T>& s = *reinterpret_cast<const detail::_m3x3<T>*>(m);
79 // inverse 3x3 matrix by cramer's rule
80 T p0 = det2x2(s.i, s.h, s.f, s.e);
81 T p1 = det2x2(s.i, s.h, s.c, s.b);
82 T p2 = det2x2(s.f, s.e, s.c, s.b);
83
84 // det = det2x2(s.a, s.d, p1, p0) + s.g * p2;
85 T det = geom::multiply_add(s.g, p2, det2x2(s.a, s.d, p1, p0));
86 if (det == 0) return false;
87
88 T inv[9] = {
89 det2x2(s.i, s.h, s.f, s.e) / det,
90 det2x2(s.h, s.i, s.b, s.c) / det,
91 det2x2(s.f, s.e, s.c, s.b) / det,
92
93 det2x2(s.g, s.i, s.d, s.f) / det,
94 det2x2(s.i, s.g, s.c, s.a) / det,
95 det2x2(s.d, s.f, s.a, s.c) / det,
96
97 det2x2(s.h, s.g, s.e, s.d) / det,
98 det2x2(s.g, s.h, s.a, s.b) / det,
99 det2x2(s.e, s.d, s.b, s.a) / det
100 };
101
102 std::copy(inv, inv + 9, out);
103 return true;
104}

118template <typename T>
119bool inv4x4(T* out, const T m[16]) {
120 const detail::_m4x4<T>& s = *reinterpret_cast<const detail::_m4x4<T>*>(m);
121 T s0 = det2x2(s.a, s.b, s.e, s.f);
122 T s1 = det2x2(s.a, s.c, s.e, s.g);
123 T s2 = det2x2(s.a, s.d, s.e, s.h);
124 T s3 = det2x2(s.b, s.c, s.f, s.g);
125 T s4 = det2x2(s.b, s.d, s.f, s.h);
126 T s5 = det2x2(s.c, s.d, s.g, s.h);
127
128 T c0 = det2x2(s.i, s.j, s.m, s.n);
129 T c1 = det2x2(s.i, s.k, s.m, s.o);
130 T c2 = det2x2(s.i, s.l, s.m, s.p);
131 T c3 = det2x2(s.j, s.k, s.n, s.o);
132 T c4 = det2x2(s.j, s.l, s.n, s.p);
133 T c5 = det2x2(s.k, s.l, s.o, s.p);
134
135 // det = s0*c5 - s1*c4 + s2*c3 + s3*c2 - s4*c1 + s5*c0;
136 T det = diff_of_products(s0, c5, s1, c4) +
137 diff_of_products(s3, c2, s4, c1) +
138 sum_of_products(s2, c3, s5, c0);
139
140 if (det == 0) return false;
141
142 T inv[16] = {
143 // cofac is: a * b - c * d + e * f
144 detail::cofac(s.f, c5, s.g, c4, s.h, c3) / det,
145 -detail::cofac(s.b, c5, s.c, c4, s.d, c3) / det,
146 detail::cofac(s.n, s5, s.o, s4, s.p, s3) / det,
147 -detail::cofac(s.j, s5, s.k, s4, s.l, s3) / det,
148
149 -detail::cofac(s.e, c5, s.g, c2, s.h, c1) / det,
150 detail::cofac(s.a, c5, s.c, c2, s.d, c1) / det,
151 -detail::cofac(s.m, s5, s.o, s2, s.p, s1) / det,
152 detail::cofac(s.i, s5, s.k, s2, s.l, s1) / det,
153
154 detail::cofac(s.e, c4, s.f, c2, s.h, c0) / det,
155 -detail::cofac(s.a, c4, s.b, c2, s.d, c0) / det,
156 detail::cofac(s.m, s4, s.n, s2, s.p, s0) / det,
157 -detail::cofac(s.i, s4, s.j, s2, s.l, s0) / det,
158
159 -detail::cofac(s.e, c3, s.f, c1, s.g, c0) / det,
160 detail::cofac(s.a, c3, s.b, c1, s.c, c0) / det,
161 -detail::cofac(s.m, s3, s.n, s1, s.o, s0) / det,
162 detail::cofac(s.i, s3, s.j, s1, s.k, s0) / det
163 };
164
165 std::copy(inv, inv + 16, out);
166 return true;
167}

187template <typename T>
188bool invNxN(T* out, T* m, index_t n) {
189 SmallStorage<index_t, 24> p(n);
190 bool parity = false;
191 if (decomp_plu(m, n, n, p.begin(), &parity) > 0) return false;
192 std::fill(out, out + n * n, (T)0);
193 // pre-apply the permutation P to the identity matrix:
194 detail::MxWrap<T,true> o = {out, n, n};
195 for (index_t i = 0; i < n; ++i) {
196 o(i, p[i]) = 1;
197 }
198 backsolve_lu(m, n, n, out);
199 return true;
200}

220template <typename T, index_t N>
221bool invNxN(T* out, T* m) {
222 index_t p[N];
223 bool parity = false;
224 if (decomp_plu(m, N, N, p, &parity) > 0) return false;
225 std::fill(out, out + N * N, (T)0);
226 // pre-apply the permutation P to the identity matrix:
227 detail::MxWrap<T,true> o = {out, N, N};
228 for (index_t i = 0; i < N; ++i) {
229 o(i, p[i]) = 1;
230 }
231 backsolve_lu<T,true>(m, N, N, out);
232 return true;
233}

241// NxN static inverse spec
242template <typename T, index_t N>
243struct _ImplMtxInvRaw {
244 template <typename Md, typename Mx>
245 static inline bool inv(Md* out, const Mx& m) {
246 // copy to a buffer having the same layout as the destination matrix:
247 SimpleMatrix<T,N,N,Md::Layout> buf(out->rows(), out->rows());
248 detail::_mtxcopy(&buf, m);
249 // do the inverse, destroying the buffer
250 return geom::invNxN<T,N>(out->data_begin(), buf.data_begin());
251 }
252};

255// dynamic matrix inverse
256template <typename T>
257struct _ImplMtxInvRaw<T,0> {
258
259 template <typename Md, typename Mx>
260 static bool inv(Md* into, const Mx& m) {
261 const index_t n = m.rows();
262 switch(n) {
263 case 2:
264 return geom::detail::_ImplMtxInvRaw<T,2>::inv(into, m);
265 case 3:
266 return geom::detail::_ImplMtxInvRaw<T,3>::inv(into, m);
267 case 4:
268 return geom::detail::_ImplMtxInvRaw<T,4>::inv(into, m);
269 default:
270 // make a buffer with the same layout as the output
271 static constexpr MatrixLayout Lyt = Md::Layout;
272 SmallStorage<T,64> buf(n * n); // try not to alloc
273 WrapperMatrix<T,0,0,Lyt> bmx(buf.get(), n, n);
274 detail::_mtxcopy(&bmx, m);
275 return geom::invNxN(into->data_begin(), bmx.data_begin(), n);
276 }
277 }
278};

371// diagonal matrices are trivial to invert
372template <typename T, index_t M, index_t N>
373class _ImplMtxInv<geom::DiagMatrix<T,M,N> > {
374public:
375
376 typedef geom::DiagMatrix<T, (M != 0 ? M : N), (M != 0 ? M : N)> return_t;
377
378 template <typename Md>
379 static bool inv(Md *into, const geom::DiagMatrix<T,M,N>& m) {
380 into->set_zero(); // for newly allocated matrices, redundant. oh well.
381 T *p = m.diagonal_begin();
382 for (index_t i = 0; p != m.diagonal_end(); ++i, ++p) {
383 T elem = *p;
384 if (elem == 0) return false;
385 into->set(i,i, 1/elem);
386 }
387 return true;
388 }
389};
