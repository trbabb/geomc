544  void clear(const node_iterator& node) {
545  KDNodeRef n = node.node;
546  KDDataRef tail = n->objects_end;
547 
548  // ascend the tree, looking for nodes that refer to the deleted range
549  // and close the hole we're about to make
550  KDNodeRef fixup = n->parent;
551  while (fixup != nodes.end() and fixup->objects_begin == n->objects_begin) {
552  // ancestor's range began with deleted range
553  if (fixup->objects_begin == fixup->objects_end) {
554  fixup->objects_begin = fixup->objects_end = tail;
555  } else {
556  fixup->objects_begin = tail;
557  }
558  // ancestor's sibling ended with deleted range
559  KDNodeRef sibling = fixup; --sibling;
560  if (sibling != nodes.end() and sibling->objects_end == n->objects_begin) {
561  sibling->objects_end = tail;
562  if (sibling->objects_begin = n->objects_begin) {
563  sibling->objects_begin = tail;
564  }
565  }
566  }
567 
568  // the right edge of the previous sibling subtree ends with the deleted range.
569  // fixup those nodes to point to the other side of the hole we just made.
570  // (nothing to do if there is no subtree to the left of us)
571  if (n->parent != nodes.end() and n != n->parent->child_begin) {
572  fixup = n; --fixup; // previous sibling (we know from test above that it exists)
573  while (true) {
574  fixup->objects_end = tail;
575  if (fixup->child_begin == fixup->child_end) break;
576  fixup = fixup->child_end; --fixup;
577  }
578  }
579 
580  // remove child nodes
581  flatten(node);
582 
583  // remove objects
584  index_t removed_objs = n->nobjs;
585  objects.erase(n->objects_begin, n->objects_end);
586  n->objects_begin = n->objects_end = tail;
587  n->bounds = Rect<T,N>();
588  n->nobjs = 0;
589 
590  // update ancestors' bounds to reflect missing objects
591  while (n->parent != nodes.end()) {
592  n = n->parent;
593  n->bounds = Rect<T,N>();
594  for (KDNodeRef i = n->child_begin; i != n->child_end; ++i) {
595  n->bounds |= i->bounds;
596  }
597  n->nobjs -= removed_objs;
598  }
599  }

609  void flatten(const node_iterator& node) {
610  KDNodeRef n = node.node;
611  if (n->child_begin == n->child_end) return;
612  for (KDNodeRef c = n->child_begin; c != n->child_end; c++) {
613  flatten(c);
614  }
615 
616  // we deleted a node that is some other node's child_end.
617  // we need to suture up the hole that we made.
618  KDNodeRef fixup = n.child_begin;
619  fixup = (--fixup)->parent;
620  KDNodeRef tail = nodes.erase(n.child_begin, n.child_end);
621  if (fixup != nodes.end()) {
622  if (fixup->child_begin == fixup->child_end) {
623  fixup->child_begin = fixup->child_end = tail;
624  } else {
625  fixup->child_end = tail;
626  }
627  }
628 
629  // we now have no children.
630  n->child_begin = n->child_end = tail;
631  }

715  void splitNode(KDNodeRef split_child, index_t depth) {
716  Vec<T,N> mean;
717  Vec<T,N> dim;
718  Vec<T,N> var;
719 
720  index_t split_axis = 0;
721  T pivot;
722 
723  // choose a split axis
724  switch (params.axis) {
725  case KDAxisChoice::AXIS_LONGEST:
726  dim = split_child->bounds.getDimensions();
727  goto CHOOSE_BIGGEST_AXIS;
728  case KDAxisChoice::AXIS_CYCLICAL:
729  split_axis = depth % N;
730  break;
731  case KDAxisChoice::AXIS_HIGHEST_VARIANCE: {
732  for (KDDataRef i = split_child->objects_begin; i != split_child->objects_end; ++i) {
733  mean += helper_t::getPoint(*i);
734  }
735  for (KDDataRef i = split_child->objects_begin; i != split_child->objects_end; ++i) {
736  Vec<T,N> p = helper_t::getPoint(*i) - mean;
737  var += p * p;
738  }
739  var /= std::max<index_t>(1, split_child->nobjs - 1); // variance with Bessel's correction
740  dim = var;
741 
742  CHOOSE_BIGGEST_AXIS:
743  split_axis = 0;
744  T big = dim[0];
745  for (index_t i = 1; i < N; i++) {
746  if (dim[i] > big) {
747  big = dim[i];
748  split_axis = i;
749  }
750  }
751  break;
752  }
753  }
754 
755  // choose a pivot
756  switch (params.pivot) {
757  case KDPivotChoice::PIVOT_MEAN:
758  if (params.axis == KDAxisChoice::AXIS_HIGHEST_VARIANCE) {
759  // already calculated mean above
760  pivot = mean[split_axis];
761  } else {
762  // calculate the mean now
763  for (KDDataRef i = split_child->objects_begin; i != split_child->objects_end; ++i) {
764  pivot += helper_t::getPoint(*i)[split_axis];
765  }
766  pivot /= split_child->nobjs;
767  }
768  break;
769  case KDPivotChoice::PIVOT_MEDIAN:
770  // xxx: todo: this
771  break;
772  }
773 
774  // make a new child with no objects and no children
775  KDNodeRef new_sibling = insertChild(split_child->parent).node;
776 
777  // partition the objects; create bbox
778  split_child->bounds = Rect<T,N>();
779  index_t nobjs = split_child->nobjs;
780  KDDataRef middle = new_sibling->objects_end = split_child->objects_end;
781  for (KDDataRef i = split_child->objects_begin; i != middle; ) {
782  Vec<T,N> v = helper_t::getPoint(*i);
783  if (v[split_axis] > pivot) {
784  // move this object to the new (upper) sibling
785  std::swap(*i, *--middle);
786  split_child->nobjs--;
787  new_sibling->bounds |= helper_t::bounds(*i);
788  // `i` now contains an unchecked object; don't increment.
789  } else {
790  split_child->bounds |= helper_t::bounds(*i);
791  ++i;
792  }
793  }
794  split_child->objects_end = new_sibling->objects_begin = middle;
795  new_sibling->nobjs = nobjs - split_child->nobjs;
796  }

804  // root node needs to calc its box
805  if (depth == 0) {
806  recalculateBounds(node);

836  object_iterator insert_impl(
837  KDNodeRef node,
838  const Object& obj,
839  const Rect<T,N>& bnd,
840  const Vec<T,N>& ctr,
841  index_t depth) {
842 
843  node->bounds |= bnd;
844  node->nobjs++;
845 
846  if (node->child_begin != node->child_end) {
847  KDNodeRef best;
848  T best_metric = std::numeric_limits<T>::max();
849  for (KDNodeRef i = node->child_begin; i != node->child_end; ++i) {
850  T metric;
851  switch (params.insert) {
852  case KDInsertionChoice::INSERT_SHORTEST_DISTANCE:
853  metric = i->bounds.dist2(ctr);
854  break;
855  case KDInsertionChoice::INSERT_SMALLEST_VOLUME_INCREASE:
856  T orig_vol = i->bounds.volume();
857  T new_vol = (i->bounds | bnd).volume();
858  metric = new_vol - orig_vol;
859  break;
860  }
861  if (metric == 0) {
862  metric = -1 / i->bounds.getCenter().dist2(ctr);
863  }
864  if (metric < best_metric) {
865  best = i;
866  best_metric = metric;
867  }
868  }
869  object_iterator out = insert_impl(best, obj, bnd, ctr, depth + 1);
870  } else {
871  KDDataRef inserted = objects.insert(node->objects_end, obj);
872 
873  if (node->objects_begin == node->objects_end) {
874  // "separate" begin and end ptrs
875  node->objects_begin = inserted;
876  // uncommon/degenerate case: multiple layers of empty nodes:
877  for (KDNodeRef p = node->parent; p != nodes.end() and p->objects_begin == p->objects_end; p = p->parent) {
878  p->objects_begin = inserted;
879  }
880  }
881 
882  // subdivide if necessary
883  if (node->nobjs > params.leaf_arity) {
884  index_t parent_arity = 0;
885  KDNodeRef parent = node->parent;
886  if (parent != nodes.end()) {
887  // count parent's children
888  for (KDNodeRef i = parent->child_begin; i != parent->child_end; ++i) {
889  ++parent_arity;
890  }
891  // new sibling, or new subtree?
892  if (parent_arity >= params.node_arity) {
893  buildTree(node, depth);
894  } else {
895  splitNode(node, depth);
896  }
897  } else {
898  // root node cannot have new silbings
899  buildTree(node, depth);
900  }
901  }

909  void nearestInNode(KDNodeRef node, const Vec<T,N>& p, KDDataRef* nearest, T* best_d2, T* worst_best) const {
910  struct SearchInfo {
911  KDNodeRef n;
912  T best_case;
913  };
914  index_t child_ct = 0;
915  index_t recurse_ct = 0;
916 
917  // todo: With a portable alloca(), this could be arbitrarily large or small.
918  // this is the only line of code where leaf arity is actually restricted.
919  // but instead we are forced to use the maximum sane amount of memory in all cases.
920  // (ironically, the argument against alloca is that it "uses too much stack").
921  SearchInfo buf[MAX_KDTREE_ARITY];
922 
923  for (KDNodeRef i = node->child_begin; i != node->child_end and recurse_ct <= MAX_KDTREE_ARITY; ++i, ++child_ct) {
924  T box_d2 = i->bounds.dist2(p);
925  T box_worst_d2 = detail::worst_case_nearest2(p, i->bounds);
926  if (box_d2 < *best_d2 and box_d2 <= *worst_best) {
927  buf[recurse_ct++] = {i, box_d2};
928  }
929  *worst_best = std::min(*worst_best, box_worst_d2);
930  }
931  // recurse
932  while (--recurse_ct >= 0) {
933  // subsequently added nodes may have better worst cases than we knew about before,
934  // or subsequent recursions may have improved the best estimate, rendering our
935  // checks obsolete and allowing us to discard more nodes. check again plz.
936  if (buf[recurse_ct].best_case < *best_d2 and buf[recurse_ct].best_case <= *worst_best) {
937  nearestInNode(buf[recurse_ct].n, p, nearest, best_d2, worst_best);
938  }
939  }
940  // is leaf?
941  // seach objects for actual nearest obj.
942  if (child_ct == 0) {
943  for (KDDataRef i = node->objects_begin; i != node->objects_end; ++i) {
944  T d2 = helper_t::dist2(*i, p);
945  if (d2 < *best_d2) {
946  *best_d2 = d2;
947  *nearest = i;
948  }
949  }
950  }
951  }
