59 bool contains(Vec<T,N> p) const {
60 Vec<T,N> a = p1 - p0;
61 Vec<T,N> b = p - p0;
62 T b2 = b.mag2();
63 T a2 = a.mag2();
64 T d = a.dot(b);
65 T x2 = d * d / a2;
66 bool in_r = b2 - x2 <= radius * radius;
67 bool in_caps = x2 < a2 and d >= 0;
68 return in_r and in_caps;
69 }

72 T sdf(Vec<T,N> p) const {
73 // approach: put p into the r, a_hat basis,
74 // where the nearest cap is at the origin.
75 // axis of the cylinder:
76 Vec<T,N> a = p1 - p0;
77 // vec from "base" of cylinder to p
78 Vec<T,N> b = p - p0;
79 T b2 = b.mag2(); // squared length of b (hypoteneuse)
80 T a2 = a.mag2(); // squared length of axis (side 1)
81 T d = a.dot(b);
82 // fractional distance along axis from 0 to 1
83 T s = d / a2;
84 // square of distance from the base to the projected axis point
85 T x2 = d * s;
86 // signed distance of p to the surface of the cylinder
87 T r = std::sqrt(b2 - x2) - radius;
88 // signed fractional distance along the axis to the nearest cap
89 s = std::max(-s, (s - 1));
90 // sdf is negative (inside shape) iff (s, r) in the negative quadrant
91 T sign = (s < 0 and r < 0) ? -1 : 1;
92 // clamped coordinates, to project orthogonally to the wall/cap
93 s = std::max(s, (T)0);
94 r = std::max(r, (T)0);
95 // squared axis-parallel distance to cap
96 x2 = s * s * a2;
97 // signed distance to the surface point
98 return sign * std::sqrt(r * r + x2);
99 }

101 Vec<T,N> project(Vec<T,N> p) const {
102 // axis of the cylinder
103 Vec<T,N> a = p1 - p0;
104 // vec from "base" of cylinder to p
105 Vec<T,N> b = p - p0;
106 T b2 = b.mag2(); // squared length of b (hypoteneuse)
107 T a2 = a.mag2(); // squared length of axis (side 1)
108 T d = a.dot(b);
109 // fractional distance along axis from 0 to 1
110 T s = d / a2;
111 // square of distance from the base to the projected axis point
112 T x2 = d * s;
113 // distance of p to the axis
114 T r_dist = std::sqrt(b2 - x2);
115 // unit length radial vector
116 Vec<T,N> r = (p - p0 - s * a) / r_dist;
117 // adjust `r` and `s` to lie on the cylinder boundary
118 if (s < 0 or s > 1) {
119 // p projects onto the cap
120 r_dist = std::min(r_dist, radius);
121 s = s < 0 ? 0 : 1;
122 } else if (r_dist < radius and
123 std::sqrt(a2) * std::min(s, 1 - s) < radius - r_dist) {
124 // p is interior and the cap is closer than the wall
125 s = s < 0.5 ? 0 : 1;
126 } else {
127 // p projects onto the cylinder wall
128 r_dist = radius;
129 }
130 return p0 + (s * a) + (r * r_dist);
131 }

137 Rect<T,N> bounds() const {
138 // construct two bounding boxes, one for each disk cap.
139 // we do this by finding the extent of the cap along each axis.
140 Vec<T,N> a = p1 - p0;
141 // square of unit normal's components
142 // aka square of projection of normal onto axis of interest
143 Vec<T,N> n2 = (a * a) / a.mag2();
144 // solve for the remaining side of the triangle
145 // which happens to be the projection of a perpendicular vector
146 Vec<T,N> c0 = radius * std::sqrt((Vec<T,N>::ones - n2));
147 // construct the disk bounds at the end cap positions
148 Rect<T,N> b0(p0 - c0, p0 + c0);
149 Rect<T,N> b1(p1 - c0, p1 + c0);
150 // box union
151 return b0 | b1;
152 }

154 Vec<T,N> convex_support(Vec<T,N> d) const {
155 Vec<T,N> a = p1 - p0;
156 Vec<T,N> perp = (d - d.project_on(a)).unit() * radius;
157 return (d.dot(a) > 0 ? p1 : p0) + perp;
158 }

161 Rect<T,1> intersect(const Ray<T,N>& ray) const {
162 Vec<T,N> a = p1 - p0;
163 Vec<T,N> b = ray.origin - p0;
164 T a2 = a.mag2();
165 T b2 = b.mag2();
166 T d = a.dot(b);
167 T s = d / a2; // fractional projection of ray origin onto axis
168 T x2 = d * s;
169 T m = ray.direction.dot(a);
170
171 // construct quadratic coeffs
172 T k0 = ray.direction.mag2() - m * m / a2;
173 T k1 = 2 * (b.dot(ray.direction) - d * m / a2);
174 T k2 = b2 - x2 - radius * radius;
175
176 // setup for cylinder intersection
177 T roots[2];
178 Rect<T,1> interval;
179
180 // construct an interval holding the ray's overlap with the infinite cylinder
181 if (k0 != 0 and quadratic_solve(roots, k0, k1, k2)) {
182 interval = Rect<T,1>::spanning_corners(roots[0], roots[1]);
183 } else if (k2 < 0) {
184 // if there is no intersection, then it might be the case that
185 // the ray points exactly down the interior of the cylinder, which
186 // can only happen if the ray origin is inside too (k2 < 0). also
187 // due to precision this manifests sometimes when there is one root:
188 // when v and a are aligned then k0 is 0, and the equation can take the form
189 // `0x^2 + (small)x + c = 0`; this should be treated the same as 'no solution'.
190 // (Tangent rays with a single root will have k0 â‰  0 and should be intersected
191 // normally).
192 interval = Rect<T,1>::full;
193 } else {
194 // no intersection
195 return Rect<T,1>::empty;
196 }
197
198 // trace the endcaps
199 if (m != 0) {
200 // trace the two cap planes
201 T z0 = d; // a.dot(ray.origin - p0);
202 T z1 = a.dot(ray.origin - p1);
203 T s0 = -z0 / m;
204 T s1 = -z1 / m;
205 // intersect the slab with the cylinder
206 interval &= Rect<T,1>::spanning_corners(s0, s1);
207 } else if (s < 0 or s > 1) {
208 // ray is parallel to the slab and outside of it
209 return Rect<T,1>::empty;
210 }
211
212 return interval;
213 }
