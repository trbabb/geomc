61  bool contains(const Vec<T,N> &p) const {
62  Vec<T,N> v = p1 - p0;
63  Vec<T,N> b = p - p0;
64  T b2 = b.mag2();
65  T v2 = v.mag2();
66  T d = v.dot(b);
67  T x2 = d * d / v2;
68  bool in_r = b2 - x2 <= radius * radius;
69  bool in_caps = x2 < v2 && d >= 0;
70  return in_r && in_caps;
71  }

77  Rect<T,N> bounds() {
78  // construct two bounding boxes, one for each disk cap.
79  // we do this by finding the extent of the cap along each axis.
80  Vec<T,N> v = p1 - p0;
81  // square of unit normal's components
82  // aka square of projection of normal onto axis of interest
83  Vec<T,N> n2 = (v * v) / v.mag2();
84  // solve for the remaining side of the triangle
85  // which happens to be the projection of a perpendicular vector
86  Vec<T,N> c0 = radius * std::sqrt((Vec<T,N>::ones - n2));
87  // construct the disk bounds at the end cap positions
88  Rect<T,N> b0(p0 - c0, p0 + c0);
89  Rect<T,N> b1(p1 - c0, p1 + c0);
90  // box union
91  return b0 | b1;
92  }

109  Hit<T,N> trace(const Ray<T,N> &ray, HitSide side, bool endcaps=true) const {
110  Hit<T,N> h;
111 
112  /* The general principle is as follows:
113  * We examine the right triangle formed by the candidate hit point P,
114  * `p0` and the cylinder axis. We may solve one of the triangle
115  * legs to obtain the distance from P to the axis. With P constructed
116  * as `o + sv`, we may solve for s such that the square of said
117  * distance equals the square of the cylinder radius. The second
118  * triangle leg gives P's projection onto the cylinder axis, which
119  * we may use to determine if we are actually on the shaft between
120  * p0 and p1.
121  */
122 
123  Vec<T,N> axis = p1 - p0;
124  Vec<T,N> k = ray.origin - p0;
125  T l = k.dot(axis);
126  T m = ray.direction.dot(axis);
127  T v2 = ray.direction.mag2();
128  T a2 = axis.mag2();
129  T k2 = k.mag2();
130 
131  T a = v2 - m * m / a2;
132  T b = 2 * (k.dot(ray.direction) - l * m / a2);
133  T c = k2 - l * l / a2 - radius * radius;
134 
135  T s;
136  T roots[2];
137  if (quadratic_solve(roots, a, b, c)) {
138 
139  if (!detail::chooseRayHit(&s, roots, &side)) {
140  // miss
141  return h;
142  }
143  T hdotaxis = l + m * s;
144  Vec<T,N> n;
145  Vec<T,N> px;
146  if (hdotaxis >= 0 and hdotaxis < a2) {
147  // hit cylinder shaft
148  h.p = ray.atMultiple(s);
149  h.n = Ray<T,N>(p0, axis).directionFromAxisTo(h.p).unit();
150  h.s = s;
151  h.side = side;
152  h.hit = true;
153  return h;
154  } else if (hdotaxis < 0) {
155  // hit shaft before p0; test for cap 0 hit
156  n = -axis;
157  px = p0;
158  } else {
159  // hit shaft after p1; test for cap 1 hit
160  n = axis;
161  px = p1;
162  }
163 
164  if (endcaps) {
165  Plane<T,N> plane(n, px); // normalizes `n`
166  if (detail::_ImplTracePlane(&s, plane, ray, &side)) {
167  Vec<T,N> p = ray.atMultiple(s);
168  // TODO: can we be sure the cap doesn't "leak" due to precision?
169  // i.e. if the entry/exit points cross the endcap axis projection,
170  // assuming the facing matches, the ray *must* hit. beware of 'start
171  // inside' case, however
172  if (p.dist2(px) <= radius * radius) {
173  // within cap radius; hit
174  h.p = p;
175  h.n = plane.normal;
176  h.s = s;
177  h.side = side;
178  h.hit = true;
179  }
180  }
181  }
182  }
183 
184  return h;
185  }
