61  bool contains(const Vec<T,N> &p) const {
62  Vec<T,N> v = p1 - p0;
63  Vec<T,N> b = p - p0;
64  T b2 = b.mag2();
65  T v2 = v.mag2();
66  T d = v.dot(b);
67  T x2 = d * d / v2;
68  bool in_r = b2 - x2 <= radius * radius;
69  bool in_caps = x2 < v2 && d >= 0;
70  return in_r && in_caps;
71  }

77  Rect<T,N> bounds() {
78  // construct two bounding boxes, one for each disk cap.
79  // we do this by finding the extent of the cap along each axis.
80  Vec<T,N> v = p1 - p0;
81  // square of unit normal's components
82  // aka square of projection of normal onto axis of interest
83  Vec<T,N> n2 = (v * v) / v.mag2();
84  // solve for the remaining side of the triangle
85  // which happens to be the projection of a perpendicular vector
86  Vec<T,N> c0 = radius * std::sqrt((Vec<T,N>::ones - n2));
87  // construct the disk bounds at the end cap positions
88  Rect<T,N> b0(p0 - c0, p0 + c0);
89  Rect<T,N> b1(p1 - c0, p1 + c0);
90  // box union
91  return b0 | b1;
92  }

94  Vec<T,N> convexSupport(Vec<T,N> d) const {
95  Vec<T,N> v = p1 - p0;
96  Vec<T,N> perp = (d - d.projectOn(v)).unit() * radius;
97  return (d.dot(v) > 0 ? p1 : p0) + perp;
98  }

108  Hit<T,N> trace(const Ray<T,N> &ray, HitSide side) const {
109  Hit<T,N> h;
110  /* The general principle is as follows:
111  * We examine the right triangle formed by the candidate hit point P,
112  * `p0` and the cylinder axis. We may solve one of the triangle
113  * legs to obtain the distance from P to the axis. With P constructed
114  * as `o + sv`, we may solve for s such that the square of said
115  * distance equals the square of the cylinder radius. We then use CSG
116  * with two planes to remove the cylinder beyond the endcaps.
117  */
118 
119  Vec<T,N> axis = p1 - p0;
120  Vec<T,N> k = ray.origin - p0;
121  T l = k.dot(axis);
122  T m = ray.direction.dot(axis);
123  T v2 = ray.direction.mag2();
124  T a2 = axis.mag2();
125  T k2 = k.mag2();
126 
127  T a = v2 - m * m / a2;
128  T b = 2 * (k.dot(ray.direction) - l * m / a2);
129  T c = k2 - l * l / a2 - radius * radius;
130 
131  T roots[2];
132  if (quadratic_solve(roots, a, b, c)) {
133  // trace endcaps
134  T pl_0 = trace_plane(p0, -axis, r);
135  T pl_1 = trace_plane(p1, axis, r);
136  T n_sign = 1;
137 
138  // order roots low -> hi
139  if (pl_0 > pl_1) {
140  std::swap(pl_0, pl_1);
141  n_sign = -1;
142  }
143  if (roots[0] > roots[1]) {
144  std::swap(roots[0], roots[1]);
145  }
146 
147  // perform csg
148  bool hit_cylinder;
149  bool hit_near;
150  h.hit = csg_intersect(
151  roots[0], roots[1],
152  pl_0, pl_1,
153  &h.s,
154  &hit_cylinder,
155  &hit_near,
156  &h.side);
157 
158  // update the hit with P, N
159  // (csg_intersect took care of the others)
160  if (h.hit) {
161  h.p = r.atMultiple(h.s);
162  if (hit_cylinder) {
163  h.n = Ray<T,N>(p0, axis).directionFromAxisTo(h.p).unit()
164  } else {
165  h.n = (((hit_near ? -1 : 1) * n_sign) * axis).unit();
166  }
167  }
168  }
169 
170  return h;
171  }
