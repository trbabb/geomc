57 bool contains(Vec<T,N> p) const {
58 Vec<T,N> a = p1 - p0;
59 Vec<T,N> b = p - p0;
60 T b2 = b.mag2();
61 T a2 = a.mag2();
62 T d = a.dot(b);
63 T x2 = d * d / a2;
64 bool in_r = b2 - x2 <= radius * radius;
65 bool in_caps = x2 < a2 and d >= 0;
66 return in_r and in_caps;
67 }

74 T sdf(Vec<T,N> p) const {
75 // approach: put p into the r, a_hat basis,
76 // where the nearest cap is at the origin.
77 // axis of the cylinder:
78 Vec<T,N> a = p1 - p0;
79 // vec from "base" of cylinder to p
80 Vec<T,N> b = p - p0;
81 T b2 = b.mag2(); // squared length of b (hypoteneuse)
82 T a2 = a.mag2(); // squared length of axis (side 1)
83 T d = a.dot(b);
84 // fractional distance along axis from 0 to 1
85 T s = d / a2;
86 // square of distance from the base to the projected axis point
87 T x2 = d * s;
88 // signed distance of p to the surface of the cylinder
89 T r = std::sqrt(b2 - x2) - radius;
90 // signed fractional distance along the axis to the nearest cap
91 s = std::max(-s, (s - 1));
92 // sdf is negative (inside shape) iff (s, r) in the negative quadrant
93 T sign = (s < 0 and r < 0) ? -1 : 1;
94 // clamped coordinates, to project orthogonally to the wall/cap
95 s = std::max(s, (T)0);
96 r = std::max(r, (T)0);
97 // squared axis-parallel distance to cap
98 x2 = s * s * a2;
99 // signed distance to the surface point
100 return sign * std::sqrt(r * r + x2);
101 }

103 Vec<T,N> project(Vec<T,N> p) const {
104 // axis of the cylinder
105 Vec<T,N> a = p1 - p0;
106 // vec from "base" of cylinder to p
107 Vec<T,N> b = p - p0;
108 T b2 = b.mag2(); // squared length of b (hypoteneuse)
109 T a2 = a.mag2(); // squared length of axis (side 1)
110 T d = a.dot(b);
111 // fractional distance along axis from 0 to 1
112 T s = d / a2;
113 // square of distance from the base to the projected axis point
114 T x2 = d * s;
115 // distance of p to the axis
116 T r_dist = std::sqrt(b2 - x2);
117 // unit length radial vector
118 Vec<T,N> r = (p - p0 - s * a) / r_dist;
119 // adjust `r` and `s` to lie on the cylinder boundary
120 if (s < 0 or s > 1) {
121 // p projects onto the cap
122 r_dist = std::min(r_dist, radius);
123 s = s < 0 ? 0 : 1;
124 } else if (r_dist < radius and
125 std::sqrt(a2) * std::min(s, 1 - s) < radius - r_dist) {
126 // p is interior and the cap is closer than the wall
127 s = s < 0.5 ? 0 : 1;
128 } else {
129 // p projects onto the cylinder wall
130 r_dist = radius;
131 }
132 return p0 + (s * a) + (r * r_dist);
133 }

135 Vec<T,N> normal(Vec<T,N> p) const {
136 // axis of the cylinder
137 auto a = p1 - p0;
138 // vec from "base" of cylinder to p
139 auto b = p - p0;
140 T b2 = b.mag2(); // squared length of b (hypoteneuse)
141 T a2 = a.mag2(); // squared length of axis (side 1)
142 T d = a.dot(b);
143 // fractional distance along axis from 0 to 1
144 T s = d / a2;
145 // square of distance from the base to the projected axis point
146 T x2 = d * s;
147 // distance of p to the axis
148 T r_dist = std::sqrt(b2 - x2);
149 // unit length radial vector
150 auto r = (p - p0 - s * a) / r_dist;
151 // choose the outward direction
152 Vec<T,N> n;
153 if (s < 0 or s > 1) {
154 // p projects onto the cap
155 if (r_dist > radius) {
156 // p projects onto the rim of the cap
157 n = p - (r * radius + p0);
158 } else {
159 // p projects to the cap face; normal is axial
160 n = s < 0 ? -a : a;
161 }
162 } else if (r_dist < radius and std::sqrt(a2) * std::min(s, 1 - s) < radius - r_dist) {
163 // p is interior and the cap is closer than the wall
164 n = s < 0.5 ? -a : a;
165 } else {
166 // p projects onto the cylinder wall
167 n = r;
168 }
169 return n.unit();
170 }

176 Rect<T,N> bounds() const {
177 // construct two bounding boxes, one for each disk cap.
178 // we do this by finding the extent of the cap along each axis.
179 Vec<T,N> a = p1 - p0;
180 // square of unit normal's components
181 // aka square of projection of normal onto axis of interest
182 Vec<T,N> n2 = (a * a) / a.mag2();
183 // solve for the remaining side of the triangle
184 // which happens to be the projection of a perpendicular vector
185 Vec<T,N> c0 = radius * std::sqrt((Vec<T,N>::ones - n2));
186 // construct the disk bounds at the end cap positions
187 Rect<T,N> b0(p0 - c0, p0 + c0);
188 Rect<T,N> b1(p1 - c0, p1 + c0);
189 // box union
190 return b0 | b1;
191 }

193 Vec<T,N> convex_support(Vec<T,N> d) const {
194 Vec<T,N> a = p1 - p0;
195 Vec<T,N> perp = (d - d.project_on(a)).unit() * radius;
196 return (d.dot(a) > 0 ? p1 : p0) + perp;
197 }

200 Rect<T,1> intersect(const Ray<T,N>& ray) const {
201 Vec<T,N> a = p1 - p0;
202 Vec<T,N> b = ray.origin - p0;
203 T a2 = a.mag2();
204 T b2 = b.mag2();
205 T d = a.dot(b);
206 T s = d / a2; // fractional projection of ray origin onto axis
207 T x2 = d * s;
208 T m = ray.direction.dot(a);
209
210 // construct quadratic coeffs
211 T k0 = ray.direction.mag2() - m * m / a2;
212 T k1 = 2 * (b.dot(ray.direction) - d * m / a2);
213 T k2 = b2 - x2 - radius * radius;
214
215 // setup for cylinder intersection
216 T roots[2];
217 Rect<T,1> interval;
218
219 // construct an interval holding the ray's overlap with the infinite cylinder
220 if (k0 != 0 and quadratic_solve(roots, k0, k1, k2)) {
221 interval = Rect<T,1>::spanning_corners(roots[0], roots[1]);
222 } else if (k2 < 0) {
223 // if there is no intersection, then it might be the case that
224 // the ray points exactly down the interior of the cylinder, which
225 // can only happen if the ray origin is inside too (k2 < 0). also
226 // due to precision this manifests sometimes when there is one root:
227 // when v and a are aligned then k0 is 0, and the equation can take the form
228 // `0x^2 + (small)x + c = 0`; this should be treated the same as 'no solution'.
229 // (Tangent rays with a single root will have k0 â‰  0 and should be intersected
230 // normally).
231 interval = Rect<T,1>::full;
232 } else {
233 // no intersection
234 return Rect<T,1>::empty;
235 }
236
237 // trace the endcaps
238 if (m != 0) {
239 // trace the two cap planes
240 T z0 = d; // a.dot(ray.origin - p0);
241 T z1 = a.dot(ray.origin - p1);
242 T s0 = -z0 / m;
243 T s1 = -z1 / m;
244 // intersect the slab with the cylinder
245 interval &= Rect<T,1>::spanning_corners(s0, s1);
246 } else if (s < 0 or s > 1) {
247 // ray is parallel to the slab and outside of it
248 return Rect<T,1>::empty;
249 }
250
251 return interval;
252 }
