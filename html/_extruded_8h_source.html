73 T sdf(Vec<T,N> p) const requires SdfObject<Shape> {
74 Vec<T,N-1> p_proj = (Vec<T,N-1>) p.template resized<N-1>();
75 T h = p[N-1];
76 // distance to the base shape within the base shape's plane
77 T sdf_base = base.sdf(p_proj);
78 // sdf of the slab that contains the height range of the shape
79 T sdf_h_slab = std::max(h - height.hi, height.lo - h);
80 if (height.contains(h)) {
81 // intersect base extrusion with the height slab
82 // because we are inside the shape, the sdf will be negative,
83 // so max() will give us a smaller absolute value— the closer surface.
84 return std::max(sdf_base, sdf_h_slab);
85 } else {
86 // nearest point must be a cap
87 return (sdf_base < 0)
88 // point projects orthogonally to the cap
89 ? sdf_h_slab
90 // point is "away" from the edge of the cap, above/below it
91 : std::sqrt(sdf_base * sdf_base + sdf_h_slab * sdf_h_slab);
92 }
93 }

95 Vec<T,N> normal(Vec<T,N> p) const requires ProjectableObject<Shape> {
96 Vec<T,N-1> p_base = (Vec<T,N-1>) p.template resized<N-1>();
97 T h = p[N-1];
98 bool inside_base = base.contains(p_base);
99 if (height.contains(h)) {
100 // p is somewhere between the caps
101 if (inside_base) {
102 // we need to know if the wall is closest, or a cap is closest
103 T cap_h = height.project(h);
104 T cap_dist = std::abs(h - cap_h);
105 T wall_dist = std::abs(base.sdf(p_base));
106 if (cap_dist < wall_dist) {
107 // the cap is nearest. use its normal
108 return {Vec<T,N-1>{}, (cap_h >= height.hi) ? 1 : -1};
109 }
110 }
111 // the wall is nearest, use its normal
112 return Vec<T,N>(base.normal(p_base), 0);
113 } else {
114 // p beyond the top or bottom cap
115 if (inside_base) {
116 // p directly above or below a cap
117 return {Vec<T,N-1>{}, (h >= height.hi) ? 1 : -1};
118 } else {
119 // p is outside the base shape
120 Vec<T,N> cap_pt {
121 base.project(p_base),
122 height.clip(h)
123 };
124 return (p - cap_pt).unit();
125 }
126 }
127 }

132 Vec<T,N> project(Vec<T,N> p) const requires ProjectableObject<Shape> {
133 const T h = p[N-1];
134 const T h_cap = std::abs(h - height.lo) < std::abs(h - height.hi)
135 ? height.lo
136 : height.hi;
137 // point in the plane of the base shape:
138 const Vec<T,N-1> p_base = p.template resized<N-1>();
139 // point projected to the boundary of the base shape:
140 const Vec<T,N-1> p_proj = base.project(p_base);
141 // nearest point on the wall:
142 Vec<T,N> wall_pt = Vec<T,N>(p_proj, height.clip(h));
143 // point on the nearer cap:
144 Vec<T,N> cap_pt = base.contains(p_base)
145 ? Vec<T,N>(p_base, h_cap)
146 : Vec<T,N>(p_proj, h_cap);
147 // pick the closer of the cap or the wall:
148 return p.dist2(wall_pt) < p.dist2(cap_pt) ? wall_pt : cap_pt;
149 }

151 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
152 Vec<T,N-1> d_ = d.template resized<N-1>();
153
154 if (d_.is_zero()) {
155 // support is normal to the endcap
156 // pick an arbitrary point on the endcap
157 d_[0] = 1;
158 }
159
160 Vec<T,N-1> p0 = base.convex_support(d_);
161
162 // top or bottom face?
163 return Vec<T,N>(p0, (d[N-1] > 0) ? height.hi : height.lo);
164 }

176 T measure_boundary() const requires BoundaryMeasurableObject<Shape> {
177 T cap_area = 0;
178 // it is possible to extrude a boundary shape to make a "tube";
179 // in that case it will not have caps. only count the cap area if
180 // the shape has an interior.
181 if constexpr (InteriorMeasurableObject<Shape>) {
182 cap_area = 2 * base.measure_interior();
183 }
184 return (
185 // walls
186 base.measure_boundary() * height.dimensions() +
187 // caps
188 cap_area
189 );
190 }

193 Rect<T,1> intersect(const Ray<T,N>& r) const requires RayIntersectableObject<Shape> {
194 Ray<T,N-1> r_base = r.template resized<N-1>();
195 Rect<T,1> interval;
196 if (r_base.direction.is_zero()) {
197 // ray is pointing directly along extrusion axis
198 if (base.contains(r_base.origin)) {
199 interval = Rect<T,1>::full;
200 }
201 } else {
202 interval = base.intersect(r_base);
203 }
204 T h = r.origin[N-1];
205 T dH = r.direction[N-1];
206 if (dH == 0) {
207 // ray is parallel to the base plane
208 return height.contains(h) ? interval : Rect<T,1>::empty;
209 } else {
210 // trace the height slab and intersect its interval
211 interval &= Rect<T,1>::from_corners(
212 (height.lo - h) / dH,
213 (height.hi - h) / dH);
214 return interval;
215 }
216 }
