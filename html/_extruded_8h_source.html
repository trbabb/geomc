68 T sdf(Vec<T,N> p) const requires SdfObject<Shape> {
69 Vec<T,N-1> p_proj = (Vec<T,N-1>) p.template resized<N-1>();
70 T h = p[N-1];
71 // distance to the base shape within the base shape's plane
72 T sdf_base = base.sdf(p_proj);
73 // sdf of the slab that contains the height range of the shape
74 T sdf_h_slab = std::max(h - height.hi, height.lo - h);
75 if (height.contains(h)) {
76 // intersect base extrusion with the height slab
77 // because we are inside the shape, the sdf will be negative,
78 // so max() will give us a smaller absolute value— the closer surface.
79 return std::max(sdf_base, sdf_h_slab);
80 } else {
81 // nearest point must be a cap
82 return (sdf_base < 0)
83 // point projects orthogonally to the cap
84 ? sdf_h_slab
85 // point is "away" from the edge of the cap, above/below it
86 : std::sqrt(sdf_base * sdf_base + sdf_h_slab * sdf_h_slab);
87 }
88 }

90 Vec<T,N> normal(Vec<T,N> p) const requires ProjectableObject<Shape> {
91 Vec<T,N-1> p_base = (Vec<T,N-1>) p.template resized<N-1>();
92 T h = p[N-1];
93 bool inside_base = base.contains(p_base);
94 if (height.contains(h)) {
95 // p is somewhere between the caps
96 if (inside_base) {
97 // we need to know if the wall is closest, or a cap is closest
98 T cap_h = height.project(h);
99 T cap_dist = std::abs(h - cap_h);
100 T wall_dist = std::abs(base.sdf(p_base));
101 if (cap_dist < wall_dist) {
102 // the cap is nearest. use its normal
103 return {Vec<T,N-1>{}, (cap_h >= height.hi) ? 1 : -1};
104 }
105 }
106 // the wall is nearest, use its normal
107 return Vec<T,N>(base.normal(p_base), 0);
108 } else {
109 // p beyond the top or bottom cap
110 if (inside_base) {
111 // p directly above or below a cap
112 return {Vec<T,N-1>{}, (h >= height.hi) ? 1 : -1};
113 } else {
114 // p is outside the base shape
115 Vec<T,N> cap_pt {
116 base.project(p_base),
117 height.clip(h)
118 };
119 return (p - cap_pt).unit();
120 }
121 }
122 }

127 Vec<T,N> project(Vec<T,N> p) const requires ProjectableObject<Shape> {
128 const T h = p[N-1];
129 const T h_cap = std::abs(h - height.lo) < std::abs(h - height.hi)
130 ? height.lo
131 : height.hi;
132 // point in the plane of the base shape:
133 const Vec<T,N-1> p_base = p.template resized<N-1>();
134 // point projected to the boundary of the base shape:
135 const Vec<T,N-1> p_proj = base.project(p_base);
136 // nearest point on the wall:
137 Vec<T,N> wall_pt = Vec<T,N>(p_proj, height.clip(h));
138 // point on the nearer cap:
139 Vec<T,N> cap_pt = base.contains(p_base)
140 ? Vec<T,N>(p_base, h_cap)
141 : Vec<T,N>(p_proj, h_cap);
142 // pick the closer of the cap or the wall:
143 return p.dist2(wall_pt) < p.dist2(cap_pt) ? wall_pt : cap_pt;
144 }

201 Rect<T,1> intersect(const Ray<T,N>& r) const requires RayIntersectableObject<Shape> {
202 Ray<T,N-1> r_base = r.template resized<N-1>();
203 Rect<T,1> interval;
204 if (r_base.direction.is_zero()) {
205 // ray is pointing directly along extrusion axis
206 if (base.contains(r_base.origin)) {
207 interval = Rect<T,1>::full;
208 }
209 } else {
210 interval = base.intersect(r_base);
211 }
212 T h = r.origin[N-1];
213 T dH = r.direction[N-1];
214 if (dH == 0) {
215 // ray is parallel to the base plane
216 return height.contains(h) ? interval : Rect<T,1>::empty;
217 } else {
218 // trace the height slab and intersect its interval
219 interval &= Rect<T,1>::from_corners(
220 (height.lo - h) / dH,
221 (height.hi - h) / dH);
222 return interval;
223 }
