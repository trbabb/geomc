72 T sdf(Vec<T,N> p) const {
73 Vec<T,N-1> p_proj = (Vec<T,N-1>) p.template resized<N-1>();
74 T h = p[N-1];
75 // distance to the base shape within the base shape's plane
76 T sdf_base = base.sdf(p_proj);
77 // sdf of the slab that contains the height range of the shape
78 T sdf_h_slab = std::max(h - height.hi, height.lo - h);
79 if (height.contains(h)) {
80 // intersect base extrusion with the height slab
81 // because we are inside the shape, the sdf will be negative,
82 // so max() will give us a smaller absolute value— the closer surface.
83 return std::max(sdf_base, sdf_h_slab);
84 } else {
85 // nearest point must be a cap
86 return (sdf_base < 0)
87 // point projects orthogonally to the cap
88 ? sdf_h_slab
89 // point is "away" from the edge of the cap, above/below it
90 : std::sqrt(sdf_base * sdf_base + sdf_h_slab * sdf_h_slab);
91 }
92 }

94 Vec<T,N> normal(Vec<T,N> p) const {
95 Vec<T,N-1> p_base = (Vec<T,N-1>) p.template resized<N-1>();
96 T h = p[N-1];
97 bool inside_base = base.contains(p_base);
98 if (height.contains(h)) {
99 // p is somewhere between the caps
100 if (inside_base) {
101 // we need to know if the wall is closest, or a cap is closest
102 T cap_h = height.project(h);
103 T cap_dist = std::abs(h - cap_h);
104 T wall_dist = std::abs(base.sdf(p_base));
105 if (cap_dist < wall_dist) {
106 // the cap is nearest. use its normal
107 return {Vec<T,N-1>{}, (cap_h >= height.hi) ? 1 : -1};
108 }
109 }
110 // the wall is nearest, use its normal
111 return Vec<T,N>(base.normal(p_base), 0);
112 } else {
113 // p beyond the top or bottom cap
114 if (inside_base) {
115 // p directly above or below a cap
116 return {Vec<T,N-1>{}, (h >= height.hi) ? 1 : -1};
117 } else {
118 // p is outside the base shape
119 Vec<T,N> cap_pt {
120 base.project(p_base),
121 height.clip(h)
122 };
123 return (p - cap_pt).unit();
124 }
125 }
126 }

131 Vec<T,N> project(Vec<T,N> p) const {
132 const T h = p[N-1];
133 const T h_cap = std::abs(h - height.lo) < std::abs(h - height.hi)
134 ? height.lo
135 : height.hi;
136 // point in the plane of the base shape:
137 const Vec<T,N-1> p_base = p.template resized<N-1>();
138 // point projected to the boundary of the base shape:
139 const Vec<T,N-1> p_proj = base.project(p_base);
140 // nearest point on the wall:
141 Vec<T,N> wall_pt = Vec<T,N>(p_proj, height.clip(h));
142 // point on the nearer cap:
143 Vec<T,N> cap_pt = base.contains(p_base)
144 ? Vec<T,N>(p_base, h_cap)
145 : Vec<T,N>(p_proj, h_cap);
146 // pick the closer of the cap or the wall:
147 return p.dist2(wall_pt) < p.dist2(cap_pt) ? wall_pt : cap_pt;
148 }

150 Vec<T,N> convex_support(Vec<T,N> d) const {
151 Vec<T,N-1> d_ = d.template resized<N-1>();
152
153 if (d_.is_zero()) {
154 // support is normal to the endcap
155 // pick an arbitrary point on the endcap
156 d_[0] = 1;
157 }
158
159 Vec<T,N-1> p0 = base.convex_support(d_);
160
161 // top or bottom face?
162 return Vec<T,N>(p0, (d[N-1] > 0) ? height.hi : height.lo);
163 }

170 Rect<T,1> intersect(const Ray<T,N>& r) const {
171 Ray<T,N-1> r_base = r.template resized<N-1>();
172 Rect<T,1> interval;
173 if (r_base.direction.is_zero()) {
174 // ray is pointing directly along extrusion axis
175 if (base.contains(r_base.origin)) {
176 interval = Rect<T,1>::full;
177 }
178 } else {
179 interval = base.intersect(r_base);
180 }
181 T h = r.origin[N-1];
182 T dH = r.direction[N-1];
183 if (dH == 0) {
184 // ray is parallel to the base plane
185 return height.contains(h) ? interval : Rect<T,1>::empty;
186 } else {
187 // trace the height slab and intersect its interval
188 interval &= Rect<T,1>::spanning_corners(
189 (height.lo - h) / dH,
190 (height.hi - h) / dH);
191 return interval;
192 }
193 }
