68  template <typename T, index_t N>
69  Vec<T,N> orthogonal(const Vec<T,N> v[N-1]) {
70  Vec<T,N> o;
71  index_t P[N];
72  T m[N * (N-1)];
73  const T* v0 = v[0].begin();
74  detail::MxWrap<T,true> mx = {m, N-1, N};
75 
76  std::copy(v0, v0 + N * (N-1), m);
77 
78  bool parity_swap = false;
79  if (decomp_plu(m, N-1, N, P, &parity_swap) > 0) {
80  // matrix is singular; nullity is > 1. return 0 vector.
81  return Vec<T,N>();
82  }
83 
84  o[N-1] = 1;
85  for (index_t r = N-2; r >= 0; r--) {
86  // back substitute.
87  for (index_t c = N-1; c > r; c--) {
88  o[r] -= o[c] * mx(r,c);
89  }
90  o[r] /= mx(r,r);
91  }
92 
93  return o;
94  }

131  template <typename T, index_t N>
132  void nullspace(const Vec<T,N> bases[], index_t n, Vec<T,N> null_basis[]) {
133  if (n >= N) return; // nothing to do
134  if (N - n == 1) {
135  null_basis[0] = orthogonal(bases);
136  return;
137  }
138 
139  T m[N * (N-2)]; // <-- N-2 is the max # bases
140  index_t P[N - 2]; // (N-1 and N cases handled above)
141  T* n0 = null_basis[0].begin();
142  const T* b0 = bases[0].begin();
143 
144  // copy the bases to temporary storage
145  std::copy(b0, b0 + N * n, m);
146  // zero-init the results
147  std::fill(n0, n0 + N * (N - n), 0);
148 
149  // each original basis is a row of:
150  detail::MxWrap<T,true> V = {m, n, N};
151  // each new null basis is a column of:
152  detail::MxWrap<T,false> X = {n0, N, N - n};
153 
154  // get V to row-echelon form
155  bool parity_swap = false;
156  if (decomp_plu(m, n, N, P, &parity_swap) > 0) return;
157 
158  // foreach null space basis
159  for (index_t b = 0; b < N - n; b++) {
160  X(b + n, b) = 1;
161  // back substitute.
162  for (index_t r = n - 1; r >= 0; r--) {
163  T& x_i = X(r, b);
164  x_i = -V(r, n + b);
165  for (index_t c = r + 1; c < n; c++) {
166  x_i -= V(r, c) * X(c, b);
167  }
168  x_i /= V(r, r);
169  }
170  }
171 
172  return;
173  }
