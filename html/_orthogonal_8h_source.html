68template <typename T, index_t N>
69Vec<T,N> orthogonal(const Vec<T,N> v[N-1]) {
70 Vec<T,N> o;
71 index_t P[N];
72 T m[N * (N-1)];
73 const T* v0 = v[0].begin();
74 detail::MxWrap<T,true> mx = {m, N-1, N};
75
76 std::copy(v0, v0 + N * (N-1), m);
77
78 bool parity_swap = false;
79 if (decomp_plu(m, N-1, N, P, &parity_swap) > 0) {
80 // matrix is singular; nullity is > 1. return 0 vector.
81 return Vec<T,N>();
82 }
83
84 o[N-1] = 1;
85 for (index_t r = N-2; r >= 0; r--) {
86 // back substitute.
87 for (index_t c = N-1; c > r; c--) {
88 o[r] -= o[c] * mx(r,c);
89 }
90 o[r] /= mx(r,r);
91 }
92
93 return o;
94}

109// todo: a different formulation could handle degeneracy in `bases` if
110// `bases` is specified to have N vectors; then the null bases
111// could fill the unused space, however many there are.
112// rn we bail if there is degeneracy, because it's excessive
113// to demand that `null_basis` always have space for N vectors.
131template <typename T, index_t N>
132bool nullspace(const Vec<T,N> bases[], index_t n, Vec<T,N> null_basis[]) {
133 if (n >= N) return true; // nothing to do
134 if (N - n == 1) {
135 // this may be optimized for some (lower) dimensions:
136 null_basis[0] = orthogonal(bases);
137 return true;
138 }
139
140 T m[N * (N-2)]; // <-- N-2 is the max # bases
141 index_t P[N - 2]; // (N-1 and N cases handled above)
142 T* n0 = null_basis[0].begin();
143 const T* b0 = bases[0].begin();
144
145 // copy the bases to temporary storage
146 std::copy(b0, b0 + N * n, m);
147 // zero-init the results
148 std::fill(n0, n0 + N * (N - n), 0);
149
150 // each original basis is a row of:
151 detail::MxWrap<T,true> V = {m, n, N};
152 // each new null basis is a column of:
153 detail::MxWrap<T,false> X = {n0, N, N - n};
154
155 // get V to row-echelon form
156 bool parity_swap = false;
157 if (decomp_plu(m, n, N, P, &parity_swap) > 0) return false;
158
159 // foreach null space basis
160 for (index_t b = 0; b < N - n; b++) {
161 X(b + n, b) = 1;
162 // back substitute.
163 for (index_t r = n - 1; r >= 0; r--) {
164 T& x_i = X(r, b);
165 x_i = -V(r, n + b);
166 for (index_t c = r + 1; c < n; c++) {
167 x_i -= V(r, c) * X(c, b);
168 }
169 x_i /= V(r, r);
170 }
171 }
172
173 return true;
174}

187template <typename T, index_t N>
188void orthogonalize(Vec<T,N> basis[], index_t n) {
189 for (index_t i = 1; i < n; i++) {
190 for (index_t j = 0; j < i; j++) {
191 // modified gram-schmidt uses the intermediate basis
192 // for better numerical stability.
193 basis[i] -= basis[i].project_on(basis[j]);
194 }
195 }
196}

215template <typename T, index_t N>
216void orthonormalize(Vec<T,N> basis[], index_t n) {
217 orthogonalize(basis, n);
218 for (index_t i = 0; i < n; i++) {
219 basis[i] /= basis[i].mag();
220 }

231template <typename T, index_t N>
232Vec<T,N> project_to_orthogonal_subspace(const Vec<T,N> bases[], Vec<T,N> x, index_t n) {
233 Vec<T,N> out;
234 for (index_t i = 0; i < n; i++) {
235 out += x.project_on(bases[i]);
236 }
237 return out;
