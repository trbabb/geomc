50  template <typename T, index_t N>
51  Vec<T,N> orthogonal(const Vec<T,N> v[N-1]) {
52  Vec<T,N> o;
53  o[N-1] = 1;
54  index_t P[N];
55  T m[N * (N-1)];
56  std::copy(v[0].begin(), v[0].begin() + N * (N-1), m);
57  bool parity_swap = false;
58 
59  if (decompLUP(m, N-1, N, P, &parity_swap)) {
60  // matrix is singular; nullity is > 1. return 0 vector.
61  return Vec<T,N>();
62  }
63 
64  for (index_t r = N-2; r >= 0; r--) {
65  // back substitute.
66  for (index_t c = N-1; c > r; c--) {
67  o[r] -= o[c] * _MxElem(m,r,c);
68  }
69  o[r] /= _MxElem(m,r,r);
70  }
71 
72  PermutationMatrix<N> Pmtx;
73  Pmtx.setRowSources(P);
74 
75  // invert the permutation.
76  // TODO: is that parity_swap correct?
77  return (parity_swap ? -1 : 1) * o * Pmtx;
78  }

157  void orthonormalize(Vec<T,N> basis[], index_t n) {
158  for (index_t i = 1; i < n; i++) {
159  for (index_t j = 0; j < i; j++) {
