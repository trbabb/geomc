222Similarity<T,N> mix(T s, const Similarity<T,N>& a, const Similarity<T,N>& b) {
223 T sign = a.sx >= 0 ? 1 : -1;
224 return Similarity<T,N>(
225 sign * std::pow(std::abs(a.sx), 1 - s) * std::pow(std::abs(b.sx), s),
226 geom::mix(s, a.rx, b.rx),
227 geom::mix(s, a.tx, b.tx)
228 );
229}
