239/*****************************************************
240 * AssignmentProxy class *
241 * *
242 * Returned by certain matrix methods in lieu of a *
243 * memory reference, when a memory reference doesn't *
244 * make sense. For example, by double-indexing *
245 * a sparse matrix (a memory location for the *
246 * requested element may not exist yet), or by *
247 * dereferencing iterators provided by such classes. *
248 * Assigning to these objects inserts an element *
249 * into the associated matrix, if possible. *
250 *****************************************************/
251
252template <typename M, typename T> class MtxAssignmentProxy {
253 M* m;
254 index_t row, col;
255public:
256
257 MtxAssignmentProxy(M* m, index_t row, index_t col):m(m), row(row), col(col) {}
258
259 // copy assignment
260 // like a reference, assign through to the pointed object.
261 // contrary to convention, we do NOT mirror this action in the copy constructor
262 // this is to mirror the behavior of references, whose ptr can only be assigned
263 // at construction.
264 inline MtxAssignmentProxy<M,T>& operator=(const MtxAssignmentProxy<M,T>& other) {
265 if (&other != this) {
266 m->set(row, col, other);
267 }
268 return *this;
269 }
270
271 // return const T& for some mtxs, T for those which use temporaries?
272 inline const typename M::elem_t operator=(typename M::elem_t val) {
273 m->set(row, col, val);
274 return val;
275 }
276
277 // TODO: this should be a reference?
278 inline operator T() const {
279 return ((const M*)m)->get(row, col);
280 }
281
282};

284/******************************************************
285 * Iterator type traits classes *
286 * *
287 * Used to inform the matrix base class about the *
288 * derived class's choice of iterator implementation. *
289 * *
290 * Because c++ is a stupid, broken language, it *
291 * cannot resolve type declaration dependencies if *
292 * the type definitions belong to the classes *
293 * themselves. *
294 * http://tinyurl.com/cvtlqow *
295 * http://tinyurl.com/bl8aznk *
296 * *
297 * C++ cannot reason about types that belong to other *
298 * "incomplete" types, even if a complete definition *
299 * for that sub-type already exists and is itself *
300 * fully defined. If instead C++ built a graph of *
301 * type declarations and topologically sorted their *
302 * dependencies, the problem would be trivially *
303 * solved. But alas, attacking problems in such a *
304 * sensible way is not part of the c++ design *
305 * philosophy. *
306 ******************************************************/
307
308// we have to pass T (rather than use M::elem_t) because, again,
309// c++ is a special child and can't type-solve its way out of wet paper bag.
310// specifically, when these are instantiated, M is an incomplete type.
311template <typename M, typename T>
312struct _ImplMtxReftype {
313 typedef MtxAssignmentProxy<M, T> reference;
314};
