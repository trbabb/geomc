171 inline Quat<T> mult(const Quat<T>& q) const {
172 Quat<T> result;
173
174 const T x = detail::VecBase<T,4>::x;
175 const T y = detail::VecBase<T,4>::y;
176 const T z = detail::VecBase<T,4>::z;
177 const T w = detail::VecBase<T,4>::w;
178
179 result.x = w*q.x + x*q.w + y*q.z - z*q.y;
180 result.y = w*q.y - x*q.z + y*q.w + z*q.x;
181 result.z = w*q.z + x*q.y - y*q.x + z*q.w;
182 result.w = w*q.w - x*q.x - y*q.y - z*q.z;
183
184 return result;
185 }

191 Vec<T,4> to_axis_angle_rotation() const {
192 T w_clamp = std::min(std::max(detail::VecBase<T,4>::w, -1.0), 1.0);
193 double alpha = 2 * std::acos(w_clamp);
194 if (detail::VecBase<T,4>::x == 0 and
195 detail::VecBase<T,4>::y == 0 and
196 detail::VecBase<T,4>::z == 0)
197 {
198 return Vec<T,4>(1, 0, 0, 0);
199 } else {
200 return Vec<T,4>(imag().unit(), alpha);
201 }

208 Vec<T,3> to_angular_velocity() const {
209 // assumes a unit quaternion.
210 T w_clamp = std::min(std::max(detail::VecBase<T,4>::w, -1.0), 1.0);
211 T alpha = 2 * std::acos(w_clamp);
212 if (imag().is_zero()) {
213 return Vec<T,3>::zeros;
214 } else {
215 return alpha * imag().unit();
216 }

223 // this may go the "long way" around. that what we want?
224 Quat<T> rotation_scale(T t) const {
225 // assumes a unit quaternion.
226 const T w = std::max(-1, std::min(Vec<T,4>::w, 1));
227 T omega = acos(w);
228 T theta = omega * t;
229 return Quat<T>(sin(theta) * imag().unit(), cos(theta));
230 }

239 Quat<T> slerp(const Quat<T> &q1, T t) const {
240 // assumed unit quaternion.
241 const Quat<T> &q0 = *this;
242 T dot = q0.dot(q1);
243 if (dot < 0) { //angle is greater than 90. Avoid "long path" rotations.
244 dot = -dot;
245 q1 = -q1;
246 }
247 dot = std::max(0, std::min<T>(1, dot)); //keep <dot> in the domain of acos.
248 T omega = acos(dot);
249 T theta = omega * t;
250 Quat<T> q2 = (q1 - q0 * dot).unit();
251 return cos(theta) * q0 + sin(theta) * q2;
252 }

281 template <typename T>
282 geom::Quat<T> exp(const geom::Quat<T>& q) {
283 T k = exp(q.w);
284 T m = q.imag().mag();
285 if (m == 0) return geom::Quat<T>(geom::Vec<T,3>::zeros, exp(q.w));
286 T c = cos(m);
287 T s = sin(m);
288 return geom::Quat<T>(k * s * q.imag() / m, k * c);
289 }

294 template <typename T>
295 geom::Quat<T> log(const geom::Quat<T>& q) {
296 T q_m = q.mag();
297 T v_m = q.imag().mag();
298 T k = acos(q.w / q_m);
299 return geom::Quat<T>(k * q.imag() / v_m, log(q_m));
300 }

305 template <typename T, typename U>
306 geom::Quat<T> pow(const geom::Quat<T> &q, U a) {
307 T mag = q.mag();
308 if (mag == 0) return geom::Quat<T>(0,0,0,1);
309 U theta = std::acos(q.real() / mag); // :(
310 geom::Vec<T,3> nhat = q.imag().unit();
311 return std::pow(mag, a) * geom::Quat<T>(
312 nhat * std::sin(a * theta),
313 std::cos(a * theta));
314 }
