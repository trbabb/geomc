158  inline Quat<T> mult(const Quat<T> &q) const {
159  Quat<T> result;
160 
161  const T &x = detail::VecBase<T,4>::x;
162  const T &y = detail::VecBase<T,4>::y;
163  const T &z = detail::VecBase<T,4>::z;
164  const T &w = detail::VecBase<T,4>::w;
165 
166  result.x = w*q.x + x*q.w + y*q.z - z*q.y;
167  result.y = w*q.y - x*q.z + y*q.w + z*q.x;
168  result.z = w*q.z + x*q.y - y*q.x + z*q.w;
169  result.w = w*q.w - x*q.x - y*q.y - z*q.z;
170 
171  return result;
172  }

222  Quat<T> slerp(const Quat<T> &q1, T t) const {
223  // assumed unit quaternion.
224  const Quat<T> &q0 = *this;
225  T dot = q0.dot(q1);
226  if (dot < 0) { //angle is greater than 90. Avoid "long path" rotations.
227  dot = -dot;
228  q1 = -q1;
229  }
230  dot = std::max(0, std::min<T>(1, dot)); //keep <dot> in the domain of acos.
231  T omega = acos(dot);
232  T theta = omega * t;
233  Quat<T> q2 = (q1 - q0 * dot).unit();
234  return cos(theta) * q0 + sin(theta) * q2;
235  }

265  geom::Quat<T> exp(const geom::Quat<T>& q) {
266  T k = exp(q.w);
267  T m = q.vectorPart().mag();
268  if (m == 0) return geom::Quat<T>(geom::Vec<T,3>::zeros, exp(q.w));
269  T c = cos(m);
270  T s = sin(m);
271  return geom::Quat<T>(k * s * q.vectorPart() / m, k * c);
