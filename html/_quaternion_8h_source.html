241 Quat<T> slerp(const Quat<T> &q1, T t) const {
242 // assumed unit quaternion.
243 const Quat<T> &q0 = *this;
244 T dot = q0.dot(q1);
245 if (dot < 0) { //angle is greater than 90. Avoid "long path" rotations.
246 dot = -dot;
247 q1 = -q1;
248 }
249 dot = std::max(0, std::min<T>(1, dot)); //keep <dot> in the domain of acos.
250 T omega = std::acos(dot);
251 T theta = omega * t;
252 Quat<T> q2 = (q1 - q0 * dot).unit();
253 return std::cos(theta) * q0 + std::sin(theta) * q2;
254 }

293template <typename T>
294geom::Quat<T> exp(const geom::Quat<T>& q) {
295 T k = std::exp(q.w);
296 T m = q.imag().mag();
297 if (m == 0) return geom::Quat<T>(geom::Vec<T,3>::zeros, std::exp(q.w));
298 T c = std::cos(m);
299 T s = std::sin(m);
300 return geom::Quat<T>(k * s * q.imag() / m, k * c);
301}

306template <typename T>
307geom::Quat<T> log(const geom::Quat<T>& q) {
308 T q_m = q.mag();
309 T v_m = q.imag().mag();
310 T k = std::acos(q.w / q_m);
311 return geom::Quat<T>(k * q.imag() / v_m, std::log(q_m));
312}
