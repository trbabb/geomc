173 Quat<T> mult(const Quat<T>& q) const {
174 Quat<T> result;
175
176 const T x = detail::VecBase<T,4>::x;
177 const T y = detail::VecBase<T,4>::y;
178 const T z = detail::VecBase<T,4>::z;
179 const T w = detail::VecBase<T,4>::w;
180
181 result.x = w*q.x + x*q.w + y*q.z - z*q.y;
182 result.y = w*q.y - x*q.z + y*q.w + z*q.x;
183 result.z = w*q.z + x*q.y - y*q.x + z*q.w;
184 result.w = w*q.w - x*q.x - y*q.y - z*q.z;
185
186 return result;
187 }

246 Quat<T> slerp(const Quat<T> &q1, T t) const {
247 // assumed unit quaternion.
248 const Quat<T> &q0 = *this;
249 T dot = q0.dot(q1);
250 if (dot < 0) { //angle is greater than 90. Avoid "long path" rotations.
251 dot = -dot;
252 q1 = -q1;
253 }
254 dot = std::max(0, std::min<T>(1, dot)); //keep <dot> in the domain of acos.
255 T omega = std::acos(dot);
256 T theta = omega * t;
257 Quat<T> q2 = (q1 - q0 * dot).unit();
258 return std::cos(theta) * q0 + std::sin(theta) * q2;
259 }

298template <typename T>
299geom::Quat<T> exp(const geom::Quat<T>& q) {
300 T k = std::exp(q.w);
301 T m = q.imag().mag();
302 if (m == 0) return geom::Quat<T>(geom::Vec<T,3>::zeros, std::exp(q.w));
303 T c = std::cos(m);
304 T s = std::sin(m);
305 return geom::Quat<T>(k * s * q.imag() / m, k * c);
306}

311template <typename T>
312geom::Quat<T> log(const geom::Quat<T>& q) {
313 T q_m = q.mag();
314 T v_m = q.imag().mag();
315 T k = std::acos(q.w / q_m);
316 return geom::Quat<T>(k * q.imag() / v_m, std::log(q_m));
317}

322template <typename T, typename U>
323geom::Quat<T> pow(const geom::Quat<T> &q, U a) {
324 T mag = q.mag();
325 if (mag == 0) return geom::Quat<T>(0,0,0,1);
326 U theta = std::acos(q.real() / mag); // :(
327 geom::Vec<T,3> nhat = q.imag().unit();
328 return std::pow(mag, a) * geom::Quat<T>(
329 nhat * std::sin(a * theta),
330 std::cos(a * theta));
331}
