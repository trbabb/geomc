33// apply the permutation map `p` to the n x m matrix `A`, overwriting `p` in the process
34// (after which p will be the identity permutation)
35template <typename T, bool RowMajor=true>
36void destructive_apply_permutation(index_t* p, T* A, index_t n, index_t m) {
37 detail::MxWrap<T,RowMajor> mx = {A, n, m};
38 for (index_t i = 0; i < n; i++) {
39 index_t cur = i;
40 index_t next = p[i]; // source of new element at this position
41 if (next == i) continue; // no swap necessary
42 // if (next < 0) {
43 // // marked as visited. restore original value
44 // p[i] = -(p[i]);
45 // continue;
46 // }
47 do {
48 for (index_t j = 0; j < m; ++j) {
49 // give v[cur] the correct element, v[next] gets the cycle's first element
50 std::swap(mx(cur, j), mx(next, j));
51 }
52 // ...and then proceed to find out what goes into the hole
53 cur = next;
54 next = p[next];
55 // mark this element as visited by mapping it to itself
56 // (we only need to do this if we might encounter the cycle again)
57 p[cur] = cur;
58
59 // mark as visited
60 // if (cur > i) {
61 // p[cur] = -(p[cur]);
62 // }
63 } while (next != i);
64 }
65}

102template <typename T, bool RowMajor=true>
103index_t decomp_lup(T* m, index_t rows, index_t cols, index_t* reorder, bool* parity) {
104 const index_t n = std::min(rows, cols);
105 index_t degenerate_ct = 0;
106 *parity = false;
107 // fill permutation array
108 if (reorder != nullptr) {
109 for (index_t i = 0; i < cols; i++){
110 reorder[i] = i;
111 }
112 }
113
114 detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
115
116 for (index_t i = 0; i < n - 1; i++) {
117 // find pivot
118 T biggest = std::abs(mx.elem(i,i));
119 index_t pvt = i;
120 for (index_t p = i + 1; p < cols; p++) {
121 T pvt_val = std::abs(mx.elem(i, p));
122 if (pvt_val > biggest) {
123 pvt = p;
124 biggest = pvt_val;
125 }
126 }
127
128 if (biggest == 0) {
129 // singular matrix
130 // could test against an epsilon to
131 // find ill-conditioned matrices
132 degenerate_ct += 1;
133 continue;
134 } else if (pvt != i) {
135 // swap col <i> with col <pvt>
136 for (index_t r = 0; r < rows; r++) {
137 std::swap(mx.elem(r, i),
138 mx.elem(r, pvt));
139 }
140 // make a note of the permutation in <P>
141 if (reorder != nullptr) std::swap(reorder[i], reorder[pvt]);
142 *parity = !*parity;
143 }
144
145 // eliminate lower elements
146 T a = mx.elem(i,i);
147 for (index_t r = i + 1; r < rows; r++) {
148 T b = mx.elem(r,i) / a;
149 if (b != 0) {
150 for (index_t c = i + 1; c < cols; c++) {
151 // R_r = R_r - b * R_i
152 T src_elem = mx.elem(i,c);
153 T dst_elem = mx.elem(r,c);
154 // mx[r,c] = dst_elem - b * src_elem;
155 mx.elem(r,c) = geom::multiply_add(-b, src_elem, dst_elem);
156 }
157 }
158 // set the lower matrix
159 mx.elem(r,i) = b;
160 }
161 }
162
163 return degenerate_ct;
164}

186template <typename T, bool RowMajor=true>
187index_t decomp_plu(T* m, index_t rows, index_t cols, index_t* reorder, bool* parity) {
188 const index_t n = std::min(rows, cols);
189 index_t degenerate_ct = 0;
190 *parity = false;
191 // fill permutation array
192 if (reorder) {
193 for (index_t i = 0; i < rows; i++){
194 reorder[i] = i;
195 }
196 }
197
198 detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
199
200 for (index_t i = 0; i < n - 1; i++) {
201 // find pivot
202 T biggest = std::abs(mx.elem(i,i));
203 index_t pvt = i;
204 for (index_t p = i + 1; p < rows; p++) {
205 T pvt_val = std::abs(mx.elem(p,i));
206 if (pvt_val > biggest) {
207 pvt = p;
208 biggest = pvt_val;
209 }
210 }
211
212 if (biggest == 0) {
213 // singular matrix. could test against an epsilon to
214 // find ill-conditioned matrices
215 degenerate_ct += 1;
216 continue;
217 } else if (pvt != i) {
218 // swap row <i> with row <pvt>
219 for (index_t c = 0; c < cols; c++) {
220 std::swap(mx.elem(i, c),
221 mx.elem(pvt, c));
222 }
223 // make a note of the permutation in <P>
224 if (reorder) std::swap(reorder[i], reorder[pvt]);
225 *parity = !*parity;
226 }
227
228 // eliminate lower elements
229 T a = mx.elem(i,i);
230 for (index_t r = i + 1; r < rows; r++) {
231 T b = mx.elem(r,i) / a;
232 if (b != 0) {
233 for (index_t c = i + 1; c < cols; c++) {
234 // R_r = R_r - b * R_i
235 T src_elem = mx.elem(i,c);
236 T dst_elem = mx.elem(r,c);
237 // mx[r,c] = dst_elem - b * src_elem;
238 mx.elem(r,c) = geom::multiply_add(-b, src_elem, dst_elem);
239 }
240 }
241 // set the lower matrix
242 mx.elem(r,i) = b;
243 }
244 }
245
246 return degenerate_ct;
247}

267template <typename T, bool RowMajor=true>
268void backsolve_plu(
269 const T* plu,
270 const index_t* p,
271 index_t n,
272 index_t m,
273 T* x,
274 const T* b,
275 index_t skip=0)
276{
277 detail::MxWrap<const T, RowMajor> mx = {plu, n, n};
278 detail::MxWrap<const T, RowMajor> B = {b, n, m};
279 // <y> and <x>'s elements are used such that
280 // y[i] is never read after x[i] is written.
281 // thus we may collapse their storage and save space:
282 detail::MxWrap<T, RowMajor> Y = {x, n, m};
283 detail::MxWrap<T, RowMajor> X = {x, n, m};
284
285 // LUX = Pb
286 // (UX) is a matrix (call it "Y")
287 // LY = Pb
288 // so let's solve for each column `i` of Y:
289 for (index_t r = 0; r < n; r++) {
290 for (index_t i = 0; i < m; ++i) {
291 Y.elem(r,i) = B.elem(p[r], i); // y[r] <- Pb[r]
292 for (index_t c = 0; c < r; c++) {
293 Y.elem(r,i) -= Y.elem(c,i) * mx.elem(r,c);
294 }
295 }
296 }
297 // now with y, we may obtain x from:
298 // Ux = y
299 for (index_t r = n - 1; r >= skip; r--) {
300 // x[r] = y[r]; // nop; x and y are the same!
301 T k = mx.elem(r,r);
302 for (index_t i = 0; i < m; ++i) {
303 for (index_t c = n - 1; c > r; c--) {
304 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
305 }
306 X.elem(r, i) /= k;
307 }
308 }
309}

328template <typename T, bool RowMajor=true>
329void backsolve_lu(
330 const T* lu,
331 index_t n,
332 index_t m,
333 T* x,
334 index_t skip=0)
335{
336 detail::MxWrap<const T, RowMajor> mx = {lu, n, n};
337 detail::MxWrap< T, RowMajor> X = {x, n, m}; // `x` is initialized with `b`
338
339 // LUx = b
340 // (Ux) is a vector (call it "y")
341 // Ly = b
342 for (index_t r = 1; r < n; r++) {
343 // do this for each column i in x:
344 for (index_t i = 0; i < m; ++i) {
345 // x[r,i] = b[r,i] // <-- already true
346 for (index_t c = 0; c < r; c++) {
347 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
348 }
349 }
350 }
351 // now with y, we may obtain x from:
352 // Ux = y
353 for (index_t r = n - 1; r >= skip; r--) {
354 T k = mx.elem(r,r);
355 // do for each column i in y
356 for (index_t i = 0; i < m; ++i) {
357 for (index_t c = n - 1; c > r; c--) {
358 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
359 }
360 X.elem(r, i) /= k;
361 }
362 }
363}

384template <typename T, bool RowMajor=true>
385inline bool linear_solve(T* a, index_t n, index_t m, T* x, index_t skip=0) {
386 SmallStorage<index_t, 24> p(n); // unlikely to need an alloc
387 bool parity;
388 if (decomp_plu<T,RowMajor>(a, n, n, p.get(), &parity) > 0) {
389 return false;
390 }
391 // pre-apply P to x
392 detail::destructive_apply_permutation<T,RowMajor>(p.get(), x, n, m);
393 // LUx = Pb
394 backsolve_lu<T,RowMajor>(a, n, m, x, skip);
395 return true;
396}

413template <typename T, index_t N>
414inline bool linear_solve(Vec<T,N> bases[N], index_t m, Vec<T,N>* x, index_t skip=0) {
415 T* _x = x->begin();
416 if (N < 5) {
417 // matrix inv is empirically faster than solve() for N < 5.
418 WrapperMatrix<T,N,N,COL_MAJOR> mx(bases->begin());
419 WrapperMatrix<T,N,0,COL_MAJOR> X(_x, m);
420 SimpleMatrix<T,N,N> m_inv;
421 if (not inv(&m_inv, mx)) return false;
422 mul(&X, m_inv, X);
423 return true;
424 } else {
425 index_t p[N];
426 T* const mx = bases[0].begin();
427 bool _parity;
428 // LU = PM (M's rows reordered)
429 if (decomp_plu<T,false>(mx, N, N, p, &_parity) > 0) {
430 return false;
431 }
432 // pre-apply P to x
433 detail::destructive_apply_permutation<T,false>(p, _x, N, m);
434 // LUx = Pb
435 backsolve_lu<T,false>(mx, N, m, _x, skip);
436 return true;
437 }
