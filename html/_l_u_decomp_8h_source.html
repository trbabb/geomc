35 index_t decompLUP(T* m, index_t rows, index_t cols, index_t *reorder, bool *swap_parity) {
36  const index_t n = std::min(rows, cols);
37  index_t degenerate_ct = 0;
38  *swap_parity = false;
39  // fill permutation array
40  for (index_t i = 0; i < cols; i++){
41  reorder[i] = i;
42  }
43 
44  for (index_t i = 0; i < n - 1; i++) {
45  // find pivot
46  T biggest = std::abs(_MxElem(i,i));
47  index_t pvt = i;
48  for (index_t p = i + 1; p < cols; p++) {
49  T pvt_val = std::abs(_MxElem(i,p));
50  if (pvt_val > biggest) {
51  pvt = p;
52  biggest = pvt_val;
53  }
54  }
55 
56  if (biggest == 0) {
57  // singular matrix
58  // could test against an epsilon to
59  // find ill-conditioned matrices
60  degenerate_ct += 1;
61  continue;
62  } else if (pvt != i) {
63  // swap col <i> with col <pvt>
64  for (index_t r = 0; r < rows; r++) {
65  std::swap(_MxElem(r, i),
66  _MxElem(r, pvt));
67  }
68  // make a note of the permutation in <P>
69  std::swap(reorder[i], reorder[pvt]);
70  *swap_parity = !*swap_parity;
71  }
72 
73  // eliminate lower elements
74  T a = _MxElem(i,i);
75  for (index_t r = i + 1; r < rows; r++) {
76  T b = _MxElem(r,i) / a;
77  for (index_t c = i + 1; c < cols; c++) {
78  // R_r = R_r - b * R_i
79  T src_elem = _MxElem(i,c);
80  T dst_elem = _MxElem(r,c);
81  _MxElem(r,c) = dst_elem - b * src_elem;
82  }
83  // set the lower matrix
84  _MxElem(r,i) = b;
85  }
86  }
87 
88  return degenerate_ct;
89 }

112  reorder[i] = i;
113  }
114 
115  for (index_t i = 0; i < n - 1; i++) {
116  // find pivot

118  index_t pvt = i;
119  for (index_t p = i + 1; p < rows; p++) {
120  T pvt_val = std::abs(_MxElem(p,i));
121  if (pvt_val > biggest) {
122  pvt = p;
123  biggest = pvt_val;
124  }
125  }
126

128  // singular matrix
129  // could test against an epsilon to
130  // find ill-conditioned matrices
131  degenerate_ct += 1;
132  continue;
133  } else if (pvt != i) {
134  // swap row <i> with row <pvt>
135  for (index_t c = 0; c < cols; c++) {
136  std::swap(_MxElem(i, c),
137  _MxElem(pvt, c));
138  }
139  // make a note of the permutation in <P>
140  std::swap(reorder[i], reorder[pvt]);
141  *swap_parity = !*swap_parity;
142  }
143 
144  // eliminate lower elements
145  T a = _MxElem(i,i);
146  for (index_t r = i + 1; r < rows; r++) {
147  T b = _MxElem(r,i) / a;
148  for (index_t c = i + 1; c < cols; c++) {
149  // R_r = R_r - b * R_i
150  T src_elem = _MxElem(i,c);
151  T dst_elem = _MxElem(r,c);
152  _MxElem(r,c) = dst_elem - b * src_elem;
153  }
154  // set the lower matrix
155  _MxElem(r,i) = b;
156  }
157  }

488  template <typename S>
489  void _linearSolve(S *dest, const S *b, bool permute=true) const {
490  const index_t n = LU.rows();
491  const index_t *p = P.getRowSources(); // permutation map
492 
493  // <y> and <dest>'s elements are used such that
494  // y[i] is never read after dest[i] is written.
495  // thus we may collapse their storage and save space:
496  S *y = dest;
497 
498  // LUx = Pb
499  // (Ux) is a vector, so let's solve for it:
500  // Ly = Pb
501  y[0] = permute ? b[p[0]] : b[0];
502  for (index_t r = 1; r < n; r++) {
503  y[r] = permute ? b[p[r]] : b[r];
504  for (index_t c = 0; c < r; c++) {
505  y[r] -= y[c] * LU.get(r,c);
506  }
507  }
508  // now with y, we may obtain x from:
509  // Ux = y
510  for (index_t r = n - 1; r >= 0; r--) {
511  // dest[r] = y[r]; // nop; dest and y are the same!
512  for (index_t c = n - 1; c > r; c--) {
513  dest[r] -= dest[c] * LU.get(r,c);
514  }
515  dest[r] /= LU.get(r,r);
516  }
517  }

520  template <typename S>
521  void _inverseTranspose(S *dest) const {
522  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
523  // Here, we use the rows of our destination matrix as though they are
524  // column vectors, and the caller will transpose.
525  const index_t *p = P.getColSources();
526  const index_t n = LU.rows();
527  std::fill(dest, dest + (n*n), 0);
528  for (index_t i = 0; i < n; i++, dest += n) {
529  dest[p[i]] = 1;
530  _linearSolve(dest, dest, false);
531  }
532  }
