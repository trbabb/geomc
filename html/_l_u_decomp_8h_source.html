96 template <typename T, bool RowMajor=true>
97 index_t decompLUP(T* m, index_t rows, index_t cols, index_t* reorder, bool* swap_parity) {
98  const index_t n = std::min(rows, cols);
99  index_t degenerate_ct = 0;
100  *swap_parity = false;
101  // fill permutation array
102  for (index_t i = 0; i < cols; i++){
103  reorder[i] = i;
104  }
105 
106  detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
107 
108  for (index_t i = 0; i < n - 1; i++) {
109  // find pivot
110  T biggest = std::abs(mx.elem(i,i));
111  index_t pvt = i;
112  for (index_t p = i + 1; p < cols; p++) {
113  T pvt_val = std::abs(mx.elem(i,p));
114  if (pvt_val > biggest) {
115  pvt = p;
116  biggest = pvt_val;
117  }
118  }
119 
120  if (biggest == 0) {
121  // singular matrix
122  // could test against an epsilon to
123  // find ill-conditioned matrices
124  degenerate_ct += 1;
125  continue;
126  } else if (pvt != i) {
127  // swap col <i> with col <pvt>
128  for (index_t r = 0; r < rows; r++) {
129  std::swap(mx.elem(r, i),
130  mx.elem(r, pvt));
131  }
132  // make a note of the permutation in <P>
133  std::swap(reorder[i], reorder[pvt]);
134  *swap_parity = !*swap_parity;
135  }
136 
137  // eliminate lower elements
138  T a = mx.elem(i,i);
139  for (index_t r = i + 1; r < rows; r++) {
140  T b = mx.elem(r,i) / a;
141  for (index_t c = i + 1; c < cols; c++) {
142  // R_r = R_r - b * R_i
143  T src_elem = mx.elem(i,c);
144  T dst_elem = mx.elem(r,c);
145  mx.elem(r,c) = dst_elem - b * src_elem;
146  }
147  // set the lower matrix
148  mx.elem(r,i) = b;
149  }
150  }
151 
152  return degenerate_ct;
153 }

178  reorder[i] = i;
179  }

184  // find pivot
185  T biggest = std::abs(mx.elem(i,i));
186  index_t pvt = i;
187  for (index_t p = i + 1; p < rows; p++) {
188  T pvt_val = std::abs(mx.elem(p,i));

192  }
193  }

196  // singular matrix
197  // could test against an epsilon to
198  // find ill-conditioned matrices
199  degenerate_ct += 1;
200  continue;
201  } else if (pvt != i) {
202  // swap row <i> with row <pvt>
203  for (index_t c = 0; c < cols; c++) {
204  std::swap(mx.elem(i, c),
205  mx.elem(pvt, c));
206  }
207  // make a note of the permutation in <P>
208  std::swap(reorder[i], reorder[pvt]);
209  *swap_parity = !*swap_parity;
210  }
211 
212  // eliminate lower elements
213  T a = mx.elem(i,i);
214  for (index_t r = i + 1; r < rows; r++) {
215  T b = mx.elem(r,i) / a;
216  for (index_t c = i + 1; c < cols; c++) {
217  // R_r = R_r - b * R_i
218  T src_elem = mx.elem(i,c);
219  T dst_elem = mx.elem(r,c);
220  mx.elem(r,c) = dst_elem - b * src_elem;
221  }

246 template <typename T, bool RowMajor=true>
247 void linearSolveLUP(const T* lup, const index_t* p, index_t n, T* x, const T* b, index_t skip=0) {
248  detail::MxWrapC<T,RowMajor> mx = {lup, n, n};
249 
250  // <y> and <x>'s elements are used such that
251  // y[i] is never read after x[i] is written.
252  // thus we may collapse their storage and save space:
253  T* y = x;
254 
255  // LUx = Pb
256  // (Ux) is a vector (call it "y"), so let's solve for it:
257  // Ly = Pb
258  y[0] = b[p[0]];
259  for (index_t r = 1; r < n; r++) {
260  y[r] = b[p[r]];
261  for (index_t c = 0; c < r; c++) {
262  y[r] -= y[c] * mx.elem(r,c);
263  }
264  }
265  // now with y, we may obtain x from:
266  // Ux = y
267  for (index_t r = n - 1; r >= skip; r--) {
268  // x[r] = y[r]; // nop; x and y are the same!
269  for (index_t c = n - 1; c > r; c--) {
270  x[r] -= x[c] * mx.elem(r,c);
271  }
272  x[r] /= mx.elem(r,r);
273  }
274 }

291 void linearSolveLU(const T* lu, index_t n, T* x, const T* b, index_t skip=0) {
292  detail::MxWrapC<T,RowMajor> mx = {lu, n, n};
293 
294  // <y> and <x>'s elements are used such that
295  // y[i] is never read after x[i] is written.
296  // thus we may collapse their storage and save space:
297  T* y = x;
298 
299  // LUx = b
300  // (Ux) is a vector, so let's solve for it:
301  // Ly = b
302  y[0] = b[0];
303  for (index_t r = 1; r < n; r++) {
304  y[r] = b[r];
305  for (index_t c = 0; c < r; c++) {
306  y[r] -= y[c] * mx.elem(r,c);
307  }
308  }
309  // now with y, we may obtain x from:
310  // Ux = y
311  for (index_t r = n - 1; r >= skip; r--) {
312  // x[r] = y[r]; // nop; x and y are the same!
313  for (index_t c = n - 1; c > r; c--) {
314  x[r] -= x[c] * mx.elem(r,c);

339  SmallStorage<index_t, 32> p(n); // probably will never have to alloc.

362 template <typename T, index_t N>
363 inline bool linearSolve(Vec<T,N>* bases, Vec<T,N>* x, const Vec<T,N>& b, index_t skip=0) {
364  if (N < 5) {
365  // matrix inv is empirically faster than solve() for N < 5.
366 
367  // m is row major, but our bases should be columns.
368  // m and its inverse are thus transposed.
369  WrapperMatrix<T,N,N> m(bases[0].begin());
370  SimpleMatrix<T,N,N> m_inv_txpose;
371  if (!inv(&m_inv_txpose, m)) return false;
372  // reverse mult order to get mul by txpose:
373  *x = b * m_inv_txpose;
374  } else {
375  index_t p[N];
376  T* const m = bases[0].begin();
377  bool parity;
378  if (decompPLU<T,false>(m, N, N, p, &parity) > 0) {
379  return false;
380  }
381  linearSolveLUP<T,false>(m, p, N, x->begin(), b.begin(), skip);
382  return true;
383  }
384 }

725  template <typename S>
726  void _inverseTranspose(S *dest) const {
727  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
728  // Here, we use the rows of our destination matrix as though they are
729  // column vectors, and the caller will transpose.
730  // todo: we can now call the templated solver and treat
731  // the input matrix as column-major directly.
732  const index_t *p = P.getColSources();
733  const index_t n = LU.rows();
734  std::fill(dest, dest + (n*n), 0);
735  for (index_t i = 0; i < n; i++, dest += n) {
736  dest[p[i]] = 1;
737  geom::linearSolveLU(LU.begin(), LU.rows(), dest, dest);
738  }
