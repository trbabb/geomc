36 index_t decompLUP(T* m, index_t rows, index_t cols, index_t *reorder, bool *swap_parity) {
37  const index_t n = std::min(rows, cols);
38  index_t degenerate_ct = 0;
39  *swap_parity = false;
40  // fill permutation array
41  for (index_t i = 0; i < cols; i++){
42  reorder[i] = i;
43  }
44 
45  for (index_t i = 0; i < n - 1; i++) {
46  // find pivot
47  T biggest = std::abs(_MxElem(i,i));
48  index_t pvt = i;
49  for (index_t p = i + 1; p < cols; p++) {
50  T pvt_val = std::abs(_MxElem(i,p));
51  if (pvt_val > biggest) {
52  pvt = p;
53  biggest = pvt_val;
54  }
55  }
56 
57  if (biggest == 0) {
58  // singular matrix
59  // could test against an epsilon to
60  // find ill-conditioned matrices
61  degenerate_ct += 1;
62  continue;
63  } else if (pvt != i) {
64  // swap col <i> with col <pvt>
65  for (index_t r = 0; r < rows; r++) {
66  std::swap(_MxElem(r, i),
67  _MxElem(r, pvt));
68  }
69  // make a note of the permutation in <P>
70  std::swap(reorder[i], reorder[pvt]);
71  *swap_parity = !*swap_parity;
72  }
73 
74  // eliminate lower elements
75  T a = _MxElem(i,i);
76  for (index_t r = i + 1; r < rows; r++) {
77  T b = _MxElem(r,i) / a;
78  for (index_t c = i + 1; c < cols; c++) {
79  // R_r = R_r - b * R_i
80  T src_elem = _MxElem(i,c);
81  T dst_elem = _MxElem(r,c);
82  _MxElem(r,c) = dst_elem - b * src_elem;
83  }
84  // set the lower matrix
85  _MxElem(r,i) = b;
86  }
87  }
88 
89  return degenerate_ct;
90 }

110  *swap_parity = false;
111  // fill permutation array
112  for (index_t i = 0; i < rows; i++){
113  reorder[i] = i;
114  }
115 
116  for (index_t i = 0; i < n - 1; i++) {
117  // find pivot
118  T biggest = std::abs(_MxElem(i,i));
119  index_t pvt = i;
120  for (index_t p = i + 1; p < rows; p++) {
121  T pvt_val = std::abs(_MxElem(p,i));
122  if (pvt_val > biggest) {
123  pvt = p;
124  biggest = pvt_val;
125  }
126  }
127 
128  if (biggest == 0) {
129  // singular matrix
130  // could test against an epsilon to
131  // find ill-conditioned matrices
132  degenerate_ct += 1;
133  continue;
134  } else if (pvt != i) {
135  // swap row <i> with row <pvt>
136  for (index_t c = 0; c < cols; c++) {
137  std::swap(_MxElem(i, c),
138  _MxElem(pvt, c));
139  }
140  // make a note of the permutation in <P>
141  std::swap(reorder[i], reorder[pvt]);
142  *swap_parity = !*swap_parity;
143  }
144 
145  // eliminate lower elements
146  T a = _MxElem(i,i);
147  for (index_t r = i + 1; r < rows; r++) {
148  T b = _MxElem(r,i) / a;
149  for (index_t c = i + 1; c < cols; c++) {
150  // R_r = R_r - b * R_i
151  T src_elem = _MxElem(i,c);
152  T dst_elem = _MxElem(r,c);
153  _MxElem(r,c) = dst_elem - b * src_elem;
154  }
155  // set the lower matrix
156  _MxElem(r,i) = b;
157  }
158  }

175 template <typename T>
176 void linearSolveLUP(const T* lup, const index_t* p, index_t n, T* x, const T* b) {
177  const T* m = lup; // so the macro works
178  const index_t cols = n; // ...
179 
180  // <y> and <x>'s elements are used such that
181  // y[i] is never read after x[i] is written.
182  // thus we may collapse their storage and save space:
183  T* y = x;
184 
185  // LUx = Pb
186  // (Ux) is a vector, so let's solve for it:
187  // Ly = Pb
188  y[0] = b[p[0]];
189  for (index_t r = 1; r < n; r++) {
190  y[r] = b[p[r]];
191  for (index_t c = 0; c < r; c++) {
192  y[r] -= y[c] * _MxElem(r,c);
193  }
194  }
195  // now with y, we may obtain x from:
196  // Ux = y
197  for (index_t r = n - 1; r >= 0; r--) {
198  // x[r] = y[r]; // nop; x and y are the same!
199  for (index_t c = n - 1; c > r; c--) {
200  x[r] -= x[c] * _MxElem(r,c);
201  }
202  x[r] /= _MxElem(r,r);
203  }
204 }

605  template <typename S>
606  void _inverseTranspose(S *dest) const {
607  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
608  // Here, we use the rows of our destination matrix as though they are
609  // column vectors, and the caller will transpose.
610  const index_t *p = P.getColSources();
