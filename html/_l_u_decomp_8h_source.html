60 template <typename T, bool RowMajor=true>
61 index_t decomp_lup(T* m, index_t rows, index_t cols, index_t* reorder, bool* swap_parity) {
62  const index_t n = std::min(rows, cols);
63  index_t degenerate_ct = 0;
64  *swap_parity = false;
65  // fill permutation array
66  for (index_t i = 0; i < cols; i++){
67  reorder[i] = i;
68  }
69 
70  detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
71 
72  for (index_t i = 0; i < n - 1; i++) {
73  // find pivot
74  T biggest = std::abs(mx.elem(i,i));
75  index_t pvt = i;
76  for (index_t p = i + 1; p < cols; p++) {
77  T pvt_val = std::abs(mx.elem(i, p));
78  if (pvt_val > biggest) {
79  pvt = p;
80  biggest = pvt_val;
81  }
82  }
83 
84  if (biggest == 0) {
85  // singular matrix
86  // could test against an epsilon to
87  // find ill-conditioned matrices
88  degenerate_ct += 1;
89  continue;
90  } else if (pvt != i) {
91  // swap col <i> with col <pvt>
92  for (index_t r = 0; r < rows; r++) {
93  std::swap(mx.elem(r, i),
94  mx.elem(r, pvt));
95  }
96  // make a note of the permutation in <P>
97  std::swap(reorder[i], reorder[pvt]);
98  *swap_parity = !*swap_parity;
99  }
100 
101  // eliminate lower elements
102  T a = mx.elem(i,i);
103  for (index_t r = i + 1; r < rows; r++) {
104  T b = mx.elem(r,i) / a;
105  if (b != 0) {
106  for (index_t c = i + 1; c < cols; c++) {
107  // R_r = R_r - b * R_i
108  T src_elem = mx.elem(i,c);
109  T dst_elem = mx.elem(r,c);
110  mx.elem(r,c) = dst_elem - b * src_elem;
111  }
112  }
113  // set the lower matrix
114  mx.elem(r,i) = b;
115  }
116  }
117 
118  return degenerate_ct;
119 }

137 template <typename T, bool RowMajor=true>
138 index_t decomp_plu(T* m, index_t rows, index_t cols, index_t* reorder, bool* swap_parity) {
139  const index_t n = std::min(rows, cols);
140  index_t degenerate_ct = 0;
141  *swap_parity = false;
142  // fill permutation array
143  for (index_t i = 0; i < rows; i++){
144  reorder[i] = i;
145  }
146 
147  detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
148 
149  for (index_t i = 0; i < n - 1; i++) {
150  // find pivot
151  T biggest = std::abs(mx.elem(i,i));
152  index_t pvt = i;
153  for (index_t p = i + 1; p < rows; p++) {
154  T pvt_val = std::abs(mx.elem(p,i));
155  if (pvt_val > biggest) {
156  pvt = p;
157  biggest = pvt_val;
158  }
159  }
160 
161  if (biggest == 0) {
162  // singular matrix
163  // could test against an epsilon to
164  // find ill-conditioned matrices
165  degenerate_ct += 1;
166  continue;
167  } else if (pvt != i) {
168  // swap row <i> with row <pvt>
169  for (index_t c = 0; c < cols; c++) {
170  std::swap(mx.elem(i, c),
171  mx.elem(pvt, c));
172  }
173  // make a note of the permutation in <P>
174  std::swap(reorder[i], reorder[pvt]);
175  *swap_parity = !*swap_parity;
176  }
177 
178  // eliminate lower elements
179  T a = mx.elem(i,i);
180  for (index_t r = i + 1; r < rows; r++) {
181  T b = mx.elem(r,i) / a;
182  if (b != 0) {
183  for (index_t c = i + 1; c < cols; c++) {
184  // R_r = R_r - b * R_i
185  T src_elem = mx.elem(i,c);
186  T dst_elem = mx.elem(r,c);
187  mx.elem(r,c) = dst_elem - b * src_elem;
188  }
189  }
190  // set the lower matrix
191  mx.elem(r,i) = b;
192  }
193  }
194 
195  return degenerate_ct;
196 }

215 template <typename T, bool RowMajor=true>
216 void linear_solve_lup(
217  const T* lup,
218  const index_t* p,
219  index_t n,
220  T* x,
221  const T* b,
222  index_t skip=0)
223 {
224  detail::MxWrap<const T, RowMajor> mx = {lup, n, n};
225 
226  // <y> and <x>'s elements are used such that
227  // y[i] is never read after x[i] is written.
228  // thus we may collapse their storage and save space:
229  T* y = x;
230 
231  // LUx = Pb
232  // (Ux) is a vector (call it "y"), so let's solve for it:
233  // Ly = Pb
234  y[0] = b[p[0]];
235  for (index_t r = 1; r < n; r++) {
236  y[r] = b[p[r]];
237  for (index_t c = 0; c < r; c++) {
238  y[r] -= y[c] * mx.elem(r,c);
239  }
240  }
241  // now with y, we may obtain x from:
242  // Ux = y
243  for (index_t r = n - 1; r >= skip; r--) {
244  // x[r] = y[r]; // nop; x and y are the same!
245  for (index_t c = n - 1; c > r; c--) {
246  x[r] -= x[c] * mx.elem(r,c);
247  }
248  x[r] /= mx.elem(r,r);
249  }
250 }

268 void linear_solve_lu(const T* lu, index_t n, T* x, const T* b, index_t skip=0) {
269  detail::MxWrap<const T,RowMajor> mx = {lu, n, n};
270 
271  // <y> and <x>'s elements are used such that
272  // y[i] is never read after x[i] is written.
273  // thus we may collapse their storage and save space:
274  T* y = x;
275 
276  // LUx = b
277  // (Ux) is a vector, so let's solve for it:
278  // Ly = b
279  y[0] = b[0];
280  for (index_t r = 1; r < n; r++) {
281  y[r] = b[r];
282  for (index_t c = 0; c < r; c++) {
283  y[r] -= y[c] * mx.elem(r,c);
284  }
285  }
286  // now with y, we may obtain x from:
287  // Ux = y
288  for (index_t r = n - 1; r >= skip; r--) {
289  // x[r] = y[r]; // nop; x and y are the same!
290  for (index_t c = n - 1; c > r; c--) {
291  x[r] -= x[c] * mx.elem(r,c);
292  }
293  x[r] /= mx.elem(r,r);
294  }

316 inline bool linear_solve(T* m, index_t n, T* x, const T* b, index_t skip=0) {
317  SmallStorage<index_t, 32> p(n); // probably will never have to alloc.
318  bool parity;
319  if (decomp_plu<T,RowMajor>(m, n, n, p.get(), &parity) > 0) {

323  return true;

340 template <typename T, index_t N>
341 inline bool linear_solve(Vec<T,N>* bases, Vec<T,N>* x, const Vec<T,N>& b, index_t skip=0) {
342  if (N < 5) {
343  // matrix inv is empirically faster than solve() for N < 5.
344 
345  // m is row major, but our bases would be columns.
346  // m and its inverse are thus transposed.
347  WrapperMatrix<T,N,N> m(bases[0].begin());
348  SimpleMatrix<T,N,N> m_inv_txpose;
349  if (!inv(&m_inv_txpose, m)) return false;
350  // reverse mult order to get mul by txpose:
351  *x = b * m_inv_txpose;
352  return true;
353  } else {
354  index_t p[N];
355  T* const m = bases[0].begin();
356  bool parity;
357  if (decomp_plu<T,false>(m, N, N, p, &parity) > 0) {
358  return false;
359  }
360  linear_solve_lup<T,false>(m, p, N, x->begin(), b.begin(), skip);
361  return true;
362  }
363 }

637  template <index_t J, index_t K, MatrixLayout Lyt, StoragePolicy SP>
638  void inverse(SimpleMatrix<T,J,K,Lyt,SP>* into) const {
639 
640 #ifdef GEOMC_MTX_CHECK_DIMS
641  _check_is_square();
642  // destination is valid?
643  if ((J * K == 0 or J != M or K != N) and
644  (into->rows() != LU.rows() or into->cols() != LU.cols())) {
645  throw DimensionMismatchException(
646  into->rows(),
647  into->cols(),
648  LU.rows(),
649  LU.cols());
650  }
651 #endif
652  T* dest = into->data_begin();
653 
654  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
655  // (If we transpose LU, also transpose PI)
656  const index_t* p = P.getColSources();
657  const index_t n = LU.rows();
658  std::fill(dest, dest + (n * n), 0);
659  for (index_t i = 0; i < n; i++, dest += n) {
660  dest[p[i]] = 1;
661  geom::linear_solve_lu<T>(LU.begin(), LU.rows(), dest, dest);
662  }
663  if (Lyt == ROW_MAJOR) {
664  // transpose the destination matrix
665  auto& m = *into;
666  for (index_t i = 0; i < n; i++) {
667  for (index_t j = 0; j < i; j++) {
668  std::swap(m(i,j), m(j,i));
669  }
670  }
671  }
672  }

678  T det() const {
679  const index_t n = LU.rows();
680  T k = get_parity();
681  for (index_t i = 0; i < n; i++) {

683  }
