68 template <typename T, bool RowMajor=true>
69 index_t decompLUP(T* m, index_t rows, index_t cols, index_t* reorder, bool* swap_parity) {
70  const index_t n = std::min(rows, cols);
71  index_t degenerate_ct = 0;
72  *swap_parity = false;
73  // fill permutation array
74  for (index_t i = 0; i < cols; i++){
75  reorder[i] = i;
76  }
77 
78  detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
79 
80  for (index_t i = 0; i < n - 1; i++) {
81  // find pivot
82  T biggest = std::abs(mx.elem(i,i));
83  index_t pvt = i;
84  for (index_t p = i + 1; p < cols; p++) {
85  T pvt_val = std::abs(mx.elem(i,p));
86  if (pvt_val > biggest) {
87  pvt = p;
88  biggest = pvt_val;
89  }
90  }
91 
92  if (biggest == 0) {
93  // singular matrix
94  // could test against an epsilon to
95  // find ill-conditioned matrices
96  degenerate_ct += 1;
97  continue;
98  } else if (pvt != i) {
99  // swap col <i> with col <pvt>
100  for (index_t r = 0; r < rows; r++) {
101  std::swap(mx.elem(r, i),
102  mx.elem(r, pvt));
103  }
104  // make a note of the permutation in <P>
105  std::swap(reorder[i], reorder[pvt]);
106  *swap_parity = !*swap_parity;
107  }
108 
109  // eliminate lower elements
110  T a = mx.elem(i,i);
111  for (index_t r = i + 1; r < rows; r++) {
112  T b = mx.elem(r,i) / a;
113  for (index_t c = i + 1; c < cols; c++) {
114  // R_r = R_r - b * R_i
115  T src_elem = mx.elem(i,c);
116  T dst_elem = mx.elem(r,c);
117  mx.elem(r,c) = dst_elem - b * src_elem;
118  }
119  // set the lower matrix
120  mx.elem(r,i) = b;
121  }
122  }
123 
124  return degenerate_ct;
125 }

154

156  // find pivot
157  T biggest = std::abs(mx.elem(i,i));
158  index_t pvt = i;
159  for (index_t p = i + 1; p < rows; p++) {

161  if (pvt_val > biggest) {

164  }
165  }
166

168  // singular matrix
169  // could test against an epsilon to

171  degenerate_ct += 1;
172  continue;
173  } else if (pvt != i) {
174  // swap row <i> with row <pvt>
175  for (index_t c = 0; c < cols; c++) {
176  std::swap(mx.elem(i, c),
177  mx.elem(pvt, c));
178  }
179  // make a note of the permutation in <P>
180  std::swap(reorder[i], reorder[pvt]);
181  *swap_parity = !*swap_parity;
182  }
183 
184  // eliminate lower elements

186  for (index_t r = i + 1; r < rows; r++) {
187  T b = mx.elem(r,i) / a;
188  for (index_t c = i + 1; c < cols; c++) {
189  // R_r = R_r - b * R_i
190  T src_elem = mx.elem(i,c);
191  T dst_elem = mx.elem(r,c);
192  mx.elem(r,c) = dst_elem - b * src_elem;

218 template <typename T, bool RowMajor=true>
219 void linearSolveLUP(const T* lup, const index_t* p, index_t n, T* x, const T* b, index_t skip=0) {
220  detail::MxWrap<const T, RowMajor> mx = {lup, n, n};
221 
222  // <y> and <x>'s elements are used such that
223  // y[i] is never read after x[i] is written.
224  // thus we may collapse their storage and save space:
225  T* y = x;
226 
227  // LUx = Pb
228  // (Ux) is a vector (call it "y"), so let's solve for it:
229  // Ly = Pb
230  y[0] = b[p[0]];
231  for (index_t r = 1; r < n; r++) {
232  y[r] = b[p[r]];
233  for (index_t c = 0; c < r; c++) {
234  y[r] -= y[c] * mx.elem(r,c);
235  }
236  }
237  // now with y, we may obtain x from:
238  // Ux = y
239  for (index_t r = n - 1; r >= skip; r--) {
240  // x[r] = y[r]; // nop; x and y are the same!
241  for (index_t c = n - 1; c > r; c--) {
242  x[r] -= x[c] * mx.elem(r,c);
243  }
244  x[r] /= mx.elem(r,r);
245  }
246 }

262 template <typename T, bool RowMajor=true>
263 void linearSolveLU(const T* lu, index_t n, T* x, const T* b, index_t skip=0) {
264  detail::MxWrap<const T,RowMajor> mx = {lu, n, n};
265 
266  // <y> and <x>'s elements are used such that
267  // y[i] is never read after x[i] is written.
268  // thus we may collapse their storage and save space:
269  T* y = x;
270 
271  // LUx = b
272  // (Ux) is a vector, so let's solve for it:
273  // Ly = b
274  y[0] = b[0];
275  for (index_t r = 1; r < n; r++) {
276  y[r] = b[r];
277  for (index_t c = 0; c < r; c++) {
278  y[r] -= y[c] * mx.elem(r,c);
279  }
280  }
281  // now with y, we may obtain x from:
282  // Ux = y
283  for (index_t r = n - 1; r >= skip; r--) {
284  // x[r] = y[r]; // nop; x and y are the same!
285  for (index_t c = n - 1; c > r; c--) {
286  x[r] -= x[c] * mx.elem(r,c);
287  }
288  x[r] /= mx.elem(r,r);
289  }

311  SmallStorage<index_t, 32> p(n); // probably will never have to alloc.

334 template <typename T, index_t N>
335 inline bool linearSolve(Vec<T,N>* bases, Vec<T,N>* x, const Vec<T,N>& b, index_t skip=0) {
336  if (N < 5) {
337  // matrix inv is empirically faster than solve() for N < 5.
338 
339  // m is row major, but our bases should be columns.
340  // m and its inverse are thus transposed.
341  WrapperMatrix<T,N,N> m(bases[0].begin());
342  SimpleMatrix<T,N,N> m_inv_txpose;
343  if (!inv(&m_inv_txpose, m)) return false;
344  // reverse mult order to get mul by txpose:
345  *x = b * m_inv_txpose;
346  } else {
347  index_t p[N];
348  T* const m = bases[0].begin();
349  bool parity;
350  if (decompPLU<T,false>(m, N, N, p, &parity) > 0) {
351  return false;
352  }
353  linearSolveLUP<T,false>(m, p, N, x->begin(), b.begin(), skip);
354  return true;
355  }
356 }

697  template <typename S>
698  void _inverseTranspose(S *dest) const {
699  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
700  // Here, we use the rows of our destination matrix as though they are
701  // column vectors, and the caller will transpose.
702  // todo: we can now call the templated solver and treat
703  // the input matrix as column-major directly.
704  const index_t *p = P.getColSources();
705  const index_t n = LU.rows();
706  std::fill(dest, dest + (n*n), 0);
707  for (index_t i = 0; i < n; i++, dest += n) {
708  dest[p[i]] = 1;
709  geom::linearSolveLU(LU.begin(), LU.rows(), dest, dest);
710  }
