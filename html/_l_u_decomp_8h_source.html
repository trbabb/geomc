34// apply the permutation map `p` to the n x m matrix `A`, overwriting `p` in the process
35// (after which p will be the identity permutation)
36template <typename T, bool RowMajor=true>
37void destructive_apply_permutation(index_t* p, T* A, index_t n, index_t m) {
38 detail::MxWrap<T,RowMajor> mx = {A, n, m};
39 for (index_t i = 0; i < n; i++) {
40 index_t cur = i;
41 index_t next = p[i]; // source of new element at this position
42 if (next == i) continue; // no swap necessary
43 // if (next < 0) {
44 // // marked as visited. restore original value
45 // p[i] = -(p[i]);
46 // continue;
47 // }
48 do {
49 for (index_t j = 0; j < m; ++j) {
50 // give v[cur] the correct element, v[next] gets the cycle's first element
51 std::swap(mx(cur, j), mx(next, j));
52 }
53 // ...and then proceed to find out what goes into the hole
54 cur = next;
55 next = p[next];
56 // mark this element as visited by mapping it to itself
57 // (we only need to do this if we might encounter the cycle again)
58 p[cur] = cur;
59
60 // mark as visited
61 // if (cur > i) {
62 // p[cur] = -(p[cur]);
63 // }
64 } while (next != i);
65 }
66}

103template <typename T, bool RowMajor=true>
104index_t decomp_lup(T* m, index_t rows, index_t cols, index_t* reorder, bool* parity) {
105 const index_t n = std::min(rows, cols);
106 index_t degenerate_ct = 0;
107 bool _parity = false;
108 if (parity == nullptr) parity = &_parity;
109 *parity = false;
110 // fill permutation array
111 if (reorder != nullptr) {
112 for (index_t i = 0; i < cols; i++){
113 reorder[i] = i;
114 }
115 }
116
117 detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
118
119 for (index_t i = 0; i < n - 1; i++) {
120 // find pivot
121 T biggest = std::abs(mx.elem(i,i));
122 index_t pvt = i;
123 for (index_t p = i + 1; p < cols; p++) {
124 T pvt_val = std::abs(mx.elem(i, p));
125 if (pvt_val > biggest) {
126 pvt = p;
127 biggest = pvt_val;
128 }
129 }
130
131 if (biggest == 0) {
132 // singular matrix
133 // could test against an epsilon to
134 // find ill-conditioned matrices
135 degenerate_ct += 1;
136 continue;
137 } else if (pvt != i) {
138 // swap col <i> with col <pvt>
139 for (index_t r = 0; r < rows; r++) {
140 std::swap(mx.elem(r, i),
141 mx.elem(r, pvt));
142 }
143 // make a note of the permutation in <P>
144 if (reorder != nullptr) std::swap(reorder[i], reorder[pvt]);
145 *parity = !*parity;
146 }
147
148 // eliminate lower elements
149 T a = mx.elem(i,i);
150 for (index_t r = i + 1; r < rows; r++) {
151 T b = mx.elem(r,i) / a;
152 if (b != 0) {
153 for (index_t c = i + 1; c < cols; c++) {
154 // R_r = R_r - b * R_i
155 T src_elem = mx.elem(i,c);
156 T dst_elem = mx.elem(r,c);
157 // mx[r,c] = dst_elem - b * src_elem;
158 mx.elem(r,c) = geom::multiply_add(-b, src_elem, dst_elem);
159 }
160 }
161 // set the lower matrix
162 mx.elem(r,i) = b;
163 }
164 }
165
166 return degenerate_ct;
167}

189template <typename T, bool RowMajor=true>
190index_t decomp_plu(T* m, index_t rows, index_t cols, index_t* reorder, bool* parity) {
191 const index_t n = std::min(rows, cols);
192 bool _parity = false;
193 if (parity == nullptr) parity = &_parity;
194 index_t degenerate_ct = 0;
195 *parity = false;
196 // fill permutation array
197 if (reorder) {
198 for (index_t i = 0; i < rows; i++){
199 reorder[i] = i;
200 }
201 }
202
203 detail::MxWrap<T,RowMajor> mx = {m, rows, cols};
204
205 for (index_t i = 0; i < n - 1; i++) {
206 // find pivot
207 T biggest = std::abs(mx.elem(i,i));
208 index_t pvt = i;
209 for (index_t p = i + 1; p < rows; p++) {
210 T pvt_val = std::abs(mx.elem(p,i));
211 if (pvt_val > biggest) {
212 pvt = p;
213 biggest = pvt_val;
214 }
215 }
216
217 if (biggest == 0) {
218 // singular matrix. could test against an epsilon to
219 // find ill-conditioned matrices
220 degenerate_ct += 1;
221 continue;
222 } else if (pvt != i) {
223 // swap row <i> with row <pvt>
224 for (index_t c = 0; c < cols; c++) {
225 std::swap(mx.elem(i, c),
226 mx.elem(pvt, c));
227 }
228 // make a note of the permutation in <P>
229 if (reorder) std::swap(reorder[i], reorder[pvt]);
230 *parity = !*parity;
231 }
232
233 // eliminate lower elements
234 T a = mx.elem(i,i);
235 for (index_t r = i + 1; r < rows; r++) {
236 T b = mx.elem(r,i) / a;
237 if (b != 0) {
238 for (index_t c = i + 1; c < cols; c++) {
239 // R_r = R_r - b * R_i
240 T src_elem = mx.elem(i,c);
241 T dst_elem = mx.elem(r,c);
242 // mx[r,c] = dst_elem - b * src_elem;
243 mx.elem(r,c) = geom::multiply_add(-b, src_elem, dst_elem);
244 }
245 }
246 // set the lower matrix
247 mx.elem(r,i) = b;
248 }
249 }
250
251 return degenerate_ct;
252}

272template <typename T, bool RowMajor=true>
273void backsolve_plu(
274 const T* plu,
275 const index_t* p,
276 index_t n,
277 index_t m,
278 T* x,
279 const T* b,
280 index_t skip=0)
281{
282 detail::MxWrap<const T, RowMajor> mx = {plu, n, n};
283 detail::MxWrap<const T, RowMajor> B = {b, n, m};
284 // <y> and <x>'s elements are used such that
285 // y[i] is never read after x[i] is written.
286 // thus we may collapse their storage and save space:
287 detail::MxWrap<T, RowMajor> Y = {x, n, m};
288 detail::MxWrap<T, RowMajor> X = {x, n, m};
289
290 // LUX = Pb
291 // (UX) is a matrix (call it "Y")
292 // LY = Pb
293 // so let's solve for each column `i` of Y:
294 for (index_t r = 0; r < n; r++) {
295 for (index_t i = 0; i < m; ++i) {
296 Y.elem(r,i) = B.elem(p[r], i); // y[r] <- Pb[r]
297 for (index_t c = 0; c < r; c++) {
298 Y.elem(r,i) -= Y.elem(c,i) * mx.elem(r,c);
299 }
300 }
301 }
302 // now with y, we may obtain x from:
303 // Ux = y
304 for (index_t r = n - 1; r >= skip; r--) {
305 // x[r] = y[r]; // nop; x and y are the same!
306 T k = mx.elem(r,r);
307 for (index_t i = 0; i < m; ++i) {
308 for (index_t c = n - 1; c > r; c--) {
309 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
310 }
311 X.elem(r, i) /= k;
312 }
313 }
314}

333template <typename T, bool RowMajor=true>
334void backsolve_lu(
335 const T* lu,
336 index_t n,
337 index_t m,
338 T* x,
339 index_t skip=0)
340{
341 detail::MxWrap<const T, RowMajor> mx = {lu, n, n};
342 detail::MxWrap< T, RowMajor> X = {x, n, m}; // `x` is initialized with `b`
343
344 // LUx = b
345 // (Ux) is a vector (call it "y")
346 // Ly = b
347 for (index_t r = 1; r < n; r++) {
348 // do this for each column i in x:
349 for (index_t i = 0; i < m; ++i) {
350 // x[r,i] = b[r,i] // <-- already true
351 for (index_t c = 0; c < r; c++) {
352 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
353 }
354 }
355 }
356 // now with y, we may obtain x from:
357 // Ux = y
358 for (index_t r = n - 1; r >= skip; r--) {
359 T k = mx.elem(r,r);
360 // do for each column i in y
361 for (index_t i = 0; i < m; ++i) {
362 for (index_t c = n - 1; c > r; c--) {
363 X.elem(r,i) -= X.elem(c,i) * mx.elem(r,c);
364 }
365 X.elem(r, i) /= k;
366 }
367 }
368}

389template <typename T, bool RowMajor=true>
390inline bool linear_solve(T* a, index_t n, index_t m, T* x, index_t skip=0) {
391 SmallStorage<index_t, 24> p(n); // unlikely to need an alloc
392 bool parity;
393 if (decomp_plu<T,RowMajor>(a, n, n, p.begin(), &parity) > 0) {
394 return false;
395 }
396 // pre-apply P to x
397 detail::destructive_apply_permutation<T,RowMajor>(p.begin(), x, n, m);
398 // LUx = Pb
399 backsolve_lu<T,RowMajor>(a, n, m, x, skip);
400 return true;
401}

418template <typename T, index_t N>
419inline bool linear_solve(Vec<T,N> bases[N], index_t m, Vec<T,N>* x, index_t skip=0) {
420 T* _x = x->begin();
421 if (N < 5) {
422 // matrix inv is empirically faster than solve() for N < 5.
423 WrapperMatrix<T,N,N,COL_MAJOR> mx(bases->begin());
424 WrapperMatrix<T,N,0,COL_MAJOR> X(_x, m);
425 SimpleMatrix<T,N,N> m_inv;
426 if (not inv(&m_inv, mx)) return false;
427 mul(&X, m_inv, X);
428 return true;
429 } else {
430 index_t p[N];
431 T* const mx = bases[0].begin();
432 bool _parity;
433 // LU = PM (M's rows reordered)
434 if (decomp_plu<T,false>(mx, N, N, p, &_parity) > 0) {
435 return false;
436 }
437 // pre-apply P to x
438 detail::destructive_apply_permutation<T,false>(p, _x, N, m);
439 // LUx = Pb
440 backsolve_lu<T,false>(mx, N, m, _x, skip);
441 return true;
442 }
443}
