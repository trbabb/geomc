106template <typename T>
107T det_destructive(T* m, index_t n) {
108 bool odd;
109 if (decomp_plu(m, n, n, nullptr, &odd) != 0) return 0;
110 T d = odd ? -1 : 1;
111 // det(A) is the product of its diagonals, if A is triangular
112 for (index_t i = 0; i < n; ++i) {
113 d *= m[i * i];
114 }
115 return d;
116}

128template <typename T>
129T detNxN(const T* m, index_t n) {
130 bool odd;
131 SmallStorage<T, 64> buf(m, n * n);
132 if (decomp_plu(buf.get(), n, n, nullptr, &odd) != 0) return 0;
133 T d = odd ? -1 : 1;
134 // det(A) is the product of its diagonal elems, if A is triangular
135 // and det(AB) = det(A) * det(B). So if M = LU,
136 // because `L` has unit diagonal, det(M) is just the diagonal elements of `U`.
137 for (index_t i = 0; i < n; ++i) {
138 d *= buf[i * i];
139 }
140 return d;
141}

189template <typename T, index_t N, index_t M>
190T det(const DiagMatrix<T,N,M>& m) {
191 static_assert(N * M == 0 or M == N, "Determinant only defined for square matrices.");
192 T d = 1;
193 if (m.rows() != m.cols()) return 0;
194 for (T* a = m.diagonal_begin(); a != m.diagonal_end(); ++a) {
195 d *= *a;
196 }
197 return d;
