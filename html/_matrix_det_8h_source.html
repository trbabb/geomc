116template <typename T>
117T det_destructive(T* m, index_t n) {
118 if (n == 1) return m[0];
119 bool odd;
120 if (decomp_plu<T>(m, n, n, nullptr, &odd) != 0) return 0;
121 T d = odd ? -1 : 1;
122 // det(A) is the product of its diagonals, if A is triangular
123 for (index_t i = 0; i < n; ++i) {
124 d *= m[i * i];
125 }
126 return d;
127}

139template <typename T>
140T detNxN(const T* m, index_t n) {
141 bool odd;
142 SmallStorage<T, 64> buf(m, n * n);
143 if (decomp_plu(buf.get(), n, n, nullptr, &odd) != 0) return 0;
144 T d = odd ? -1 : 1;
145 // det(A) is the product of its diagonal elems, if A is triangular
146 // and det(AB) = det(A) * det(B). So if M = LU,
147 // because `L` has unit diagonal, det(M) is just the diagonal elements of `U`.
148 for (index_t i = 0; i < n; ++i) {
149 d *= buf[i * i];
150 }
151 return d;
152}

200template <typename T, index_t N, index_t M>
201T det(const DiagMatrix<T,N,M>& m) {
202 static_assert(N * M == 0 or M == N, "Determinant only defined for square matrices.");
203 T d = 1;
204 if (m.rows() != m.cols()) return 0;
205 for (T* a = m.diagonal_begin(); a != m.diagonal_end(); ++a) {
206 d *= *a;
