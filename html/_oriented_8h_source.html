74  Rect<T,N> bounds() const {
75  // the base class implementation would work; but we specialize
76  // here for a little extra performance (avoid stupidly re-computing `d`).
77  Rect<T,N> r;
78  for (index_t axis = 0; axis < N; ++axis) {
79  // we want to test along a world cardinal axis in shape-space coordinates.
80  // but this direction is a normal, which is transformed
81  // with the inverse transpose. so we take a row of xf.mat instead of
82  // a col from xf.inv:
83  Vec<T,N> basis(xf.mat.row(axis));
84  // find the body-space extreme and transform it to world space:
85  Vec<T,N> p_hi = xf * shape.convex_support( basis);
86  Vec<T,N> p_lo = xf * shape.convex_support(-basis);
87  // update the extremes along the test axis
88  r.hi[axis] = p_hi[axis];
89  r.lo[axis] = p_lo[axis];
90  }
91  return r;
92  }
93 
95  inline Oriented<Rect<T,N>> oriented_bounds() const {
96  return xf * shape.bounds();
97  }

130  Rect<T,N> bounds() const {
131  // faster than using convex_support, somewhat surprisingly.
132  Vec<T,N> pts[2] = {shape.lo, shape.hi};
133  Vec<T,N> lo = xf * shape.lo;
134  Vec<T,N> hi = lo;
135 
136  // for each corner
137  for (index_t i = 1; i < (1 << N); i++) {
138  Vec<T,N> p;
139  // for each coordinate
140  for (index_t j = 0; j < N; j++) {
141  p[j] = pts[(i & (1 << j)) != 0][j];
142  }
143  p = xf * p;
144  lo = std::min(lo, p);
145  hi = std::max(hi, p);
146  }
147 
148  return Rect<T,N>(lo, hi);
149  }

154  }

157  Vec<T,N> convex_support(Vec<T,N> d) const {
158  Vec<T,N> d_body = xf.applyInverseNormal(d);
159  Vec<T,N> o;
160  for (index_t i = 0; i < N; i++) {
161  o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
162  }
163  return xf * o;
164  }

192  inline bool intersects(const Oriented< Rect<T,N> >& b1) {
193  // we will make ourselves axis-aligned; this is the same as applying
194  // the inverse of our xf. We apply the inverse of xf to b1 too,
195  // to preserve the relationhip between the us. From this, we
196  // fallback to an ORect <-> Rect test.
197  Oriented< Rect<T,N> > b1_in_b0 = b1 / xf;
198 
199  return b1_in_b0.intersects(shape);
200  }

212  Ray<T,N> rbody = r / xf;
213  Hit<T,N> h = shape.trace(rbody, side);
214  if (h.hit) {
215  h.p = xf * h.p;

217  }
