38template <typename Shape>
39class Oriented:
40 public Convex <typename Shape::elem_t, Shape::N, Oriented<Shape>>,
41 public RayIntersectable<typename Shape::elem_t, Shape::N, Oriented<Shape>>
42{
43public:
45 typedef typename Shape::elem_t T;
47 static constexpr size_t N = Shape::N;
48
50 Shape shape;
52 AffineTransform<T,N> xf;
53
55 Oriented() {}
56
58 explicit Oriented(const Shape& s):shape(s) {}
59
61 Oriented(const Shape& s, const AffineTransform<T,N>& xf):
62 shape(s),
63 xf(xf) {}
64
65 bool operator==(const Oriented& other) const {
66 return shape == other.shape && xf == other.xf;
67 }
68
70 bool contains(Vec<T,N> p) const {
71 p = p / xf;
72 return shape.contains(p);
73 }
74
75 Vec<T,N> convex_support(Vec<T,N> d) const {
76 d = xf.apply_inverse_normal(d);
77 Vec<T,N> p = shape.convex_support(d);
78 return xf * p;
79 }
80
81 Rect<T,N> bounds() const {
82 // the base class implementation would work; but we specialize
83 // here for a little extra performance (avoid stupidly re-computing `d`).
84 Rect<T,N> r;
85 for (index_t axis = 0; axis < N; ++axis) {
86 // we want to test along a world cardinal axis in shape-space coordinates.
87 // but this direction is a normal, which is transformed
88 // with the inverse transpose. so we take a row of xf.mat instead of
89 // a col from xf.inv:
90 Vec<T,N> basis(xf.mat.row(axis));
91 // find the body-space extreme and transform it to world space:
92 Vec<T,N> p_hi = xf * shape.convex_support( basis);
93 Vec<T,N> p_lo = xf * shape.convex_support(-basis);
94 // update the extremes along the test axis
95 r.hi[axis] = p_hi[axis];
96 r.lo[axis] = p_lo[axis];
97 }
98 return r;
99 }
100
102 Rect<T,1> intersect(const Ray<T,N>& r) const {
103 return shape.intersect(r / xf);
104 }
105
107 inline Oriented<Rect<T,N>> oriented_bounds() const {
108 return xf * shape.bounds();
109 }

112}; // class Oriented

119template <typename T, index_t N>
120class Oriented<Rect<T,N>>:
121 public Convex<T,N,Oriented<Rect<T,N>>>,
122 public RayIntersectable<T,N,Oriented<Rect<T,N>>>
123{
124public:
125
127 Rect<T,N> shape;
129 AffineTransform<T,N> xf; // moves object points to world positions
130
132 Oriented() {}
133
135 explicit Oriented(const Rect<T,N>& box):shape(box) {}
136
138 Oriented(const Rect<T,N>& box, const AffineTransform<T,N>& xf):
139 shape(box),
140 xf(xf) {}
141
142 bool operator==(const Oriented& other) const {
143 return shape == other.shape && xf == other.xf;
144 }
145
147 Rect<T,N> bounds() const {
148 // faster than using convex_support, somewhat surprisingly.
149 Vec<T,N> pts[2] = {shape.lo, shape.hi};
150 Vec<T,N> lo = xf * shape.lo;
151 Vec<T,N> hi = lo;
152
153 // for each corner
154 for (index_t i = 1; i < (1 << N); i++) {
155 Vec<T,N> p;
156 // for each coordinate
157 for (index_t j = 0; j < N; j++) {
158 p[j] = pts[(i & (1 << j)) != 0][j];
159 }
160 p = xf * p;
161 lo = std::min(lo, p);
162 hi = std::max(hi, p);
163 }
164
165 return Rect<T,N>(lo, hi);
166 }
167
169 inline Oriented< Rect<T,N> > oriented_bounds() const {
170 return *this;
171 }
172
173 Vec<T,N> convex_support(Vec<T,N> d) const {
174 Vec<T,N> d_body = xf.apply_inverse_normal(d);
175 Vec<T,N> o;
176 for (index_t i = 0; i < N; i++) {
177 o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
178 }
179 return xf * o;
180 }
181
186 bool contains(Vec<T,N> p) const {
187 p = p / xf; // world coord to object coord
188 return shape.contains(p);
189 }
190
197 bool intersects(const Rect<T,N>& r) const {
198 return detail::RectIntersector<T,N>::intersect(*this, r);
199 }
200
207 bool intersects(const Oriented< Rect<T,N> >& b1) {
208 // we will make ourselves axis-aligned; this is the same as applying
209 // the inverse of our xf. We apply the inverse of xf to b1 too,
210 // to preserve the relationhip between the us. From this, we
211 // fallback to an ORect <-> Rect test.
212 Oriented< Rect<T,N> > b1_in_b0 = b1 / xf;
213
214 return b1_in_b0.intersects(shape);
215 }
216
218 Rect<T,1> intersect(const Ray<T,N>& r) const {
219 return shape.intersect(r / xf);
220 }
221
223 T volume() const {
224 SimpleMatrix<T,N,N> mx;
225 for (index_t i = 0; i < N; ++i) {
226 // mult this basis vector by the length of the rect along that axis
227 T s = shape.hi[i] - shape.lo[i];
228 for (index_t j = 0; j < N; ++j) {
229 mx(j,i) = s * xf.mat(j,i);
230 }
231 }
232 return det_destructive(mx.data_begin(), N);
233 }
