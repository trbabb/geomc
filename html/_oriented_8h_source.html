39template <typename Shape>
40class Oriented:
41 public Convex <typename Shape::elem_t, Shape::N, Oriented<Shape>>,
42 public RayIntersectable<typename Shape::elem_t, Shape::N, Oriented<Shape>>
43{
44public:
46 typedef typename Shape::elem_t T;
48 static constexpr size_t N = Shape::N;
49
51 Shape shape;
53 AffineTransform<T,N> xf;
54
56 Oriented() {}
57
59 Oriented(const Shape& s):shape(s) {}
60
62 Oriented(const Shape& s, const AffineTransform<T,N>& xf):
63 shape(s),
64 xf(xf) {}
65
67 bool contains(Vec<T,N> p) const {
68 p = p / xf;
69 return shape.contains(p);
70 }
71
72 Vec<T,N> convex_support(Vec<T,N> d) const {
73 d = xf.apply_inverse_normal(d);
74 Vec<T,N> p = shape.convex_support(d);
75 return xf * p;
76 }
77
78 Rect<T,N> bounds() const {
79 // the base class implementation would work; but we specialize
80 // here for a little extra performance (avoid stupidly re-computing `d`).
81 Rect<T,N> r;
82 for (index_t axis = 0; axis < N; ++axis) {
83 // we want to test along a world cardinal axis in shape-space coordinates.
84 // but this direction is a normal, which is transformed
85 // with the inverse transpose. so we take a row of xf.mat instead of
86 // a col from xf.inv:
87 Vec<T,N> basis(xf.mat.row(axis));
88 // find the body-space extreme and transform it to world space:
89 Vec<T,N> p_hi = xf * shape.convex_support( basis);
90 Vec<T,N> p_lo = xf * shape.convex_support(-basis);
91 // update the extremes along the test axis
92 r.hi[axis] = p_hi[axis];
93 r.lo[axis] = p_lo[axis];
94 }
95 return r;
96 }
97
99 Rect<T,1> intersect(const Ray<T,N>& r) const {
100 return shape.intersect(r / xf);
101 }
102
104 inline Oriented<Rect<T,N>> oriented_bounds() const {
105 return xf * shape.bounds();
106 }

116template <typename T, index_t N>
117class Oriented<Rect<T,N>>:
118 public Convex<T,N,Oriented<Rect<T,N>>>,
119 public RayIntersectable<T,N,Oriented<Rect<T,N>>>
120{
121public:
122
124 Rect<T,N> shape;
126 AffineTransform<T,N> xf; // moves object points to world positions
127
129 Oriented() {}
130
132 Oriented(const Rect<T,N>& box):shape(box) {}
133
135 Oriented(const Rect<T,N>& box, const AffineTransform<T,N>& xf):
136 shape(box),
137 xf(xf) {}
138
139
141 Rect<T,N> bounds() const {
142 // faster than using convex_support, somewhat surprisingly.
143 Vec<T,N> pts[2] = {shape.lo, shape.hi};
144 Vec<T,N> lo = xf * shape.lo;
145 Vec<T,N> hi = lo;
146
147 // for each corner
148 for (index_t i = 1; i < (1 << N); i++) {
149 Vec<T,N> p;
150 // for each coordinate
151 for (index_t j = 0; j < N; j++) {
152 p[j] = pts[(i & (1 << j)) != 0][j];
153 }
154 p = xf * p;
155 lo = std::min(lo, p);
156 hi = std::max(hi, p);
157 }
158
159 return Rect<T,N>(lo, hi);
160 }
161
163 inline Oriented< Rect<T,N> > oriented_bounds() const {
164 return *this;
165 }
166
167 Vec<T,N> convex_support(Vec<T,N> d) const {
168 Vec<T,N> d_body = xf.apply_inverse_normal(d);
169 Vec<T,N> o;
170 for (index_t i = 0; i < N; i++) {
171 o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
172 }
173 return xf * o;
174 }
175
180 inline bool contains(Vec<T,N> p) const {
181 p = p / xf; // world coord to object coord
182 return shape.contains(p);
183 }
184
191 inline bool intersects(const Rect<T,N>& r) {
192 return detail::RectIntersector<T,N>::intersect(*this, r);
193 }
194
201 inline bool intersects(const Oriented< Rect<T,N> >& b1) {
202 // we will make ourselves axis-aligned; this is the same as applying
203 // the inverse of our xf. We apply the inverse of xf to b1 too,
204 // to preserve the relationhip between the us. From this, we
205 // fallback to an ORect <-> Rect test.
206 Oriented< Rect<T,N> > b1_in_b0 = b1 / xf;
207
208 return b1_in_b0.intersects(shape);
209 }
210
212 Rect<T,1> intersect(const Ray<T,N>& r) const {
213 return shape.intersect(r / xf);
214 }
215
217 T volume() const {
218 SimpleMatrix<T,N,N> mx;
219 for (index_t i = 0; i < N; ++i) {
220 // mult this basis vector by the length of the rect along that axis
221 T s = shape.hi[i] - shape.lo[i];
222 for (index_t j = 0; j < N; ++j) {
223 mx(j,i) = s * xf.mat(j,i);
224 }
225 }
226 return det_destructive(mx.data_begin(), N);
227 }
