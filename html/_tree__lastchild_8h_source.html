349  item_iterator next_item = m_items.erase(item);
350 
351  // first, we must update the item count of all direct ancestors.
352  // second, if the deleted item is a "boundary", it may be a boundary for some
353  // ancestors as well. those ancestors must be updated to refer to the new boundary.
354  NodeRef cur_node = parent;
355  while (cur_node != m_nodes.end()) {
356  --cur_node->n_items;
357  if (cur_node->items_begin == item) {
358  cur_node->items_begin = next_item
359  }
360  if (cur_node->items_end == item) {
361  cur_node->items_end = next_item;
362  }
363  cur_node = cur_node->parent;
364  }
365 
366  // the deleted item may be the last_item for left siblings of `parent`,
367  // and therefore the last_item of their ancestors also,
368  // and also the first_item for those with no children.
369 
370  // for each sibling of `parent`
371  NodeRef aunt = parent;
372  while (true) {
373  bool aunt_is_edge = false;
374  cur_node = --aunt;
375 
376  // ascend the chain of ancestors, removing references to the deleted item.
377  while (cur_node != m_nodes.end()) {
378  bool cur_is_edge = false;
379

381  cur_is_edge = true;
382  cur_node->items_begin = next_item;
383  }
384  if (cur_node->items_end == item) {
385  cur_is_edge = true;
386  cur_node->items_end = next_item;
387  }
388
