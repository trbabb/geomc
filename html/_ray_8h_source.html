87  T dist2FromAxisTo(const Vec<T,3> &p) const {
88  Vec<T,N> b = p - origin;
89  T b2 = b.mag2();
90  T d = direction.dot(b);
91  return b2 - (d * d / b2);

98  Vec<T,N> directionFromAxisTo(const Vec<T,N> &p) const {
99  Vec<T,N> b = p - origin;
100  T d = b.dot(direction);
101  return b - (d / direction.mag2()) * direction;
102  }
