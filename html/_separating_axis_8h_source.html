82  /********** N=3 Axis Iterator **********/
83 
84  template <typename T>
85  class RectAxisHelper<T,3> : public RectAxisHelperBase<T,3> {
86  private:
87 
88  typedef RectAxisHelperBase<T,3> base_t;
89 
90  public:
91 
92  static const bool SAT_SUPPORTED = true;
93 
94  RectAxisHelper(const AffineTransform<T,3> &xf) : base_t(xf) {}
95 
96  // we don't include the canonical axes, because we handle
97  // them separately for speed. A more user-facing implementation
98  // would iterate over all 15 axes.
99 
100  Vec<T,3> axis() {
101  index_t i = base_t::i;
102  if (i < 3) {
103  // test each of the principal axes in the oriented box
104  return this->_getNormal(i);
105  } else { // we are assuming `not done()`
106  // test all combinations of principal axes between
107  // the two boxes. We assume one is axis-aligned.
108  Vec<T,3> b0_axis;
109  Vec<T,3> b1_axis = this->_getXfAxis(i % 3);
110  b0_axis[i / 3 - 1] = 1;
111  return b0_axis ^ b1_axis;
112  }
113  }
114 
115  inline bool done() {
116  return base_t::i >= 12; // 3 Xf axes, plus 3 * 3 axis combinations.
117  }
118  };

150  template <typename T, index_t N, typename Enable=void>
151  class RectIntersector {
152 
153  static inline bool intersect(const OrientedRect<T,N> &b0, const Rect<T,N> &b1) {
154  return gjk_intersect(b0, b1);
155  }
156  };
157 
158  template <typename T, index_t N>
159  class RectIntersector<T, N, typename boost::enable_if_c<RectAxisHelper<T,N>::SAT_SUPPORTED>::type> {
160  public:
161 
162  static bool intersect(const OrientedRect<T,N> &b0, const Rect<T,N> &b1) {
163  // here we apply the separating axis theorem.
164  // all edge axes, face normals, and edge-edge normals must be tested.
165  // the first two cases are one and the same; they are the principal
166  // axes of each rect; amounting to 2N tests. The third case is
167  // 9 axes for N=3 and 0 axes for N=2. Other dimensions may be
168  // supported by extending detail::RectAxisHelper to higher N.
169  const static index_t n_corners = 1 << N;
170  Vec<T,N> b0_pts[n_corners];
171  Vec<T,N> b1_pts[n_corners];
172  Vec<T,N> b0_body_extreme[2] = { b0.box.min(), b0.box.max() };
173  Vec<T,N> b1_body_extreme[2] = { b1.min(), b1.max() };
174 
175  // compute world-space points
176  for (index_t c = 0; c < n_corners; c++) {
177  for (index_t i = 0; i < N; i++) {
178  int min_or_max = (c & (1 << i)) != 0;
179  b0_pts[c][i] = b0_body_extreme[min_or_max][i];
180  b1_pts[c][i] = b1_body_extreme[min_or_max][i];
181  }
182  b0_pts[c] = b0.xf * b0_pts[c];
183  }
184 
185  // compare along world (i.e. b1's) axes
186  // special case because no projection needed.
187  for (index_t i = 0; i < N; i++) {
188  T lo = std::numeric_limits<T>::max();
189  T hi = std::numeric_limits<T>::lowest();
190  for (index_t j = 0; j < n_corners; j++) {
191  lo = std::min(lo, b0_pts[j][i]);
192  hi = std::max(hi, b0_pts[j][i]);
193  }
194  // if no overlap on this axis, we've found a separating axis.
195  if (lo >= b1.max()[i] or hi < b1.min()[i]) return false;
196  }
197 
198  // now test all the remaining axes. For N=2, this is simply
199  // the two principal axes of the oriented box. For N=3, we must also
200  // test each axis which is the cross product of two principal axes,
201  // for up to 12 additional axis tests (we have covered N already).
202  for (detail::RectAxisHelper<T,N> helper(b0.xf); not helper.done(); helper.next()) {
203  Vec<T,N> axis = helper.axis();
204  if (axis == Vec<T,N>::zeros) continue; // redundant axis
205  Rect<T,1> b0_range, b1_range;
206  b0_range = b1_range = Rect<T,1>(std::numeric_limits<T>::max(),
207  std::numeric_limits<T>::lowest());
208  for (index_t corner = 0; corner < n_corners; corner++) {
209  T b0_proj = axis.dot(b0_pts[corner]);
210  T b1_proj = axis.dot(b1_pts[corner]);
211  b0_range |= b0_proj;
212  b1_range |= b1_proj;
213  }
214  // no overlap on this axis; separating axis found.
215  if (not b0_range.intersects(b1_range)) return false;
216  }
217 
218  // overlap on all candidate axes.
219  return true;
220  }
221 
222  };
