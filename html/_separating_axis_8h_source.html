14/************************************************************
15 * For small dimensions, we use the separating axis theorem *
16 * to perform box-box intersection tests. The number of *
17 * axes we must test and the strategy for choosing them *
18 * changes based on the dimension; these classes are *
19 * specialized for those dimensions. The cases get far more *
20 * complicated as N increases; only N=2 and N=3 are *
21 * implemented. *
22 ************************************************************/

30 * Iterator base class *

33template <typename T, index_t N>
34class RectAxisHelperBase {
35
36 public:
37
38 const AffineTransform<T,N>& xf;
39 index_t i;
40
41 RectAxisHelperBase(const AffineTransform<T,N>& xf) : xf(xf), i(0) {}
42
43 inline void next() {
44 i += 1;
45 }
46
47 protected:
48
49 Vec<T,N> _getXfAxis(index_t j) {
50 Vec<T,N> a;
51 for (index_t k = 0; k < N; k++) {
52 a[k] = xf.mat[k][j];
53 }
54 return a;
55 }
56
57
58 Vec<T,N> _getNormal(index_t j) {
59 Vec<T,N> n;
60 n[j] = 1;
61 return xf.apply_normal(n);
62 }
63
64};

83/********** N=3 Axis Iterator **********/
84
85template <typename T>
86class RectAxisHelper<T,3> : public RectAxisHelperBase<T,3> {
87 private:
88
89 typedef RectAxisHelperBase<T,3> base_t;
90
91 public:
92
93 static const bool SAT_SUPPORTED = true;
94
95 RectAxisHelper(const AffineTransform<T,3>& xf) : base_t(xf) {}
96
97 // we don't include the canonical axes, because we handle
98 // them separately for speed. A more user-facing implementation
99 // would iterate over all 15 axes.
100
101 Vec<T,3> axis() {
102 index_t i = base_t::i;
103 if (i < 3) {
104 // test each of the principal axes in the oriented box
105 return this->_getNormal(i);
106 } else { // we are assuming `not done()`
107 // test all combinations of principal axes between
108 // the two boxes. We assume one is axis-aligned.
109 Vec<T,3> b0_axis;
110 Vec<T,3> b1_axis = this->_getXfAxis(i % 3);
111 b0_axis[i / 3 - 1] = 1;
112 return b0_axis ^ b1_axis;
113 }
114 }
115
116 inline bool done() {
117 return base_t::i >= 12; // 3 Xf axes, plus 3 * 3 axis combinations.
118 }
119};

146/***********************************************************
147 * Intersector class. Use separating axis theorem if *
148 * supported; else gjk. *
149 ***********************************************************/
150
151template <typename T, index_t N, typename Enable=void>
152class RectIntersector {
153
154 static inline bool intersect(const Transformed<Rect<T,N> >& b0, const Rect<T,N>& b1) {
155 return gjk_intersect(b0, b1);
156 }
157};
158
159template <typename T, index_t N>
160class RectIntersector<T, N,
161 typename std::enable_if<RectAxisHelper<T,N>::SAT_SUPPORTED>::type>
162{
163 public:
164
165 static bool intersect(const Transformed<Rect<T,N> >& b0, const Rect<T,N>& b1) {
166 // here we apply the separating axis theorem.
167 // all edge axes, face normals, and edge-edge normals must be tested.
168 // the first two cases are one and the same; they are the principal
169 // axes of each rect; amounting to 2N tests. The third case is
170 // 9 axes for N=3 and 0 axes for N=2. Other dimensions may be
171 // supported by extending detail::RectAxisHelper to higher N,
172 // though this is unlikely to be simpler than simply falling back to GJK.
173 constexpr index_t n_corners = 1 << N;
174 Vec<T,N> b0_pts[n_corners];
175 Vec<T,N> b1_pts[n_corners];
176 Vec<T,N> b0_body_extreme[2] = { b0.shape.lo, b0.shape.hi };
177 Vec<T,N> b1_body_extreme[2] = { b1.lo, b1.hi };
178
179 // compute world-space points
180 for (index_t c = 0; c < n_corners; c++) {
181 for (index_t i = 0; i < N; i++) {
182 int min_or_max = (c & (1 << i)) != 0;
183 b0_pts[c][i] = b0_body_extreme[min_or_max][i];
184 b1_pts[c][i] = b1_body_extreme[min_or_max][i];
185 }
186 b0_pts[c] = b0.xf * b0_pts[c];
187 }
188
189 // compare along world (i.e. b1's) axes.
190 // special case because no projection needed.
191 // this is basically testing if the two AABBs overlap.
192 for (index_t i = 0; i < N; i++) {
193 T lo = std::numeric_limits<T>::max();
194 T hi = std::numeric_limits<T>::lowest();
195 for (index_t j = 0; j < n_corners; j++) {
196 lo = std::min(lo, b0_pts[j][i]);
197 hi = std::max(hi, b0_pts[j][i]);
198 }
199 // if no overlap on this axis, we've found a separating axis.
200 if (lo > b1.hi[i] or hi < b1.lo[i]) return false;
201 }
202
203 // now test all the remaining axes. For N=2, this is simply
204 // the two principal axes of the oriented box. For N=3, we must also
205 // test each axis which is the cross product of two principal axes,
206 // for up to 12 additional axis tests (we have covered N already).
207 for (detail::RectAxisHelper<T,N> helper(b0.xf); not helper.done(); helper.next()) {
208 Vec<T,N> axis = helper.axis();
209 if (axis == Vec<T,N>::zeros) continue; // redundant axis
210 Rect<T,1> b0_range, b1_range;
211 b0_range = b1_range = Rect<T,1>(std::numeric_limits<T>::max(),
212 std::numeric_limits<T>::lowest());
213 for (index_t corner = 0; corner < n_corners; corner++) {
214 T b0_proj = axis.dot(b0_pts[corner]);
215 T b1_proj = axis.dot(b1_pts[corner]);
216 b0_range |= b0_proj;
217 b1_range |= b1_proj;
218 }
219 // no overlap on this axis; separating axis found.
220 if (not b0_range.intersects(b1_range)) return false;
221 }
222
223 // overlap on all candidate axes.
224 return true;
225 }
226
227};
