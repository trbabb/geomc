23class Sphere:

91 inline point_t project(point_t p) const {

101 Rect<T,1> intersect(const Ray<T,N>& ray) const {
102 T r2 = r * r;
103 Vec<T,N> dir = ray.direction;
104 Vec<T,N> x0 = center - ray.origin;
105 // solve for s such that ||s * ray - ctr|| == radius
106 T a = dir.mag2();
107 T b = -2 * dir.dot(x0);
108 T c = x0.mag2() - r2;
109 T roots[2];
110 if (quadratic_solve(roots, a, b, c)) {
111 return Rect<T,1>::spanning_corners(roots[0], roots[1]);
112 } else {
113 // empty interval
114 return Rect<T,1>();
115 }
116 }
