105 Rect<T,1> intersect(const Ray<T,N>& ray) const {
106 T r2 = r * r;
107 Vec<T,N> dir = ray.direction;
108 Vec<T,N> x0 = center - ray.origin;
109 // solve for s such that ||s * ray - ctr|| == radius
110 T a = dir.mag2();
111 T b = -2 * dir.dot(x0);
112 T c = x0.mag2() - r2;
113 T roots[2];
114 if (quadratic_solve(roots, a, b, c)) {
115 return Rect<T,1>::spanning_corners(roots[0], roots[1]);
116 } else {
117 // empty interval
118 return Rect<T,1>();
