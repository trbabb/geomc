106 Vec<T,N> x0 = center - ray.origin;
107 // solve for s such that ||s * ray - ctr|| == radius
108 T a = dir.mag2();
109 T b = -2 * dir.dot(x0);
110 T c = x0.mag2() - r2;
111 T roots[2];
112 if (quadratic_solve(roots, a, b, c)) {
