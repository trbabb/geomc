84  Hit<T,N> trace(const Ray<T,N> &ray, HitSide side) const {
85  Hit<T,N> h = Hit<T,N>(ray, side); // defaults to miss
86  T r2 = r*r;
87  // if inside, we are guaranteed to hit the back. return miss if not tracing backface.
88  if ((ray.origin.dist2(center) < r2) and !(side & HIT_BACK)) return h;
89  const Vec<T,N> &dir = ray.direction; //for shorthand
90  Vec<T,N> x0 = center - ray.origin;
91  // solve for s such that ||s*ray - ctr|| == radius
92  T a = dir.dot(dir);
93  T b = -2*dir.dot(x0);
94  T c = x0.dot(x0) - r2;
95  T roots[2];
96  if (quadratic_solve(roots,a,b,c)) {
97  T s;
98  if (detail::chooseRayHit(&s, roots, &side)) {
99  // successful hit
100  h.p = ray.atMultiple(s);
101  h.n = (h.p - center).unit();
102  h.s = s;
103  h.side = side;
104  h.hit = true;
105  return h;
106  }
107  } else {
108  // no intersection; return miss.
109  return h;
110  }
111  }

129  bool trace_sphere(const Ray<T,N> &ray, T *s0, T *s1, const Vec<T,N> &center, T r) {
130  T r2 = r*r;
131  const Vec<T,N> &dir = ray.direction; //for shorthand
132  Vec<T,N> x0 = center - ray.origin;
133  // solve for s such that ||s*ray - ctr|| == radius
134  T a = dir.dot(dir);
135  T b = -2 * dir.dot(x0);
136  T c = x0.dot(x0) - r2;
137  T roots[2];
138  if (quadratic_solve(roots,a,b,c)) {
139  // order the roots along the ray, from -inf; s0 first.
140  if (roots[0] < roots[1]) {
141  *s0 = roots[0];
142  *s1 = roots[1];
143  } else {
144  *s0 = roots[1];
145  *s1 = roots[0];
146  }
147  return true;
148  }
149  return false;
150  }
