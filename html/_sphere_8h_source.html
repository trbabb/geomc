26template <typename T>
27constexpr T measure_sphere_boundary(index_t d, T r) {
28 constexpr T k = 2 * std::numbers::pi_v<T>;
29 if (d == 0) return 1;
30 if (d == 1) return 2;
31 if (d == 2) return k * r;
32 return k * r * r * measure_sphere_boundary(d - 2, r) / (T) (d - 2);
33}

130 Rect<T,1> intersect(const Ray<T,N>& ray) const {
131 T r2 = r * r;
132 Vec<T,N> dir = ray.direction;
133 Vec<T,N> x0 = center - ray.origin;
134 // solve for s such that ||s * ray - ctr|| == radius
135 T a = dir.mag2();
136 T b = -2 * dir.dot(x0);
137 T c = x0.mag2() - r2;
138 T roots[2];
139 if (quadratic_solve(roots, a, b, c)) {
140 return Rect<T,1>::from_corners(roots[0], roots[1]);
141 } else {
142 // empty interval
143 return Rect<T,1>();
144 }
145 }
