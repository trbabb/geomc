55 static Rotation<T,2> align_vectors(const Vec<T,2>& dir, const Vec<T,2>& align_with) {
56 // numerically stable formula from https://scicomp.stackexchange.com/a/27769
57 Vec<T,2> a = align_with.unit();
58 Vec<T,2> b = dir.unit();
59 T d_a = 2. * std::atan2((a + b).mag(), (a - b).mag());
