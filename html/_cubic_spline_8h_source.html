15 if constexpr (N > 1) {
16 return S {

49template <typename S, typename T, index_t N>
50concept CubicSplineObject = requires (S s, const VecType<T,N> control_pts[4]) {
51 S(control_pts);
52 S(control_pts[0], control_pts[1], control_pts[2], control_pts[3]);
53 { S::basis() } -> std::convertible_to<const SimpleMatrix<T,4,4>&>;
54 { S::inverse_basis() } -> std::convertible_to<const SimpleMatrix<T,4,4>&>;
55 { s.control_points() } -> std::convertible_to<const VecType<T,N>*>;
56 { s.bounds() } -> std::same_as<Rect<T,N>>;
57 { s.velocity(T(0)) } -> std::same_as<VecType<T,N>>;
58 { s(T(0)) } -> std::same_as<VecType<T,N>>;
59};

72template <typename T, index_t N>
73class PolynomialSpline : public Dimensional<T, N> {
74public:
76 VecType<T,N> k0;
78 VecType<T,N> k1;
80 VecType<T,N> k2;
82 VecType<T,N> k3;
83
84 static constexpr SimpleMatrix<T,4,4> ControlToCoefficients = {
85 { 1, 0, 0, 0},
86 { 0, 1, 0, 0},
87 { 0, 0, 1, 0},
88 { 0, 0, 0, 1},
89 };
90
91 static constexpr SimpleMatrix<T,4,4> CoefficientsToControl = {
92 { 1, 0, 0, 0},
93 { 0, 1, 0, 0},
94 { 0, 0, 1, 0},
95 { 0, 0, 0, 1},
96 };
97
98 constexpr PolynomialSpline() {
99 k1[0] = 1;
100 }
101
103 constexpr PolynomialSpline(VecType<T,N> k0, VecType<T,N> k1, VecType<T,N> k2, VecType<T,N> k3):
104 k0(k0), k1(k1), k2(k2), k3(k3) {}
105
106 constexpr PolynomialSpline(const VecType<T,N> pts[4]):
107 k0(pts[0]), k1(pts[1]), k2(pts[2]), k3(pts[3]) {}
108
110 template <CubicSplineObject<T,N> Spline>
111 constexpr operator Spline() const {
112 WrapperMatrix<T,4,N> b = {PointType<T,N>::iterator(control_points()[0])};
113 // (4 x 4) * (4 x N) = (4 x N)
114 SimpleMatrix<T,4,N> coeffs = Spline::inverse_basis() * b;
115 return detail::_make_spline<T,N,Spline>(coeffs);
116 }
117
119 constexpr const VecType<T,N>* control_points() const { return &k0; }
121 constexpr VecType<T,N>* control_points() { return &k0; }
122
124 static constexpr SimpleMatrix<T,4,4> basis() {
125 return ControlToCoefficients;
126 }
127
129 static constexpr SimpleMatrix<T,4,4> inverse_basis() {
130 return CoefficientsToControl;
131 }
132
134 Rect<T,N> bounds() const {
135 Rect<T,N> bounds = Rect<T,N>::from_corners(
136 k0, // s = 0 point
137 k0 + k1 + k2 + k3 // s = 1 point
138 );
139 // look for inflection points between the endpoints.
140 // where does each coordinate reach an extreme?
141 for (index_t axis = 0; axis < N; ++axis) {
142 Vec<T,4> k = {
143 coord(k0, axis),
144 coord(k1, axis),
145 coord(k2, axis),
146 coord(k3, axis)
147 };
148 // find the zeroes of the derivative
149 T results[2];
150 bool ok = geom::quadratic_solve(results, k[3] * 3, k[2] * 2, k[1]);
151 if (ok) {
152 for (index_t i = 0; i < 2; ++i) {
153 T s = results[i];
154 if (s > 0 and s < 1) {
155 T s2 = s * s;
156 T v = k.dot({1, s, s2, s2 * s});
157 coord(bounds.lo, axis) = std::min(bounds.lo, v);
158 coord(bounds.hi, axis) = std::max(bounds.hi, v);
159 }
160 }
161 }
162 }
163 return bounds;
164 }
165
167 constexpr VecType<T,N> operator()(T s) const {
168 T s2 = s * s;
169 return k0 + k1 * s + k2 * s2 + k3 * s2 * s;
170 }
171
173 constexpr VecType<T,N> velocity(T s) const {
174 return k1 + k2 * 2 * s + k3 * 3 * s * s;
175 }
176
178 constexpr VecType<T,N> acceleration(T s) const {
179 return k2 * 2 + k3 * 6 * s;
180 }
181
182};

188class CubicSpline : public Dimensional<T,N> {
189

193public:

196 template <CubicSplineObject<T,N> Spline>
197 requires (not std::same_as<Spline, PolynomialSpline<T,N>>)
198 constexpr operator Spline() const {
199 using ptype = PointType<T,N>;
200 WrapperMatrix<T,4,N> b = {ptype::iterator(derived().control_points()[0])};
201 // (4 x 4) * (4 x N) = (4 x N)
202 SimpleMatrix<T,4,N> coeffs = Spline::inverse_basis() * Derived::basis() * b;
203 return detail::_make_spline<T,N,Spline>(coeffs);

208 using ptype = PointType<T,N>;
209 WrapperMatrix<T,4,N> b = {ptype::iterator(derived().control_points()[0])};
210 SimpleMatrix<T,4,N> coeffs = Derived::basis() * b;
211 return detail::_make_spline<T,N,PolynomialSpline<T,N>>(coeffs);
212 }
213
215 constexpr Vec<T,N> operator()(T s) const {
216 PolynomialSpline<T,N> poly {*this};
217 return poly(s);
218 }
219
221 constexpr Vec<T,N> velocity(T s) const {
222 PolynomialSpline<T,N> poly {*this};
223 return poly.velocity(s);
224 }
225
227 constexpr Vec<T,N> acceleration(T s) const {
228 PolynomialSpline<T,N> poly {*this};
229 return poly.acceleration(s);
230 }

233 constexpr Rect<T,N> bounds() const {
234 PolynomialSpline<T,N> poly {*this};
