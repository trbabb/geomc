31struct MxWrap {
32
33 T* m;
34 const index_t rows;
35 const index_t cols;
36
37 inline T& elem(index_t r, index_t c) {

43 }

91 * These are used to track whether a matrix has a *

93 * turn affects the type of the storage ID buffer *

102 * A compound matrix, like AugmentedMatrix, for *
103 * example, may have either static or dynamic storage *

105 ******************************************************/

131/*****************************************************
132 * Matrix dimension agreement *
133 * *
134 * Check if two matrices' dimensions match *
135 *****************************************************/
136
137template <typename Ma, typename Mb, typename Enable=void>
138struct MatrixDimensionMatch {
139 static const bool isStaticMatch = false;
140
141 static inline bool isMatch(const Ma &ma, const Mb &mb) {
142 return false;
143 }
144
145 static inline void check(const Ma &ma, const Mb &mb) {
146 throw DimensionMismatchException(ma.rows(), ma.cols(), mb.rows(), mb.cols());
147 }
148};
149
150template <typename Ma, typename Mb>
151struct MatrixDimensionMatch <Ma, Mb,
152 typename std::enable_if<
153 ((Ma::ROWDIM == Mb::ROWDIM or Ma::ROWDIM * Mb::ROWDIM == 0) and
154 (Ma::COLDIM == Mb::COLDIM or Ma::COLDIM * Mb::COLDIM == 0))
155 >::type> {
156
157 static const bool isStaticMatch = true;
158
159 static inline bool isMatch(const Ma &ma, const Mb &mb) {
160 if ((Ma::ROWDIM * Mb::ROWDIM == 0 and ma.rows() != mb.rows()) or
161 (Ma::COLDIM * Mb::COLDIM == 0 and ma.cols() != mb.cols())) {
162 return false;
163 }
164 return true;
165 }
166
167 static inline void check(const Ma& ma, const Mb &mb) {
168 if (not isMatch(ma, mb)) {
169 throw DimensionMismatchException(ma.rows(), ma.cols(), mb.rows(), mb.cols());
170 }
171 }
172};
