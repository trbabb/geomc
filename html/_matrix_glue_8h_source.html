23 /****************************************************
24  * Storage class *
25  * *
26  * Matrices may be of either static or dynamic *
27  * dimensions. Statically-sized matrices may use *
28  * statically-allocated storage and avoid expensive *
29  * mallocs() upon creation and duplication. This *
30  * class allows matrix templates to choose between *
31  * the two methods of storage based on <N>. *
32  ****************************************************/

37 
38  Storage() {}
39  explicit Storage(index_t n) {}
40 
41  inline T* get() { return data; }
42  inline const T* get() const { return data; }
43  inline index_t size() const { return N; }
44 };

101 /****************************************************
102  * Dimension classes *
103  * *
104  * Let's say we have a Storage object, which may *
105  * wrap either a static or a dynamic array. If the *
106  * array is dynamic, we must store the length. We *
107  * don't want to pay for that length counter if the *
108  * wrapped memory is a static array, however, since *
109  * the length is known at compile time. This class *
110  * allocates storage for one size counter if <N> is *
111  * DYNAMIC_DIMENSION, otherwise no storage is *
112  * allocated and <N> is the reported array length. *
113  ****************************************************/
114 
115 template <index_t N>
116 struct Dimension {
117  typedef index_t storage_t[0];
118  static inline void set(storage_t s, index_t v) {}
119  static inline index_t value(const storage_t v) { return N; }
120 
121 };
122 
123 template <>
124 struct Dimension<DYNAMIC_DIM> {
125  typedef index_t storage_t[1];
126  static inline void set(storage_t s, index_t v) { s[0] = v; }
127  static inline index_t value(const storage_t v) { return *v; }
128 
129 };
