114 template <typename Ma, typename Mb>
115 struct MatrixDimensionMatch <Ma, Mb,
116  typename boost::enable_if_c<
117  ((Ma::ROWDIM == Mb::ROWDIM or Ma::ROWDIM * Mb::ROWDIM == 0) and
118  (Ma::COLDIM == Mb::COLDIM or Ma::COLDIM * Mb::COLDIM == 0))
119  >::type> {
120 
121  static const bool isStaticMatch = true;
122 
123  static inline bool isMatch(const Ma &ma, const Mb &mb) {
124  if ((Ma::ROWDIM * Mb::ROWDIM == 0 and ma.rows() != mb.rows()) or
125  (Ma::COLDIM * Mb::COLDIM == 0 and ma.cols() != mb.cols())) {
126  return false;
127  }
128  return true;
129  }
130 
131  static inline void check(const Ma& ma, const Mb &mb) {
132  if (not isMatch(ma, mb)) {
133  throw DimensionMismatchException(ma.rows(), ma.cols(), mb.rows(), mb.cols());
134  }
135  }
