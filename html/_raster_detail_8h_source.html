94template <typename I, typename O, index_t N, index_t Channels, EdgeBehavior Edge>
95class _ImplRasterSample<I,O,N,Channels,Edge,INTERP_LINEAR> {
96public:
97 typedef typename Raster<I,O,N,Channels>::coord_t coord_t;
98 typedef typename Raster<I,O,N,Channels>::grid_t grid_t;
99 typedef typename Raster<I,O,N,Channels>::sample_t sample_t;
100
101 static inline sample_t sample(const Raster<I,O,N,Channels> *r, const coord_t &pt) {
102 sample_t buf[1<<N];
103 grid_t gridPt = (grid_t)pt;
104 coord_t s = pt - ((coord_t)gridPt);
105
106 // copy surrounding 2^N sample pts into a contiguous buffer
107 r->template copy<Edge>(buf, Rect<typename grid_t::elem_t, N>(gridPt, gridPt + grid_t(1)));
108
109 return interp_linear(PointType<I,N>::iterator(s), buf, N);
110 }
111};

137class _ImplRasterIndex {
138public:
139
140 static inline index_t index(const grid_t &extent, const grid_t &c) {
141 index_t dim = 1;
142 index_t idx = 0;
143 for (index_t i = 0; i < N; i++){
144 index_t x = detail::_ImplEdge<Edge>::coord(c[i], extent[i]);
145 idx += x*dim;
146 dim *= extent[i];
147 }
148 return idx;
149 }
150};
