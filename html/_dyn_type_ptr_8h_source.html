1/*
2 * AnyPtr.h
3 *
4 * This class stores a pointer to an object of unknown type.
5 * It's safer than void*, because it stores the type_info of the
6 * object that was assigned to it. This is like a boost::any, except
7 * that the user provides the storage-- thus there are no malloc()s/news,
8 * so this class has the potential to be much faster.
9 *
10 *
11 * CAREFUL: polymorphism.
12 * - A is the base of B
13 * - We store a B*
14 * - We want to use our stored B* as an A*
15 * - Conversion fails because type(B) != type(A)
16 * - Will dynamic_cast work on void*? sounds sketchy.
17 * - A: No.
18 * - Need access to the vtable, but c++ is bad at being reflective.
19 * - This is pretty much impossible without manual input of some kind.
20 *
21 * Created on: Aug 7, 2011
22 * Author: tbabb
23 */
24

44class DynTypePtr {
45public:

49 DynTypePtr():
50 ptr(0),
51 tinfo(&typeid(void)){
52 //do nothing else
53 }
54
55 template <typename T> DynTypePtr(T *var):
56 ptr(var),
57 tinfo(&typeid(T)){
58 //do nothing else
59 }
60
61 virtual ~DynTypePtr(){}
62
64
65 bool isNull(){
66 return ptr == 0;
67 }
68
69 const type_info& getType(){
70 return *tinfo;
71 }
72
73 //return a reference to the pointed object
74 template <typename T> T& item() {
75 if (typeid(T) == *tinfo){
76 return *(T*)ptr;
77 } else {
78 throw bad_dyntype_ptr_cast();
79 }
80 }
81
82 //return a const reference to the pointed object
83 template <typename T> const T& item() const {
84 if (typeid(T) == *tinfo){
85 return *(T*)ptr;
86 } else {
87 throw bad_dyntype_ptr_cast();
88 }
89 }
90
91 //return the raw pointer, as the correct type.
92 template <typename T> T* get() const {
93 if (typeid(T) == *tinfo){
94 return (T*)ptr;
95 } else {
96 throw bad_dyntype_ptr_cast();
97 }
98 }
99
100 /*
101 * You would really like to implement operator*(), wouldn't you?
102 * But you can't, because it would be templated, and template inference
103 * doesn't happen when the return type is the only way to infer the
104 * template parameter. So you would have to write obj.operator*<Type>()
105 * literally everwhere, which is just ugly.
106 */
107
108 template <typename T> DynTypePtr& operator=(T *var){
109 ptr = var;
110 tinfo = &typeid(T);
111 return *this;
112 }
113
114protected:
115 void *ptr;
116 const type_info *tinfo;
117
118};
