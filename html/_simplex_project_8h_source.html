9template <typename T, index_t N>
10struct SimplexFace {
11
12 // basis layout:
13 // [spanning] [nullspace] [normal]
14 // as the simplex shrinks, the nullspace grows. This layout ensures:
15 // (1) that we don't have to copy the entire nullspace when we exclude a basis, and
16 // (2) that the existing nullspace and the spanning basis are contiguous,
17 // which means we can compute the nullspace by passing a single pointer.
18
19 index_t n = 0; // number of bases in the face (one less than the number of verts)
20 Vec<T,N> bases[N];
21 index_t included[N + 1];
22
23 SimplexFace() = default;
24
25 SimplexFace(const SimplexFace<T,N>& other) = default;

27 SimplexFace(const Simplex<T,N>& s, bool compute_nullspace=true):n(s.n - 1) {
28 Vec<T,N>* basis = spanning();
29 // the origin of the simplex is its final vertex:
30 for (index_t i = 0; i < n; ++i) {
31 basis[i] = s.pts[i] - s.pts[n];
32 included[i] = i;
33 }
34 included[n] = n;
35 if (n < N and compute_nullspace) {
36 // compute the orthogonal complement
37 geom::nullspace(basis, n, nullspace());
38 }
39 }

42 return N - n;
43 }

46 return bases[N - 1];

69 SimplexFace<T,N> exclude(index_t i) const {
70 SimplexFace<T,N> out;
71 out.n = n - 1;
72 std::copy(included, included + N + 1, out.included);
73 Vec<T,N> ejected;
74 if (i < n) {
75 std::copy(bases, bases + N, out.bases);
76 // perform this shuffle:
77 // ejected <- final basis <- normal
78 out.bases[i] = bases[n - 1];
79 out.bases[n - 1] = normal();
80 // update the index map. note that the root point is always included
81 // and must stay at the end, so we move the final non-root point into
82 // the vacated hole, and then shift the root point down by one:
83 out.included[i] = included[n - 1];
84 out.included[n - 1] = included[n];
85 ejected = bases[i];
86 } else {
87 // the excluded vertex is the current origin of the basis.
88 // the new origin is the previous final basis; re-root all the axes
89 for (index_t k = 0; k < out.n; ++k) {
90 out.bases[k] = bases[k] - bases[n - 1];
91 }
92 // copy the nullspace
93 std::copy(bases + n, bases + N, out.bases + out.n);
94 // the ejected basis points from the new origin toward the old origin
95 ejected = -bases[n - 1];
96 }
97 if (out.n > 0) {
98 // compute the normal to the new face. it's orthogonal to the new basis,
99 // and also lies in the space of the parent face (orthogonal to the parent's
100 // nullspace).
101 out.normal() = orthogonal(out.bases);
102 // make sure the normal faces outward:
103 if (out.normal().dot(ejected) >= 0) {
104 out.normal() = -out.normal();
105 }
106 } else {
107 // the simplex is a point. we have only one choice left for the
108 // new normal, and it's in the space of the parent simplex; i.e.
109 // the basis we just ejected.
110 out.normal() = -ejected;
111 }
112 return out;
113 }

129template <typename T, index_t N>
130Vec<T,N> _project_to_face(
131 const Simplex<T,N>& simplex,
132 SimplexFace<T,N> face,
133 Vec<T,N> point,
134 Vec<T,N> to_p)
135{
136 Vec<T,N> out;
137 // we pass this bc the facefinding algorithm already has it:
138 // to_p = p - s.pts[best_face->included[best_n]];
139 if (face.n == 0) {
140 // projects to a vertex; the projection is the vertex itself.
141 out = simplex[face.included[0]];
142 } else if (face.n == 1) {
143 // projects to an edge
144 const Vec<T,N>& v = face.spanning()[0];
145 out = to_p.project_on(v) + simplex.pts[face.included[1]];
146 } else if (N > 2 and face.n == N - 1) {
147 // p projects to a plane.
148 // subtract the normal component of the vector from the face to the point
149 out = point - to_p.project_on(face.normal());
150 } else if (face.n == N) {
151 // projects to the interior of the simplex; p unchanged
152 out = point;
153 } else if constexpr (N > 3) {
154 // this case is only possible with N > 3.
155 // project to the lowest-dimension subspace (nullspace or spanning basis).
156 index_t n_null = face.n_null();
157 bool proj_null = n_null < face.n;
158 Vec<T,N>* proj_basis = proj_null ? face.nullspace() : face.spanning();
159 index_t proj_n = proj_null ? n_null : face.n;
160 if (simplex.n < N + 1 or not proj_null) {
161 // the projection basis is not orthogonal if:
162 // (a) the root simplex is not full, and we generated the nullspace
163 // on inititalization
164 // (b) we're projecting to the simplex basis, which is in general
165 // not orthogonal
166 orthogonalize(proj_basis, proj_n);
167 }
168 // project P onto the chosen basis
169 for (index_t i = 0; i < proj_n; ++i) {
170 out += to_p.project_on(proj_basis[i]);
171 }
172 if (proj_null) {
173 // if we projected to the nullspace,
174 // then subtract that component from the original vector,
175 // leaving a vector in the space of the simplex.
176 out = to_p - out;
177 }
178 // add back the origin of the simplex
179 out += simplex.pts[face.included[face.n]];
180 }
181
182 return out;
183}

186template <typename T, index_t N>
187struct ProjectionResult {
188 // the candidate/resultant face to which the point projects
189 SimplexFace<T,N> face;
190 // intermediate candidate has been found?
191 bool exists = false;
192 // the point on the face to which the input point projects
193 Vec<T,N> projected_point;
194 // squared distance from the point to its projection on the face
195 T distance2 = std::numeric_limits<T>::max();
196 // in the process of search, has a degenerate face been encountered?
197 bool is_degenerate = false;
198 // whether the point is inside a full simplex
199 bool contains = false;
200 // whether the point is on the back side of a boundary face.
201 // this can happen if we are not yet sure whether the point is inside
202 // a full simplex
203 bool backfacing = false;
204
205 operator bool() const {
206 return exists;
207 }
208
209 void update(T d2, Vec<T,N> point, const SimplexFace<T,N>& face, bool new_backfacing) {
210 if (not exists // no previous candidates
211 // this candidate is closer
212 or (d2 < distance2 and new_backfacing <= backfacing)
213 // frontface always beats backface
214 or (not new_backfacing and backfacing))
215 {
216 this->exists = true;
217 this->face = face;
218 this->projected_point = point;
219 this->distance2 = d2;
220 this->backfacing = new_backfacing;
221 }
222 }

259protected:
260
261 // return whether the point is interior to the subface
262 bool _find_nearest_face(const SimplexFace<T,N>& face, Vec<T,N> to_p, int depth) {
263 bool all_inside = true;
264 bool backfacing = depth > 0 and face.normal().dot(to_p) < 0;
265 if (backfacing and op == ProjectionOp::PROJECT and depth == 1 and face.n == N - 1) {
266 // we're on the interior side of a top-level boundary face, and we might project
267 // the point to one such face if we're inside a full simplex. we don't need to check
268 // its subfaces, though, because you can't project to a subface (e.g., an edge)
269 // from the interior of a simplex. we also shouldn't early exit because we're on
270 // the backside! (in the case where we're clipping, interior faces are not
271 // projection candidates).
272
273 // because we don't boundary check in this case, we could in principle project
274 // to the face's plane well outside the simplex. if we are truly inside
275 // the simplex, then some later face will be closer and occlude that hit. but we
276 // have to be sure that any frontfacing hit always clobbers a backfacing best-hit,
277 // since the backfacing plane might skim closer to the point than the frontfacing
278 // face it should project to.
279 } else {
280 if (backfacing) {
281 // point is on the interior side of a boundary,
282 // so it does not project to this subface.
283 return true;
284 }
285 if (face.n == 0) {
286 // simplex is a point and has no subfaces to check
287 result.update(to_p.mag2(), simplex[face.included[0]], face, false);
288 return false;
289 }
290
291 SimplexFace<T,N> nearest_face;
292 index_t n_to_check = face.n + (check_faces == SimplexFaces::SKIP_BACKFACE ? 0 : 1);
293 for (index_t i = 0; i < n_to_check; ++i) {
294 // check each subface for ownership of the projected point.
295 SimplexFace<T,N> subface = face.exclude(i);
296
297 // re-root p if we're checking the backface
298 Vec<T,N> to_sub_p = (i == face.n) ? (to_p - face.bases[i - 1]) : to_p;
299 if (subface.normal().mag2() == 0) [[unlikely]] {
300 result.is_degenerate = true;
301 continue; // degenerate subface
302 }
303
304 bool is_interior = _find_nearest_face(subface, to_sub_p, depth + 1);
305
306 if (not is_interior) all_inside = false;
307 }
308 }
309
310 if (all_inside) {
311 // the point is on the interior of all our subfaces, therefore it projects to us.
312 if (face.n < N) {
313 Vec<T,N> projected_pt = _project_to_face(simplex, face, p, to_p);
314 T d2 = p.dist2(projected_pt);
315 result.update(d2, projected_pt, face, backfacing);
316 } else {
317 // if we're a full simplex, we don't project at all:
318 // - if projecting, we project to a top-level boundary face, not the root
319 // - if clipping, no change to the point at all.
320 // we do know we're inside the simplex, though:
321 result.contains = true;
322 if (op == ProjectionOp::CLIP) {
323 result.update(0, p, face, true);
324 }
325 }
326 }
327 // this can be backfacing when we're checking the top-level boundary faces;
328 // i.e. the point is interior to a 2D tri or a 3D tet
329 return backfacing;
330 }

340 Vec<T,N> normal_direction() {
341 Vec<T,N> out_normal = p - result.projected_point;
342 index_t best_n = result.face.n; // dimension of the projection target simplex
343 Vec<T,N> to_p = p - simplex.pts[result.face.included[best_n]];
344 if (best_n == 1) {
345 // projects to an edge
346 const Vec<T,N>& v = result.face.spanning()[0];
347 if constexpr (N == 3) {
348 // we can use the cross product to get a direction to the edge.
349 // numerically this is better than subtracting from the projected point
350 // (catastrophic cancellation)
351 Vec<T,N> c = v ^ to_p;
352 out_normal = c ^ v;
353 } else if constexpr (N == 2) {
354 // there are only two orthogonal directions to a line in 2D
355 Vec<T,N> d = v.right_perpendicular();
356 out_normal = d.dot(to_p) >= 0 ? d : -d;
357 }
358 } else if (N > 2 and best_n == N - 1) {
359 // p projects to a hyperplane. return its normal, which we explicitly know
360 Vec<T,N> normal = result.face.nullspace()[0];
361 out_normal = normal.dot(to_p) >= 0 ? normal : -normal;
362 }
363
364 return out_normal;
365 }

371 Simplex<T,N> out;
372 for (index_t i = 0; i <= result.face.n; ++i) {
373 out.pts[i] = simplex.pts[result.face.included[i]];
374 }
375 out.n = result.face.n + 1;
376 return out;
377 }
