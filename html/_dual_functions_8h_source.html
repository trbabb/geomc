384template <typename T, geom::DiscontinuityPolicy P>
385inline geom::Dual<T,P> pow(const geom::Dual<T,P> &base, const geom::Dual<T,P> &xp) {
386 // here we use the chain rule for partial derivatives:
387 // f(x,y) = x ^ y
388 // x = g(t); y = h(t)
389 // then:
390 // df / dt = (df / dx) * (dx / dt) + (df / dy) * (dy / dt)
391 // = (df / dx) * g`(t) + (df / dy) * h`(t)
392 // and:
393 // df / dx = d/dx (x ^ y) = y * x ^ (y - 1)
394 // df / dy = d/dy (x ^ y) = log(x) * (x ^ y)
395 T a_c = pow(base.x, xp.x);
396 return geom::Dual<T,P>(
397 a_c,
398 base.dx * xp.x * a_c / xp.x +
399 xp.dx * a_c * std::log(base.x)
400 );
401
402}

453template <typename T, geom::DiscontinuityPolicy P>
454inline geom::Dual<T,P> abs(const geom::Dual<T,P> &d) {
455 bool neg = d.x < 0;
456 T dx;
457
458 if (d.x == 0) {
459 dx = geom::Dual<T,P>::discontinuity_policy::resolve(-d.dx, d.dx);
460 } else {
461 dx = neg ? -d.dx: d.dx;
462 }
463
464 return geom::Dual<T,P>(neg ? -d.x : d.x, dx);
465}

472template <typename T, geom::DiscontinuityPolicy P>
473inline geom::Dual<T,P> ceil(const geom::Dual<T,P> &d) {
474 T x = ceil(d.x);
475 T dx;
476 if (P == geom::DiscontinuityPolicy::Inf and x == d.x) {
477 const T inf = std::numeric_limits<T>::infinity();
478 // jump is in the same direction that the function is changing
479 dx = std::copysign(inf, d.dx);
480 } else if (P == geom::DiscontinuityPolicy::NaN and x == d.x) {
481 dx = std::numeric_limits<T>::quiet_NaN();
482 } else {
483 dx = 0;
484 }
485 return geom::Dual<T,P>(x, dx);
486}

514template <typename T, geom::DiscontinuityPolicy P>
515inline geom::Dual<T,P> min(const geom::Dual<T,P> &d1, const geom::Dual<T,P> &d2) {
516 bool lt = d1.x <= d2.x;
517 T x = lt ? d1.x : d2.x;
518 T dx;
519 if (d1.x == d2.x) {
520 dx = geom::Dual<T,P>::discontinuity_policy::resolve(
521 // faster increasing function wins min() on the left
522 std::max(d1.dx, d2.dx),
523 std::min(d1.dx, d2.dx));
524 } else {
525 dx = lt ? d1.dx : d2.dx;
526 }
527 return geom::Dual<T,P>(x,dx);
528}

535template <typename T, geom::DiscontinuityPolicy P>
536inline geom::Dual<T,P> max(const geom::Dual<T,P> &d1, const geom::Dual<T,P> &d2) {
537 bool gt = d1.x >= d2.x;
538 T x = gt ? d1.x : d2.x;
