432  } else {

447  SmallStorage<T,N>& operator=(SmallStorage<T,N>& other) {
448  if (&other == this) return; // no self-assign
449  // don't bother to realloc if the buffer is already the correct size.
450  if (sz > N and data and sz != other.sz) delete [] data;
451  if (other.sz <= N) {
452  data = buf;
453  } else if (other.sz != sz or not data) {
454  data = new T[other.sz];
455  }

483  void resize(index_t n) {
484  if (n > N and n > sz) {
485  T* new_data = new T[n];
486  // move the data rather than copy-construct for efficiency:
487  std::copy(
488  std::make_move_iterator(data),
