366template <typename T, index_t N>
367class SmallStorage {
368
369 T buf[N];
370 T* data;
371 index_t sz;
372
373public:
374
376 explicit SmallStorage(index_t n) : sz(n) {
377 if (sz > N) {
378 data = new T[sz];
379 } else {
380 data = buf;
381 }
382 }
383
385 ~SmallStorage() {
386 if (sz > N) delete [] data;
387 sz = 0;
388 data = NULL;
389 }
390
392 SmallStorage(SmallStorage<T,N>& other) : sz(other.sz) {
393 if (sz > N) {
394 data = new T[sz];
395 } else {
396 data = buf;
397 }
398 std::copy(other.data, other.data + sz, data);
399 }
400
402 SmallStorage(const T* array, index_t n) : sz(n) {
403 if (sz > N) {
404 data = new T[sz];
405 } else {
406 data = buf;
407 }
408 std::copy(array, array + sz, data);
409 }
410
412 SmallStorage(SmallStorage<T,N>&& other) noexcept : sz(other.sz) {
413 if (sz > N) {
414 data = other.data;
415 } else {
416 data = buf;
417 std::copy(
418 std::make_move_iterator(other.data),
419 std::make_move_iterator(other.data + sz),
420 data);
421 }
422 other.data = NULL;
423 other.sz = 0;
424 }
425
427 SmallStorage<T,N>& operator=(SmallStorage<T,N>&& other) noexcept {
428 if (sz > N and data) delete [] data;
429 sz = other.sz;
430 if (sz > N) {
431 data = other.data;
432 } else {
433 data = buf;
434 std::copy(
435 std::make_move_iterator(other.data),
436 std::make_move_iterator(other.data + sz),
437 data);
438 }
439 other.data = NULL;
440 other.sz = 0;
441 return *this;
442 }
443
445 SmallStorage<T,N>& operator=(SmallStorage<T,N>& other) {
446 if (&other == this) return; // no self-assign
447 // don't bother to realloc if the buffer is already the correct size.
448 if (sz > N and data and sz != other.sz) delete [] data;
449 if (other.sz <= N) {
450 data = buf;
451 } else if (other.sz != sz or not data) {
452 data = new T[other.sz];
453 }
454 sz = other.sz;
455 std::copy(other.data, other.data + sz, data);
456 return *this;
457 }
458
460 inline T* get() { return data; }
462 inline const T* get() const { return data; }
464 inline index_t size() const { return sz; }
465
467 inline T& operator[](index_t i) { return data[i]; }
469 inline T operator[](index_t i) const { return data[i]; }
470
481 void resize(index_t n) {
482 if (n > N and n > sz) {
483 T* new_data = new T[n];
484 // move the data rather than copy-construct for efficiency:
485 std::copy(
486 std::make_move_iterator(data),
487 std::make_move_iterator(data + sz), new_data);
488 std::swap(new_data, data); // a little more atomic
489 if (sz > N) {
490 delete [] new_data;
491 }
492 sz = n;
493 }
494 }
495
496};
