29template <typename T>
30void transpose_square_matrix(T* m, index_t n) {
31 detail::MxWrap<T,true> mx = {m, n, n};
32 for (index_t i = 0; i < n; ++i) {
33 for (index_t j = i + 1; j < n; ++j) {
34 std::swap(mx(i,j), mx(j,i));
35 }
36 }
37}

91template <typename Md, typename M>
92inline void mtx_copy_txpose(Md* into, const M& m) {
93 for (index_t c = 0; c < into->cols(); c++) {
94 for (index_t r = 0; r < into->rows(); r++) {
95 into->set(r, c, m(c, r));
96 }
97 }
98}

186template <typename Ma, typename Mb>
187bool mtxequal(const Ma& a, const Mb& b) {
188 typename Ma::const_iterator ai = a.begin();
189 typename Mb::const_iterator bi = b.begin();
190
191 for (; ai != a.end(); ++ai, ++bi) {
192 if (*ai != *bi) {
193 return false;
194 }
195 }
196
197 return true;
198}
