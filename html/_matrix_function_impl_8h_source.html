73 template <typename Md, typename M>
74 inline void mtx_copy_txpose(Md *into, const M& m) {
75  for (index_t c = 0; c < into->cols(); c++) {
76  for (index_t r = 0; r < into->rows(); r++) {
77  into->set(r, c, m.get(c, r));
78  }
79  }
80 }
