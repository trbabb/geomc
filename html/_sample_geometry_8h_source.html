30template <typename T, index_t N>
31struct SampleShape<Simplex<T,N>> : public ShapeDistribution<Simplex<T,N>> {
32 Simplex<T,N> shape;
33
34 SampleShape(const Simplex<T,N>& s):ShapeDistribution(s) {}
35
36 template <typename Generator>
37 Vec<T,N> operator()(Generator& rng) {
38 // algorithm from: https://projecteuclid.org/download/pdf_1/euclid.pjm/1102911301
39 // we pick barycentric coordinates and then normalize so they sum to 1.
40 std::exponential_distribution<T> e(1);
41 T s[N + 1];
42 T sum = 0;
43 for (index_t i = 0; i < N + 1; ++i) {
44 T xi = e(rng);
45 s[i] = xi;
46 sum += xi;
47 }
48 Vec<T,N> p;
49 for (index_t i = 0; i < N + 1; ++i) {
50 T xi = s[i] / sum;
51 p += xi * shape.pts[i];
52 }
53 return p;
54 }
55};

66 Vec<T,N> p;
67 for (index_t i = 0; i < N; ++i) {
68 p[i] = shape.lo[i] + u(rng) * (shape.hi[i] - shape.lo[i]);
