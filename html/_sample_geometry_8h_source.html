87template <typename T, index_t N>
88struct SampleShape<Simplex<T,N>> : public detail::ShapeDistribution<Simplex<T,N>> {
89 using detail::ShapeDistribution<Simplex<T,N>>::shape;
90 using typename Dimensional<T,N>::point_t;
91
92private:
93 std::exponential_distribution<T> e = 1;
94public:
95
96 template <typename Generator>
97 point_t operator()(Generator& rng) {
98 // algorithm from: https://projecteuclid.org/download/pdf_1/euclid.pjm/1102911301
99 // we pick barycentric coordinates and then normalize so they sum to 1.
100 T s[N + 1];
101 T sum = 0;
102 DenseUniformDistribution<T> u(0, 1);
103 for (index_t i = 0; i < N + 1; ++i) {
104 T xi = u(rng);
105 s[i] = xi;
106 sum += xi;
107 }
108 point_t p;
109 for (index_t i = 0; i < N + 1; ++i) {
110 T xi = s[i] / sum;
111 p += xi * shape.pts[i];
112 }
113 return p;
114 }
115
116 bool operator==(const SampleShape& other) const {
117 return shape == other.shape;
118 }
119};

154template <typename T, index_t N>
155struct SampleShape<Sphere<T,N>> : public detail::ShapeDistribution<Sphere<T,N>> {
156 using detail::ShapeDistribution<Sphere<T,N>>::shape;
157 using typename Dimensional<T,N>::point_t;
158
159private:
160 using ptype = PointType<T,N>;
161 using gauss_t = std::conditional_t<(N <= 3), int[0], std::normal_distribution<T>>;
162 gauss_t _gauss;
163public:
164
165 template <typename Generator>
166 point_t operator()(Generator& rng) {
167 if constexpr (N <= 3) {
168 // rejection sampling
169 DenseUniformDistribution<T> unif {-1,1};
170 point_t out;
171 do {
172 // generate a point in the signed unit box
173 for (index_t i = 0; i < N; ++i) {
174 ptype::iterator(out)[i] = unif(rng);
175 }
176 // if the point is outside the unit sphere, reject it and try again
177 } while (ptype::mag2(out) > 1);
178 return shape.r * out + shape.center;
179 } else {
180 // draw a multivariate gaussian, then project it onto the sphere
181 DenseUniformDistribution<T> u_01(0,1);
182 Vec<T,N> p;
183 for (index_t i = 0; i < N; ++i) {
184 p[i] = _gauss(rng);
185 }
186 return p.unit() * shape.r * std::pow(u_01(rng), 1 / (T)N) + shape.center;
187 }
188 }
189
190 bool operator==(const SampleShape& other) const {
191 return shape == other.shape;
192 }
193};

243template <typename T, index_t N>
244struct SampleShape<SphericalShell<T,N>> : public detail::ShapeDistribution<SphericalShell<T,N>> {
245 using detail::ShapeDistribution<SphericalShell<T,N>>::shape;
246 using typename Dimensional<T,N>::point_t;
247
248private:
249 using ptype = PointType<T,N>;
250 using gauss_t = std::conditional_t<(N <= 3), int[0], std::normal_distribution<T>>;
251 gauss_t _gauss;
252public:
253
254 template <typename Generator>
255 point_t operator()(Generator& rng) {
256 // recall that SphericalShell is Dilated<Hollow<Sphere>>
257 // dilated.hollow.base
258 const Sphere<T,N>& sphere = shape.shape.shape;
259 T r0 = std::max<T>(sphere.radius - shape.dilation, 0);
260 T r1 = sphere.radius + shape.dilation;
261 DenseUniformDistribution<T> u_dist(r0, r1);
262 point_t p;
263 if constexpr (N <= 3) {
264 // rejection sampling
265 DenseUniformDistribution<T> unif {-1,1};
266 do {
267 // generate a point in the signed unit box
268 for (index_t i = 0; i < N; ++i) {
269 ptype::iterator(p)[i] = unif(rng);
270 }
271 // if the point is outside the unit sphere, reject it and try again
272 } while (ptype::mag2(p) > 1);
273 } else {
274 // draw a multivariate gaussian, then project it onto the sphere
275 for (index_t i = 0; i < N; ++i) {
276 p[i] = _gauss(rng);
277 }
278 }
279 p = p.unit();
280 T r = std::pow(u_dist(rng), 1 / (T)N);
281 return p * r + sphere.center;
282 }

306 constexpr point_t _Z = {VecType<T,N-1>((T)0), (T)1};

330template <typename T, index_t N>
331requires (N == 2 or N == 3)
332struct SampleShape<SphericalCap<T,N>> : public detail::ShapeDistribution<SphericalCap<T,N>> {
333 using detail::ShapeDistribution<SphericalCap<T,N>>::shape;
334 using typename Dimensional<T,N>::point_t;
335
336 template <typename Generator>
337 point_t operator()(Generator& rng) {
338 DenseUniformDistribution<T> uniform {0, 1};
339 constexpr T pi = std::numbers::pi_v<T>;
340 T half_angle = std::clamp<T>(shape.half_angle_radians, 0, pi);
341 if constexpr (N == 2) {
342 T u = uniform(rng);
343 // pick theta in [-half_angle, +half_angle]
344 T theta = u * half_angle - (1 - u) * half_angle;
345 return {
346 std::sin(theta),
347 std::cos(theta)
348 };
349 } else if constexpr (N == 3) {
350 T u = uniform(rng);
351 T v = uniform(rng);
352 // we pick a height with uniform probability
353 // (because a slice through the sphere with thickness t at height h
354 // has unchanging area regardless of h).
355 T min_h = std::cos(half_angle);
356 T h = u * min_h + (1 - u);
357 T r = std::sqrt(1 - h * h);
358 T theta = v * 2 * pi;
359 return {
360 r * std::cos(theta),
361 r * std::sin(theta),
362 h
363 };
364 }
365 }

487template <typename T, index_t N>
488struct SampleShape<Hollow<Rect<T,N>>> : public detail::ShapeDistribution<Hollow<Rect<T,N>>> {
489 using detail::ShapeDistribution<Hollow<Rect<T,N>>>::shape;
490 using typename Dimensional<T,N>::point_t;
491
492private:
493 point_t _face_areas;
494 T _face_area_sum;
495public:
496
497 SampleShape<Hollow<Rect<T,N>>>() { param(shape); }
498 SampleShape(const Hollow<Rect<T,N>>& s) { param(s); }
499
500 template <typename Generator>
501 point_t operator()(Generator& rng) {
502 using ptype = PointType<T,N>;
503 DenseUniformDistribution<T> u(0, 1);
504 // choose a point inside the box
505 point_t p;
506 for (index_t i = 0; i < N; ++i) {
507 T s = u(rng);
508 ptype::iterator(p)[i] = s * shape.lo[i] + (1 - s) * shape.hi[i];
509 }
510 // choose a face in proportion to its area
511 T sum = 0;
512 index_t side = 0;
513 T s = u(rng) * _face_area_sum;
514 while (sum < s and side < N) {
515 sum += _face_areas[side];
516 ++side;
517 }
518 // project the point onto the chosen face
519 T midpt = shape.lo[side] / 2 + shape.hi[side] / 2; // avoid overflow
520 p[side] = p[side] > midpt ? shape.hi[side] : shape.lo[side];
521 return p;
522 }
523
524 void param(const Hollow<Rect<T,N>>& s) {
525 shape = s;
526 _face_areas = shape.face_areas();
527 _face_area_sum = _face_areas.sum();
528 }
529
530 bool operator==(const SampleShape& other) const = default;
531};

555 DenseUniformDistribution<T> u(0, 1);
556 const Extruded<Shape>& extrusion = shape.shape;
557 T s = u(rng) * (_cap_area + _wall_area);
558 T h = u(rng);
559 if (s < _cap_area) {
560 // choose a point on the cap
561 return {
562 _face_shape_sampler(rng),
563 h > 0.5 ? extrusion.height.hi : extrusion.height.lo

565 } else {
566 // choose a point on the wall

569 extrusion.height.remap(h)

577 _face_shape_sampler.param(face_shape);
578 _wall_sampler.param(face_shape);
579 _cap_area = face_shape.measure_interior();
580 _wall_area = face_shape.measure_boundary() * s.shape.height.dimensions();
