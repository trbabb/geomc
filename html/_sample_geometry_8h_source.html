78template <typename T, index_t N>
79struct SampleShape<Simplex<T,N>> : public detail::ShapeDistribution<Simplex<T,N>> {
80 using detail::ShapeDistribution<Simplex<T,N>>::shape;
81 using typename Dimensional<T,N>::point_t;
82
83private:
84 std::exponential_distribution<T> e = 1;
85public:
86
87 template <typename Generator>
88 point_t operator()(Generator& rng) {
89 // algorithm from: https://projecteuclid.org/download/pdf_1/euclid.pjm/1102911301
90 // we pick barycentric coordinates and then normalize so they sum to 1.
91 T s[N + 1];
92 T sum = 0;
93 for (index_t i = 0; i < N + 1; ++i) {
94 T xi = e(rng);
95 s[i] = xi;
96 sum += xi;
97 }
98 point_t p;
99 for (index_t i = 0; i < N + 1; ++i) {
100 T xi = s[i] / sum;
101 p += xi * shape.pts[i];
102 }
103 return p;
104 }
105
106 bool operator==(const SampleShape& other) const {
107 return shape == other.shape;
108 }
109};

123 using ptype = PointType<T,N>;
124 DenseUniformDistribution<T> u(0, 1);
125 point_t p;
126 for (index_t i = 0; i < N; ++i) {

140template <typename T, index_t N>
141struct SampleShape<Sphere<T,N>> : public detail::ShapeDistribution<Sphere<T,N>> {
142 using detail::ShapeDistribution<Sphere<T,N>>::shape;
143 using typename Dimensional<T,N>::point_t;
144
145private:
146 using ptype = PointType<T,N>;
147 using gauss_t = std::conditional_t<(N <= 3), int[0], std::normal_distribution<T>>;
148 gauss_t _gauss;
149public:
150
151 template <typename Generator>
152 point_t operator()(Generator& rng) {
153 if constexpr (N <= 3) {
154 // rejection sampling
155 DenseUniformDistribution<T> unif {-1,1};
156 point_t out;
157 do {
158 // generate a point in the signed unit box
159 for (index_t i = 0; i < N; ++i) {
160 ptype::iterator(out)[i] = unif(rng);
161 }
162 // if the point is outside the unit sphere, reject it and try again
163 } while (ptype::mag2(out) > 1);
164 return shape.r * out + shape.center;
165 } else {
166 // draw a multivariate gaussian, then project it onto the sphere
167 DenseUniformDistribution<T> u_01(0,1);
168 Vec<T,N> p;
169 for (index_t i = 0; i < N; ++i) {
170 p[i] = _gauss(rng);
171 }
172 return p.unit() * shape.r * std::pow(u_01(rng), 1 / (T)N) + shape.center;
173 }
174 }
175
176 bool operator==(const SampleShape& other) const {
177 return shape == other.shape;
178 }
179};

231template <typename T, index_t N>
232struct SampleShape<SphericalShell<T,N>> : public detail::ShapeDistribution<SphericalShell<T,N>> {
233 using detail::ShapeDistribution<SphericalShell<T,N>>::shape;
234 using typename Dimensional<T,N>::point_t;
235
236private:
237 using ptype = PointType<T,N>;
238 using gauss_t = std::conditional_t<(N <= 3), int[0], std::normal_distribution<T>>;
239 gauss_t _gauss;
240public:
241
242 template <typename Generator>
243 point_t operator()(Generator& rng) {
244 // recall that SphericalShell is Dilated<Hollow<Sphere>>
245 // dilated.hollow.base
246 const Sphere<T,N>& sphere = shape.shape.shape;
247 T r0 = std::max<T>(sphere.radius - shape.dilation, 0);
248 T r1 = sphere.radius + shape.dilation;
249 DenseUniformDistribution<T> u_dist(r0, r1);
250 point_t p;
251 if constexpr (N <= 3) {
252 // rejection sampling
253 DenseUniformDistribution<T> unif {-1,1};
254 do {
255 // generate a point in the signed unit box
256 for (index_t i = 0; i < N; ++i) {
257 ptype::iterator(p)[i] = unif(rng);
258 }
259 // if the point is outside the unit sphere, reject it and try again
260 } while (ptype::mag2(p) > 1);
261 } else {
262 // draw a multivariate gaussian, then project it onto the sphere
263 for (index_t i = 0; i < N; ++i) {
264 p[i] = _gauss(rng);
265 }
266 }
267 p = p.unit();
268 T r = std::pow(u_dist(rng), 1 / (T)N);
269 return p * r + sphere.center;
270 }

273 return shape == other.shape;

282template <typename T, index_t N>
283struct SampleShape<Cylinder<T,N>> : public detail::ShapeDistribution<Cylinder<T,N>> {
284 using detail::ShapeDistribution<Cylinder<T,N>>::shape;
285 using typename Dimensional<T,N>::point_t;
286
287private:
288 SampleShape<Sphere<T,N-1>> _sphere;
289public:
290
291 template <typename Generator>
292 point_t operator()(Generator& rng) {
293 constexpr point_t _Z = {{(T) 0}, (T) 1};
294 DenseUniformDistribution<T> unif {0,1};
295 // pick a point on the cap
296 auto base = _sphere(rng) * shape.radius;
297 point_t axis = shape.p1 - shape.p0;
298 // reorient the cap distribution to be perpendicular to the axis
299 base = base.align(_Z, axis.unit());
300 // pick a distance along the axis direction and follow it away from the base
301 return base + unif(rng) * axis + shape.p0;
302 }
303
304 void reset() {
305 _sphere.reset();
306 }
307
308 bool operator==(const SampleShape& other) const {
309 return shape == other.shape;
