35 typedef typename std::aligned_storage<sizeof(T), alignof(T)>::type storage_t;
36
37 // `head + size` scheme chosen over `head + tail` to avoid
38 // ambiguity of "full vs. empty buffer"
39
40 // We usage storage_t both to admit objects with no default constructor,
41 // and also to avoid unnecessary constructions/destructions of items
42 // in the empty area of the buffer.
43
44 // We manage our own static/dynamic buffer instead of delegating to SmallStorage
45 // because we need to manage the liveness of the elements in the buffer.
46 // A SmallStorage<storage_t> won't properly call move, copy, or assignment
47 // operators because storage_t hides the overlying class, and of course
48 // SmallStorage does not know which items are live.

105 Deque(Deque<T,N>&& other):
106 _data((other._capacity > N) ? other._data : _buf),
107 _capacity(other._capacity),
108 _head((other._capacity > N) ? other._head : 0),
109 _size(other._size)
110 {
111 if (other._capacity <= N) {
112 // move-construct individual items.
113 for (index_t i = 0; i < _size; ++i) {
114 T* p = get() + i;
115 new (p) T(std::move(other[i]));
116 other[i].~T();
117 }
118 } else {
119 // we took ownership of `other`'s allocated array.
120 // turn `other` back into a static buffer.
121 other._capacity = N;
122 other._data = other._buf;
123 }
124 other._size = 0;
125 other._head = 0;
126 }

142 Deque& operator=(const Deque& other) {
143 index_t old_size = size();
144 index_t new_size = other.size();
145 index_t n = std::min(old_size, new_size);
146 reserve(other._size);
147 // delegate assignment to T
148 for (index_t i = 0; i < n; ++i) {
149 this->operator[](i) = other[i];
150 }
151 // copy-construct new items
152 for (index_t i = n; i < new_size; ++i) {
153 T* p = item(i);
154 new (p) T(other[i]);
155 }
156 // destroy extra items
157 for (index_t i = n; i < old_size; ++i) {
158 T* p = item(i);
159 p->~T();
160 }
161 _size = other._size;
162
163 return *this;
164 }

167 Deque& operator=(Deque&& other) {
168 // destroy all our stuff
169 for (index_t i = 0; i < _size; ++i) {
170 item(i)->~T();
171 }
172 if (other._capacity > N) {
173 // `other` has a dynamic array, just take ownership of it
174 if (_capacity > N) {
175 // delete our own dynamic array
176 delete [] _data;
177 }
178 _data = other._data;
179 _capacity = other._capacity;
180 _head = other._head;
181 // turn `other` back into a static buffer
182 other._data = other._buf;
183 other._capacity = N;
184 } else {
185 // `other`'s array cannot be moved; copy items instead.
186 // we do not need to reserve(), obviously, because _size < N.
187 // if we have space for more than N, that's fine too.
188 _head = 0;
189 for (index_t i = 0; i < other._size; ++i) {
190 T* p = get() + i;
191 // move-construct a new item in our own buffer
192 new (p) T(std::move(other.operator[](i)));
193 // destroy its source
194 other.operator[](i).~T();
195 }
196 }
197
198 _size = other._size;
199 other._size = 0;
200 other._head = 0;
201
202 return *this;
203 }

354 void reserve(index_t new_cap) {
355 if (new_cap > _capacity) {
356 index_t resize_to = std::max(_capacity * 2, new_cap);
357 storage_t* array = new storage_t[resize_to];
358
359 // move the items into the new array
360 for (index_t i = 0; i < _size; ++i) {
361 T* p = reinterpret_cast<T*>(array + i);
362 // move-construct
363 new (p) T(std::move(*item(i)));
364 // destroy the source
365 item(i)->~T();
366 }
367
368 if (_capacity > N) {
369 delete [] _data;
370 }
371
372 _data = array;
373 _head = 0;
374 _capacity = resize_to;
375 }
376 }
