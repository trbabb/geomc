123 // returns an upper bound on the squared distance to the nearest object
124 // to `p` in mimimum bounding box `r`.
125 template <typename T, index_t N>
126 T worst_case_nearest2(const Vec<T,N>& p, const Rect<T,N>& r) {
127  T result = std::numeric_limits<T>::max();
128 
129  // each face of the box must have an object touching it, if it is a minimal box.
130  // each face has a farthest corner where such an object could be hiding,
131  // and we want the nearest of these.
132 
133  Vec<T,N> far_extremes;
134  Vec<T,N> near_extremes;
135 
136  for (index_t axis = 0; axis < N; axis++) {
137  T lodist = std::abs(p[axis] - r.min()[axis]);
138  T hidist = std::abs(p[axis] - r.max()[axis]);
139 
140  far_extremes[axis] = lodist > hidist ? r.min()[axis] : r.max()[axis];
141  near_extremes[axis] = lodist > hidist ? r.max()[axis] : r.min()[axis];
142  }
143 
144  for (index_t axis = 0; axis < N; axis++) {
145  Vec<T,N> face_extreme = far_extremes;
146  // along this axis, we only need the nearer face:
147  face_extreme[axis] = near_extremes[axis];
148  result = std::min(result, (face_extreme - p).mag2());
149  }
150  return result;
151 }
