132  template <index_t N> Vec<T,N> unit() {
133  Vec<T,N> v;
134  do {
135  for (index_t i = 0; i < N; i++) {
136  // pick a point inside the unit cube
137  v[i] = this->rng->template rand<T>(-1,1);
138  }
139  // discard points outside the unit ball
140  } while (v.mag2() > 1);
141  return v.unit(); // return point on ball surface
142  }

246  Vec<T,3> cap(T half_angle_radians) {
247  // we pick a height with uniform probability
248  // (because a slice through the sphere with thickness t at height h
249  // has unchanging area regardless of h).
250  T min_h = cos(half_angle_radians);
251  T h = this->rng->template rand<T>(min_h, 1);
252  T r = sqrt(1 - h * h);
253  T theta = this->rng->template rand<T>(2*M_PI);
254  return Vec<T,3>(r * cos(theta), r * sin(theta), h);
255  }

264  inline Vec<T,3> cap(Vec<T,3> dir, T half_angle_radians) {
265  Vec<T,3> v = this->cap(half_angle_radians);
266  Vec<T,3> rotAxis = Vec<T,3>(0,0,1).cross(dir);
