58 template <typename T> void permute(T objs[], size_t len, Random& rng=*getRandom()) {
59  for (size_t i = 0; i < len; i++) {
60  size_t swapIdx = rng.rand(len - i) + i;
61  T tmp = objs[swapIdx];
62  objs[swapIdx] = objs[i];
63  objs[i] = tmp;

132  template <index_t N>
133  Vec<T,N> unit() {
134  Vec<T,N> v;
135  do {
136  for (index_t i = 0; i < N; i++) {
137  // pick a point inside the unit cube
138  v[i] = this->rng->template rand<T>(-1,1);
139  }
140  // discard points outside the unit ball
141  } while (v.mag2() > 1);
142  return v.unit(); // return point on ball surface
143  }

254  Vec<T,3> cap(T half_angle_radians) {
255  // we pick a height with uniform probability
256  // (because a slice through the sphere with thickness t at height h
257  // has unchanging area regardless of h).
258  T min_h = cos(half_angle_radians);
259  T h = this->rng->template rand<T>(min_h, 1);
260  T r = sqrt(1 - h * h);
261  T theta = this->rng->template rand<T>(2*M_PI);
262  return Vec<T,3>(r * cos(theta), r * sin(theta), h);
263  }

272  inline Vec<T,3> cap(Vec<T,3> dir, T half_angle_radians) {
273  Vec<T,3> v = this->cap(half_angle_radians);
274  Vec<T,3> rotAxis = Vec<T,3>(0,0,1).cross(dir);
275  if (rotAxis.isZero()) return v;
276  return v.rotate(rotAxis, std::acos(dir.z/dir.mag()));
277  }
