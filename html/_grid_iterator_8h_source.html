140  inline void increment() {
141  for (index_t i = dim_first;
142  i != dim_end and ++(pt[i]) > region.hi[i];
143  i += dim_increment) {
144  if (i != dim_last) pt[i] = region.lo[i];
145  }
146  }
147 
148  inline void decrement() {
149  for (index_t i = dim_first;
150  i != dim_end and --(pt[i]) < region.lo[i];
151  i += dim_increment) {
152  if (i != dim_last) pt[i] = region.hi[i] - 1;
153  }
154  }
155 
156  inline void advance(index_t n) {
157  point_t dest = point_t(0);
158  point_t dim = region.dimensions();
159  index_t vol = region.volume();
160 
161  for (index_t i = dim_last; i != dim_first - dim_increment; i -= dim_increment) {
162  vol = vol / dim[i];
163  dest[i] = n / vol;
164  n = n % vol;
165  }
166 
167  pt = dest;
168  }

170  inline index_t distance_to(const GridIterator<T,N,Order> &other) const {
171  // TODO: xxx: test this. handling of volume's sign is suspicious.
172  // TODO: xxx: handling of reverse case is suspicious too.
173  point_t dim = region.dimensions();
174  point_t jump = other.pt - pt;
175  index_t dist = 0;
176  index_t vol = 1;
177  for (index_t i = dim_first; i != dim_end; i += dim_increment) {
178  dist += vol * jump[i];
179  vol *= std::max(dim[i], (T)0);
180  }
181  return dist;
182  }
