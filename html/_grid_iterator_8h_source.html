135  inline void increment() {
136  for (index_t i = dim_first;
137  i != dim_end and ++(pt[i]) >= region.max()[i];
138  i += dim_increment) {
139  if (i != dim_last) pt[i] = region.min()[i];
140  }
141  }

143  inline void decrement() {
144  for (index_t i = dim_first;
145  i != dim_end and --(pt[i]) < region.min()[i];
146  i += dim_increment) {
147  if (i != dim_last) pt[i] = region.max()[i] - 1;
148  }
149  }

151  inline void advance(index_t n) {
152  point_t dest = point_t(0);
153  point_t dim = region.getDimensions();
154  index_t vol = region.volume();
155 
156  for (index_t i = dim_last; i != dim_first - dim_increment; i -= dim_increment) {
157  vol = vol / dim[i];
158  dest[i] = n / vol;
159  n = n % vol;
160  }
161 
162  pt = dest;
163  }

165  inline index_t distance_to(const GridIterator<T,N,Order> &other) const {
166  // TODO: xxx: test this. handling of volume's sign is suspicious.
167  // TODO: xxx: handling of reverse case is suspicious too.
168  point_t dim = region.getDimensions();
169  point_t jump = other.pt - pt;
170  index_t dist = 0;
171  index_t vol = 1;
172  for (index_t i = dim_first; i != dim_end; i += dim_increment) {
173  dist += vol * jump[i];
174  vol *= dim[i];
175  }
