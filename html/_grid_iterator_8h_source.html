48template <typename T, index_t N, ArrayOrder Order>
49class GridIterator :
50 public boost::iterator_facade
51 <
52 GridIterator<T,N,Order>, // self type
53 typename PointType<T,N>::point_t, // value (pointed to) type
54 std::random_access_iterator_tag, // implemented concepts (all)
55 const typename PointType<T,N>::point_t& // reference to elem type (may be a proxy)
56 >
57{
58
59 static constexpr bool RowMaj = (Order == ARRAYORDER_FIRST_DIM_CONSECUTIVE);
60 static constexpr index_t dim_first = RowMaj ? 0 : (N - 1);
61 static constexpr index_t dim_last = RowMaj ? (N - 1) : 0;
62 static constexpr index_t dim_end = RowMaj ? N : -1;
63 static constexpr index_t dim_increment = RowMaj ? 1 : -1;
64
65public:
67 static constexpr ArrayOrder order = Order;
69 typedef typename PointType<T,N>::point_t point_t;
70
72 const Rect<T,N> region;
74 point_t pt;
75
79 GridIterator(const Rect<T,N> &r):
80 region(r),
81 pt(region.lo) {}
87 GridIterator(const Rect<T,N> &r, const point_t &p):
88 region(r),
89 pt(p) {}
90
97 GridIterator(const point_t &lo, const point_t &hi):
98 region(lo, hi - point_t(1)),
99 pt(lo) {}
100
104 inline GridIterator<T,N,Order> begin() const {
105 GridIterator<T,N,Order> other = *this;
106 other.pt = region.lo;
107 return other;
108 }
109
114 inline GridIterator<T,N,Order> end() const {
115 GridIterator<T,N,Order> other = *this;
116 other.pt = region.lo;
117 // one beyond the highest cell. i.e. first cell in
118 // (N-1)D block just beyond the end of this region.
119 other.pt[dim_last] = region.hi[dim_last] + 1;
120 return other;
121 }
122
123private:
124
125 friend class boost::iterator_core_access;
126
127 inline bool equal(const GridIterator<T,N,Order> &other) const {
128 return pt == other.pt;
129 }
130
131 inline void increment() {
132 for (index_t i = dim_first;
133 i != dim_end and ++(pt[i]) > region.hi[i];
134 i += dim_increment) {
135 if (i != dim_last) pt[i] = region.lo[i];
136 }
137 }
138
139 inline void decrement() {
140 for (index_t i = dim_first;
141 i != dim_end and --(pt[i]) < region.lo[i];
142 i += dim_increment) {
143 if (i != dim_last) pt[i] = region.hi[i] - 1;
144 }
145 }
146
147 inline void advance(index_t n) {
148 point_t dest = point_t(0);
149 point_t dim = region.dimensions();
150 index_t vol = region.volume();
151
152 for (index_t i = dim_last; i != dim_first - dim_increment; i -= dim_increment) {
153 vol = vol / dim[i];
154 dest[i] = n / vol;
155 n = n % vol;
156 }
157
158 pt = dest;
159 }
160
161 inline index_t distance_to(const GridIterator<T,N,Order> &other) const {
162 point_t dim = region.dimensions();
163 point_t jump = other.pt - pt;
164 index_t dist = 0;
165 index_t vol = 1;
166 for (index_t i = dim_first; i != dim_end; i += dim_increment) {
167 dist += vol * jump[i];
168 vol *= std::max(dim[i], (T)0);
169 }
170 return dist;
171 }
172
173 inline const point_t& dereference() const {
174 return pt;
175 }
176
177};
