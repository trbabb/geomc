56 T sdf(point_t p) const {
57 if (p.is_zero()) return 1;
58 if (is_inside_angle(p)) {
59 // distance to the sphere surface
60 return p.mag() - 1;
61 } else {
62 // distance to the boundary of the cap

64 T s = std::sin(half_angle_radians);
65 T c = std::cos(half_angle_radians);
66 T z = p[N - 1];
67 T x = mag(p_proj);
68 T dx = x - s;
69 T dz = z - c;

147 Rect<T,1> intersect(const Ray<T,N>& ray) const {
148 Rect<T,1> sphere_interval = Sphere<T,N>().intersect(ray);
149 if (sphere_interval.is_empty()) return sphere_interval;
150 T ab[2] = {sphere_interval.lo, sphere_interval.hi};

152 for (index_t i = 0; i < 2; ++i) {

184 bool is_inside_angle(point_t p) const {
185 T m2 = p.mag2();
186 T z = p[N - 1];
187 if (m2 == 0) return true;
188 T t = geom::positive_mod<T>(half_angle_radians / std::numbers::pi_v<T>, 2);
189 // half-turns from the axis, in [-0.5, 0.5]
190 // 0 is the equatorial plane, -0.5 is +pole, 0.5 is -pole
191 T u = 0.5 - std::abs(1 - t);
192 // split into two cases for better numerical precision
193 if (std::abs(u) < 0.25){
194 // point is closer to the equatorial plane;
195 // test based on the height above the equatorial plane,
196 // where angle mostly affects height
197 T c = std::cos(half_angle_radians);
198 T s_cap = c < 0 ? -1 : 1;
199 T s_pt = z < 0 ? -1 : 1;
200 return s_pt * z * z / m2 >= s_cap * c * c;
201 } else {
202 // point is closer to the axis;
203 // test based on the distance from the axis,
204 // where angle mostly affects radius
205 T s = std::sin(half_angle_radians);
206 VecType<T,N-1> p_base = p.template resized<N-1>();
207 T k = mag2(p_base) / m2;
208 if (u < 0) {
209 // angle is above the equator; area nearest the axis is included.
210 // everything below the equator is excluded.
211 return k <= s * s and z > 0;
212 } else {
213 // angle is below the equator; area nearest the axis is excluded.
214 // everything above the equator is included.
215 return k >= s * s or z > 0;
216 }
217 }
218 }
