42 template <typename T>
43 Hit<T,3> trace_tri(const Vec<T,3> &p0,
44  const Vec<T,3> &p1,
45  const Vec<T,3> &p2,
46  const Ray<T,3> &r,
47  HitSide sides=HIT_FRONT){
48  Hit<T,3> hit(r,sides);
49 
50  Vec<T,3> dP[2] = {p1 - p0, p2 - p0};
51  Plane<T,3> plane = Plane<T,3>::from_basis(dP, p0);
52  T s;
53 
54  if (!detail::_ImplTracePlane(&s, plane, r, &sides)) {
55  return hit;
56  }
57 
58  Vec<T,3> p = r.atMultiple(s);
59 
60  // dPdu, dPdv, and N form a 3d basis.
61  // thus we can solve:
62  // u*dPdu + v*dPdv + t*N = P
63  // t should essentially be zero; we can ignore it.
64 
65  // 3d matrix inversion is faster than you think.
66  // also, gcc tends to unroll those copy ops.
67  SimpleMatrix<T,3,3> mtx;
68  std::copy(dP[0].begin(), dP[0].end(), mtx.col(0));
69  std::copy(dP[1].begin(), dP[1].end(), mtx.col(1));
70  std::copy(plane.normal.begin(), plane.normal.end(), mtx.col(2));
71  inv(&mtx, mtx);
72  Vec<T,3> uvn = mtx * (p-p0);
73 
74  T u = uvn.x;
75  T v = uvn.y;
76 
77  if (u < 0 || v < 0 || u + v > 1){
78  //intersection outside of triangle
79  return hit;
80  }
81 
82  hit.p = p;
83  hit.n = plane.normal;
84  hit.s = s;
85  hit.side = sides; //set by ImplTracePlane
86  hit.hit = true;
87 
88  return hit;
89 }

92 Hit<T,3> trace_planar_quad(const Vec<T,3> &dPdu, const Vec<T,3> &dPdv, const Vec<T,3> &origin, const Ray<T,3> &ray, HitSide sides){
93  Hit<T,3> hit(ray,sides);
94  Vec<T,3> dP[2] = {dPdu, dPdv};
95  Plane<T,3> pl = Plane<T,3>::from_basis(dP, origin);
96  T s;
97 
98  if (!detail::_ImplTracePlane(&s, pl, ray, &sides)){
99  // miss
100  return hit;
101  }
102 
103  Vec<T,3> p = ray.atMultiple(s);
104 
105  // solve for u and v
106  SimpleMatrix<T,3,3> mtx;
107  std::copy(dPdu.begin(), dPdu.end(), mtx.col(0));
108  std::copy(dPdv.begin(), dPdv.end(), mtx.col(1));
109  std::copy(pl.normal.begin(), pl.normal.end(), mtx.col(2));
110  inv(&mtx, mtx);
111  Vec<T,3> uvn = mtx * (p-origin);
112 
113  T u = uvn.x;
114  T v = uvn.y;
115 
116  if (u < 0 or u > 1 or v < 0 or v > 1){
117  // missed quad area
118  return hit;
119  }
120 
121  hit.p = p;
122  hit.n = pl.normal;
123  hit.s = s;
124  hit.side = sides; // set by ImplTracePlane
125  hit.hit = true;
