148 template <typename T, index_t N>
149 bool trace_simplex(const Vec<T,N> verts[N], const Ray<T,N>& ray, Vec<T,N-1>* uv, T* s) {
150  /* Solve for u,v coordinates along the edges radiating from `verts[0]`:
151  A + u(B-A) + v(C-A) = o + sV
152  u(B-A) + v(C-A) - sV = o - A
153  Mx = o - A
154  */
155 
156  // todo: cramer's rule might be used in 3d for faster linear solve.
157  // (note: is unstable).
158 
159  // compute bases
160  Vec<T,N> bases[N];
161  for (index_t i = 1; i < N; ++i) {
162  bases[i - 1] = verts[i] - verts[0];
163  }
164  bases[N - 1] = -ray.direction;
165 
166  // linear solve
167  Vec<T,N> x;
168  Vec<T,N> b = ray.origin - verts[0];
169  if (not linearSolve(bases, &x, b)) return false;
170 
171  // inside simplex?
172  T sum = 0;
173  for (index_t i = 0; i < N - 1; ++i) {
174  if (x[i] < 0) return false;
175  sum += x[i];
176  }
177  if (sum > 1) return false;
178 
179  // output result
180  *s = x[N];
181  *uv = x.template resized<N-1>();
182  return true;
183 }

194 template <typename T>
195 Hit<T,3> trace_tri(const Vec<T,3> &p0,
196  const Vec<T,3> &p1,
197  const Vec<T,3> &p2,
198  const Ray<T,3> &r,
199  HitSide sides=HIT_FRONT){
200  Hit<T,3> hit(r,sides);
201 
202  Vec<T,3> dP[2] = {p1 - p0, p2 - p0};
203  Plane<T,3> plane = Plane<T,3>::from_basis(dP, p0);
204  T s;
205 
206  if (!detail::_ImplTracePlane(&s, plane, r, &sides)) {
207  return hit;
208  }
209 
210  Vec<T,3> p = r.atMultiple(s);
211 
212  // dPdu, dPdv, and N form a 3d basis.
213  // thus we can solve:
214  // u*dPdu + v*dPdv + t*N = P
215  // t should essentially be zero; we can ignore it.
216  // xxx: ^ that reeks of stupid. can we solve this in one go?
217 
218  // 3d matrix inversion is faster than you think.
219  // also, gcc tends to unroll those copy ops.
220  SimpleMatrix<T,3,3> mtx;
221  std::copy(dP[0].begin(), dP[0].end(), mtx.col(0));
222  std::copy(dP[1].begin(), dP[1].end(), mtx.col(1));
223  std::copy(plane.normal.begin(), plane.normal.end(), mtx.col(2));
224  inv(&mtx, mtx);
225  Vec<T,3> uvn = mtx * (p-p0);
226 
227  T u = uvn.x;
228  T v = uvn.y;
229 
230  if (u < 0 || v < 0 || u + v > 1) {
231  //intersection outside of triangle
232  return hit;
233  }
234 
235  hit.p = p;
236  hit.n = plane.normal;
237  hit.s = s;
238  hit.side = sides; //set by ImplTracePlane
239  hit.hit = true;
240 
241  return hit;
242 }

246 Hit<T,3> trace_planar_quad(
247  const Vec<T,3> &dPdu,
248  const Vec<T,3> &dPdv,
249  const Vec<T,3> &origin,
250  const Ray<T,3> &ray,
251  HitSide sides) {
252  Hit<T,3> hit(ray,sides);

254  Plane<T,3> pl = Plane<T,3>::from_basis(dP, origin);
255  T s;
256 
257  if (!detail::_ImplTracePlane(&s, pl, ray, &sides)){
258  // miss
259  return hit;
260  }
261 
262  Vec<T,3> p = ray.atMultiple(s);
263 
264  // solve for u and v
265  SimpleMatrix<T,3,3> mtx;
266  std::copy(dPdu.begin(), dPdu.end(), mtx.col(0));
267  std::copy(dPdv.begin(), dPdv.end(), mtx.col(1));
268  std::copy(pl.normal.begin(), pl.normal.end(), mtx.col(2));
269  inv(&mtx, mtx);
270  Vec<T,3> uvn = mtx * (p-origin);
271 
272  T u = uvn.x;
273  T v = uvn.y;
274 
275  if (u < 0 or u > 1 or v < 0 or v > 1){
276  // missed quad area
277  return hit;
278  }
279 
280  hit.p = p;
281  hit.n = pl.normal;
282  hit.s = s;
283  hit.side = sides; // set by ImplTracePlane
284  hit.hit = true;

286  return hit;

308 template <typename T, index_t N>
309 bool trace_sphere(const Vec<T,N>& center, T radius, const Ray<T,N> &ray, T* s) {
310  T r2 = radius * radius;
311  // if inside, we are guaranteed to hit the back. return miss if not tracing backface.
312  const Vec<T,N>& dir = ray.direction; //for shorthand
313  Vec<T,N> x0 = center - ray.origin;
314  // solve for s such that ||s * ray - ctr|| == radius
315  T a = dir.dot(dir);
316  T b = -2 * dir.dot(x0);
317  T c = x0.dot(x0) - r2;
318  T roots[2];
319  if (quadratic_solve(roots, a, b, c)) {
320  if (roots[1] < roots[0]) {
321  std::swap(roots[0], roots[1]);
322  }
323  T positive_root = (roots[0] >= 0) ? roots[0] : roots[1];
324  if (positive_root >= 0) {
325  *s = positive_root;
326  return true;
327  }
328  }
329  // no intersection; return miss.
330  return false;
331 }
