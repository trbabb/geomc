99 Rect<T,N> bounds() const requires BoundedObject<Shape> {

102 // test along a cardinal axis in shape-space
103 Vec<T,N> d;
104 d[axis] = 1;
105 d = xf.apply_inverse_direction(d);
106 Vec<T,N> p_lo = xf * shape.convex_support( d);
107 Vec<T,N> p_hi = xf * shape.convex_support(-d);
108 r.lo[axis] = p_lo[axis];
109 r.hi[axis] = p_hi[axis];
110 }
