39template <typename Shape>
40class Transformed:
41 public Convex <typename Shape::elem_t, Shape::N, Transformed<Shape>>,
42 public RayIntersectable<typename Shape::elem_t, Shape::N, Transformed<Shape>>
43{
44public:
46 using T = typename Shape::elem_t;
48 static constexpr size_t N = Shape::N;
49
51 Shape shape;
53 AffineTransform<T,N> xf;
54
56 Transformed() {}
57
59 explicit Transformed(const Shape& s):shape(s) {}
60
62 Transformed(const Shape& s, const AffineTransform<T,N>& xf):
63 shape(s),
64 xf(xf) {}
65
66 bool operator==(const Transformed& other) const {
67 return shape == other.shape && xf == other.xf;
68 }
69
71 bool contains(Vec<T,N> p) const {
72 p = p / xf;
73 return shape.contains(p);
74 }
75
76 Vec<T,N> convex_support(Vec<T,N> d) const {
77 d = xf.apply_inverse_normal(d);
78 Vec<T,N> p = shape.convex_support(d);
79 return xf * p;
80 }
81
82 Rect<T,N> bounds() const {
83 // the base class implementation would work; but we specialize
84 // here for a little extra performance (avoid stupidly re-computing `d`).
85 Rect<T,N> r;
86 for (index_t axis = 0; axis < N; ++axis) {
87 // we want to test along a world cardinal axis in shape-space coordinates.
88 // but this direction is a normal, which is transformed
89 // with the inverse transpose. so we take a row of xf.mat instead of
90 // a col from xf.inv:
91 Vec<T,N> basis(xf.mat.row(axis));
92 // find the body-space extreme and transform it to world space:
93 Vec<T,N> p_hi = xf * shape.convex_support( basis);
94 Vec<T,N> p_lo = xf * shape.convex_support(-basis);
95 // update the extremes along the test axis
96 r.hi[axis] = p_hi[axis];
97 r.lo[axis] = p_lo[axis];
98 }
99 return r;
100 }
101
103 Rect<T,1> intersect(const Ray<T,N>& r) const {
104 return shape.intersect(r / xf);
105 }
106
108 inline Transformed<Rect<T,N>> transformed_bounds() const {
109 return xf * shape.bounds();
110 }
111
112
113}; // class Transformed

120template <typename T, index_t N>
121class Transformed<Rect<T,N>>:
122 public Convex<T,N,Transformed<Rect<T,N>>>,
123 public RayIntersectable<T,N,Transformed<Rect<T,N>>>
124{
125public:
126
128 Rect<T,N> shape;
130 AffineTransform<T,N> xf; // moves object points to world positions
131
133 Transformed() {}
134
136 explicit Transformed(const Rect<T,N>& box):shape(box) {}
137
139 Transformed(const Rect<T,N>& box, const AffineTransform<T,N>& xf):
140 shape(box),
141 xf(xf) {}
142
143 bool operator==(const Transformed& other) const {
144 return shape == other.shape && xf == other.xf;
145 }
146
148 Rect<T,N> bounds() const {
149 // faster than using convex_support, somewhat surprisingly.
150 Vec<T,N> pts[2] = {shape.lo, shape.hi};
151 Vec<T,N> lo = xf * shape.lo;
152 Vec<T,N> hi = lo;
153
154 // for each corner
155 for (index_t i = 1; i < (1 << N); i++) {
156 Vec<T,N> p;
157 // for each coordinate
158 for (index_t j = 0; j < N; j++) {
159 p[j] = pts[(i & (1 << j)) != 0][j];
160 }
161 p = xf * p;
162 lo = std::min(lo, p);
163 hi = std::max(hi, p);
164 }
165
166 return Rect<T,N>(lo, hi);
167 }
168
170 inline Transformed<Rect<T,N>> transformed_bounds() const {
171 return *this;
172 }
173
174 Vec<T,N> convex_support(Vec<T,N> d) const {
175 Vec<T,N> d_body = xf.apply_inverse_normal(d);
176 Vec<T,N> o;
177 for (index_t i = 0; i < N; i++) {
178 o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
179 }
180 return xf * o;
181 }
182
187 bool contains(Vec<T,N> p) const {
188 p = p / xf; // world coord to object coord
189 return shape.contains(p);
190 }
191
198 bool intersects(const Rect<T,N>& r) const {
199 return detail::RectIntersector<T,N>::intersect(*this, r);
200 }
201
208 bool intersects(const Transformed< Rect<T,N> >& b1) {
209 // we will make ourselves axis-aligned; this is the same as applying
210 // the inverse of our xf. We apply the inverse of xf to b1 too,
211 // to preserve the relationhip between the us. From this, we
212 // fallback to an ORect <-> Rect test.
213 Transformed< Rect<T,N> > b1_in_b0 = b1 / xf;
214
215 return b1_in_b0.intersects(shape);
216 }
217
219 Rect<T,1> intersect(const Ray<T,N>& r) const {
220 return shape.intersect(r / xf);
221 }
222
224 T volume() const {
225 SimpleMatrix<T,N,N> mx;
226 for (index_t i = 0; i < N; ++i) {
227 // mult this basis vector by the length of the rect along that axis
228 T s = shape.hi[i] - shape.lo[i];
229 for (index_t j = 0; j < N; ++j) {
230 mx(j,i) = s * xf.mat(j,i);
231 }
232 }
233 return det_destructive(mx.data_begin(), N);
234 }

236}; // class Transformed<Rect>
