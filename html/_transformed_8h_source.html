44template <typename Shape>
45class Transformed: public Dimensional<typename Shape::elem_t, Shape::N> {
46public:
48 using T = typename Shape::elem_t;
50 static constexpr size_t N = Shape::N;
51
53 Shape shape;
55 AffineTransform<T,N> xf;
56
58 Transformed() {}
59
61 explicit Transformed(const Shape& s):shape(s) {}
62
64 Transformed(const Shape& s, const AffineTransform<T,N>& xf):
65 shape(s),
66 xf(xf) {}
67
68 static constexpr bool admits_cusps() { return true; }
69
70 bool operator==(const Transformed& other) const {
71 return shape == other.shape && xf == other.xf;
72 }
73
75 bool contains(Vec<T,N> p) const requires RegionObject<Shape> {
76 p = p / xf;
77 return shape.contains(p);
78 }
79
80 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
81 d = xf.apply_inverse_normal(d);
82 Vec<T,N> p = shape.convex_support(d);
83 return xf * p;
84 }

86 Rect<T,N> bounds() const requires BoundedObject<Shape> {
87 // the base class implementation would work; but we specialize
88 // here for a little extra performance (avoid stupidly re-computing `d`).
89 Rect<T,N> r;
90 for (index_t axis = 0; axis < N; ++axis) {
91 // we want to test along a world cardinal axis in shape-space coordinates.
92 // but this direction is a normal, which is transformed
93 // with the inverse transpose. so we take a row of xf.mat instead of
94 // a col from xf.inv:
95 Vec<T,N> basis(xf.mat.row(axis));
96 // find the body-space extreme and transform it to world space:
97 Vec<T,N> p_hi = xf * shape.convex_support( basis);
98 Vec<T,N> p_lo = xf * shape.convex_support(-basis);
99 // update the extremes along the test axis
100 r.hi[axis] = p_hi[axis];
101 r.lo[axis] = p_lo[axis];
102 }
103 return r;
104 }
105
107 T measure_interior() const requires InteriorMeasurableObject<Shape> {
108 // exclude the translation component
109 SimpleMatrix<T,N,N> m = xf.direction_matrix();
110 T s = 1;
111 if constexpr (N <= 4) {
112 // in-place det is defined and fast for 4x4 and smaller
113 s = geom::det(m);
114 } else {
115 // avoid copying the matrix in the case where it's large
116 s = geom::det_destructive(m.data_begin(), N);
117 }
118 return std::abs(s) * shape.measure_interior();
119 }
120
121 template <ConvexObject S>
122 requires NDimensional<S,T,N>
123 bool intersects(const S& other) const {
124 return geom::intersects(
125 as_any_convex(*this),
126 as_any_convex(other)
127 );
128 }
129
131 Rect<T,1> intersect(const Ray<T,N>& r) const requires RayIntersectableObject<Shape> {
132 return shape.intersect(r / xf);
133 }
134
136 inline Transformed<Rect<T,N>> transformed_bounds() const requires BoundedObject<Shape> {
137 return xf * shape.bounds();
138 }

140}; // class Transformed

147template <typename T, index_t N>
148class Transformed<Rect<T,N>> : public Dimensional<T,N> {
149public:
150
152 Rect<T,N> shape;
154 AffineTransform<T,N> xf; // moves object points to world positions
155
157 Transformed() {}
158
160 explicit Transformed(const Rect<T,N>& box):shape(box) {}
161
163 Transformed(const Rect<T,N>& box, const AffineTransform<T,N>& xf):
164 shape(box),
165 xf(xf) {}
166
167 bool operator==(const Transformed& other) const {
168 return shape == other.shape && xf == other.xf;
169 }
170
172 Rect<T,N> bounds() const {
173 // faster than using convex_support, somewhat surprisingly.
174 Vec<T,N> pts[2] = {shape.lo, shape.hi};
175 Vec<T,N> lo = xf * shape.lo;
176 Vec<T,N> hi = lo;
177
178 // for each corner
179 for (index_t i = 1; i < (1 << N); i++) {
180 Vec<T,N> p;
181 // for each coordinate
182 for (index_t j = 0; j < N; j++) {
183 p[j] = pts[(i & (1 << j)) != 0][j];
184 }
185 p = xf * p;
186 lo = std::min(lo, p);
187 hi = std::max(hi, p);
188 }
189
190 return Rect<T,N>(lo, hi);
191 }
192
194 inline Transformed<Rect<T,N>> transformed_bounds() const {
195 return *this;
196 }
197
198 Vec<T,N> convex_support(Vec<T,N> d) const {
199 Vec<T,N> d_body = xf.apply_inverse_normal(d);
200 Vec<T,N> o;
201 for (index_t i = 0; i < N; i++) {
202 o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
203 }
204 return xf * o;
205 }
206
211 bool contains(Vec<T,N> p) const {
212 p = p / xf; // world coord to object coord
213 return shape.contains(p);
214 }
215
216 template <ConvexObject S>
217 requires NDimensional<S,T,N>
218 bool intersects(const S& other) const {
219 return geom::intersects(
220 as_any_convex(*this),
221 as_any_convex(other)
222 );
223 }
224
231 bool intersects(const Rect<T,N>& r) const {
232 return detail::RectIntersector<T,N>::intersect(*this, r);
233 }
234
241 bool intersects(const Transformed<Rect<T,N>>& b1) {
242 // we will make ourselves axis-aligned; this is the same as applying
243 // the inverse of our xf. We apply the inverse of xf to b1 too,
244 // to preserve the relationhip between the us. From this, we
245 // fallback to an ORect <-> Rect test.
246 Transformed<Rect<T,N>> b1_in_b0 = b1 / xf;
247
248 return b1_in_b0.intersects(shape);
249 }
250
252 Vec<T,N> corner(index_t i) const {
253 return xf * shape.corner(i);
254 }
255
257 Rect<T,1> intersect(const Ray<T,N>& r) const {
258 return shape.intersect(r / xf);
259 }
260
262 T measure_interior() const {
263 SimpleMatrix<T,N,N> mx;
264 for (index_t i = 0; i < N; ++i) {
265 // mult this basis vector by the length of the rect along that axis
266 T s = shape.hi[i] - shape.lo[i];
267 for (index_t j = 0; j < N; ++j) {
268 mx(j,i) = s * xf.mat(j,i);
269 }
270 }
271 return det_destructive(mx.data_begin(), N);
272 }
