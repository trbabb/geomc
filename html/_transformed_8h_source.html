44template <typename Shape>
45class Transformed:
46 public Convex <typename Shape::elem_t, Shape::N, Transformed<Shape>>,
47 public RayIntersectable<typename Shape::elem_t, Shape::N, Transformed<Shape>>
48{
49public:
51 using T = typename Shape::elem_t;
53 static constexpr size_t N = Shape::N;
54
56 Shape shape;
58 AffineTransform<T,N> xf;
59
61 Transformed() {}
62
64 explicit Transformed(const Shape& s):shape(s) {}
65
67 Transformed(const Shape& s, const AffineTransform<T,N>& xf):
68 shape(s),
69 xf(xf) {}
70
71 static constexpr bool admits_cusps() { return true; }
72
73 bool operator==(const Transformed& other) const {
74 return shape == other.shape && xf == other.xf;
75 }
76
78 bool contains(Vec<T,N> p) const requires RegionObject<Shape> {
79 p = p / xf;
80 return shape.contains(p);
81 }
82
83 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
84 d = xf.apply_inverse_normal(d);
85 Vec<T,N> p = shape.convex_support(d);
86 return xf * p;
87 }
88
89 Rect<T,N> bounds() const requires BoundedObject<Shape> {
90 // the base class implementation would work; but we specialize
91 // here for a little extra performance (avoid stupidly re-computing `d`).
92 Rect<T,N> r;
93 for (index_t axis = 0; axis < N; ++axis) {
94 // we want to test along a world cardinal axis in shape-space coordinates.
95 // but this direction is a normal, which is transformed
96 // with the inverse transpose. so we take a row of xf.mat instead of
97 // a col from xf.inv:
98 Vec<T,N> basis(xf.mat.row(axis));
99 // find the body-space extreme and transform it to world space:
100 Vec<T,N> p_hi = xf * shape.convex_support( basis);
101 Vec<T,N> p_lo = xf * shape.convex_support(-basis);
102 // update the extremes along the test axis
103 r.hi[axis] = p_hi[axis];
104 r.lo[axis] = p_lo[axis];
105 }
106 return r;
107 }
108
110 T measure_interior() const requires InteriorMeasurableObject<Shape> {
111 return std::abs(geom::det(xf.mat)) * shape.measure_interior();
112 }
113
115 Rect<T,1> intersect(const Ray<T,N>& r) const requires RayIntersectableObject<Shape> {
116 return shape.intersect(r / xf);
117 }
118
120 inline Transformed<Rect<T,N>> transformed_bounds() const requires BoundedObject<Shape> {
121 return xf * shape.bounds();
122 }
123
124}; // class Transformed

131template <typename T, index_t N>
132class Transformed<Rect<T,N>>:
133 public Convex<T,N,Transformed<Rect<T,N>>>,
134 public RayIntersectable<T,N,Transformed<Rect<T,N>>>
135{
136public:
137
139 Rect<T,N> shape;
141 AffineTransform<T,N> xf; // moves object points to world positions
142
144 Transformed() {}
145
147 explicit Transformed(const Rect<T,N>& box):shape(box) {}
148
150 Transformed(const Rect<T,N>& box, const AffineTransform<T,N>& xf):
151 shape(box),
152 xf(xf) {}
153
154 bool operator==(const Transformed& other) const {
155 return shape == other.shape && xf == other.xf;
156 }
157
159 Rect<T,N> bounds() const {
160 // faster than using convex_support, somewhat surprisingly.
161 Vec<T,N> pts[2] = {shape.lo, shape.hi};
162 Vec<T,N> lo = xf * shape.lo;
163 Vec<T,N> hi = lo;
164
165 // for each corner
166 for (index_t i = 1; i < (1 << N); i++) {
167 Vec<T,N> p;
168 // for each coordinate
169 for (index_t j = 0; j < N; j++) {
170 p[j] = pts[(i & (1 << j)) != 0][j];
171 }
172 p = xf * p;
173 lo = std::min(lo, p);
174 hi = std::max(hi, p);
175 }
176
177 return Rect<T,N>(lo, hi);
178 }
179
181 inline Transformed<Rect<T,N>> transformed_bounds() const {
182 return *this;
183 }
184
185 Vec<T,N> convex_support(Vec<T,N> d) const {
186 Vec<T,N> d_body = xf.apply_inverse_normal(d);
187 Vec<T,N> o;
188 for (index_t i = 0; i < N; i++) {
189 o[i] = ((d_body[i] < 0) ? shape.lo : shape.hi)[i];
190 }
191 return xf * o;
192 }
193
198 bool contains(Vec<T,N> p) const {
199 p = p / xf; // world coord to object coord
200 return shape.contains(p);
201 }
202
203 using Convex<T,N,Transformed<Rect<T,N>>>::intersects;
204
211 bool intersects(const Rect<T,N>& r) const {
212 return detail::RectIntersector<T,N>::intersect(*this, r);
213 }
214
221 bool intersects(const Transformed< Rect<T,N> >& b1) {
222 // we will make ourselves axis-aligned; this is the same as applying
223 // the inverse of our xf. We apply the inverse of xf to b1 too,
224 // to preserve the relationhip between the us. From this, we
225 // fallback to an ORect <-> Rect test.
226 Transformed< Rect<T,N> > b1_in_b0 = b1 / xf;
227
228 return b1_in_b0.intersects(shape);
229 }
230
232 Rect<T,1> intersect(const Ray<T,N>& r) const {
233 return shape.intersect(r / xf);
234 }
235
237 T measure_interior() const {
238 SimpleMatrix<T,N,N> mx;
239 for (index_t i = 0; i < N; ++i) {
240 // mult this basis vector by the length of the rect along that axis
241 T s = shape.hi[i] - shape.lo[i];
242 for (index_t j = 0; j < N; ++j) {
243 mx(j,i) = s * xf.mat(j,i);
244 }
245 }
246 return det_destructive(mx.data_begin(), N);
247 }
