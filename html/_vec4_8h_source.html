38 template <typename T>
39 const Vec<T,4> composite_over(Vec<T,4> fg, Vec<T,4> bg) {
40 Vec<T,3> c_fg = fg.template resized<3>();
41 Vec<T,3> c_bg = bg.template resized<3>();
42 T a_fg = fg.w;
43 T a_bg = bg.w;
44 T a_out = a_fg + (1 - a_fg) * a_bg;
45 if (a_out == 0) {
46 return fg + bg;
47 }
48 return {
49 (c_fg * a_fg + c_bg * a_bg * (1 - a_fg)) / a_out,
50 a_out
51 };
52 }
