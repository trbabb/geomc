21 template <typename T> const Vec<T,4> from_argb(int aRGB) {
22 T a = ((aRGB & 0xff000000) >> 24) / detail::_RGBChannelConversion<T>::factor;
23 T r = ((aRGB & 0x00ff0000) >> 16) / detail::_RGBChannelConversion<T>::factor;
24 T g = ((aRGB & 0x0000ff00) >> 8) / detail::_RGBChannelConversion<T>::factor;
25 T b = (aRGB & 0x000000ff) / detail::_RGBChannelConversion<T>::factor;
26 return Vec<T,4>(r,g,b,a);
27 }
