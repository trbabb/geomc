170 SmallStorage<T,N>& operator=(SmallStorage<T,N>&& other) {
171 _destroy(_data, _size);
172 // if we have allocated storage, free it. we're going to clobber
173 if (_capacity > N) delete [] _data;
174 _capacity = other._capacity;

177 // the donor container has allocated storage, just grab it
