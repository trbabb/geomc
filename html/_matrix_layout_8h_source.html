46 // compute the new memory location

48 // handle "off-end" correctly:

50 index_t j = i % s;
51 index_t k = i / s;
52 // compute transposed offset:
53 return base + (j % minor) * major + (j / minor) + k * s;

85 inline index_t distance_to(const TransposeIterator<T,K>& other) const {
86 // note the symmetry: this is the same as offs(), but major & minor are swapped.
87 // that scans: to invert a transpose, transpose again!
88 index_t s = major * minor;
89 index_t q = other.p - base;
90 index_t j = q % s;

92 index_t o_i = (j % major) * minor + (j / major) + k * s;
