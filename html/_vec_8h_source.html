200/* Declare the Vec type that we'll actually use, as a subclass of the hidden base template
201 * class. We do this so that all sizes of vector can share common code, since a template that
202 * is merely specialized does not implicitly inherit functionality from the base template.
203 * With this method, Vec<T,2> can be a specialization of Vec<T,N>, but still keep the
204 * functionality of VecBase<T,N>.
205 *
206 * All vectors are sizeof(T)*N and can be packed.
207 */
208
268template <typename T, index_t N>
269class Vec : public detail::VecCommon< T,N,Vec<T,N> > {
270public:
271
272 // to prevent c++20 ambiguity warnings:
273 using detail::VecCommon<T,N,Vec<T,N>>::operator==;
274
278 constexpr Vec():detail::VecCommon< T,N,Vec<T,N> >() {}
279
285 constexpr Vec(T a):detail::VecCommon< T,N,Vec<T,N> >(a) {}
286
291 constexpr Vec(const T a[N]):detail::VecCommon< T,N,Vec<T,N> >(a) {}
292
299 template <typename U>
300 constexpr Vec(const Vec<U,N-1> &v, T a) {
301 std::copy(v.begin(), v.end(), detail::VecBase<T,N>::begin());
302 this->get(N-1) = a;
303 }
304
309 template <typename Mx, typename Ref>
310 Vec(detail::MtxColIterator<Mx,Ref> mtx_col) {
311 T *p = this->begin();
312 const index_t n = std::min(N, mtx_col->mtx->cols());
313 for (index_t i = 0; i < n; i++, p++, mtx_col++) {
314 *p = *mtx_col;
315 }
316 }

324#if __cplusplus >= 201103L or PARSING_DOXYGEN
325 Vec(const std::initializer_list<T>& items):
326 detail::VecCommon< T,N,Vec<T,N> >(items.begin())
327 {
328#if __cplusplus >= 201402L
329 // items.size() not constexpr in c++11 D:<
330 static_assert(items.size() == N, "Vector must be initialized with N items");
331#endif
332 }
333#endif

335}; /* class Vec */
