202/* Declare the Vec type that we'll actually use, as a subclass of the hidden base template
203 * class. We do this so that all sizes of vector can share common code, since a template that
204 * is merely specialized does not implicitly inherit functionality from the base template.
205 * With this method, Vec<T,2> can be a specialization of Vec<T,N>, but still keep the
206 * functionality of VecBase<T,N>.
207 *
208 * All vectors are sizeof(T)*N and can be packed.
209 */
210
270template <typename T, index_t N> class Vec : public detail::VecCommon< T,N,Vec<T,N> > {
271public:
272
276 constexpr Vec():detail::VecCommon< T,N,Vec<T,N> >() {}
277
283 constexpr Vec(T a):detail::VecCommon< T,N,Vec<T,N> >(a) {}
284
289 constexpr Vec(const T a[N]):detail::VecCommon< T,N,Vec<T,N> >(a) {}
290
297 template <typename U>
298 constexpr Vec(const Vec<U,N-1> &v, T a) {
299 std::copy(v.begin(), v.end(), detail::VecBase<T,N>::begin());
300 this->get(N-1) = a;
301 }
302
307 template <typename Mx, typename Ref>
308 Vec(detail::MtxColIterator<Mx,Ref> mtx_col) {
309 T *p = this->begin();
310 const index_t n = std::min(N, mtx_col->mtx->cols());
311 for (index_t i = 0; i < n; i++, p++, mtx_col++) {
312 *p = *mtx_col;
313 }
314 }

323 Vec(const std::initializer_list<T>& items):

325 {
326#if __cplusplus >= 201402L
327 // items.size() not constexpr in c++11 D:<
328 static_assert(items.size() == N, "Vector must be initialized with N items");
329#endif
330 }
