411 // compute the signature of this permutation, by counting the length of the cycles.
412 // because we can decompose a permutation into a product of cyclical permutations,
413 // we can decompose the sign as the product of the sign of those cycles.
414 // a cycle has odd sign (-1) if the number of elements in the cycle is even.
415 // of course we do not actually have to keep track of the count; only its parity.
416 // note that inverting/transposing a permutation does not change its sign,
417 // so we can interchange row_src and row_dst at our convenience without changing the answer.
418 // whichever one we are given, we use the other as a buffer to mark which elements
419 // have been visited; this allows us to avoid a dynamic memory alloc. for that reason,
420 // we must call this function after one index map has been filled, but before the other has been computed.
421 // runs in O(n) time.
422
423 bool _compute_sign(index_t* p, index_t* buf, index_t n) {
424 std::fill(buf, buf+n, 0);
425
426 // follow the cycles.
427 bool odd = false;
428 for (index_t i = 0; i < n; ++i) {
429 if (buf[i] == 1) continue; // not strictly necessary, but saves us an indirection
430 else buf[i] = 1;
431 for (index_t j = p[i]; buf[j] != 1; j = p[j]) {
432 buf[j] = 1; // mark that this element has been visited.
433 odd = !odd; // the product of the sign of the cycles has flipped.
434 }
435 }
436 return odd;
437 }

451template <typename RandomAccessIterator>
452void permutation_apply(index_t* p, index_t n, RandomAccessIterator v) {
453 for (index_t i = 0; i < n; i++) {
454 index_t cur = i;
455 index_t next = p[i]; // source of new element at this position
456 if (next == i) continue; // no swap necessary
457 do {
458 // give v[cur] the correct element, v[next] gets the cycle's first element
459 std::swap(v[cur], v[next]);
460 // ...and then proceed to find out what goes into the hole
461 cur = next;
462 next = p[next];
463 // mark this element as visited by mapping it to itself
464 // (we only need to do this if we might encounter the cycle again)
465 p[cur] = cur;
466 } while (next != i);
467 }
468}

488 index_t next = p[cur];
