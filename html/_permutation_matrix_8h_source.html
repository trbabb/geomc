390  // compute the signature of this permutation, by counting the length of the cycles.
391  // because we can decompose a permutation into a product of cyclical permutations,
392  // we can decompose the sign as the product of the sign of those cycles.
393  // a cycle has odd sign (-1) if the number of elements in the cycle is even.
394  // of course we do not actually have to keep track of the count; only its parity.
395  // note that inverting/transposing a permutation does not change its sign,
396  // so we can interchange row_src and row_dst at our convenience without changing the answer.
397  // whichever one we are given, we use the other as a buffer to mark which elements
398  // have been visited; this allows us to avoid a dynamic memory alloc. for that reason,
399  // we must call this function after one index map has been filled, but before the other has been computed.
400  // runs in O(n) time.
401 
402  bool computeSign(index_t* p, index_t* buf, index_t n) {
403  std::fill(buf, buf+n, 0);
404 
405  // follow the cycles.
406  bool odd = false;
407  for (index_t i = 0; i < n; ++i) {
408  if (buf[i] == 1) continue; // not strictly necessary, but saves us an indirection
409  else buf[i] = 1;
410  for (index_t j = p[i]; buf[j] != 1; j = p[j]) {
411  buf[j] = 1; // mark that this element has been visited.
412  odd = !odd; // the product of the sign of the cycles has flipped.
413  }
414  }
415  return odd;
416  }
