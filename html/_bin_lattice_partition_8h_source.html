BinLatticePartition.h
1 /*
2  * BinLatticePartition.h
3  *
4  * Implements an N-dimensional, gridded, non-hierarchical spatial partition of cells.
5  * Put another way, objects are kept in a grid of "buckets".
6  *
7  * Good for promximity/neighbor searches, when the density of objects
8  * is roughly known.
9  *
10  * Spatial searches over many bins are likely to be inefficient, especially
11  * with larger dimensions, as every bin within the search region is queried
12  * regardless of whether it has contents.
13  *
14  * Bins are sparsely allocated and there is no memory cost for having
15  * empty bins, and the structure itself does not have limited bounds.
16  *
17  * Multiple/duplicate objects may be placed at the same location.
18  *
19  * Plays nice in one dimension; point_t will revert to T instead of Vec<T,1>
20  *
21  * Created on: Aug 19, 2012
22  * Author: tbabb
23  */

162  void increment() {
163  if (!off_end){
164  bin_t cur_cell = cur_cell_iter->first;
165  cur_cell_iter++;
166  // we must test whether we have hopped out of this bin; or the iterator will just keep
167  // going on to another random one. which I find quite silly.
168  if ((cur_cell_iter == grid->_cells.end() or cur_cell_iter->first != cur_cell) and !gotoNextNonemptyCell(cur_cell)){
169  off_end = true;

186  // iterate over cells in the region, checking each for existence.
187  // return false if no nonempty cells exist within the region.
188  bool gotoNextNonemptyCell(bin_t startCell) {
189  bin_t thisCellCoords = startCell;
190  while (true) {
191  // increment cell within block
192  bool cellWithinRegion = false;
193  for (index_t inc_coord = 0; inc_coord < N; inc_coord++){
194  if (thisCellCoords[inc_coord] + 1 >= cellblock.max()[inc_coord]){
195  // don't exceeded maximum along this axis; can't increment.
196  // rollover this axis and attempt to increment the next.
197  thisCellCoords[inc_coord] = cellblock.min()[inc_coord];
198  } else {
199  thisCellCoords[inc_coord]++;
200  cellWithinRegion = true; // successfully incremented cell w/o going off edge
201  break;
202  }
203  }
204 
205  // see if this cell has contents.
206  if (cellWithinRegion){
207  typename cellmap_t::iterator i = grid->_cells.find(thisCellCoords);
208  if (i != grid->_cells.end()){
209  cur_cell_iter = i;
210  return true;
211  }
212  } else {
213  break;
214  }
215  }
216  return false;
217  }
