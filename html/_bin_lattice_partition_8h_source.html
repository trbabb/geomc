BinLatticePartition.h
1 /*
2  * BinLatticePartition.h
3  *
4  * Implements an N-dimensional, gridded, non-hierarchical spatial partition of cells.
5  * Put another way, objects are kept in a grid of "buckets".
6  *
7  * Good for promximity/neighbor searches, when the density of objects
8  * is roughly known.
9  *
10  * Spatial searches over many bins are likely to be inefficient, especially
11  * with larger dimensions, as every bin within the search region is queried
12  * regardless of whether it has contents.
13  *
14  * Bins are sparsely allocated and there is no memory cost for having
15  * empty bins, and the structure itself does not have limited bounds.
16  *
17  * Multiple/duplicate objects may be placed at the same location.
18  *
19  * Plays nice in one dimension; point_t will revert to T instead of Vec<T,1>
20  *
21  * Created on: Aug 19, 2012
22  * Author: tbabb
23  */

171  void increment() {
172  if (!off_end){
173  bin_t cur_cell = cur_cell_iter->first;
174  cur_cell_iter++;
175  // we must test whether we have hopped out of this bin; or the iterator will just keep
176  // going on to another random one. which I find quite silly.
177  if ((cur_cell_iter == grid->_cells.end() or cur_cell_iter->first != cur_cell) and !gotoNextNonemptyCell(cur_cell)){
178  off_end = true;
179  }
180  }
181  }
182 
183  bool equal(region_iterator const & other) const {
184  if (off_end and other.off_end){
185  return true;
186  } else {
187  return other.cur_cell_iter == cur_cell_iter;
188  }
189  }
190 
191  item_t& dereference() const {
192  return cur_cell_iter->second;
193  }
194 
195  // iterate over cells in the region, checking each for existence.
196  // return false if no nonempty cells exist within the region.
197  bool gotoNextNonemptyCell(bin_t startCell) {
198  bin_t thisCellCoords = startCell;
199  while (true) {
200  // increment cell within block
201  bool cellWithinRegion = false;
202  for (index_t inc_coord = 0; inc_coord < N; inc_coord++){
203  if (thisCellCoords[inc_coord] + 1 >= cellblock.max()[inc_coord]){
204  // don't exceeded maximum along this axis; can't increment.
205  // rollover this axis and attempt to increment the next.
206  thisCellCoords[inc_coord] = cellblock.min()[inc_coord];
207  } else {
208  thisCellCoords[inc_coord]++;
209  cellWithinRegion = true; // successfully incremented cell w/o going off edge
210  break;
211  }
212  }
213 
214  // see if this cell has contents.
215  if (cellWithinRegion){
216  typename cellmap_t::iterator i = grid->_cells.find(thisCellCoords);
217  if (i != grid->_cells.end()){
218  cur_cell_iter = i;
219  return true;
220  }
221  } else {
222  break;
223  }
224  }
225  return false;
226  }

232  typename cellmap_t::iterator cur_cell_iter; //iterator of currently inspected cell.
