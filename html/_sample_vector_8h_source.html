54 // sample from {-1, 1}
55 std::uniform_int_distribution<int> coin(0, 1);
56 return coin(rng) ? 1 : -1;
57 } else if constexpr (N <= 3) {
58 // rejection sampling
59 DenseUniformDistribution<T> unif {-1,1};

63 for (index_t i = 0; i < N; ++i) {
