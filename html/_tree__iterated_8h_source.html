49  // internally, nodes are kept in "sibling-first" order:
50  // - visit a node
51  // - visit its next sibling, if there is one
52  // - visit its first child
53  // as such, all direct children of a node are contiguous, and come
54  // after (though not necessarily *immediately* after) their parent.
55  // all the descendents of a leftward child come before all the
56  // descendents of the child to its right. Therefore, any given subtree
57  // (excluding its root) is contiguous. It is this recursive pattern:
58 
59  // [node] ... [c1 c2 c3 c4 ...]
60  // [children of c1] [descendents of c1...]
61  // [children of c2] [descendents of c2...]
62  // ...

700  if (node == m_nodes.begin() or node.node_count() > 0) return m_nodes.end();
701 
702  node_iterator new_node = insert_child_node(node.node->parent, node, obj);
703  NodeRef hi = node.node;

706  // xxx: can the formulation of this be flipped so that
707  // the new node is higher (rightward / more recent),
708  // rather than lower? it should act more like an `append`.
709 
710  // traverse `node` looking for items to move to the new `lo` node.
711  // because `lo` is adjacent to `hi`, we can add items to `lo` by moving the
712  // boundary between `lo` and `hi` and swapping the contents of the iterators.
713  // we traverse backwards so that our swapping does not unsort our items.
714  for (ItemRef i = hi->items_last; i != std::prev(hi->items_first); ) {
715  if (compare(*i, pivot) < 0) {
716  // move the item to the lower node
717 
718  // grab one item from the beginning of `hi`.
719  // swap `*i` and the stolen item, so that the
720  // tested item lies in the low node where it belongs,
721  // and the stolen item lies under `i`. we'll check it
722  // on the next iteration.
723 
724  if (lo->n_items == 0) {
725  lo->items_first = lo->items_last = hi->items_first;
726  } else {
727  lo->items_last = hi->items_first;
728  }
729 
730  ++lo->n_items;
731  --hi->n_items;
732  ++hi->items_first;
733  std::swap(*i, *(lo->items_last));
734 
735  // we do not decrement i, because we've just moved
736  // an untested item underneath it. we should test
737  // it first before moving on to the next one.
738  } else {
739  // `*i` is in the right place.
740  --i;

1146  // use all the child counts to recompute the first/last
1147  // references in internal nodes. useful when taking ownership
1148  // of another tree's internal node list.
1149  void recalculate_references() {
1150  NodeRef nod = m_nodes.begin();
1151  NodeRef begin_child = nod; ++begin_child;
1152  ItemRef begin_items = m_items.begin(); // next unassigned leaf item
1153  ItemRef last_item = m_items.end(); // last / most recent assigned leaf item in the list
1154 
1155  // in essence we "build" the tree by simply taking ownership of
1156  // the nodes/items that already exist. we walk over the nodes in
1157  // tree order, which means that child nodes are taken in order too.
1158  // this spares us recursion (i.e. no possibility of stack-blowing)
1159  // and some logic keeping track of descent/ascent.
1160  for (; nod != m_nodes.end(); ++nod) {
1161  nod->child_first = nod->child_last = (nod->n_children > 0) ? begin_child : m_nodes.end();
1162  nod->items_first = nod->items_last = (nod->n_items > 0) ? begin_items : m_items.end();
1163 
1164  if (nod->n_children > 0) {
1165  nod->child_first->parent = nod;
1166 
1167  // acquire direct children
1168  for (index_t c = 0; c < nod->n_children - 1; ++c) {
1169  ++nod->child_last;
1170  nod->child_last->parent = nod;
1171  }
1172 
1173  // move the node "free space"
1174  begin_child = nod->child_last; ++begin_child;
1175  } else {
1176  if (nod->n_items > 0) {
1177  for (index_t i = 0; i < nod->n_items - 1; ++i) {
1178  ++nod->items_last;
1179  }
1180  // move the item "free space"
1181  begin_items = last_item = nod->items_last; ++begin_items;
1182  }
1183 
1184  // fixup the items_end along the right edge of this subtree.
1185  NodeRef descendent = nod;
1186  for (NodeRef ancestor = nod->parent;
1187  ancestor != m_nodes.end();
1188  descendent = ancestor, ancestor = ancestor->parent) {
1189  if (ancestor->child_last == descendent) {
1190  if (ancestor->n_items > 0) ancestor->items_last = last_item;
1191  } else {
1192  break;
1193  }
1194  }
1195  }
1196  }
1197  }
1198 
1199 
1200  // find the position of the node that follows `node`'s subtree in "sibling-first" order.
1201  // the successor to `node` occurs just before the next subtree to the right.
1202  // an `O(k * log(n))` operation, on `n` the size of the tree, and `k` the arity.
1203  // this is different from `++node` because `++node` may descend into the
1204  // current node's subtree; we want to exactly hop over it.
1205  NodeRef node_successor(const NodeRef& node) const {
1206  NodeRef ancestor = node->parent;
1207  NodeRef child = node;
1208  NodeRef successor = nodes_end().node; // if there is no subtree to the right,
1209  // this is the conceptual successor.
1210  while (ancestor != m_nodes.end()) {
1211  NodeRef off_end_child = ancestor->child_last;
1212  if (ancestor->n_children > 0) ++off_end_child;
1213  // look for a populated node to the right of us
1214  while (++child != off_end_child) {
1215  if (child->n_children > 0) {
1216  successor = child->child_first;
1217  break;
1218  }
1219  }
1220  if (successor != m_nodes.end()) {
1221  // found the successor
1222  break;
1223  } else {
1224  // ascend the tree
1225  child = ancestor;
1226  ancestor = ancestor->parent;
1227  }
1228  }
1229  return successor;
1230  }
1231 
1232 
1233  // find the first item that occurs after where the last item owned by
1234  // the given node would go, in tree order.
1235  ItemRef item_successor(NodeRef node) const {
1236  for (NodeRef parent = node->parent;
1237  parent != m_nodes.end();
1238  node = parent, parent = node->parent) {
1239  NodeRef end_c = parent->child_last;
1240  if (parent->n_children != 0) ++end_c;
1241  for (; node != end_c; ++node) {
1242  if (node->n_items > 0) {
1243  return node->items_first;
1244  }
1245  }
1246  }
1247  return items_end().item;
1248  }
