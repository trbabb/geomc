18 bool _ImplTracePlane(T* s, const Plane<T,N> &p, const Ray<T,N> &r, HitSide *side) {
19  T dot = p.normal.dot(r.direction);
20 
21  if (dot < 0 and (*side & HIT_FRONT)) {
22  *side = HIT_FRONT;
23  } else if (dot >= 0 and (*side & HIT_BACK)) {
24  *side = HIT_BACK;
25  } else {
26  // backface culled.
27  return false;
28  }
29 
30  T d = p.distance(r.origin);
31  *s = -d / dot; // similar triangles to get to the plane
32 
33  if (*s <= 0 || dot == 0) {
34  // hit point behind ray origin, or ray perfectly parallel to plane (prevent NaN)
35  return false;
36  }

41 // choose a ray hit for a fully convex object (i.e. one with only a single ray
42 // entry and exit each), accounting for hit side preference.
43 template <typename T>
44 bool chooseRayHit(T *s, T roots[2], HitSide *side) {
45  T s0, s1;
46 
47  // order the roots along the ray, from -inf; s0 first.
48  if (roots[0] < roots[1]) {
49  s0 = roots[0];
50  s1 = roots[1];
51  } else {
52  s0 = roots[1];
53  s1 = roots[0];
54  }
55 
56  // lesser root will always be the frontside, greater will be the back.
57  // if tracing both front and back, and both hit, return the nearer (front).
58  if (s0 > 0 && (*side & HIT_FRONT)) {
59  *s = s0;
60  *side = HIT_FRONT;
61  } else if (s1 > 0 && (*side & HIT_BACK)) {
62  *s = s1;
63  *side = HIT_BACK;
64  } else {
65  return false; // hit behind origin; return miss.
66  }
67  return true;
68 }
