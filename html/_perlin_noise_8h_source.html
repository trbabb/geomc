72  T eval(point_t pt) {
73  point_t p0f = std::floor(pt);
74  grid_t p0 = (grid_t)p0f; //grid pt
75  point_t p_modulus = pt - p0f; //pos. within grid
76  const index_t corners = 1 << N;
77  T planeVals[corners];
78  for (index_t c = 0; c < corners; c++) {
79  grid_t cPt;
80  for (index_t axis = 0; axis < N; axis++) {
81  cPt[axis] = (c & (1 << axis)) != 0;
82  }
83  const point_t &grid_grad = get_grid_gradient(p0 + cPt);
84  planeVals[c] = (p_modulus - (point_t)cPt).dot(grid_grad);
85  }
86 
87  for (index_t axis = 0; axis < N; axis++) {
88  T t = fade(p_modulus[axis]); //interpolation parameter
89  for (index_t pair = 0; pair < (1 << (N-axis-1)); pair++) {
90  planeVals[pair] = mix(t, planeVals[pair*2], planeVals[pair*2+1]);
91  }
92  }
93 
94  return planeVals[0];
95  }

108  point_t gradient(point_t pt) {
109  point_t g;
110  //central difference method
111  for (index_t axis = 0; axis < N; axis++) {
112  point_t dx;
113  dx[axis] = PERLIN_EPSILON;
114  g[axis] = eval(pt + dx) - eval(pt - dx);
115  }
116  return g;
117  }

121  inline const point_t& get_grid_gradient(const grid_t &pt) {
122  index_t idx = 0;
123  for (index_t axis = 0; axis < N; axis++) {
124  idx = p[(idx + pt[axis]) & ((index_t)PERLIN_MODULO_MASK)];
125  }
126  return gradients[idx];
127  }
