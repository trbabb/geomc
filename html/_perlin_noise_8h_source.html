78 T eval(point_t pt) const {
79 const index_t corners = 1 << N;
80 T f_x[corners];
81
82 // we use floor() because a bare cast will round toward zero:
83 point_t p0f = std::floor(pt);
84 grid_t p0 = (grid_t)p0f; // grid pt
85 point_t p_mod = pt - p0f; // pos. within grid
86
87 // initialize our interpolants with (hyper-)planar values emanating from the corners.
88 for (index_t c = 0; c < corners; c++) {
89 grid_t cnr = corner(c);
90 const point_t& grid_grad = get_grid_gradient(p0 + cnr);
91 f_x[c] = pointtype::dot((p_mod - (point_t)cnr), grid_grad);
92 }
93
94 // we start by interpolating pairs of values along an arbitrary axis (axis 0).
95 // those interpolated values will in turn be interpolated along the next axis,
96 // as in bi-/trilinear interpolation. each interpolation reduces the number
97 // of values by half, until we have one value.
98
99 for (index_t axis = 0; axis < N; axis++) {
100 T x = pointtype::iterator(p_mod)[axis];
101 T t = fade(x);
102 for (index_t pair = 0; pair < (1 << (N - axis - 1)); pair++) {
103 f_x[pair] = mix(t, f_x[pair * 2], f_x[pair * 2 + 1]);
104 }
105 }
106
107 return f_x[0];
108 }

116 std::pair<T, point_t> gradient(point_t pt) const {
117 const index_t corners = 1 << N;
118 T f[corners]; // values
119 point_t df_dx[corners]; // gradients
120
121 point_t p0f = std::floor(pt);
122 grid_t p0 = (grid_t)p0f;
123 point_t p_mod = pt - p0f;
124
125 // useful facts:
126 // mix(t, a, b) = (1 - t) * a + t * b
127 // d/dt mix(t, a, b) = b - a
128 // d/da mix(t, a, b) = 1 - t
129 // d/db mix(t, a, b) = t
130
131 // the multivariate chain rule:
132 // df/dt f(a(t), b(t), c(t)) =
133 // df_da(a(t), ...) * da_dt(t) +
134 // df_db(..., b(t), ...) * db_dt(t) + ...
135
136 // start with planar functions from each corner;
137 // remember the derivatives of these functions:
138 for (index_t c = 0; c < corners; ++c) {
139 grid_t cnr = corner(c);
140 df_dx[c] = get_grid_gradient(p0 + cnr);
141 f[c] = pointtype::dot((p_mod - (point_t)cnr), df_dx[c]);
142 }
143
144 // now start interpolating these functions, updating the derivative along the way
145 for (index_t axis = 0; axis < N; axis++) {
146 T x = pointtype::iterator(p_mod)[axis];
147 T t = fade(x);
148 T dt_dx = dfade_dt(x); // (1) note that this is zero along other axes,
149 // as this is only a fuction of x[axis].
150
151 // as in `eval()`, we begin by interpolating pairs of points along the x axis
152 // and reducing the number of values by half with successive interpolations.
153 for (index_t pair = 0; pair < (1 << (N - axis - 1)); pair++) {
154 index_t k = pair * 2;
155 T a = f[k];
156 T b = f[k + 1];
157 point_t da_dx = df_dx[k];
158 point_t db_dx = df_dx[k + 1];
159
160 // the expression to be differentiated:
161 f[pair] = mix(t, a, b);
162
163 // apply the multivariate chain rule:
164 point_t w;
165 // dt/dx is only nonzero along the current axis:
166 T* w_i = pointtype::iterator(w) + axis;
167 // dmix_dt(t, ...) * dt/dx
168 *w_i += (b - a) * dt_dx;
169 // dmix_da(..., a, ...) * da/dx
170 w += (1 - t) * da_dx;
171 // dmix_db(..., ..., b) * db/dx
172 w += t * db_dx;
173 df_dx[pair] = w;
174 }
175 }
176
177 return std::pair<T, point_t>(f[0], df_dx[0]);
178 }

214 static inline grid_t corner(index_t i) {

222 static inline T fade(T t) {
