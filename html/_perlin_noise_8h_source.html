78  T eval(point_t pt) const {
79  const index_t corners = 1 << N;
80  T f_x[corners];
81 
82  // we use floor() because a bare cast will round toward zero:
83  point_t p0f = std::floor(pt);
84  grid_t p0 = (grid_t)p0f; // grid pt
85  point_t p_mod = pt - p0f; // pos. within grid
86 
87  // initialize our interpolants with (hyper-)planar values emanating from the corners.
88  for (index_t c = 0; c < corners; c++) {
89  grid_t cnr = corner(c);
90  const point_t& grid_grad = get_grid_gradient(p0 + cnr);
91  f_x[c] = pointtype::dot((p_mod - (point_t)cnr), grid_grad);
92  }
93 
94  // we start by interpolating pairs of values along an arbitrary axis (axis 0).
95  // those interpolated values will in turn be interpolated along the next axis,
96  // as in bi-/trilinear interpolation. each interpolation reduces the number
97  // of values by half, until we have one value.
98 
99  for (index_t axis = 0; axis < N; axis++) {
100  T x = pointtype::iterator(p_mod)[axis];
101  T t = fade(x);
102  for (index_t pair = 0; pair < (1 << (N - axis - 1)); pair++) {
103  f_x[pair] = mix(t, f_x[pair * 2], f_x[pair * 2 + 1]);
104  }
105  }
106 
107  return f_x[0];
108  }

121  std::pair<T, point_t> gradient(point_t pt) const {
122  const index_t corners = 1 << N;
123  T f[corners]; // values
124  point_t df_dx[corners]; // gradients
125 
126  point_t p0f = std::floor(pt);
127  grid_t p0 = (grid_t)p0f;
128  point_t p_mod = pt - p0f;
129 
130  // useful facts:
131  // mix(t, a, b) = (1 - t) * a + t * b
132  // d/dt mix(t, a, b) = b - a
133  // d/da mix(t, a, b) = 1 - t
134  // d/db mix(t, a, b) = t
135 
136  // the multivariate chain rule:
137  // df/dt f(a(t), b(t), c(t)) =
138  // df_da(a(t), ...) * da_dt(t) +
139  // df_db(..., b(t), ...) * db_dt(t) + ...
140 
141  // start with planar functions from each corner;
142  // remember the derivatives of these functions:
143  for (index_t c = 0; c < corners; ++c) {
144  grid_t cnr = corner(c);
145  df_dx[c] = get_grid_gradient(p0 + cnr);
146  f[c] = pointtype::dot((p_mod - (point_t)cnr), df_dx[c]);
147  }
148 
149  // now start interpolating these functions, updating the derivative along the way
150  for (index_t axis = 0; axis < N; axis++) {
151  T x = pointtype::iterator(p_mod)[axis];
152  T t = fade(x);
153  T dt_dx = dfade_dt(x); // (1) note that this is zero along other axes,
154  // as this is only a fuction of x[axis].
155 
156  // as in `eval()`, we begin by interpolating pairs of points along the x axis
157  // and reducing the number of values by half with successive interpolations.
158  for (index_t pair = 0; pair < (1 << (N - axis - 1)); pair++) {
159  index_t k = pair * 2;
160  T a = f[k];
161  T b = f[k + 1];
162  point_t da_dx = df_dx[k];
163  point_t db_dx = df_dx[k + 1];
164 
165  // the expression to be differentiated:
166  f[pair] = mix(t, a, b);
167 
168  // apply the multivariate chain rule:
169  point_t w;
170  // dt/dx is only nonzero along the current axis:
171  T* w_i = pointtype::iterator(w) + axis;
172  // dmix_dt(t, ...) * dt/dx
173  *w_i += (b - a) * dt_dx;
174  // dmix_da(..., a, ...) * da/dx
175  w += (1 - t) * da_dx;
176  // dmix_db(..., b) * db/dx
177  w += t * db_dx;
178  df_dx[pair] = w;
179  }
180  }
181 
182  return std::pair<T, point_t>(f[0], df_dx[0]);
183  }

187  inline const point_t& get_grid_gradient(const grid_t& pt) const {
188  index_t idx = 0;
189  if (N > 1) {
190  // iterate on the permutation table, offsetting with each coordinate.
191  for (index_t axis = 0; axis < N; axis++) {
192  idx = p[positive_mod(idx + gridtype::iterator(pt)[axis], (index_t)PERLIN_NUM_GRADIENTS)];
193  }
194  } else {
195  // 1D case: indexing into the permutation table would not be iterated,
196  // so the noise would be periodic with period `num_gradients`. instead,
197  // do a linear congruential scramble, using Knuth's constants:
198  idx = gridtype::iterator(pt)[0];
199  idx = 6364136223846793005LL * idx + 1442695040888963407LL;
200  idx = positive_mod(idx, (index_t)PERLIN_NUM_GRADIENTS);
201  }
202  return gradients[idx];
203  }

210  }

214  static inline T fade(T t) {
215  return t * t * t * (t * (t * 6 - 15) + 10);
216  }

218  static inline T dfade_dt(T t) {
219  T w = t - 1;

221  }
