88template <typename S, typename T>
89inline typename std::common_type<S,T>::type mix(S t, T a, T b) {
90 typedef typename std::common_type<S,T>::type U;
91 // (1 - t) * a + t * b
92 return sum_of_products<U>(1 - t, a, t, b);

103template <typename S, typename T>
104T interp_linear(const S s[], const T p[], int dim) {
105 if (dim <= 1){
106 return mix<S,T>(*s, p[0], p[1]);
107 } else {
108 int next_dim = dim - 1;
109 int blocksize = 1 << next_dim;
110 T p0 = interp_linear(s, p, next_dim);
111 T p1 = interp_linear(s, p + blocksize, next_dim);
112 return mix<S,T>(s[next_dim], p0, p1);
113 }
114}

123template <typename S, typename T>
124T interp_cubic(S s, const T p[4]) {
125 return p[1] + 0.5 *
126 s*(p[2] - p[0] +
127 s*(2.0*p[0] - 5.0*p[1] + 4.0*p[2] - p[3] +
128 s*(3.0*(p[1] - p[2]) + p[3] - p[0])));
129}

139template <typename S, typename T>
140T interp_cubic(const S s[], const T p[], int dim) {
141 if (dim <= 1){
142 return interp_cubic(*s, p);
143 } else {
144 T buf[4];
145 int next_dim = dim - 1;
146 int blocksize = 1 << (2 * next_dim); //TODO: is this right?
147 buf[0] = interp_cubic(s, p, next_dim);
148 buf[1] = interp_cubic(s, p + blocksize, next_dim);
149 buf[2] = interp_cubic(s, p + 2 * blocksize, next_dim);
150 buf[3] = interp_cubic(s, p + 3 * blocksize, next_dim);
151 return interp_cubic(s[next_dim], buf);
152 }

169template <typename T>
170bool quadratic_solve(T results[2], T a, T b, T c){
171 T descr = diff_of_products(b, b, 4 * a, c); // b^2 - 4ac
172 // no real solution, or an equation of the form "c = 0"
173 if (descr < 0 or (a == 0 and b == 0)) return false;
174 // addition cannot cancel, since b and descr are chosen to have the same sign
175 T q = -0.5 * (b + std::copysign(std::sqrt(descr), b));
176 T x0 = q / a; // note: client could pass a = 0
177 T x1 = c / q; // q is zero iff b and c are zero
178 results[0] = (a != 0) ? x0 : x1;
179 results[1] = (q != 0) ? x1 : x0;
180 return true;
181}
