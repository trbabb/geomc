101 typedef typename std::common_type<S,T>::type U;

114template <typename S, typename T>
115inline T interp_linear(const S s[], const T p[], int dim) {
116 if (dim <= 1){
117 return mix<S,T>(*s, p[0], p[1]);
118 } else {
119 int next_dim = dim - 1;
120 int blocksize = 1 << next_dim;
121 T p0 = interp_linear(s, p, next_dim);
122 T p1 = interp_linear(s, p + blocksize, next_dim);
123 return mix<S,T>(s[next_dim], p0, p1);
124 }
125}

136 return p[1] + 0.5 *

138 s*(2.0*p[0] - 5.0*p[1] + 4.0*p[2] - p[3] +

180template <typename T>
181inline bool quadratic_solve(T results[2], T a, T b, T c){
182 T descr = diff_of_products(b, b, 4 * a, c); // b^2 - 4ac
183 // no real solution, or an equation of the form "c = 0"
184 if (descr < 0 or (a == 0 and b == 0)) return false;
185 // addition cannot cancel, since b and descr are chosen to have the same sign
186 T q = -0.5 * (b + std::copysign(std::sqrt(descr), b));
187 T x0 = q / a; // note: client could pass a = 0
188 T x1 = c / q; // q is zero iff b and c are zero
189 results[0] = (a != 0) ? x0 : x1;
190 results[1] = (q != 0) ? x1 : x0;
191 return true;
192}

199template <typename T>
200inline T angle_to(T radians_0, T radians_1) {
201 constexpr T turn = 2 * M_PI;
202 T a0 = positive_mod<T>(radians_0, turn);
203 T a1 = positive_mod<T>(radians_1, turn);
204 // one of three images of a0 is closest to a1:
205 // ···----o------|----o-->*----|---o---···
206 // a0 - τ 0 a0 a1 τ a0 + τ
207 T da = a1 - a0;
208 T db = a1 - (a0 + turn);
209 T dc = a1 - (a0 - turn);
210 // find the difference of smallest magnitude:
211 T err = std::abs(da) < std::abs(db) ? da : db;
212 err = std::abs(err) < std::abs(dc) ? err : dc;
213 return err;
214}
