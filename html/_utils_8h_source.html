98template <typename S, typename T>
99inline typename std::common_type<S,T>::type mix(S t, T a, T b) {
100 typedef typename std::common_type<S,T>::type U;
101 // (1 - t) * a + t * b
102 return sum_of_products<U>(1 - t, a, t, b);

113template <typename S, typename T>
114inline T interp_linear(const S s[], const T p[], int dim) {
115 if (dim <= 1){
116 return mix<S,T>(*s, p[0], p[1]);
117 } else {
118 int next_dim = dim - 1;
119 int blocksize = 1 << next_dim;
120 T p0 = interp_linear(s, p, next_dim);
121 T p1 = interp_linear(s, p + blocksize, next_dim);
122 return mix<S,T>(s[next_dim], p0, p1);
123 }
124}

133template <typename S, typename T>
134inline T interp_cubic(S s, const T p[4]) {
135 return p[1] + 0.5 *
136 s*(p[2] - p[0] +
137 s*(2.0*p[0] - 5.0*p[1] + 4.0*p[2] - p[3] +
138 s*(3.0*(p[1] - p[2]) + p[3] - p[0])));
139}

179template <typename T>
180inline bool quadratic_solve(T results[2], T a, T b, T c){
181 T descr = diff_of_products(b, b, 4 * a, c); // b^2 - 4ac
182 // no real solution, or an equation of the form "c = 0"
183 if (descr < 0 or (a == 0 and b == 0)) return false;
184 // addition cannot cancel, since b and descr are chosen to have the same sign
185 T q = -0.5 * (b + std::copysign(std::sqrt(descr), b));
186 T x0 = q / a; // note: client could pass a = 0
187 T x1 = c / q; // q is zero iff b and c are zero
188 results[0] = (a != 0) ? x0 : x1;
189 results[1] = (q != 0) ? x1 : x0;
190 return true;
191}

198template <typename T>
199inline T angle_to(T radians_0, T radians_1) {
200 constexpr T turn = 2 * M_PI;
201 T a0 = positive_mod<T>(radians_0, turn);
202 T a1 = positive_mod<T>(radians_1, turn);
203 // one of three images of a0 is closest to a1:
204 // ···----o------|----o-->*----|---o---···
205 // a0 - τ 0 a0 a1 τ a0 + τ
206 T da = a1 - a0;
207 T db = a1 - (a0 + turn);
208 T dc = a1 - (a0 - turn);
209 // find the difference of smallest magnitude:
210 T err = std::abs(da) < std::abs(db) ? da : db;
211 err = std::abs(err) < std::abs(dc) ? err : dc;
212 return err;
213}
