114 template <typename T>
115 T spherical_harmonic_coeff(index_t l, index_t m, T cos_alt, T azi) {
116  index_t m_a = std::abs(m);
117  static const T sr2 = std::sqrt((T)2);
118  static const T pi = boost::math::constants::pi<T>();
119 
120  T P_lm = legendre(l, m_a, cos_alt);
121 
122  // k contains a sub-expression commonly written: (l-m)!/(l+m)!
123  // which is vulnerable to precision/overflow issues, as very large
124  // intermediate numbers must be stored. here, we compute the reduced
125  // fraction directly:
126  T z = 1;
127  if (l != 0) for (index_t i = l - m_a + 1; i <= l + m_a; i++) { z *= i; }
128 
129  T k = std::sqrt((1/z) * ((2 * l + 1) / (4 * pi)));
130 
131  if (m > 0) {
132  return sr2 * k * std::cos( m * azi) * P_lm;
133  } else if (m < 0) {
134  return sr2 * k * std::sin(-m * azi) * P_lm;
135  } else {
136  return k * legendre(l, 0, cos_alt);
137  }
138 }

315  // protected implementation takes cos(alt), which is just z
316  // for a unit vector. this formulation may better preserve precision when
317  // z is available directly.
318  T _eval(T cos_alt, T azi) const {
319  static const T sr2 = std::sqrt((T)2);
320  static const T pi = boost::math::constants::pi<T>();
321  const index_t l = bands();
322  const T x = cos_alt;
323  T sum = 0;
324  for (index_t m = 0; m < l; m++) {
325  T c_ma = std::cos(m * azi);
326  T s_ma = std::sin(m * azi);
327  T pmm0 = 1;
328  T pmm1 = 1;
329  for (index_t l_i = m; l_i < l; l_i++) {
330  T P_lm = 1;
331 
332  // evaluate P_{l_i}_m(x) from previous values (if extant)
333  if (l_i == m) {
334  if (m > 0) {
335  // compute double factorial of m, with (-1)^m factor
336  for (index_t i = 1; i <= m; i++) {
337  pmm0 *= -(i * 2 - 1);
338  }
339  pmm0 *= std::pow(1 - x * x, m / 2.0);
340  }
341  P_lm *= pmm0;
342  } else if (l_i == m + 1) {
343  pmm1 = x * (2 * m + 1) * pmm0;
344  P_lm *= pmm1;
345  } else {
346  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
347  (l_i - m);
348  pmm0 = pmm1;
349  pmm1 = pmi;
350  P_lm *= pmi;
351  }
352  // P_lm now contains the legendre polynomial value at x
353 
354  // evaluate ((l-m)!/(l+m)!)^-1
355  T z = 1;
356  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
357 
358  // fold Legendre values into SH basis functions
359  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
360  if (m != 0) {
361  sum += P_lm * sr2 * k * c_ma * coeff(l_i, m);
362  sum += P_lm * sr2 * k * s_ma * coeff(l_i, -m);
363  } else {
364  sum += P_lm * k * coeff(l_i, 0);
365  }
366  }
367  }
368  return sum;
369  }

372  void _project(T cos_alt, T azi, T val) {
373  static const T sr2 = std::sqrt((T)2);
374  static const T pi = boost::math::constants::pi<T>();
375  const index_t l = bands();
376  const T x = cos_alt;
377  for (index_t m = 0; m < l; m++) {
378  T c_ma = std::cos(m * azi);
379  T s_ma = std::sin(m * azi);
380  T pmm0 = 1;
381  T pmm1 = 1;
382  for (index_t l_i = m; l_i < l; l_i++) {
383  T P_lm = 1;
384  // evaluate P_{l_i}_m(x) from previous values (if extant)
385  if (l_i == m) {
386  if (m > 0) {
387  for (index_t i = 1; i <= m; i++) {
388  pmm0 *= -(i * 2 - 1);
389  }
390  pmm0 *= std::pow(1 - x * x, m / 2.0);
391  }
392  P_lm = pmm0;
393  } else if (l_i == m + 1) {
394  pmm1 = x * (2 * m + 1) * pmm0;
395  P_lm = pmm1;
396  } else {
397  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
398  (l_i - m);
399  pmm0 = pmm1;
400  pmm1 = pmi;
401  P_lm = pmi;
402  }
403  // P_lm now contains the legendre polynomial value at x
404 
405  // evaluate ((l-m)!/(l+m)!)^-1
406  // this could be done with a LUT for less numerical precision
407  // but I prefer "correct" over "moderately faster".
408  T z = 1;
409  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
410 
411  // fold Legendre values into SH basis functions
412  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
413  if (m != 0) {
414  coeff(l_i, m) += P_lm * sr2 * k * c_ma * val;
415  coeff(l_i, -m) += P_lm * sr2 * k * s_ma * val;
416  } else {
417  coeff(l_i, 0) += P_lm * k * val;
418  }
419  }
420  }

678  void generate_lambertian(index_t order=-1) {
679  // from "On the relationship between radiance and irradiance:
680  // determining the illumination from images of a convex Lambertian
681  // object" by Ramamoorthi and Hanrahan. Note that the coefficients
682  // described therein are those of the spherical harmonics, not zonal
683  // harmonics, so there is an extra factor of sqrt((2l+1)/4pi) applied here.
684  const index_t ct = order < 1 ? bands() : std::min(order, bands());
685  clear();
686  T fac = 1;
687  coeffs[0] = 0.25;
688  if (bands() > 1) {
689  coeffs[1] = 0.5;
690  for (index_t n = 2; n < ct; n += 2) { // odd bands are 0.
691  fac *= n * (n - 1);
692  T fac_num = 1;
693  T fac_den = 1;
694  for (index_t i = 1; i <= n / 2; i++) {
695  fac_num *= i + n/2;
696  fac_den *= i;
697  }
698  T k = (2 * n + 1) / (T)2;
699  T A_n = k / (T)((n + 2) * (n - 1));
700  A_n *= std::pow(2, -n) * fac_num / fac_den;
701  if (!(n & (index_t)2)) A_n *= -1;
702  coeffs[n] = A_n;
703  }
704  }

713  void generate_spot(T cos_angle) {
714  // the indefinite integral of the legendre polynomials are
715  // analytic and recurrent. We use these to integrate over the range
716  // of cos_angle for which the spot function is nonzero.
717  const index_t n = bands();
718  const T x = std::min((T)1, std::max((T)-1, cos_angle));
719  coeffs[0] = (2 - x - 1) / 2;
720  T p_0 = 1; // P(i - 1, x)
721  T p_1 = x; // P(i, x)
722  T p_i = x; // P(i + 1, x)
723  for (index_t i = 1; i < n; i++) {
724  T k = (2 * i + 1) / ((T)2);
725  p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
726  coeffs[i] = -k * (p_i - p_0) / (2 * i + 1);
727  p_0 = p_1;
728  p_1 = p_i;
729  }
730  }
