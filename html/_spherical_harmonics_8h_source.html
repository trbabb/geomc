113template <typename T>
114inline T spherical_harmonic_coeff(index_t l, index_t m, T cos_alt, T azi) {
115 index_t m_a = std::abs(m);
116 static const T sr2 = std::sqrt((T)2);
117 static const T pi = std::numbers::pi_v<T>;
118
119 T P_lm = legendre(l, m_a, cos_alt);
120
121 // k contains a sub-expression commonly written: (l-m)!/(l+m)!
122 // which is vulnerable to precision/overflow issues, as very large
123 // intermediate numbers must be stored. here, we compute the reduced
124 // fraction directly:
125 T z = 1;
126 if (l != 0) for (index_t i = l - m_a + 1; i <= l + m_a; i++) { z *= i; }
127
128 T k = std::sqrt((1/z) * ((2 * l + 1) / (4 * pi)));
129
130 if (m > 0) {
131 return sr2 * k * std::cos( m * azi) * P_lm;
132 } else if (m < 0) {
133 return sr2 * k * std::sin(-m * azi) * P_lm;
134 } else {
135 return k * legendre(l, 0, cos_alt);
136 }
137}

315 // for a unit vector. this formulation may better preserve precision when

317 T _eval(T cos_alt, T azi) const {
318 static const T sr2 = std::sqrt((T)2);
319 static const T pi = std::numbers::pi_v<T>;
320 const index_t l = bands();
321 const T x = cos_alt;
322 T sum = 0;
323 for (index_t m = 0; m < l; m++) {
324 T c_ma = std::cos(m * azi);
325 T s_ma = std::sin(m * azi);
326 T pmm0 = 1;
327 T pmm1 = 1;
328 for (index_t l_i = m; l_i < l; l_i++) {
329 T P_lm = 1;
330
331 // evaluate P_{l_i}_m(x) from previous values (if extant)
332 if (l_i == m) {
333 if (m > 0) {
334 // compute double factorial of m, with (-1)^m factor
335 for (index_t i = 1; i <= m; i++) {
336 pmm0 *= -(i * 2 - 1);
337 }
338 pmm0 *= std::pow(1 - x * x, m / 2.0);
339 }
340 P_lm *= pmm0;
341 } else if (l_i == m + 1) {
342 pmm1 = x * (2 * m + 1) * pmm0;
343 P_lm *= pmm1;
344 } else {
345 T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
346 (l_i - m);
347 pmm0 = pmm1;
348 pmm1 = pmi;
349 P_lm *= pmi;
350 }
351 // P_lm now contains the legendre polynomial value at x
352
353 // evaluate ((l-m)!/(l+m)!)^-1
354 T z = 1;
355 if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
356
357 // fold Legendre values into SH basis functions
358 T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
359 if (m != 0) {
360 sum += P_lm * sr2 * k * c_ma * coeff(l_i, m);
361 sum += P_lm * sr2 * k * s_ma * coeff(l_i, -m);
362 } else {
363 sum += P_lm * k * coeff(l_i, 0);
364 }
365 }
366 }
367 return sum;
368 }

371 void _project(T cos_alt, T azi, T val) {
372 static const T sr2 = std::sqrt((T)2);
373 static const T pi = std::numbers::pi_v<T>;
374 const index_t l = bands();
375 const T x = cos_alt;
376 for (index_t m = 0; m < l; m++) {
377 T c_ma = std::cos(m * azi);
378 T s_ma = std::sin(m * azi);
379 T pmm0 = 1;
380 T pmm1 = 1;
381 for (index_t l_i = m; l_i < l; l_i++) {
382 T P_lm = 1;
383 // evaluate P_{l_i}_m(x) from previous values (if extant)
384 if (l_i == m) {
385 if (m > 0) {
386 for (index_t i = 1; i <= m; i++) {
387 pmm0 *= -(i * 2 - 1);
388 }
389 pmm0 *= std::pow(1 - x * x, m / 2.0);
390 }
391 P_lm = pmm0;
392 } else if (l_i == m + 1) {
393 pmm1 = x * (2 * m + 1) * pmm0;
394 P_lm = pmm1;
395 } else {
396 T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
397 (l_i - m);
398 pmm0 = pmm1;
399 pmm1 = pmi;
400 P_lm = pmi;
401 }
402 // P_lm now contains the legendre polynomial value at x
403
404 // evaluate ((l-m)!/(l+m)!)^-1
405 // this could be done with a LUT for less numerical precision
406 // but I prefer "correct" over "moderately faster".
407 T z = 1;
408 if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
409
410 // fold Legendre values into SH basis functions
411 T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
412 if (m != 0) {
413 coeff(l_i, m) += P_lm * sr2 * k * c_ma * val;
414 coeff(l_i, -m) += P_lm * sr2 * k * s_ma * val;
415 } else {
416 coeff(l_i, 0) += P_lm * k * val;
417 }
418 }
419 }

677 void generate_lambertian(index_t order=-1) {
678 // from "On the relationship between radiance and irradiance:
679 // determining the illumination from images of a convex Lambertian
680 // object" by Ramamoorthi and Hanrahan. Note that the coefficients
681 // described therein are those of the spherical harmonics, not zonal
682 // harmonics, so there is an extra factor of sqrt((2l+1)/4pi) applied here.
683 const index_t ct = order < 1 ? bands() : std::min(order, bands());
684 clear();
685 T fac = 1;
686 coeffs[0] = 0.25;
687 if (bands() > 1) {
688 coeffs[1] = 0.5;
689 for (index_t n = 2; n < ct; n += 2) { // odd bands are 0.
690 fac *= n * (n - 1);
691 T fac_num = 1;
692 T fac_den = 1;
693 for (index_t i = 1; i <= n / 2; i++) {
694 fac_num *= i + n/2;
695 fac_den *= i;
696 }
697 T k = (2 * n + 1) / (T)2;
698 T A_n = k / (T)((n + 2) * (n - 1));
699 A_n *= std::pow(2, -n) * fac_num / fac_den;
700 if (!(n & (index_t)2)) A_n *= -1;
701 coeffs[n] = A_n;
702 }
703 }

712 void generate_spot(T cos_angle) {
713 // the indefinite integral of the legendre polynomials are
714 // analytic and recurrent. We use these to integrate over the range
715 // of cos_angle for which the spot function is nonzero.
716 const index_t n = bands();
717 const T x = std::min((T)1, std::max((T)-1, cos_angle));
718 coeffs[0] = (2 - x - 1) / 2;
719 T p_0 = 1; // P(i - 1, x)
720 T p_1 = x; // P(i, x)
721 T p_i = x; // P(i + 1, x)
722 for (index_t i = 1; i < n; i++) {
723 T k = (2 * i + 1) / ((T)2);
724 p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
725 coeffs[i] = -k * (p_i - p_0) / (2 * i + 1);
726 p_0 = p_1;
727 p_1 = p_i;
728 }
729 }

736 T eval(T cos_alt) const {
737 const T x = cos_alt;
738 T sum = coeffs[0];
739 T p_0 = 1; // P(i - 1, x)
740 T p_1 = x; // P(i, x)
741 if (bands() > 1) sum += coeffs[1] * x;
742 for (index_t i = 1; i < bands() - 1; i++) {
743 // P(i + 1, x):
744 T p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
745 p_0 = p_1;
746 p_1 = p_i;
747 sum += coeffs[i + 1] * p_i;
748 }
749 return sum;
