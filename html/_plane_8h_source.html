152 void apply(const AffineTransform<T,N> &xf) {
153 Vec<T,N> p0 = normal*d; // a point on the plane
154 Vec<T,N> p1 = xf.apply(p0); // the new position of that point

195 bool intersects(const Vec<T,N> *hull, index_t npts) const {
196 bool negative;
197 for (index_t i = 0; i < npts; i++){
198 T dist = distance(hull[i]);
199 if (i > 0){
200 // if a point lies on the plane, or another point lies on the
201 // other side of the plane, then this shape crosses the plane.
202 if (dist == 0 || ((dist < 0) != negative)){
203 return false;
204 }
205 } else {
206 negative = dist < 0;
207 }
208 }
209 return false;
210 }
