148  void apply(const AffineTransform<T,N> &xf) {
149  Vec<T,N> p0 = normal*d; // a point on the plane
150  Vec<T,N> p1 = xf.apply(p0); // the new position of that point
151  normal = xf.applyNormal(normal).unit(); // construct a plane with transformed normal
152  d = p1.dot(normal); // and transformed position
153  }

159  void applyInverse(const AffineTransform<T,N> &xf) {
160  Vec<T,N> p0 = normal*d;

162  normal = xf.applyInverseNormal(normal).unit();
163  d = p1.dot(normal);

190  bool intersects(const Vec<T,N> *hull, index_t npts) const {
191  bool negative;
192  for (index_t i = 0; i < npts; i++){
193  T dist = distance(hull[i]);
194  if (i > 0){
195  // if a point lies on the plane, or another point lies on the
196  // other side of the plane, then this shape crosses the plane.
197  if (dist == 0 || ((dist < 0) != negative)){
198  return false;
199  }
200  } else {
201  negative = dist < 0;
202  }
203  }
