159 void apply(const AffineTransform<T,N> &xf) {
160 Vec<T,N> p0 = normal*d; // a point on the plane
161 Vec<T,N> p1 = xf.apply(p0); // the new position of that point
162 normal = xf.apply_normal(normal).unit(); // construct a plane with transformed normal
163 d = p1.dot(normal); // and transformed position
164 }
165
170 void apply_inverse(const AffineTransform<T,N> &xf) {
171 Vec<T,N> p0 = normal*d;
172 Vec<T,N> p1 = xf.apply_inverse(p0);
173 normal = xf.apply_inverse_normal(normal).unit();
174 d = p1.dot(normal);
175 }

181 template <typename ConvexShape>
182 inline bool intersects(const ConvexShape& shape) const {
183 return shape.convex_support( normal).dot(normal) + d > 0 and
184 shape.convex_support(-normal).dot(normal) + d <= 0;
185 }

191 inline bool intersects(const Sphere<T,N> &s) const {

202 bool intersects(const Vec<T,N> *hull, index_t npts) const {
203 bool negative;
204 for (index_t i = 0; i < npts; i++){
205 T dist = distance(hull[i]);
206 if (i > 0){
207 // if a point lies on the plane, or another point lies on the
208 // other side of the plane, then this shape crosses the plane.
209 if (dist == 0 || ((dist < 0) != negative)){
210 return false;
211 }
212 } else {
213 negative = dist < 0;
214 }
215 }
216 return false;
217 }
