145  void apply(const AffineTransform<T,N> &xf) {
146  Vec<T,N> p0 = normal*d; // a point on the plane
147  Vec<T,N> p1 = xf.apply(p0); // the new position of that point

170  inline bool intersects(const Vec<T,N> *hull, index_t npts) const {
171  bool negative;
172  for (index_t i = 0; i < npts; i++){
173  T dist = distance(hull[i]);
174  if (i > 0){
175  // if a point lies on the plane, or another point lies on the
176  // other side of the plane, then this shape crosses the plane.
177  if (dist == 0 || ((dist < 0) != negative)){
178  return false;
179  }
180  } else {
181  negative = dist < 0;
182  }
183  }
