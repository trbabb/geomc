152  void apply(const AffineTransform<T,N> &xf) {
153  Vec<T,N> p0 = normal*d; // a point on the plane
154  Vec<T,N> p1 = xf.apply(p0); // the new position of that point
155  normal = xf.applyNormal(normal).unit(); // construct a plane with transformed normal
156  d = p1.dot(normal); // and transformed position
157  }

163  void applyInverse(const AffineTransform<T,N> &xf) {
164  Vec<T,N> p0 = normal*d;
165  Vec<T,N> p1 = xf.applyInverse(p0);
166  normal = xf.applyInverseNormal(normal).unit();
167  d = p1.dot(normal);
168  }

194  bool intersects(const Vec<T,N> *hull, index_t npts) const {
195  bool negative;
196  for (index_t i = 0; i < npts; i++){
197  T dist = distance(hull[i]);
198  if (i > 0){
199  // if a point lies on the plane, or another point lies on the
200  // other side of the plane, then this shape crosses the plane.
201  if (dist == 0 || ((dist < 0) != negative)){
202  return false;
203  }
204  } else {
205  negative = dist < 0;
206  }
207  }
208  return false;
209  }
