48template <typename Ma, typename Mb, typename Enable=void>
49class _ImplMtxMul {
50public:
51
52 //TODO: this should probably use T = decltype(Ma::elem_t * Mb::elem_t)
53 typedef SimpleMatrix<typename Ma::elem_t, Ma::ROWDIM, Mb::COLDIM> return_t;
54
55 template <typename Md>
56 static void mul(Md* d, const Ma& a, const Mb& b) {
57 typedef typename Md::elem_t T;
58 for (index_t r = 0; r < a.rows(); r++) {
59 for (index_t c = 0; c < b.cols(); c++) {
60 T sum = 0;
61 for (index_t n = 0; n < a.cols(); n++) {
62 sum += a(r, n) * b(n, c);
63 }
64 d->set(r, c, sum);
65 }
66 }
67 }
68
69 template <typename Md>
70 static void mul_acc(Md* d, const Ma& a, const Mb& b) {
71 typedef typename Md::elem_t T;
72 for (index_t r = 0; r < a.rows(); r++) {
73 for (index_t c = 0; c < b.cols(); c++) {
74 T sum = (*d)(r, c);
75 for (index_t n = 0; n < a.cols(); n++) {
76 sum += a(r, n) * b(n, c);
77 }
78 d->set(r, c, sum);
79 }
80 }
81 }
82};

86template <typename Mat, typename T, index_t M, index_t N>
87class _ImplMtxMul<Mat, geom::DiagMatrix<T,M,N>, REQUIRE_MATRIX_T(Mat)> {
88public:
89
90 typedef SimpleMatrix<T, Mat::ROWDIM, N> return_t;
91
92 // (a x M) * (M x N) -> (a x N)
93 template <typename Md>
94 static void mul(Md* d, const Mat& a, const geom::DiagMatrix<T,M,N>& b) {
95 const T* dp = b.diagonal_begin();
96 for (index_t c = 0; c < b.cols(); ++c, ++dp) {
97 for (index_t r = 0; r < a.rows(); ++r) {
98 d->set(r,c, (*dp) * a(r,c));
99 }
100 }
101 }
102};

126template <typename T, index_t L, index_t M1, index_t M2, index_t N>
127class _ImplMtxMul<geom::DiagMatrix<T,L,M1>, geom::DiagMatrix<T,M2,N>, void> {
128public:
129
130 typedef geom::DiagMatrix<T,L,N> return_t;
131
132 // diagonal destination matrix
133 template <typename U, index_t I, index_t J>
134 static void mul(geom::DiagMatrix<U,I,J>* d, const geom::DiagMatrix<T,L,M1>& a, const geom::DiagMatrix<T,M2,N>& b) {
135 T* pa = a.diagonal_begin();
136 T* pb = b.diagonal_begin();
137 T* pc = d->diagonal_begin();
138 for (index_t i = 0; i < a.rows(); ++i, ++pa, ++pb, ++pc) {
139 *pc = (*pa) * (*pb);
140 }
141 }
142
143 template <typename U, index_t I, index_t J>
144 static void mul_acc(geom::DiagMatrix<U,I,J>* d, const geom::DiagMatrix<T,L,M1>& a, const geom::DiagMatrix<T,M2,N>& b) {
145 T* pa = a.diagonal_begin();
146 T* pb = b.diagonal_begin();
147 T* pc = d->diagonal_begin();
148 for (index_t i = 0; i < a.rows(); ++i, ++pa, ++pb, ++pc) {
149 *pc += (*pa) * (*pb);
150 }
151 }
152
153 // arbitrary destination matrix
154 template <typename Md>
155 static void mul(Md* d, const geom::DiagMatrix<T,L,M1>& a, const geom::DiagMatrix<T,M2,N>& b) {
156 T* pa = a.diagonal_begin();
157 T* pb = b.diagonal_begin();
158 d->set_zero();
159 for (index_t i = 0; i < a.rows(); ++i, ++pa, ++pb) {
160 d->set(i, i, (*pa) * (*pb));
161 }
162 }
163
164 template <typename Md>
165 static void mul_acc(Md* d, const geom::DiagMatrix<T,L,M1>& a, const geom::DiagMatrix<T,M2,N>& b) {
166 T* pa = a.diagonal_begin();
167 T* pb = b.diagonal_begin();
168 for (index_t i = 0; i < a.rows(); ++i, ++pa, ++pb) {
169 d->set(i, i, (*d)(i,i) + (*pa) * (*pb));
170 }
171 }
172};

326template <index_t M, index_t N>
327class _ImplMtxMul<geom::PermutationMatrix<M>, geom::PermutationMatrix<N>, REQUIRE_MATRIX_T(geom::PermutationMatrix<M>)> { // require is not "neccessary", but helps disambiguate template solving
328public:
329
330 typedef geom::PermutationMatrix<M> Ma;
331 typedef geom::PermutationMatrix<N> Mb;
332 typedef geom::PermutationMatrix<((M < N) ? M : N)> return_t;
333
334 template <typename Md>
335 static void mul(Md* d, const Ma& a, const Ma& b) {
336 const index_t* d_a = a.getRowDestinations();
337 const index_t* d_b = b.getRowDestinations();
338
339 d->set_zero();
340 for (index_t i = 0; i < a.rows(); i++) {
341 d->set(i, d_a[d_b[i]], 1);
342 }
343 }
344
345 template <index_t K>
346 static void mul(geom::PermutationMatrix<K>* d, const Ma& a, const Mb& b) {
347 const index_t* s_a = a.getRowSources();
348 const index_t* s_b = b.getRowSources();
349 const index_t* d_a = a.getRowDestinations();
350 const index_t* d_b = b.getRowDestinations();
351
352 index_t* d_rd = d->getSrcData();
353 index_t* d_cd = d->getDstData();
354
355 // the best way to understand this is to draw a picture with little
356 // arrows and then look at it.
357 for (index_t i = 0; i < a.rows(); i++) {
358 d_rd[i] = s_b[s_a[i]];
359 d_cd[i] = d_a[d_b[i]];
360 }
361 }
362};
