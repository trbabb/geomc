447  template <typename I, typename O, typename A, typename B> ExprPtr(I,O) binopx(ExprPtr(I,A) a, ExprPtr(I,B) b, O (*func)(A,B), string opname){
448  ExprKindPtr(ConstExpr,I,A) c1;
449  ExprKindPtr(ConstExpr,I,B) c2;
450  if ((c1 = boost::dynamic_pointer_cast<ConstExpr<I,A> >(a)) && (c2 = boost::dynamic_pointer_cast<ConstExpr<I,B> >(b))){
451  return expr<I,O>((*func)(c1->v,c2->v));
452  } else {
453  return ExprPtr(I,O)(new BinopExpr<I,O,A,B>(a,b,func,opname));
454  }
455  }

457  template <typename I, typename O, typename E> ExprPtr(I,O) unopx(ExprPtr(I,E) a, O (*func)(E), string opname){
458  ExprKindPtr(ConstExpr,I,E) c;
459  if ((c = boost::dynamic_pointer_cast<ConstExpr<I,E> >(a)) != NULL){
460  return expr<I,O>((*func)(c->v));
461  } else {
462  return ExprPtr(I,O)(new UnaryOpExpr<I,O,E>(a,func,opname));
463  }
464  }
