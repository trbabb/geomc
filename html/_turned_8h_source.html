52 std::pair<PointType<T,N-1>,T> point_and_angle(Vec<T,N> p) const {
53 T x = p[0];
54 T y = p[N-1];
55 T r = std::sqrt(geom::sum_of_products(x, x, y, y));
56 T theta = std::atan2(y, x);
57 // put theta in 0, 2π:
58 if (theta < 0) {
59 theta += 2 * std::numbers::pi_v<T>;
60 }

62 if constexpr (N == 2) {
63 q = r;
64 } else {
65 p[0] = r;
66 q = p.template resized<N-1>();
67 }

69 }

76 T sdf(Vec<T,N> p) const requires SdfObject<Shape> {
77 auto [q, theta] = point_and_angle(p);
78 T a = canonical_angle();
79 if (theta <= a) {
80 // point is within the sweep arc
81 return shape.sdf(q);
82 } else {
83 // point is near a cap
84 Vec<T,N> p_near_cap = p;
85 T b = a / 2 + std::numbers::pi_v<T>;
86 if (theta < b) {
87 // p is near the high angle cap.
88 // rotate p to have the same relationship to the cross sectional plane.
89 T phi = theta - a;
90 T s = std::sin(phi);
91 T c = std::cos(phi);
92 p_near_cap[0] = c * q[0];
93 p_near_cap[N-1] = s * q[0];
94 }
95 // project the test point to the cross section
96 // and find the distance within the plane
97 PointType<T,N-1> p_plane = p_near_cap.template resized<N-1>();
98 T d = shape.sdf(p_plane);
99 T h = p_near_cap[N-1];
100 // and compute the N-D distance
101 return std::sqrt(geom::sum_of_products(d, d, h, h));
102 }
103 }
