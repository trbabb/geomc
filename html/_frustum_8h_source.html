104 Rect<T,1> intersect(const Ray<T,N>& ray) const {
105 // todo: check N=2 case
106 // s-values for the top and bottom slabs
107 Rect<T,1> slab_range = clipped_height().intersect(
108 Ray<T,1>(ray.origin[N-1], ray.direction[N-1]));
109
110 // we want to project the ray to the h=1 plane. the family of rays that
111 // homogeneously project to a common line on h=1 belong to the plane spanned by
112 // `o` (the ray origin) and `v`, so we will work in that subspace. we want to choose
113 // an `o` that is close to the origin for stability, so we ⟂ project Z+ to the (o,v)
114 // subspace.
115
116 // first make our ray subspace basis orthogonal:
117 Vec<T,N> o_ortho = ray.origin - ray.origin.project_on(ray.direction);
118 Vec<T,N> z_plus;
119 z_plus[N-1] = 1;
120 // project the Z+ axis to the ray subspace. (the point on the projected ray
121 // which is closest to the origin is a multiple of this point)
122 // todo: this can probably be optimized:
123 Vec<T,N> o_base = z_plus.project_on(ray.direction) + z_plus.project_on(o_ortho);
124 // project V to the h=0 plane
125 Vec<T,N> v_base = ray.direction - ray.direction.project_on(o_base);
126
127 // handle degenerate cases
128 if (o_base[N-1] == 0) {
129 // the ray lies in the h=0 plane
130 if (o_base == (T)0) [[unlikely]] {
131 // the ray passes through the origin, and intersects the
132 // frustum tip, if it's included in the height range
133 T s = ray.direction.fraction_on(ray.origin);
134 return Rect<T,1>(s, s) & slab_range;
135 } else {
136 // the ray does not pass through the origin
137 return Rect<T,1>::empty;
138 }
139 } else if (v_base == (T)0) {
140 // ray passes through the origin. the frustum intersection is full iff the ray is
141 // inside the the frustum; else it is empty.
142 return base.contains(o_base.template resized<N-1>())
143 ? slab_range // full range & slab_range
144 : Rect<T,1>::empty; // empty range & slab_range
145 }
146 // put `o` onto the h=1 plane
147 o_base /= o_base[N-1];
148
149 // form the intersection in the base plane
150 Ray<T,N-1> ray_base = Ray<T,N-1>(
151 o_base.template resized<N-1>(),
152 v_base.template resized<N-1>()
153 );
154 Rect<T,1> s = base.intersect(ray_base);
155
156 // short circuit: if the projected ray does not intersect
157 // the base shape, then it does not intersect the frustum.
158 if (s.is_empty()) {
159 return Rect<T,1>::empty;
160 }
161
162 /* find the hit point `p` on the h=0 plane; the final hit point `q` will lie on some
163 multiple of `p`. The trick is to write `p` and `q` as a points in the (o,v) basis,
164 and compute the multiple of `v` that takes us to a multiple of `p`.
165 We are solving this geometrical problem:
166
167 origin
168 • ⎫ ⎫
169 ╱ ╲ │ ⎬ p_y
170 ╱ • p ⎬ 1.0 ⎭
171 ╱ · │
172 o •───────•──> ⎭
173 q v
174
175 ╰──────────╯ 1.0
176 ╰───╯ k * p_x
177 ╰───╯ o_x
178 ╰───────╯ s <<< want to find this
179 */
180 Vec<T,N> lo {ray_base.at_multiple(s.lo), 1};
181 Vec<T,N> hi {ray_base.at_multiple(s.hi), 1};
182 T o_x = ray.origin.fraction_on(ray.direction);
183 T p_lo_y = lo.fraction_on(o_ortho);
184 T p_hi_y = hi.fraction_on(o_ortho);
185 T p_lo_x = lo.fraction_on(ray.direction);
186 T p_hi_x = hi.fraction_on(ray.direction);
187
188 // construct the interval of overlap with the infinite frustum.
189 s = Rect<T,1>::spanning_corners(
190 p_lo_x / p_lo_y - o_x,
191 p_hi_x / p_hi_y - o_x
192 );
193
194 T h0 = ray.origin[N-1] + s.lo * ray.direction[N-1];
195 T h1 = ray.origin[N-1] + s.hi * ray.direction[N-1];
196 if ((h0 < 0) != (h1 < 0)) {
197 // if the two hits are on either side of h=0, then the interval is inverted.
198 // (we have already handled the ray through the origin case).
199 // which of the two half-infinite intervals should be intersected with the slab?
200 s = (slab_range.lo < 0)
201 ? Rect<T,1>(std::numeric_limits<T>::lowest(), s.lo)
202 : Rect<T,1>(s.hi, std::numeric_limits<T>::max());
203 }
204 // intersect the slab with the frustum interval.
205 return s & slab_range;
206 }

208 Vec<T,N> convex_support(Vec<T,N> d) const {
209 Rect<T,1> h = clipped_height();
210 T sign = h.lo < 0 ? -1 : 1; // ← the shape is flipped below the origin
211
212 // find the appropriate direction to check in shape-space.
213 base_point_t d_s = sign * d.template resized<N-1>();
214 if (d_s == (T)0) d_s[0] = 1; // ← pick an arbitrary direction
215
216 // find the relevant point on the base shape, and place it on the h=1 plane
217 Vec<T,N> p(base.convex_support(d_s), 1);
218
219 // top or bottom face?
220 // the support point definitely lies on the line passing through
221 // the origin and `p`. which of the two vertices on that line is the
222 // point? the one in the +p direction, or the -p direction?
223 T z = (p.dot(d) > 0) ? h.hi : h.lo;
224
225 // rescale the point appropriately:
226 return z * p;
227 }
228
229 Rect<T,N> bounds() const {
230 Rect<T,N-1> b0 = base.bounds();
231 Rect<T,1> h = clipped_height();
232 // make two rects for the top and bottom faces;
233 // union them; extrude by the height:
234 return ((b0 * h.lo) | (b0 * h.hi)) * h; // purdy!!
235 }

238 inline Rect<T,1> clipped_height() const {
239 const Rect<T,1>& h = height; // shorthand
240 return Rect<T,1>((h.lo < 0 and h.hi > 0) ? 0 : h.lo, h.hi);
241 }
