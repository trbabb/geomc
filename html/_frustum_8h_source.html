91 if (not clipped_height().contains(p[N-1])) return false;

98 Rect<T,1> intersect(const Ray<T,N>& ray) const {
99 // todo: check N=2 case
100 // s-values for the top and bottom slabs
101 Rect<T,1> slab_range = clipped_height().intersect(
102 Ray<T,1>(ray.origin[N-1], ray.direction[N-1]));
103
104 // we want to project the ray to the h=1 plane. the family of rays that
105 // homogeneously project to a common line on h=1 belong to the plane spanned by
106 // `o` (the ray origin) and `v`, so we will work in that subspace. we want to choose
107 // an `o` that is close to the origin for stability, so we ⟂ project Z+ to the (o,v)
108 // subspace.
109
110 // first make our ray subspace basis orthogonal:
111 Vec<T,N> o_ortho = ray.origin - ray.origin.project_on(ray.direction);
112 Vec<T,N> z_plus;
113 z_plus[N-1] = 1;
114 // project the Z+ axis to the ray subspace. (the point on the projected ray
115 // which is closest to the origin is a multiple of this point)
116 // todo: this can probably be optimized:
117 Vec<T,N> o_base = z_plus.project_on(ray.direction) + z_plus.project_on(o_ortho);
118 // project V to the h=0 plane
119 Vec<T,N> v_base = ray.direction - ray.direction.project_on(o_base);
120
121 // handle degenerate cases
122 if (o_base[N-1] == 0) {
123 // the ray lies in the h=0 plane
124 if (o_base == (T)0) {
125 // the ray passes through the origin, and intersects the
126 // frustum tip, if it's included in the height range
127 T s = ray.direction.fraction_on(ray.origin);
128 return Rect<T,1>(s, s) & slab_range;
129 } else {
130 // the ray does not pass through the origin
131 return Rect<T,1>::empty;
132 }
133 } else if (v_base == (T)0) {
134 // ray passes through the origin. the frustum intersection is full iff the ray is
135 // inside the the frustum; else it is empty.
136 return base.contains(o_base.template resized<N-1>())
137 ? slab_range // full range & slab_range
138 : Rect<T,1>::empty; // empty range & slab_range
139 }
140 // put `o` onto the h=1 plane
141 o_base /= o_base[N-1];
142
143 // form the intersection in the base plane
144 Ray<T,N-1> ray_base = Ray<T,N-1>(
145 o_base.template resized<N-1>(),
146 v_base.template resized<N-1>());
147 Rect<T,1> s = base.intersect(ray_base);
148
149 // short circuit: if the projected ray does not intersect
150 // the base shape, then it does not intersect the frustum.
151 if (s.is_empty()) {
152 return Rect<T,1>::empty;
153 }
154
155 /* find the hit point `p` on the h=0 plane; the final hit point `q` will lie on some
156 multiple of `p`. The trick is to write `p` and `q` as a points in the (o,v) basis,
157 and compute the multiple of `v` that takes us to a multiple of `p`.
158 We are solving this geometrical problem:
159
160 origin
161 • ⎫ ⎫
162 ╱ ╲ │ ⎬ p_y
163 ╱ • p ⎬ 1.0 ⎭
164 ╱ · │
165 o •───────•──> ⎭
166 q v
167
168 ╰──────────╯ 1.0
169 ╰───╯ k * p_x
170 ╰───╯ o_x
171 ╰───────╯ s <<< want to find this
172 */
173 Vec<T,N> lo {ray_base.at_multiple(s.lo), 1};
174 Vec<T,N> hi {ray_base.at_multiple(s.hi), 1};
175 T o_x = ray.origin.fraction_on(ray.direction);
176 T p_lo_y = lo.fraction_on(o_ortho);
177 T p_hi_y = hi.fraction_on(o_ortho);
178 T p_lo_x = lo.fraction_on(ray.direction);
179 T p_hi_x = hi.fraction_on(ray.direction);
180
181 // construct the interval of overlap with the infinite frustum.
182 s = Rect<T,1>::spanning_corners(
183 p_lo_x / p_lo_y - o_x,
184 p_hi_x / p_hi_y - o_x);
185
186 T h0 = ray.origin[N-1] + s.lo * ray.direction[N-1];
187 T h1 = ray.origin[N-1] + s.hi * ray.direction[N-1];
188 if ((h0 < 0) != (h1 < 0)) {
189 // if the two hits are on either side of h=0, then the interval is inverted.
190 // (we have already handled the ray through the origin case).
191 // which of the two half-infinite intervals should be intersected with the slab?
192 s = (slab_range.lo < 0)
193 ? Rect<T,1>(std::numeric_limits<T>::lowest(), s.lo)
194 : Rect<T,1>(s.hi, std::numeric_limits<T>::max());
195 }
196 // intersect the slab with the frustum interval.
197 return s & slab_range;
198 }

200 Vec<T,N> convex_support(Vec<T,N> d) const {
201 Rect<T,1> h = clipped_height();
202 T sign = h.lo < 0 ? -1 : 1; // ← the shape is flipped below the origin
203
204 // find the appropriate direction to check in shape-space.
205 base_point_t d_s = sign * d.template resized<N-1>();
206 if (d_s == (T)0) d_s[0] = 1; // ← pick an arbitrary direction
207
208 // find the relevant point on the base shape, and place it on the h=1 plane
209 Vec<T,N> p(base.convex_support(d_s), 1);
210
211 // top or bottom face?
212 // the support point definitely lies on the line passing through
213 // the origin and `p`. which of the two vertices on that line is the
214 // point? the one in the +p direction, or the -p direction?
215 T z = (p.dot(d) > 0) ? h.hi : h.lo;
216
217 // rescale the point appropriately:
218 return z * p;
219 }
220
221 Rect<T,N> bounds() const {
222 Rect<T,N-1> b0 = base.bounds();
223 Rect<T,1> h = clipped_height();
224 // make two rects for the top and bottom faces;
225 // union them; extrude by the height:
226 return ((b0 * h.lo) | (b0 * h.hi)) * h; // purdy!!
227 }
228
230 inline Rect<T,1> clipped_height() const {
231 const Rect<T,1>& h = height; // shorthand
232 return Rect<T,1>((h.lo < 0 and h.hi > 0) ? 0 : h.lo, h.hi);
233 }
