65  Vec<T,N> convexSupport(Vec<T,N> d) const {
66  typedef PointType<T,N-1> Pt;
67 
68  d = xf.applyInverseNormal(d);
69 
70  T hmax = height.max();
71  T hmin = height.min();
72  T sign = (hmax < 0) ? -1 : 1;
73 
74  // find the base corner
75  Vec<T,N> p;
76  for (index_t i = 0; i < N-1; i++) {
77  if (sign * d[i] > 0) p[i] = Pt::iterator(base.max())[i];
78  else p[i] = Pt::iterator(base.min())[i];
79  }
80  p[N-1] = 1;
81 
82  // top or bottom corner?
83  p *= (p.dot(d) > 0) ? hmax : ((hmax > 0 and hmin < 0) ? 0 : hmin);
84 
85  return xf * p;
86  }

93  void getCorners(Vec<T,N> p[1 << N]) const {
94  typedef PointType<T,N-1> Pt;
95  typename Pt::point_t extreme[2] = { base.min(), base.max() };
96 
97  const T hmax = height.max();
98  T hmin = height.min();
99  hmin = (hmax > 0 and hmin < 0) ? 0 : hmin;
100 
101  for (index_t k = 0; k < 2; k++) {
102  // for lower and upper extents
103  T h = k == 0 ? hmin : hmax;
104  for (index_t c = 0; c < (1 << (N-1)); c++) {
105  // for each corner of the base rect
106  Vec<T,N> pt;
107  for (index_t i = 0; i < N-1; i++) {
108  // for each coordinate of the corner point
109  pt[i] = Pt::iterator(extreme[(c & (1 << i)) != 0])[i];
110  }
111  pt[N-1] = 1;
112  pt *= h;
113  pt = xf * pt;
114  p[c + ((k > 0) ? (1 <<(N-1)) : 0)] = pt;
115  }
116  }
