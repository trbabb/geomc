8// todo: point projection and sdf is unsolved
9// - can't project point to base plane and recurse
10// - example case: consider a shape far from origin
11// - pt will project to the wall directly "underneath" it, in the limit
12// - it's not just the Z component that's distorted; the direction to the shape
13// surface is also distorted
14// - in general, projection vector field is "stretched" away from origin by this effect
15// - answer: we can perform the projection if we can project to the surface of a
16// *perspective transformation* of the base shape. that is, put the base shape on a piece
17// of glass, shine a point light at it, and let the shadow fall onto an arbitrarily-oriented
18// plane. We want to project to the shape of the shadow.
19// - the plane we are projecting to is sensitive to the point we are testing. that is, the
20// plane is the one normal to the test point vector (i.e., a vector with its tail at
21// the origin / frustum tip). proof is too small for margin to contain.
22// - sdf() draws a circle around the point which contains none of the shape;
23// this circle becomes a conic section in the perspective-transformed space. the
24// actual closest point may be inside the circle that circumscribes the conic section;
25// we do not know in which direction that point lies from sdf() or project() alone.
26// - in general this cannot be written in terms of the SDF or projection of the base shape
27// without some kind of iterative solution.

108 Rect<T,1> intersect(const Ray<T,N>& ray) const requires RayIntersectableObject<Shape> {
109 // todo: check N=2 case
110 // s-values for the top and bottom slabs
111 Rect<T,1> slab_range = clipped_height().intersect(
112 Ray<T,1>(ray.origin[N-1], ray.direction[N-1]));
113
114 // we want to project the ray to the h=1 plane. the family of rays that
115 // homogeneously project to a common line on h=1 belong to the plane spanned by
116 // `o` (the ray origin) and `v`, so we will work in that subspace. we want to choose
117 // an `o` that is close to the origin for stability, so we ⟂ project Z+ to the (o,v)
118 // subspace.
119
120 // first make our ray subspace basis orthogonal:
121 Vec<T,N> o_ortho = ray.origin - ray.origin.project_on(ray.direction);
122 Vec<T,N> z_plus;
123 z_plus[N-1] = 1;
124 // project the Z+ axis to the ray subspace. (the point on the projected ray
125 // which is closest to the origin is a multiple of this point)
126 // todo: this can probably be optimized:
127 Vec<T,N> o_base = z_plus.project_on(ray.direction) + z_plus.project_on(o_ortho);
128 // project V to the h=0 plane
129 Vec<T,N> v_base = ray.direction - ray.direction.project_on(o_base);
130
131 // handle degenerate cases
132 if (o_base[N-1] == 0) {
133 // the ray lies in the h=0 plane
134 if (o_base == (T)0) [[unlikely]] {
135 // the ray passes through the origin, and intersects the
136 // frustum tip, if it's included in the height range
137 T s = ray.direction.fraction_on(ray.origin);
138 return Rect<T,1>(s, s) & slab_range;
139 } else {
140 // the ray does not pass through the origin
141 return Rect<T,1>::empty;
142 }
143 } else if (v_base == (T)0) {
144 // ray passes through the origin. the frustum intersection is full iff the ray is
145 // inside the the frustum; else it is empty.
146 return base.contains(o_base.template resized<N-1>())
147 ? slab_range // full range & slab_range
148 : Rect<T,1>::empty; // empty range & slab_range
149 }
150 // put `o` onto the h=1 plane
151 o_base /= o_base[N-1];
152
153 // form the intersection in the base plane
154 Ray<T,N-1> ray_base = Ray<T,N-1>(
155 o_base.template resized<N-1>(),
156 v_base.template resized<N-1>()
157 );
158 Rect<T,1> s = base.intersect(ray_base);
159
160 // short circuit: if the projected ray does not intersect
161 // the base shape, then it does not intersect the frustum.
162 if (s.is_empty()) {
163 return Rect<T,1>::empty;
164 }
165
166 /* find the hit point `p` on the h=0 plane; the final hit point `q` will lie on some
167 multiple of `p`. The trick is to write `p` and `q` as a points in the (o,v) basis,
168 and compute the multiple of `v` that takes us to a multiple of `p`.
169 We are solving this geometrical problem:
170
171 origin
172 • ⎫ ⎫
173 ╱ ╲ │ ⎬ p_y
174 ╱ • p ⎬ 1.0 ⎭
175 ╱ · │
176 o •───────•──> ⎭
177 q v
178
179 ╰──────────╯ 1.0
180 ╰───╯ k * p_x
181 ╰───╯ o_x
182 ╰───────╯ s <<< want to find this
183 */
184 Vec<T,N> lo {ray_base.at_multiple(s.lo), 1};
185 Vec<T,N> hi {ray_base.at_multiple(s.hi), 1};
186 T o_x = ray.origin.fraction_on(ray.direction);
187 T p_lo_y = lo.fraction_on(o_ortho);
188 T p_hi_y = hi.fraction_on(o_ortho);
189 T p_lo_x = lo.fraction_on(ray.direction);
190 T p_hi_x = hi.fraction_on(ray.direction);
191
192 // construct the interval of overlap with the infinite frustum.
193 s = Rect<T,1>::from_corners(
194 p_lo_x / p_lo_y - o_x,
195 p_hi_x / p_hi_y - o_x
196 );
197
198 T h0 = ray.origin[N-1] + s.lo * ray.direction[N-1];
199 T h1 = ray.origin[N-1] + s.hi * ray.direction[N-1];
200 if ((h0 < 0) != (h1 < 0)) {
201 // if the two hits are on either side of h=0, then the interval is inverted.
202 // (we have already handled the ray through the origin case).
203 // which of the two half-infinite intervals should be intersected with the slab?
204 s = (slab_range.lo < 0)
205 ? Rect<T,1>(std::numeric_limits<T>::lowest(), s.lo)
206 : Rect<T,1>(s.hi, std::numeric_limits<T>::max());
207 }
208 // intersect the slab with the frustum interval.
209 return s & slab_range;
210 }

212 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
213 Rect<T,1> h = clipped_height();
214 T sign = h.lo < 0 ? -1 : 1; // ← the shape is flipped below the origin
215
216 // find the appropriate direction to check in shape-space.
217 base_point_t d_s = sign * d.template resized<N-1>();
218 if (d_s == (T)0) d_s[0] = 1; // ← pick an arbitrary direction
219
220 // find the relevant point on the base shape, and place it on the h=1 plane
221 Vec<T,N> p(base.convex_support(d_s), 1);
222
223 // top or bottom face?
224 // the support point definitely lies on the line passing through
225 // the origin and `p`. which of the two vertices on that line is the
226 // point? the one in the +p direction, or the -p direction?
227 T z = (p.dot(d) > 0) ? h.hi : h.lo;
228
229 // rescale the point appropriately:
230 return z * p;
231 }

233 Rect<T,N> bounds() const requires BoundedObject<Shape> {
234 Rect<T,N-1> b0 = base.bounds();
235 Rect<T,1> h = clipped_height();
236 // make two rects for the top and bottom faces;
237 // union them; extrude by the height:
238 return ((b0 * h.lo) | (b0 * h.hi)) * h; // purdy!!
