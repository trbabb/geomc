83  bool contains(Vec<T,N> p) const {
84  if (not clipped_height().contains(p[N-1])) return false;
85  if (p[N-1] == 0) return p.isZero();
86  p /= p[N-1];
87  return base.contains(p.template resized<N-1>());
88  }

91  Vec<T,N> convex_support(Vec<T,N> d) const {
92  Rect<T,1> h = clipped_height();
93  T sign = h.lo < 0 ? -1 : 1; // ← the shape is flipped below the origin
94 
95  // find the appropriate direction to check in shape-space.
96  Vec<T,N-1> d_s = sign * d.template resized<N-1>();
97  if (d_s.isZero()) d_s[0] = 1; // ← pick an arbitrary direction
98 
99  // find the relevant point on the base shape, and place it on the h=1 plane
100  Vec<T,N> p(base.convex_support(d_s), 1);
101 
102  // top or bottom face?
103  // the support point definitely lies on the line passing through
104  // the origin and `p`. which of the two vertices on that line is the
105  // point? the one in the +p direction, or the -p direction?
106  T z = (p.dot(d) > 0) ? h.hi : h.lo;
107 
108  // rescale the point appropriately:
109  return z * p;
110  }
111 
112 
113  Rect<T,N> bounds() const {
114  Rect<T,N-1> b0 = base.bounds();
115  Rect<T,1> h = clipped_height();
116  // make two rects for the top and bottom faces;
117  // union them; extrude by the height:
118  return ((b0 * h.lo) | (b0 * h.hi)) * h; // purdy!!
119  }
