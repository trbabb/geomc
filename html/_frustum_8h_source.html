8// todo: point projection and sdf is unsolved
9// - can't project point to base plane and recurse
10// - example case: consider a shape far from origin
11// - pt will project to the wall directly "underneath" it, in the limit
12// - it's not just the Z component that's distorted; the direction to the shape
13// surface is also distorted
14// - in general, projection vector field is "stretched" away from origin by this effect
15// - answer: we can perform the projection if we can project to the surface of a
16// *perspective transformation* of the base shape. that is, put the base shape on a piece
17// of glass, shine a point light at it, and let the shadow fall onto an arbitrarily-oriented
18// plane. We want to project to the shape of the shadow.
19// - the plane we are projecting to is sensitive to the point we are testing. that is, the
20// plane is the one normal to the test point vector (i.e., a vector with its tail at
21// the origin / frustum tip). proof is too small for margin to contain.
22// - sdf() draws a circle around the point which contains none of the shape;
23// this circle becomes a conic section in the perspective-transformed space. the
24// actual closest point may be inside the circle that circumscribes the conic section;
25// we do not know in which direction that point lies from sdf() or project() alone.
26// - in general this cannot be written in terms of the SDF or projection of the base shape
27// without some kind of iterative solution.

111 Rect<T,1> intersect(const Ray<T,N>& ray) const requires RayIntersectableObject<Shape> {
112 // todo: check N=2 case
113 // s-values for the top and bottom slabs
114 Rect<T,1> slab_range = clipped_height().intersect(
115 Ray<T,1>(ray.origin[N-1], ray.direction[N-1]));
116
117 // we want to project the ray to the h=1 plane. the family of rays that
118 // homogeneously project to a common line on h=1 belong to the plane spanned by
119 // `o` (the ray origin) and `v`, so we will work in that subspace. we want to choose
120 // an `o` that is close to the origin for stability, so we ⟂ project Z+ to the (o,v)
121 // subspace.
122
123 // first make our ray subspace basis orthogonal:
124 Vec<T,N> o_ortho = ray.origin - ray.origin.project_on(ray.direction);
125 Vec<T,N> z_plus;
126 z_plus[N-1] = 1;
127 // project the Z+ axis to the ray subspace. (the point on the projected ray
128 // which is closest to the origin is a multiple of this point)
129 // todo: this can probably be optimized:
130 Vec<T,N> o_base = z_plus.project_on(ray.direction) + z_plus.project_on(o_ortho);
131 // project V to the h=0 plane
132 Vec<T,N> v_base = ray.direction - ray.direction.project_on(o_base);
133
134 // handle degenerate cases
135 if (o_base[N-1] == 0) {
136 // the ray lies in the h=0 plane
137 if (o_base == (T)0) [[unlikely]] {
138 // the ray passes through the origin, and intersects the
139 // frustum tip, if it's included in the height range
140 T s = ray.direction.fraction_on(ray.origin);
141 return Rect<T,1>(s, s) & slab_range;
142 } else {
143 // the ray does not pass through the origin
144 return Rect<T,1>::empty;
145 }
146 } else if (v_base == (T)0) {
147 // ray passes through the origin. the frustum intersection is full iff the ray is
148 // inside the the frustum; else it is empty.
149 return base.contains(o_base.template resized<N-1>())
150 ? slab_range // full range & slab_range
151 : Rect<T,1>::empty; // empty range & slab_range
152 }
153 // put `o` onto the h=1 plane
154 o_base /= o_base[N-1];
155
156 // form the intersection in the base plane
157 Ray<T,N-1> ray_base = Ray<T,N-1>(
158 o_base.template resized<N-1>(),
159 v_base.template resized<N-1>()
160 );
161 Rect<T,1> s = base.intersect(ray_base);
162
163 // short circuit: if the projected ray does not intersect
164 // the base shape, then it does not intersect the frustum.
165 if (s.is_empty()) {
166 return Rect<T,1>::empty;
167 }
168
169 /* find the hit point `p` on the h=0 plane; the final hit point `q` will lie on some
170 multiple of `p`. The trick is to write `p` and `q` as a points in the (o,v) basis,
171 and compute the multiple of `v` that takes us to a multiple of `p`.
172 We are solving this geometrical problem:
173
174 origin
175 • ⎫ ⎫
176 ╱ ╲ │ ⎬ p_y
177 ╱ • p ⎬ 1.0 ⎭
178 ╱ · │
179 o •───────•──> ⎭
180 q v
181
182 ╰──────────╯ 1.0
183 ╰───╯ k * p_x
184 ╰───╯ o_x
185 ╰───────╯ s <<< want to find this
186 */
187 Vec<T,N> lo {ray_base.at_multiple(s.lo), 1};
188 Vec<T,N> hi {ray_base.at_multiple(s.hi), 1};
189 T o_x = ray.origin.fraction_on(ray.direction);
190 T p_lo_y = lo.fraction_on(o_ortho);
191 T p_hi_y = hi.fraction_on(o_ortho);
192 T p_lo_x = lo.fraction_on(ray.direction);
193 T p_hi_x = hi.fraction_on(ray.direction);
194
195 // construct the interval of overlap with the infinite frustum.
196 s = Rect<T,1>::from_corners(
197 p_lo_x / p_lo_y - o_x,
198 p_hi_x / p_hi_y - o_x
199 );
200
201 T h0 = ray.origin[N-1] + s.lo * ray.direction[N-1];
202 T h1 = ray.origin[N-1] + s.hi * ray.direction[N-1];
203 if ((h0 < 0) != (h1 < 0)) {
204 // if the two hits are on either side of h=0, then the interval is inverted.
205 // (we have already handled the ray through the origin case).
206 // which of the two half-infinite intervals should be intersected with the slab?
207 s = (slab_range.lo < 0)
208 ? Rect<T,1>(std::numeric_limits<T>::lowest(), s.lo)
209 : Rect<T,1>(s.hi, std::numeric_limits<T>::max());
210 }
211 // intersect the slab with the frustum interval.
212 return s & slab_range;
213 }

215 Vec<T,N> convex_support(Vec<T,N> d) const requires ConvexObject<Shape> {
216 Rect<T,1> h = clipped_height();
217 T sign = h.lo < 0 ? -1 : 1; // ← the shape is flipped below the origin
218
219 // find the appropriate direction to check in shape-space.
220 base_point_t d_s = sign * d.template resized<N-1>();
221 if (d_s == (T)0) d_s[0] = 1; // ← pick an arbitrary direction
222
223 // find the relevant point on the base shape, and place it on the h=1 plane
224 Vec<T,N> p(base.convex_support(d_s), 1);
225
226 // top or bottom face?
227 // the support point definitely lies on the line passing through
228 // the origin and `p`. which of the two vertices on that line is the
229 // point? the one in the +p direction, or the -p direction?
230 T z = (p.dot(d) > 0) ? h.hi : h.lo;
231
232 // rescale the point appropriately:
233 return z * p;
234 }

236 Rect<T,N> bounds() const requires BoundedObject<Shape> {
237 Rect<T,N-1> b0 = base.bounds();
238 Rect<T,1> h = clipped_height();
239 // make two rects for the top and bottom faces;
240 // union them; extrude by the height:
241 return ((b0 * h.lo) | (b0 * h.hi)) * h; // purdy!!
242 }

245 inline Rect<T,1> clipped_height() const {
246 const Rect<T,1>& h = height; // shorthand
247 return Rect<T,1>((h.lo < 0 and h.hi > 0) ? 0 : h.lo, h.hi);
248 }
