47  Rect<T,N> bounds() {
48  // faster than using convexSupport, somewhat surprisingly.
49  Vec<T,N> pts[2] = {box.min(), box.max()};
50  Vec<T,N> lo = xf * box.min();
51  Vec<T,N> hi = lo;
52 
53  // for each corner
54  for (index_t i = 1; i < (1 << N); i++) {
55  Vec<T,N> p;
56  // for each coordinate
57  for (index_t j = 0; j < N; j++) {
58  p[j] = pts[(i & (1 << j)) != 0][j];
59  }
60  p = xf * p;
61  lo = std::min(lo, p);
62  hi = std::max(hi, p);
63  }
64 
65  return Rect<T,N>(lo, hi);
66  }

69  Vec<T,N> convexSupport(Vec<T,N> d) const {
70  Vec<T,N> d_body = xf.applyInverseNormal(d);
71  Vec<T,N> o;
72  for (index_t i = 0; i < N; i++) {
73  o[i] = ((d_body[i] < 0) ? box.min() : box.max())[i];
74  }
75  return xf * o;
76  }

84  void getCorners(Vec<T,N> out[1 << N]) {
85  const static index_t n_corners = 1 << N;
86  Vec<T,N> extreme[2] = { box.min(), box.max() };
87  for (index_t c = 0; c < n_corners; c++) {
88  Vec<T,N> pt;
89  for (index_t i = 0; i < N; i++) {
90  pt[i] = extreme[(c & (1 << i)) != 0][i];
91  }
92  out[c] = xf * pt;
93  }
94  }

119  inline bool intersects(const OrientedRect<T,N> &b1) {
120  // we will make ourselves axis-aligned; this is the same as applying
121  // the inverse of our xf. We apply the inverse of xf to b1 too,
122  // to preserve the relationhip between the us. From this, we
123  // fallback to an ORect <-> Rect test.
124  OrientedRect<T,N> b1_in_b0 = b1 / xf;
125 
126  return b1_in_b0.intersects(box);
127  }

138  Hit<T,N> trace(const Ray<T,N> &r, HitSide side) {
139  Ray<T,N> rbody = r / xf;
140  Hit<T,N> h = box.trace(rbody, side);
141  if (h.hit) {
142  h.p = xf * h.p;
143  h.n = xf.applyNormal(h.n);
144  }
145  return h;
146  }
