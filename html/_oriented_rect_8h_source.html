47  Rect<T,N> bounds() {
48  Vec<T,N> pts[2] = {box.min(), box.max()};
49  Vec<T,N> lo = xf * box.min();
50  Vec<T,N> hi = lo;
51 
52  // for each corner
53  for (index_t i = 1; i < (1 << N); i++) {
54  Vec<T,N> p;
55  // for each coordinate
56  for (index_t j = 0; j < N; j++) {
57  p[j] = pts[(i & (1 << j)) != 0][j];
58  }
59  p = xf * p;
60  lo = std::min(lo, p);
61  hi = std::max(hi, p);
62  }
63 
64  return Rect<T,N>(lo,hi);
65  }

83  void getCorners(Vec<T,N> out[1 << N]) {

118  inline bool intersects(const OrientedRect<T,N> &b1) {
119  // we will make ourselves axis-aligned; this is the same as applying
120  // the inverse of our xf. We apply the inverse of xf to b1 too,
121  // to preserve the relationhip between the us. From this, we
122  // fallback to an ORect <-> Rect test.
123  OrientedRect<T,N> b1_in_b0 = b1 / xf;
